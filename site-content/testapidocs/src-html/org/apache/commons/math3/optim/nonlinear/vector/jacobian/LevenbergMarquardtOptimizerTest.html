172    private void checkEstimate(ModelFunction problem,
173                               ModelFunctionJacobian problemJacobian,
174                               double initialStepBoundFactor, int maxCostEval,
175                               double costRelativeTolerance, double parRelativeTolerance,
176                               double orthoTolerance, boolean shouldFail) {
177        try {
178            LevenbergMarquardtOptimizer optimizer
179                = new LevenbergMarquardtOptimizer(initialStepBoundFactor,
180                                                  costRelativeTolerance,
181                                                  parRelativeTolerance,
182                                                  orthoTolerance,
183                                                  Precision.SAFE_MIN);
184            optimizer.optimize(new MaxEval(maxCostEval),
185                               problem,
186                               problemJacobian,
187                               new Target(new double[] { 0, 0, 0, 0, 0 }),
188                               new Weight(new double[] { 1, 1, 1, 1, 1 }),
189                               new InitialGuess(new double[] { 98.680, 47.345 }));
190            Assert.assertTrue(!shouldFail);
191        } catch (DimensionMismatchException ee) {
192            Assert.assertTrue(shouldFail);
193        } catch (TooManyEvaluationsException ee) {
194            Assert.assertTrue(shouldFail);
195        }
196    }

199     * Non-linear test case: fitting of decay curve (from Chapter 8 of
200     * Bevington's textbook, "Data reduction and analysis for the physical sciences").
201     * XXX The expected ("reference") values may not be accurate and the tolerance too
202     * relaxed for this test to be currently really useful (the issue is under
203     * investigation).

205    @Test
206    public void testBevington() {
207        final double[][] dataPoints = {
208            // column 1 = times
209            { 15, 30, 45, 60, 75, 90, 105, 120, 135, 150,
210              165, 180, 195, 210, 225, 240, 255, 270, 285, 300,
211              315, 330, 345, 360, 375, 390, 405, 420, 435, 450,
212              465, 480, 495, 510, 525, 540, 555, 570, 585, 600,
213              615, 630, 645, 660, 675, 690, 705, 720, 735, 750,
214              765, 780, 795, 810, 825, 840, 855, 870, 885, },
215            // column 2 = measured counts
216            { 775, 479, 380, 302, 185, 157, 137, 119, 110, 89,
217              74, 61, 66, 68, 48, 54, 51, 46, 55, 29,
218              28, 37, 49, 26, 35, 29, 31, 24, 25, 35,
219              24, 30, 26, 28, 21, 18, 20, 27, 17, 17,
220              14, 17, 24, 11, 22, 17, 12, 10, 13, 16,
221              9, 9, 14, 21, 17, 13, 12, 18, 10, },
222        };
223
224        final BevingtonProblem problem = new BevingtonProblem();
225
226        final int len = dataPoints[0].length;
227        final double[] weights = new double[len];
228        for (int i = 0; i < len; i++) {
229            problem.addPoint(dataPoints[0][i],
230                             dataPoints[1][i]);
231
232            weights[i] = 1 / dataPoints[1][i];
233        }
234
235        final LevenbergMarquardtOptimizer optimizer
236            = new LevenbergMarquardtOptimizer();
237
238        final PointVectorValuePair optimum
239            = optimizer.optimize(new MaxEval(100),
240                                 problem.getModelFunction(),
241                                 problem.getModelFunctionJacobian(),
242                                 new Target(dataPoints[1]),
243                                 new Weight(weights),
244                                 new InitialGuess(new double[] { 10, 900, 80, 27, 225 }));
245
246        final double[] solution = optimum.getPoint();
247        final double[] expectedSolution = { 10.4, 958.3, 131.4, 33.9, 205.0 };
248
249        final double[][] covarMatrix = optimizer.computeCovariances(solution, 1e-14);
250        final double[][] expectedCovarMatrix = {
251            { 3.38, -3.69, 27.98, -2.34, -49.24 },
252            { -3.69, 2492.26, 81.89, -69.21, -8.9 },
253            { 27.98, 81.89, 468.99, -44.22, -615.44 },
254            { -2.34, -69.21, -44.22, 6.39, 53.80 },
255            { -49.24, -8.9, -615.44, 53.8, 929.45 }
256        };
257
258        final int numParams = expectedSolution.length;
259
260        // Check that the computed solution is within the reference error range.
261        for (int i = 0; i < numParams; i++) {
262            final double error = FastMath.sqrt(expectedCovarMatrix[i][i]);
263            Assert.assertEquals("Parameter " + i, expectedSolution[i], solution[i], error);
264        }
265
266        // Check that each entry of the computed covariance matrix is within 10%
267        // of the reference matrix entry.
268        for (int i = 0; i < numParams; i++) {
269            for (int j = 0; j < numParams; j++) {
270                Assert.assertEquals("Covariance matrix [" + i + "][" + j + "]",
271                                    expectedCovarMatrix[i][j],
272                                    covarMatrix[i][j],
273                                    FastMath.abs(0.1 * expectedCovarMatrix[i][j]));
274            }
275        }
276    }

278    @Test
279    public void testCircleFitting2() {
280        final double xCenter = 123.456;
281        final double yCenter = 654.321;
282        final double xSigma = 10;
283        final double ySigma = 15;
284        final double radius = 111.111;
285        // The test is extremely sensitive to the seed.
286        final long seed = 59421061L;
287        final RandomCirclePointGenerator factory
288            = new RandomCirclePointGenerator(xCenter, yCenter, radius,
289                                             xSigma, ySigma,
290                                             seed);
291        final CircleProblem circle = new CircleProblem(xSigma, ySigma);
292
293        final int numPoints = 10;
294        for (Vector2D p : factory.generate(numPoints)) {
295            circle.addPoint(p.getX(), p.getY());
296        }
297
298        // First guess for the center's coordinates and radius.
299        final double[] init = { 90, 659, 115 };
300
301        final LevenbergMarquardtOptimizer optimizer
302            = new LevenbergMarquardtOptimizer();
303        final PointVectorValuePair optimum = optimizer.optimize(new MaxEval(100),
304                                                                circle.getModelFunction(),
305                                                                circle.getModelFunctionJacobian(),
306                                                                new Target(circle.target()),
307                                                                new Weight(circle.weight()),
308                                                                new InitialGuess(init));
309
310        final double[] paramFound = optimum.getPoint();
311
312        // Retrieve errors estimation.
313        final double[] asymptoticStandardErrorFound = optimizer.computeSigma(paramFound, 1e-14);
314
315        // Check that the parameters are found within the assumed error bars.
316        Assert.assertEquals(xCenter, paramFound[0], asymptoticStandardErrorFound[0]);
317        Assert.assertEquals(yCenter, paramFound[1], asymptoticStandardErrorFound[1]);
318        Assert.assertEquals(radius, paramFound[2], asymptoticStandardErrorFound[2]);
319    }

321    private static class BevingtonProblem {
322        private List<Double> time;
323        private List<Double> count;
324
325        public BevingtonProblem() {
326            time = new ArrayList<Double>();
327            count = new ArrayList<Double>();
328        }
329
330        public void addPoint(double t, double c) {
331            time.add(t);
332            count.add(c);
333        }
334
335        public ModelFunction getModelFunction() {
336            return new ModelFunction(new MultivariateVectorFunction() {
337                    public double[] value(double[] params) {
338                        double[] values = new double[time.size()];
339                        for (int i = 0; i < values.length; ++i) {
340                            final double t = time.get(i);
341                            values[i] = params[0] +
342                                params[1] * FastMath.exp(-t / params[3]) +
343                                params[2] * FastMath.exp(-t / params[4]);
344                        }
345                        return values;
346                    }
347                });
348        }
349
350        public ModelFunctionJacobian getModelFunctionJacobian() {
351            return new ModelFunctionJacobian(new MultivariateMatrixFunction() {
352                    public double[][] value(double[] params) {
353                        double[][] jacobian = new double[time.size()][5];
354
355                        for (int i = 0; i < jacobian.length; ++i) {
356                            final double t = time.get(i);
357                            jacobian[i][0] = 1;
358
359                            final double p3 =  params[3];
360                            final double p4 =  params[4];
361                            final double tOp3 = t / p3;
362                            final double tOp4 = t / p4;
363                            jacobian[i][1] = FastMath.exp(-tOp3);
364                            jacobian[i][2] = FastMath.exp(-tOp4);
365                            jacobian[i][3] = params[1] * FastMath.exp(-tOp3) * tOp3 / p3;
366                            jacobian[i][4] = params[2] * FastMath.exp(-tOp4) * tOp4 / p4;
367                        }
368                        return jacobian;
369                    }
370                });
371        }
372    }
