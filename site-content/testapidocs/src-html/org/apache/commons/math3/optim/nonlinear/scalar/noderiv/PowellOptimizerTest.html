036    @Test(expected=MathUnsupportedOperationException.class)
037    public void testBoundsUnsupported() {
038        final MultivariateFunction func = new SumSincFunction(-1);
039        final PowellOptimizer optim = new PowellOptimizer(1e-8, 1e-5,
040                                                          1e-4, 1e-4);
041
042        optim.optimize(new MaxEval(100),
043                       new ObjectiveFunction(func),
044                       GoalType.MINIMIZE,
045                       new InitialGuess(new double[] { -3, 0 }),
046                       new SimpleBounds(new double[] { -5, -1 },
047                                        new double[] { 5, 1 }));
048    }

050    @Test
051    public void testSumSinc() {
052        final MultivariateFunction func = new SumSincFunction(-1);
053
054        int dim = 2;
055        final double[] minPoint = new double[dim];
056        for (int i = 0; i < dim; i++) {
057            minPoint[i] = 0;
058        }
059
060        double[] init = new double[dim];
061
062        // Initial is minimum.
063        for (int i = 0; i < dim; i++) {
064            init[i] = minPoint[i];
065        }
066        doTest(func, minPoint, init, GoalType.MINIMIZE, 1e-9, 1e-9);
067
068        // Initial is far from minimum.
069        for (int i = 0; i < dim; i++) {
070            init[i] = minPoint[i] + 3;
071        }
072        doTest(func, minPoint, init, GoalType.MINIMIZE, 1e-9, 1e-5);
073        // More stringent line search tolerance enhances the precision
074        // of the result.
075        doTest(func, minPoint, init, GoalType.MINIMIZE, 1e-9, 1e-9, 1e-7);
076    }

140    /**
141     * Ensure that we do not increase the number of function evaluations when
142     * the function values are scaled up.
143     * Note that the tolerances parameters passed to the constructor must
144     * still hold sensible values because they are used to set the line search
145     * tolerances.
146     */
147    @Test
148    public void testRelativeToleranceOnScaledValues() {
149        final MultivariateFunction func = new MultivariateFunction() {
150                public double value(double[] x) {
151                    final double a = x[0] - 1;
152                    final double b = x[1] - 1;
153                    return a * a * FastMath.sqrt(FastMath.abs(a)) + b * b + 1;
154                }
155            };
156
157        int dim = 2;
158        final double[] minPoint = new double[dim];
159        for (int i = 0; i < dim; i++) {
160            minPoint[i] = 1;
161        }
162
163        double[] init = new double[dim];
164        // Initial is far from minimum.
165        for (int i = 0; i < dim; i++) {
166            init[i] = minPoint[i] - 20;
167        }
168
169        final double relTol = 1e-10;
170
171        final int maxEval = 1000;
172        // Very small absolute tolerance to rely solely on the relative
173        // tolerance as a stopping criterion
174        final PowellOptimizer optim = new PowellOptimizer(relTol, 1e-100);
175
176        final PointValuePair funcResult = optim.optimize(new MaxEval(maxEval),
177                                                         new ObjectiveFunction(func),
178                                                         GoalType.MINIMIZE,
179                                                         new InitialGuess(init));
180        final double funcValue = func.value(funcResult.getPoint());
181        final int funcEvaluations = optim.getEvaluations();
182
183        final double scale = 1e10;
184        final MultivariateFunction funcScaled = new MultivariateFunction() {
185                public double value(double[] x) {
186                    return scale * func.value(x);
187                }
188            };
189
190        final PointValuePair funcScaledResult = optim.optimize(new MaxEval(maxEval),
191                                                               new ObjectiveFunction(funcScaled),
192                                                               GoalType.MINIMIZE,
193                                                               new InitialGuess(init));
194        final double funcScaledValue = funcScaled.value(funcScaledResult.getPoint());
195        final int funcScaledEvaluations = optim.getEvaluations();
196
197        // Check that both minima provide the same objective funciton values,
198        // within the relative function tolerance.
199        Assert.assertEquals(1, funcScaledValue / (scale * funcValue), relTol);
200
201        // Check that the numbers of evaluations are the same.
202        Assert.assertEquals(funcEvaluations, funcScaledEvaluations);
203    }

205    /**
206     * @param func Function to optimize.
207     * @param optimum Expected optimum.
208     * @param init Starting point.
209     * @param goal Minimization or maximization.
210     * @param fTol Tolerance (relative error on the objective function) for
211     * "Powell" algorithm.
212     * @param pointTol Tolerance for checking that the optimum is correct.
213     */
214    private void doTest(MultivariateFunction func,
215                        double[] optimum,
216                        double[] init,
217                        GoalType goal,
218                        double fTol,
219                        double pointTol) {
220        final PowellOptimizer optim = new PowellOptimizer(fTol, Math.ulp(1d));
221
222        final PointValuePair result = optim.optimize(new MaxEval(1000),
223                                                     new ObjectiveFunction(func),
224                                                     goal,
225                                                     new InitialGuess(init));
226        final double[] point = result.getPoint();
227
228        for (int i = 0, dim = optimum.length; i < dim; i++) {
229            Assert.assertEquals("found[" + i + "]=" + point[i] + " value=" + result.getValue(),
230                                optimum[i], point[i], pointTol);
231        }
232    }

234    /**
235     * @param func Function to optimize.
236     * @param optimum Expected optimum.
237     * @param init Starting point.
238     * @param goal Minimization or maximization.
239     * @param fTol Tolerance (relative error on the objective function) for
240     * "Powell" algorithm.
241     * @param fLineTol Tolerance (relative error on the objective function)
242     * for the internal line search algorithm.
243     * @param pointTol Tolerance for checking that the optimum is correct.
244     */
245    private void doTest(MultivariateFunction func,
246                        double[] optimum,
247                        double[] init,
248                        GoalType goal,
249                        double fTol,
250                        double fLineTol,
251                        double pointTol) {
252        final PowellOptimizer optim = new PowellOptimizer(fTol, Math.ulp(1d),
253                                                          fLineTol, Math.ulp(1d));
254
255        final PointValuePair result = optim.optimize(new MaxEval(1000),
256                                                     new ObjectiveFunction(func),
257                                                     goal,
258                                                     new InitialGuess(init));
259        final double[] point = result.getPoint();
260
261        for (int i = 0, dim = optimum.length; i < dim; i++) {
262            Assert.assertEquals("found[" + i + "]=" + point[i] + " value=" + result.getValue(),
263                                optimum[i], point[i], pointTol);
264        }
265
266        Assert.assertTrue(optim.getIterations() > 0);
267    }
