036public class MultiStartMultivariateOptimizerTest {
037    @Test
038    public void testCircleFitting() {
039        CircleScalar circle = new CircleScalar();
040        circle.addPoint( 30.0,  68.0);
041        circle.addPoint( 50.0,  -6.0);
042        circle.addPoint(110.0, -20.0);
043        circle.addPoint( 35.0,  15.0);
044        circle.addPoint( 45.0,  97.0);
045        // TODO: the wrapper around NonLinearConjugateGradientOptimizer is a temporary hack for
046        // version 3.1 of the library. It should be removed when NonLinearConjugateGradientOptimizer
047        // will officially be declared as implementing MultivariateDifferentiableOptimizer
048        GradientMultivariateOptimizer underlying
049            = new NonLinearConjugateGradientOptimizer(NonLinearConjugateGradientOptimizer.Formula.POLAK_RIBIERE,
050                                                      new SimpleValueChecker(1e-10, 1e-10));
051        JDKRandomGenerator g = new JDKRandomGenerator();
052        g.setSeed(753289573253l);
053        RandomVectorGenerator generator
054            = new UncorrelatedRandomVectorGenerator(new double[] { 50, 50 },
055                                                    new double[] { 10, 10 },
056                                                    new GaussianRandomGenerator(g));
057        int nbStarts = 10;
058        MultiStartMultivariateOptimizer optimizer
059            = new MultiStartMultivariateOptimizer(underlying, nbStarts, generator);
060        PointValuePair optimum
061            = optimizer.optimize(new MaxEval(1000),
062                                 circle.getObjectiveFunction(),
063                                 circle.getObjectiveFunctionGradient(),
064                                 GoalType.MINIMIZE,
065                                 new InitialGuess(new double[] { 98.680, 47.345 }));
066        Assert.assertEquals(1000, optimizer.getMaxEvaluations());
067        PointValuePair[] optima = optimizer.getOptima();
068        Assert.assertEquals(nbStarts, optima.length);
069        for (PointValuePair o : optima) {
070            // we check the results of all intermediate restarts here (there are 10 such results)
071            Vector2D center = new Vector2D(o.getPointRef()[0], o.getPointRef()[1]);
072            Assert.assertTrue(69.9592 < circle.getRadius(center));
073            Assert.assertTrue(69.9602 > circle.getRadius(center));
074            Assert.assertTrue(96.0745 < center.getX());
075            Assert.assertTrue(96.0762 > center.getX());
076            Assert.assertTrue(48.1344 < center.getY());
077            Assert.assertTrue(48.1354 > center.getY());
078        }
079
080        Assert.assertTrue(optimizer.getEvaluations() > 850);
081        Assert.assertTrue(optimizer.getEvaluations() < 900);
082
083        Assert.assertEquals(3.1267527, optimum.getValue(), 1e-8);
084    }
085
086    @Test
087    public void testRosenbrock() {
088        Rosenbrock rosenbrock = new Rosenbrock();
089        SimplexOptimizer underlying
090            = new SimplexOptimizer(new SimpleValueChecker(-1, 1e-3));
091        NelderMeadSimplex simplex = new NelderMeadSimplex(new double[][] {
092                { -1.2,  1.0 },
093                { 0.9, 1.2 } ,
094                {  3.5, -2.3 }
095            });
096        JDKRandomGenerator g = new JDKRandomGenerator();
097        g.setSeed(16069223052l);
098        RandomVectorGenerator generator
099            = new UncorrelatedRandomVectorGenerator(2, new GaussianRandomGenerator(g));
100        int nbStarts = 10;
101        MultiStartMultivariateOptimizer optimizer
102            = new MultiStartMultivariateOptimizer(underlying, nbStarts, generator);
103        PointValuePair optimum
104            = optimizer.optimize(new MaxEval(1100),
105                                 new ObjectiveFunction(rosenbrock),
106                                 GoalType.MINIMIZE,
107                                 simplex,
108                                 new InitialGuess(new double[] { -1.2, 1.0 }));
109        Assert.assertEquals(nbStarts, optimizer.getOptima().length);
110
111        Assert.assertEquals(rosenbrock.getCount(), optimizer.getEvaluations());
112        Assert.assertTrue(optimizer.getEvaluations() > 900);
113        Assert.assertTrue(optimizer.getEvaluations() < 1200);
114        Assert.assertTrue(optimum.getValue() < 5e-5);
115    }
116
117    private static class Rosenbrock implements MultivariateFunction {
118        private int count;
119
120        public Rosenbrock() {
121            count = 0;
122        }
123
124        public double value(double[] x) {
125            ++count;
126            double a = x[1] - x[0] * x[0];
127            double b = 1 - x[0];
128            return 100 * a * a + b * b;
129        }
130
131        public int getCount() {
132            return count;
133        }
134    }
