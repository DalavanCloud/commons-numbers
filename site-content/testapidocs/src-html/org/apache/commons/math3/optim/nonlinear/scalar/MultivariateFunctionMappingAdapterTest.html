019import org.apache.commons.math3.analysis.MultivariateFunction;
020import org.apache.commons.math3.optim.InitialGuess;

025import org.apache.commons.math3.optim.nonlinear.scalar.noderiv.SimplexOptimizer;

029public class MultivariateFunctionMappingAdapterTest {
030    @Test
031    public void testStartSimplexInsideRange() {
032        final BiQuadratic biQuadratic = new BiQuadratic(2.0, 2.5, 1.0, 3.0, 2.0, 3.0);
033        final MultivariateFunctionMappingAdapter wrapped
034            = new MultivariateFunctionMappingAdapter(biQuadratic,
035                                                     biQuadratic.getLower(),
036                                                     biQuadratic.getUpper());
037
038        SimplexOptimizer optimizer = new SimplexOptimizer(1e-10, 1e-30);
039        final AbstractSimplex simplex = new NelderMeadSimplex(new double[][] {
040                wrapped.boundedToUnbounded(new double[] { 1.5, 2.75 }),
041                wrapped.boundedToUnbounded(new double[] { 1.5, 2.95 }),
042                wrapped.boundedToUnbounded(new double[] { 1.7, 2.90 })
043            });
044
045        final PointValuePair optimum
046            = optimizer.optimize(new MaxEval(300),
047                                 new ObjectiveFunction(wrapped),
048                                 simplex,
049                                 GoalType.MINIMIZE,
050                                 new InitialGuess(wrapped.boundedToUnbounded(new double[] { 1.5, 2.25 })));
051        final double[] bounded = wrapped.unboundedToBounded(optimum.getPoint());
052
053        Assert.assertEquals(biQuadratic.getBoundedXOptimum(), bounded[0], 2e-7);
054        Assert.assertEquals(biQuadratic.getBoundedYOptimum(), bounded[1], 2e-7);
055    }

142    private static class BiQuadratic implements MultivariateFunction {
143
144        private final double xOptimum;
145        private final double yOptimum;
146
147        private final double xMin;
148        private final double xMax;
149        private final double yMin;
150        private final double yMax;
151
152        public BiQuadratic(final double xOptimum, final double yOptimum,
153                           final double xMin, final double xMax,
154                           final double yMin, final double yMax) {
155            this.xOptimum = xOptimum;
156            this.yOptimum = yOptimum;
157            this.xMin     = xMin;
158            this.xMax     = xMax;
159            this.yMin     = yMin;
160            this.yMax     = yMax;
161        }
162
163        public double value(double[] point) {
164            // the function should never be called with out of range points
165            Assert.assertTrue(point[0] >= xMin);
166            Assert.assertTrue(point[0] <= xMax);
167            Assert.assertTrue(point[1] >= yMin);
168            Assert.assertTrue(point[1] <= yMax);
169
170            final double dx = point[0] - xOptimum;
171            final double dy = point[1] - yOptimum;
172            return dx * dx + dy * dy;
173
174        }
175
176        public double[] getLower() {
177            return new double[] { xMin, yMin };
178        }
179
180        public double[] getUpper() {
181            return new double[] { xMax, yMax };
182        }
183
184        public double getBoundedXOptimum() {
185            return (xOptimum < xMin) ? xMin : ((xOptimum > xMax) ? xMax : xOptimum);
186        }
187
188        public double getBoundedYOptimum() {
189            return (yOptimum < yMin) ? yMin : ((yOptimum > yMax) ? yMax : yOptimum);
190        }
191    }
