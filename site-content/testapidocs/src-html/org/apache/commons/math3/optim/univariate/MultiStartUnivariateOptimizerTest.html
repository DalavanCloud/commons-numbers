052    @Test
053    public void testSinMin() {
054        UnivariateFunction f = new Sin();
055        UnivariateOptimizer underlying = new BrentOptimizer(1e-10, 1e-14);
056        JDKRandomGenerator g = new JDKRandomGenerator();
057        g.setSeed(44428400075l);
058        MultiStartUnivariateOptimizer optimizer = new MultiStartUnivariateOptimizer(underlying, 10, g);
059        optimizer.optimize(new MaxEval(300),
060                           new UnivariateObjectiveFunction(f),
061                           GoalType.MINIMIZE,
062                           new SearchInterval(-100.0, 100.0));
063        UnivariatePointValuePair[] optima = optimizer.getOptima();
064        for (int i = 1; i < optima.length; ++i) {
065            double d = (optima[i].getPoint() - optima[i-1].getPoint()) / (2 * FastMath.PI);
066            Assert.assertTrue(FastMath.abs(d - FastMath.rint(d)) < 1.0e-8);
067            Assert.assertEquals(-1.0, f.value(optima[i].getPoint()), 1.0e-10);
068            Assert.assertEquals(f.value(optima[i].getPoint()), optima[i].getValue(), 1.0e-10);
069        }
070        Assert.assertTrue(optimizer.getEvaluations() > 200);
071        Assert.assertTrue(optimizer.getEvaluations() < 300);
072    }
073
074    @Test
075    public void testQuinticMin() {
076        // The quintic function has zeros at 0, +-0.5 and +-1.
077        // The function has extrema (first derivative is zero) at 0.27195613 and 0.82221643,
078        UnivariateFunction f = new QuinticFunction();
079        UnivariateOptimizer underlying = new BrentOptimizer(1e-9, 1e-14);
080        JDKRandomGenerator g = new JDKRandomGenerator();
081        g.setSeed(4312000053L);
082        MultiStartUnivariateOptimizer optimizer = new MultiStartUnivariateOptimizer(underlying, 5, g);
083
084        UnivariatePointValuePair optimum
085            = optimizer.optimize(new MaxEval(300),
086                                 new UnivariateObjectiveFunction(f),
087                                 GoalType.MINIMIZE,
088                                 new SearchInterval(-0.3, -0.2));
089        Assert.assertEquals(-0.27195613, optimum.getPoint(), 1e-9);
090        Assert.assertEquals(-0.0443342695, optimum.getValue(), 1e-9);
091
092        UnivariatePointValuePair[] optima = optimizer.getOptima();
093        for (int i = 0; i < optima.length; ++i) {
094            Assert.assertEquals(f.value(optima[i].getPoint()), optima[i].getValue(), 1e-9);
095        }
096        Assert.assertTrue(optimizer.getEvaluations() >= 50);
097        Assert.assertTrue(optimizer.getEvaluations() <= 100);
098    }
099
100    @Test
101    public void testBadFunction() {
102        UnivariateFunction f = new UnivariateFunction() {
103                public double value(double x) {
104                    if (x < 0) {
105                        throw new LocalException();
106                    }
107                    return 0;
108                }
109            };
110        UnivariateOptimizer underlying = new BrentOptimizer(1e-9, 1e-14);
111        JDKRandomGenerator g = new JDKRandomGenerator();
112        g.setSeed(4312000053L);
113        MultiStartUnivariateOptimizer optimizer = new MultiStartUnivariateOptimizer(underlying, 5, g);
114 
115        try {
116            optimizer.optimize(new MaxEval(300),
117                               new UnivariateObjectiveFunction(f),
118                               GoalType.MINIMIZE,
119                               new SearchInterval(-0.3, -0.2));
120            Assert.fail();
121        } catch (LocalException e) {
122            // Expected.
123        }
124
125        // Ensure that the exception was thrown because no optimum was found.
126        Assert.assertTrue(optimizer.getOptima()[0] == null);
127    }
128
129    private static class LocalException extends RuntimeException {
130        private static final long serialVersionUID = 1194682757034350629L;
131    }
