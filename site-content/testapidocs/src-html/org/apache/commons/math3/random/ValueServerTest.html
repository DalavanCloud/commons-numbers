048    /**
049      * Generate 1000 random values and make sure they look OK.<br>
050      * Note that there is a non-zero (but very small) probability that
051      * these tests will fail even if the code is working as designed.
052      */
053    @Test
054    public void testNextDigest() throws Exception {
055        double next = 0.0;
056        double tolerance = 0.1;
057        vs.computeDistribution();
058        Assert.assertTrue("empirical distribution property",
059            vs.getEmpiricalDistribution() != null);
060        SummaryStatistics stats = new SummaryStatistics();
061        for (int i = 1; i < 1000; i++) {
062            next = vs.getNext();
063            stats.addValue(next);
064        }
065        Assert.assertEquals("mean", 5.069831575018909, stats.getMean(), tolerance);
066        Assert.assertEquals("std dev", 1.0173699343977738, stats.getStandardDeviation(),
067            tolerance);
068
069        vs.computeDistribution(500);
070        stats = new SummaryStatistics();
071        for (int i = 1; i < 1000; i++) {
072            next = vs.getNext();
073            stats.addValue(next);
074        }
075        Assert.assertEquals("mean", 5.069831575018909, stats.getMean(), tolerance);
076        Assert.assertEquals("std dev", 1.0173699343977738, stats.getStandardDeviation(),
077            tolerance);
078    }

080    /**
081     * Verify that when provided with fixed seeds, stochastic modes
082     * generate fixed sequences.  Verifies the fix for MATH-654.
083     */
084    @Test 
085    public void testFixedSeed() throws Exception {
086        ValueServer valueServer = new ValueServer();
087        URL url = getClass().getResource("testData.txt");
088        valueServer.setValuesFileURL(url);
089        valueServer.computeDistribution();
090        checkFixedSeed(valueServer, ValueServer.DIGEST_MODE);
091        checkFixedSeed(valueServer, ValueServer.EXPONENTIAL_MODE);
092        checkFixedSeed(valueServer, ValueServer.GAUSSIAN_MODE);
093        checkFixedSeed(valueServer, ValueServer.UNIFORM_MODE);
094    }
095    
096    /**
097     * Do the check for {@link #testFixedSeed()}
098     * @param mode ValueServer mode
099     */
100    private void checkFixedSeed(ValueServer valueServer, int mode) throws Exception {
101        valueServer.reSeed(1000);
102        valueServer.setMode(mode);
103        double[][] values = new double[2][100];
104        for (int i = 0; i < 100; i++) {
105            values[0][i] = valueServer.getNext();
106        }
107        valueServer.reSeed(1000);
108        for (int i = 0; i < 100; i++) {
109            values[1][i] = valueServer.getNext();
110        }
111        Assert.assertTrue(Arrays.equals(values[0], values[1])); 
112    }

152    /**
153     * Test ValueServer REPLAY_MODE using values in testData file.<br>
154     * Check that the values 1,2,1001,1002 match data file values 1 and 2.
155     * the sample data file.
156     */
157    @Test
158    public void testReplay() throws Exception {
159        double firstDataValue = 4.038625496201205;
160        double secondDataValue = 3.6485326248346936;
161        double tolerance = 10E-15;
162        double compareValue = 0.0d;
163        vs.setMode(ValueServer.REPLAY_MODE);
164        vs.resetReplayFile();
165        compareValue = vs.getNext();
166        Assert.assertEquals(compareValue,firstDataValue,tolerance);
167        compareValue = vs.getNext();
168        Assert.assertEquals(compareValue,secondDataValue,tolerance);
169        for (int i = 3; i < 1001; i++) {
170           compareValue = vs.getNext();
171        }
172        compareValue = vs.getNext();
173        Assert.assertEquals(compareValue,firstDataValue,tolerance);
174        compareValue = vs.getNext();
175        Assert.assertEquals(compareValue,secondDataValue,tolerance);
176        vs.closeReplayFile();
177        // make sure no NPE
178        vs.closeReplayFile();
179    }

181    /**
182     * Test other ValueServer modes
183     */
184    @Test
185    public void testModes() throws Exception {
186        vs.setMode(ValueServer.CONSTANT_MODE);
187        vs.setMu(0);
188        Assert.assertEquals("constant mode test",vs.getMu(),vs.getNext(),Double.MIN_VALUE);
189        vs.setMode(ValueServer.UNIFORM_MODE);
190        vs.setMu(2);
191        double val = vs.getNext();
192        Assert.assertTrue(val > 0 && val < 4);
193        vs.setSigma(1);
194        vs.setMode(ValueServer.GAUSSIAN_MODE);
195        val = vs.getNext();
196        Assert.assertTrue("gaussian value close enough to mean",
197            val < vs.getMu() + 100*vs.getSigma());
198        vs.setMode(ValueServer.EXPONENTIAL_MODE);
199        val = vs.getNext();
200        Assert.assertTrue(val > 0);
201        try {
202            vs.setMode(1000);
203            vs.getNext();
204            Assert.fail("bad mode, expecting IllegalStateException");
205        } catch (IllegalStateException ex) {
206            // ignored
207        }
208    }
