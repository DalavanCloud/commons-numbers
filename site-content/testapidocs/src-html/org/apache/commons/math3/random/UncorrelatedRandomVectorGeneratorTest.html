018package org.apache.commons.math3.random;
019
020import org.apache.commons.math3.linear.RealMatrix;
021import org.apache.commons.math3.stat.descriptive.moment.VectorialCovariance;
022import org.apache.commons.math3.stat.descriptive.moment.VectorialMean;
023
024import org.junit.Test;
025import org.junit.Assert;
026
027public class UncorrelatedRandomVectorGeneratorTest {
028    private double[] mean;
029    private double[] standardDeviation;
030    private UncorrelatedRandomVectorGenerator generator;
031
032    public UncorrelatedRandomVectorGeneratorTest() {
033        mean              = new double[] {0.0, 1.0, -3.0, 2.3};
034        standardDeviation = new double[] {1.0, 2.0, 10.0, 0.1};
035        RandomGenerator rg = new JDKRandomGenerator();
036        rg.setSeed(17399225432l);
037        generator =
038            new UncorrelatedRandomVectorGenerator(mean, standardDeviation,
039                    new GaussianRandomGenerator(rg));
040    }
041
042    @Test
043    public void testMeanAndCorrelation() {
044
045        VectorialMean meanStat = new VectorialMean(mean.length);
046        VectorialCovariance covStat = new VectorialCovariance(mean.length, true);
047        for (int i = 0; i < 10000; ++i) {
048            double[] v = generator.nextVector();
049            meanStat.increment(v);
050            covStat.increment(v);
051        }
052
053        double[] estimatedMean = meanStat.getResult();
054        double scale;
055        RealMatrix estimatedCorrelation = covStat.getResult();
056        for (int i = 0; i < estimatedMean.length; ++i) {
057            Assert.assertEquals(mean[i], estimatedMean[i], 0.07);
058            for (int j = 0; j < i; ++j) {
059                scale = standardDeviation[i] * standardDeviation[j];
060                Assert.assertEquals(0, estimatedCorrelation.getEntry(i, j) / scale, 0.03);
061            }
062            scale = standardDeviation[i] * standardDeviation[i];
063            Assert.assertEquals(1, estimatedCorrelation.getEntry(i, i) / scale, 0.025);
064        }
065    }
066}
