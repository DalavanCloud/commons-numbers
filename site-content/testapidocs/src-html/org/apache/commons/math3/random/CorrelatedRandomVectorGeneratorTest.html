039    public CorrelatedRandomVectorGeneratorTest() {
040        mean = new double[] { 0.0, 1.0, -3.0, 2.3 };
041
042        RealMatrix b = MatrixUtils.createRealMatrix(4, 3);
043        int counter = 0;
044        for (int i = 0; i < b.getRowDimension(); ++i) {
045            for (int j = 0; j < b.getColumnDimension(); ++j) {
046                b.setEntry(i, j, 1.0 + 0.1 * ++counter);
047            }
048        }
049        RealMatrix bbt = b.multiply(b.transpose());
050        covariance = MatrixUtils.createRealMatrix(mean.length, mean.length);
051        for (int i = 0; i < covariance.getRowDimension(); ++i) {
052            covariance.setEntry(i, i, bbt.getEntry(i, i));
053            for (int j = 0; j < covariance.getColumnDimension(); ++j) {
054                double s = bbt.getEntry(i, j);
055                covariance.setEntry(i, j, s);
056                covariance.setEntry(j, i, s);
057            }
058        }
059
060        RandomGenerator rg = new JDKRandomGenerator();
061        rg.setSeed(17399225432l);
062        GaussianRandomGenerator rawGenerator = new GaussianRandomGenerator(rg);
063        generator = new CorrelatedRandomVectorGenerator(mean,
064                                                        covariance,
065                                                        1.0e-12 * covariance.getNorm(),
066                                                        rawGenerator);
067    }

074    @Test
075    public void testMath226() {
076        double[] mean = { 1, 1, 10, 1 };
077        double[][] cov = {
078                { 1, 3, 2, 6 },
079                { 3, 13, 16, 2 },
080                { 2, 16, 38, -1 },
081                { 6, 2, -1, 197 }
082        };
083        RealMatrix covRM = MatrixUtils.createRealMatrix(cov);
084        JDKRandomGenerator jg = new JDKRandomGenerator();
085        jg.setSeed(5322145245211l);
086        NormalizedRandomGenerator rg = new GaussianRandomGenerator(jg);
087        CorrelatedRandomVectorGenerator sg =
088            new CorrelatedRandomVectorGenerator(mean, covRM, 0.00001, rg);
089
090        double[] min = new double[mean.length];
091        Arrays.fill(min, Double.POSITIVE_INFINITY);
092        double[] max = new double[mean.length];
093        Arrays.fill(max, Double.NEGATIVE_INFINITY);
094        for (int i = 0; i < 10; i++) {
095            double[] generated = sg.nextVector();
096            for (int j = 0; j < generated.length; ++j) {
097                min[j] = FastMath.min(min[j], generated[j]);
098                max[j] = FastMath.max(max[j], generated[j]);
099            }
100        }
101        for (int j = 0; j < min.length; ++j) {
102            Assert.assertTrue(max[j] - min[j] > 2.0);
103        }
104
105    }

118    @Test
119    public void testMeanAndCovariance() {
120
121        VectorialMean meanStat = new VectorialMean(mean.length);
122        VectorialCovariance covStat = new VectorialCovariance(mean.length, true);
123        for (int i = 0; i < 5000; ++i) {
124            double[] v = generator.nextVector();
125            meanStat.increment(v);
126            covStat.increment(v);
127        }
128
129        double[] estimatedMean = meanStat.getResult();
130        RealMatrix estimatedCovariance = covStat.getResult();
131        for (int i = 0; i < estimatedMean.length; ++i) {
132            Assert.assertEquals(mean[i], estimatedMean[i], 0.07);
133            for (int j = 0; j <= i; ++j) {
134                Assert.assertEquals(covariance.getEntry(i, j),
135                                    estimatedCovariance.getEntry(i, j),
136                                    0.1 * (1.0 + FastMath.abs(mean[i])) * (1.0 + FastMath.abs(mean[j])));
137            }
138        }
139
140    }

173    private CorrelatedRandomVectorGenerator createSampler(double[][] cov) {
174        RealMatrix matrix = new Array2DRowRealMatrix(cov);
175        double small = 10e-12 * matrix.getNorm();
176        return new CorrelatedRandomVectorGenerator(
177                new double[cov.length],
178                matrix,
179                small,
180                new GaussianRandomGenerator(new JDKRandomGenerator()));
181    }
182
183    private void testSampler(final double[][] covMatrix, int samples, double epsilon) {
184        CorrelatedRandomVectorGenerator sampler = createSampler(covMatrix);
185        
186        StorelessCovariance cov = new StorelessCovariance(covMatrix.length);
187        for (int i = 0; i < samples; ++i) {
188            cov.increment(sampler.nextVector());
189        }
190
191        double[][] sampleCov = cov.getData();
192        for (int r = 0; r < covMatrix.length; ++r) {
193            TestUtils.assertEquals(covMatrix[r], sampleCov[r], epsilon);
194        }
195
196    }
