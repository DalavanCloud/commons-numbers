032    public void testEqualsDouble() {
033        final double x = 1234.5678;
034        Assert.assertTrue(MathUtils.equals(x, x));
035        Assert.assertFalse(MathUtils.equals(x, -x));
036
037        // Special cases (cf. semantics of JDK's "Double").
038        // 1. NaN
039        Assert.assertTrue(MathUtils.equals(Double.NaN, Double.NaN));
040        // 2. Negative zero
041        final double mZero = -0d;
042        final double zero = 0d;
043        Assert.assertTrue(MathUtils.equals(zero, zero));
044        Assert.assertTrue(MathUtils.equals(mZero, mZero));
045        Assert.assertFalse(MathUtils.equals(mZero, zero));
046    }

048    @Test
049    public void testHash() {
050        double[] testArray = {
051            Double.NaN,
052            Double.POSITIVE_INFINITY,
053            Double.NEGATIVE_INFINITY,
054            1d,
055            0d,
056            1E-14,
057            (1 + 1E-14),
058            Double.MIN_VALUE,
059            Double.MAX_VALUE };
060        for (int i = 0; i < testArray.length; i++) {
061            for (int j = 0; j < testArray.length; j++) {
062                if (i == j) {
063                    Assert.assertEquals(MathUtils.hash(testArray[i]), MathUtils.hash(testArray[j]));
064                    Assert.assertEquals(MathUtils.hash(testArray[j]), MathUtils.hash(testArray[i]));
065                } else {
066                    Assert.assertTrue(MathUtils.hash(testArray[i]) != MathUtils.hash(testArray[j]));
067                    Assert.assertTrue(MathUtils.hash(testArray[j]) != MathUtils.hash(testArray[i]));
068                }
069            }
070        }
071    }

091     * Make sure that permuted arrays do not hash to the same value.
092     */
093    @Test
094    public void testPermutedArrayHash() {
095        double[] original = new double[10];
096        double[] permuted = new double[10];
097        RandomDataGenerator random = new RandomDataGenerator();
098
099        // Generate 10 distinct random values
100        for (int i = 0; i < 10; i++) {
101            final RealDistribution u = new UniformRealDistribution(i + 0.5, i + 0.75);
102            original[i] = u.sample();
103        }
104
105        // Generate a random permutation, making sure it is not the identity
106        boolean isIdentity = true;
107        do {
108            int[] permutation = random.nextPermutation(10, 10);
109            for (int i = 0; i < 10; i++) {
110                if (i != permutation[i]) {
111                    isIdentity = false;
112                }
113                permuted[i] = original[permutation[i]];
114            }
115        } while (isIdentity);
116
117        // Verify that permuted array has different hash
118        Assert.assertFalse(MathUtils.hash(original) == MathUtils.hash(permuted));
119    }

149    @Test
150    public void testNormalizeAngle() {
151        for (double a = -15.0; a <= 15.0; a += 0.1) {
152            for (double b = -15.0; b <= 15.0; b += 0.2) {
153                double c = MathUtils.normalizeAngle(a, b);
154                Assert.assertTrue((b - FastMath.PI) <= c);
155                Assert.assertTrue(c <= (b + FastMath.PI));
156                double twoK = FastMath.rint((a - c) / FastMath.PI);
157                Assert.assertEquals(c, a - twoK * FastMath.PI, 1.0e-14);
158            }
159        }
160    }

162    @Test
163    public void testReduce() {
164        final double period = -12.222;
165        final double offset = 13;
166
167        final double delta = 1.5;
168
169        double orig = offset + 122456789 * period + delta;
170        double expected = delta;
171        Assert.assertEquals(expected,
172                            MathUtils.reduce(orig, period, offset),
173                            1e-7);
174        Assert.assertEquals(expected,
175                            MathUtils.reduce(orig, -period, offset),
176                            1e-7);
177
178        orig = offset - 123356789 * period - delta;
179        expected = FastMath.abs(period) - delta;
180        Assert.assertEquals(expected,
181                            MathUtils.reduce(orig, period, offset),
182                            1e-6);
183        Assert.assertEquals(expected,
184                            MathUtils.reduce(orig, -period, offset),
185                            1e-6);
186
187        orig = offset - 123446789 * period + delta;
188        expected = delta;
189        Assert.assertEquals(expected,
190                            MathUtils.reduce(orig, period, offset),
191                            1e-6);
192        Assert.assertEquals(expected,
193                            MathUtils.reduce(orig, -period, offset),
194                            1e-6);
195
196        Assert.assertTrue(Double.isNaN(MathUtils.reduce(orig, Double.NaN, offset)));
197        Assert.assertTrue(Double.isNaN(MathUtils.reduce(Double.NaN, period, offset)));
198        Assert.assertTrue(Double.isNaN(MathUtils.reduce(orig, period, Double.NaN)));
199        Assert.assertTrue(Double.isNaN(MathUtils.reduce(orig, period,
200                Double.POSITIVE_INFINITY)));
201        Assert.assertTrue(Double.isNaN(MathUtils.reduce(Double.POSITIVE_INFINITY,
202                period, offset)));
203        Assert.assertTrue(Double.isNaN(MathUtils.reduce(orig,
204                Double.POSITIVE_INFINITY, offset)));
205        Assert.assertTrue(Double.isNaN(MathUtils.reduce(orig,
206                Double.POSITIVE_INFINITY, Double.POSITIVE_INFINITY)));
207        Assert.assertTrue(Double.isNaN(MathUtils.reduce(Double.POSITIVE_INFINITY,
208                period, Double.POSITIVE_INFINITY)));
209        Assert.assertTrue(Double.isNaN(MathUtils.reduce(Double.POSITIVE_INFINITY,
210                Double.POSITIVE_INFINITY, offset)));
211        Assert.assertTrue(Double.isNaN(MathUtils.reduce(Double.POSITIVE_INFINITY,
212                Double.POSITIVE_INFINITY,  Double.POSITIVE_INFINITY)));
213    }

215    @Test
216    public void testReduceComparedWithNormalizeAngle() {
217        final double tol = Math.ulp(1d);
218        final double period = 2 * Math.PI;
219        for (double a = -15; a <= 15; a += 0.5) {
220            for (double center = -15; center <= 15; center += 1) {
221                final double nA = MathUtils.normalizeAngle(a, center);
222                final double offset = center - Math.PI;
223                final double r = MathUtils.reduce(a, period, offset);
224                Assert.assertEquals(nA, r + offset, tol);
225            }
226        }
227    }

318    @Test
319    public void testCopySignByte() {
320        byte a = MathUtils.copySign(Byte.MIN_VALUE, (byte) -1);
321        Assert.assertEquals(Byte.MIN_VALUE, a);
322
323        final byte minValuePlusOne = Byte.MIN_VALUE + (byte) 1;
324        a = MathUtils.copySign(minValuePlusOne, (byte) 1);
325        Assert.assertEquals(Byte.MAX_VALUE, a);
326
327        a = MathUtils.copySign(Byte.MAX_VALUE, (byte) -1);
328        Assert.assertEquals(minValuePlusOne, a);
329
330        final byte one = 1;
331        byte val = -2;
332        a = MathUtils.copySign(val, one);
333        Assert.assertEquals(-val, a);
334
335        final byte minusOne = -one;
336        val = 2;
337        a = MathUtils.copySign(val, minusOne);
338        Assert.assertEquals(-val, a);
339
340        val = 0;
341        a = MathUtils.copySign(val, minusOne);
342        Assert.assertEquals(val, a);
343
344        val = 0;
345        a = MathUtils.copySign(val, one);
346        Assert.assertEquals(val, a);
347    }
