017package org.apache.commons.math3.util;

019import java.math.BigDecimal;
020import java.math.BigInteger;
021import java.math.MathContext;
022
023import org.apache.commons.math3.TestUtils;
024import org.apache.commons.math3.exception.MathArithmeticException;
025import org.junit.Assert;
026import org.junit.Test;
027
028public class BigRealTest {
029
030    @Test
031    public void testConstructor() {
032        Assert.assertEquals(1.625,
033                            new BigReal(new BigDecimal("1.625")).doubleValue(),
034                            1.0e-15);
035        Assert.assertEquals(-5.0,
036                            new BigReal(new BigInteger("-5")).doubleValue(),
037                            1.0e-15);
038        Assert.assertEquals(-5.0, new BigReal(new BigInteger("-5"),
039                                              MathContext.DECIMAL64)
040            .doubleValue(), 1.0e-15);
041        Assert
042            .assertEquals(0.125,
043                          new BigReal(new BigInteger("125"), 3).doubleValue(),
044                          1.0e-15);
045        Assert.assertEquals(0.125, new BigReal(new BigInteger("125"), 3,
046                                               MathContext.DECIMAL64)
047            .doubleValue(), 1.0e-15);
048        Assert.assertEquals(1.625, new BigReal(new char[] {
049            '1', '.', '6', '2', '5'
050        }).doubleValue(), 1.0e-15);
051        Assert.assertEquals(1.625, new BigReal(new char[] {
052            'A', 'A', '1', '.', '6', '2', '5', '9'
053        }, 2, 5).doubleValue(), 1.0e-15);
054        Assert.assertEquals(1.625, new BigReal(new char[] {
055            'A', 'A', '1', '.', '6', '2', '5', '9'
056        }, 2, 5, MathContext.DECIMAL64).doubleValue(), 1.0e-15);
057        Assert.assertEquals(1.625, new BigReal(new char[] {
058            '1', '.', '6', '2', '5'
059        }, MathContext.DECIMAL64).doubleValue(), 1.0e-15);
060        Assert.assertEquals(1.625, new BigReal(1.625).doubleValue(), 1.0e-15);
061        Assert.assertEquals(1.625, new BigReal(1.625, MathContext.DECIMAL64)
062            .doubleValue(), 1.0e-15);
063        Assert.assertEquals(-5.0, new BigReal(-5).doubleValue(), 1.0e-15);
064        Assert.assertEquals(-5.0, new BigReal(-5, MathContext.DECIMAL64)
065            .doubleValue(), 1.0e-15);
066        Assert.assertEquals(-5.0, new BigReal(-5l).doubleValue(), 1.0e-15);
067        Assert.assertEquals(-5.0, new BigReal(-5l, MathContext.DECIMAL64)
068            .doubleValue(), 1.0e-15);
069        Assert.assertEquals(1.625, new BigReal("1.625").doubleValue(), 1.0e-15);
070        Assert.assertEquals(1.625, new BigReal("1.625", MathContext.DECIMAL64)
071            .doubleValue(), 1.0e-15);
072    }
073
074    @Test
075    public void testCompareTo() {
076        BigReal first = new BigReal(1.0 / 2.0);
077        BigReal second = new BigReal(1.0 / 3.0);
078        BigReal third = new BigReal(1.0 / 2.0);
079
080        Assert.assertEquals(0, first.compareTo(first));
081        Assert.assertEquals(0, first.compareTo(third));
082        Assert.assertEquals(1, first.compareTo(second));
083        Assert.assertEquals(-1, second.compareTo(first));
084
085    }
086
087    @Test
088    public void testAdd() {
089        BigReal a = new BigReal("1.2345678");
090        BigReal b = new BigReal("8.7654321");
091        Assert.assertEquals(9.9999999, a.add(b).doubleValue(), 1.0e-15);
092    }

101    @Test
102    public void testNegate() {
103        BigReal a = new BigReal("1.2345678");
104        BigReal zero = new BigReal("0.0000000");
105        Assert.assertEquals(a.negate().add(a), zero);
106        Assert.assertEquals(a.add(a.negate()), zero);
107        Assert.assertEquals(zero, zero.negate());
108    }

124    @Test
125    public void testReciprocal() {
126        BigReal a = new BigReal("1.2345678");
127        double eps = FastMath.pow(10., -a.getScale());
128        BigReal one = new BigReal("1.0000000");
129        BigReal b = a.reciprocal();
130        BigReal r = one.subtract(a.multiply(b));
131        Assert.assertTrue(FastMath.abs(r.doubleValue()) <= eps);
132        r = one.subtract(b.multiply(a));
133        Assert.assertTrue(FastMath.abs(r.doubleValue()) <= eps);
134    }

141    @Test

143        BigReal a = new BigReal("1024.0");

156    public void testBigDecimalValue() {
157        BigDecimal pi = new BigDecimal(

163
164    @Test
165    public void testEqualsAndHashCode() {
166        BigReal zero = new BigReal(0.0);
167        BigReal nullReal = null;
168        Assert.assertTrue(zero.equals(zero));
169        Assert.assertFalse(zero.equals(nullReal));
170        Assert.assertFalse(zero.equals(Double.valueOf(0)));
171        BigReal zero2 = new BigReal(0.0);
172        Assert.assertTrue(zero.equals(zero2));
173        Assert.assertEquals(zero.hashCode(), zero2.hashCode());
174        BigReal one = new BigReal(1.0);
175        Assert.assertFalse((one.equals(zero) || zero.equals(one)));
176        Assert.assertTrue(one.equals(BigReal.ONE));
177    }

181        BigReal[] Reals = {
182            new BigReal(3.0), BigReal.ONE, BigReal.ZERO, new BigReal(17),
183            new BigReal(FastMath.PI), new BigReal(-2.5)
184        };

186            Assert.assertEquals(Real, TestUtils.serializeAndRecover(Real));
187        }
