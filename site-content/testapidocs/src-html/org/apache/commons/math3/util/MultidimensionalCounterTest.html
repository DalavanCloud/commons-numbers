020import java.util.NoSuchElementException;

027/**

031public class MultidimensionalCounterTest {
032    @Test
033    public void testPreconditions() {
034        MultidimensionalCounter c;
035
036        try {
037            c = new MultidimensionalCounter(0, 1);
038            Assert.fail("NotStrictlyPositiveException expected");
039        } catch (NotStrictlyPositiveException e) {
040            // Expected.
041        }
042        try {
043            c = new MultidimensionalCounter(2, 0);
044            Assert.fail("NotStrictlyPositiveException expected");
045        } catch (NotStrictlyPositiveException e) {
046            // Expected.
047        }
048        try {
049            c = new MultidimensionalCounter(-1, 1);
050            Assert.fail("NotStrictlyPositiveException expected");
051        } catch (NotStrictlyPositiveException e) {
052            // Expected.
053        }
054
055        c = new MultidimensionalCounter(2, 3);
056        try {
057            c.getCount(1, 1, 1);
058            Assert.fail("DimensionMismatchException expected");
059        } catch (DimensionMismatchException e) {
060            // Expected.
061        }
062        try {
063            c.getCount(3, 1);
064            Assert.fail("OutOfRangeException expected");
065        } catch (OutOfRangeException e) {
066            // Expected.
067        }
068        try {
069            c.getCount(0, -1);
070            Assert.fail("OutOfRangeException expected");
071        } catch (OutOfRangeException e) {
072            // Expected.
073        }
074        try {
075            c.getCounts(-1);
076            Assert.fail("OutOfRangeException expected");
077        } catch (OutOfRangeException e) {
078            // Expected.
079        }
080        try {
081            c.getCounts(6);
082            Assert.fail("OutOfRangeException expected");
083        } catch (OutOfRangeException e) {
084            // Expected.
085        }
086    }

105    @Test
106    public void testIterator() {
107        final int dim1 = 3;
108        final int dim2 = 4;
109
110        final MultidimensionalCounter.Iterator iter
111            = new MultidimensionalCounter(dim1, dim2).iterator();
112
113        final int max = dim1 * dim2;
114        for (int i = 0; i < max; i++) {
115            Assert.assertTrue(iter.hasNext());
116
117            // Should not throw.
118            iter.next();
119        }
120
121        Assert.assertFalse(iter.hasNext());
122    }
123
124    @Test(expected=NoSuchElementException.class)
125    public void testIteratorNoMoreElements() {
126        final MultidimensionalCounter.Iterator iter
127            = new MultidimensionalCounter(4, 2).iterator();
128
129        while (iter.hasNext()) {

156    @Test
157    public void testIterationConsistency() {
158        final MultidimensionalCounter c = new MultidimensionalCounter(2, 3, 4);
159        final int[][] expected = new int[][] {
160            { 0, 0, 0 },
161            { 0, 0, 1 },
162            { 0, 0, 2 },
163            { 0, 0, 3 },
164            { 0, 1, 0 },
165            { 0, 1, 1 },
166            { 0, 1, 2 },
167            { 0, 1, 3 },
168            { 0, 2, 0 },
169            { 0, 2, 1 },
170            { 0, 2, 2 },
171            { 0, 2, 3 },
172            { 1, 0, 0 },
173            { 1, 0, 1 },
174            { 1, 0, 2 },
175            { 1, 0, 3 },
176            { 1, 1, 0 },
177            { 1, 1, 1 },
178            { 1, 1, 2 },
179            { 1, 1, 3 },
180            { 1, 2, 0 },
181            { 1, 2, 1 },
182            { 1, 2, 2 },
183            { 1, 2, 3 }
184        };
185
186        final int totalSize = c.getSize();
187        Assert.assertEquals(expected.length, totalSize);
188
189        final int nDim = c.getDimension();
190        final MultidimensionalCounter.Iterator iter = c.iterator();
191        for (int i = 0; i < totalSize; i++) {
192            if (!iter.hasNext()) {
193                Assert.fail("Too short");
194            }
195            final int uniDimIndex = iter.next().intValue();
196            Assert.assertEquals("Wrong iteration at " + i, i, uniDimIndex);
197
198            for (int dimIndex = 0; dimIndex < nDim; dimIndex++) {
199                Assert.assertEquals("Wrong multidimensional index for [" + i + "][" + dimIndex + "]",
200                                    expected[i][dimIndex], iter.getCount(dimIndex));
201            }
202
203            Assert.assertEquals("Wrong unidimensional index for [" + i + "]",
204                                c.getCount(expected[i]), uniDimIndex);
205
206            final int[] indices = c.getCounts(uniDimIndex);
207            for (int dimIndex = 0; dimIndex < nDim; dimIndex++) {
208                Assert.assertEquals("Wrong multidimensional index for [" + i + "][" + dimIndex + "]",
209                                    expected[i][dimIndex], indices[dimIndex]);
210            }
211        }
212
213        if (iter.hasNext()) {
214            Assert.fail("Too long");
215        }
216    }
