144    /**
145     * Verifies that the iterator generates a lexicographically
146     * increasing sequence of b(n,k) arrays, each having length k
147     * and each array itself increasing.
148     * 
149     * @param c Combinations.
150     */
151    private void checkLexicographicIterator(Combinations c) {
152        final Comparator<int[]> comp = c.comparator();
153        final int n = c.getN();
154        final int k = c.getK();
155
156        int[] lastIterate = null;
157
158        long numIterates = 0;
159        for (int[] iterate : c) {
160            Assert.assertEquals(k, iterate.length);
161
162            // Check that the sequence of iterates is ordered.
163            if (lastIterate != null) {
164                Assert.assertTrue(comp.compare(iterate, lastIterate) == 1);
165            }
166
167            // Check that each iterate is ordered.
168            for (int i = 1; i < iterate.length; i++) {
169                Assert.assertTrue(iterate[i] > iterate[i - 1]);
170            }
171
172            lastIterate = iterate;
173            ++numIterates;
174        }
175
176        // Check the number of iterates.
177        Assert.assertEquals(CombinatoricsUtils.binomialCoefficient(n, k),
178                            numIterates);
179    }
