041    @Test(expected = NotStrictlyPositiveException.class)
042    public void testNonEmptyData() {
043        // Should not accept empty data
044        new MultivariateNormalMixtureExpectationMaximization(new double[][] {});
045    }

047    @Test(expected = DimensionMismatchException.class)
048    public void testNonJaggedData() {
049        // Reject data with nonconstant numbers of columns
050        double[][] data = new double[][] {
051                { 1, 2, 3 },
052                { 4, 5, 6, 7 },
053        };
054        new MultivariateNormalMixtureExpectationMaximization(data);
055    }

058    public void testMultipleColumnsRequired() {
059        // Data should have at least 2 columns

066    @Test(expected = NotStrictlyPositiveException.class)
067    public void testMaxIterationsPositive() {
068        // Maximum iterations for fit must be positive integer
069        double[][] data = getTestSamples();
070        MultivariateNormalMixtureExpectationMaximization fitter =
071                new MultivariateNormalMixtureExpectationMaximization(data);
072
073        MixtureMultivariateNormalDistribution
074            initialMix = MultivariateNormalMixtureExpectationMaximization.estimate(data, 2);
075
076        fitter.fit(initialMix, 0, 1E-5);
077    }

079    @Test(expected = NotStrictlyPositiveException.class)
080    public void testThresholdPositive() {
081        // Maximum iterations for fit must be positive
082        double[][] data = getTestSamples();
083        MultivariateNormalMixtureExpectationMaximization fitter =
084                new MultivariateNormalMixtureExpectationMaximization(
085                    data);

087        MixtureMultivariateNormalDistribution

093    @Test(expected = ConvergenceException.class)
094    public void testConvergenceException() {
095        // ConvergenceException thrown if fit terminates before threshold met
096        double[][] data = getTestSamples();
097        MultivariateNormalMixtureExpectationMaximization fitter
098            = new MultivariateNormalMixtureExpectationMaximization(data);
099
100        MixtureMultivariateNormalDistribution
101            initialMix = MultivariateNormalMixtureExpectationMaximization.estimate(data, 2);
102
103        // 5 iterations not enough to meet convergence threshold
104        fitter.fit(initialMix, 5, 1E-5);
105    }

107    @Test(expected = DimensionMismatchException.class)
108    public void testIncompatibleIntialMixture() {
109        // Data has 3 columns
110        double[][] data = new double[][] {
111                { 1, 2, 3 }, { 4, 5, 6 }, { 7, 8, 9 }
112        };
113        double[] weights = new double[] { 0.5, 0.5 };
114
115        // These distributions are compatible with 2-column data, not 3-column
116        // data
117        MultivariateNormalDistribution[] mvns = new MultivariateNormalDistribution[2];
118
119        mvns[0] = new MultivariateNormalDistribution(new double[] {
120                        -0.0021722935000328823, 3.5432892936887908 },
121                        new double[][] {
122                                { 4.537422569229048, 3.5266152281729304 },
123                                { 3.5266152281729304, 6.175448814169779 } });
124        mvns[1] = new MultivariateNormalDistribution(new double[] {
125                        5.090902706507635, 8.68540656355283 }, new double[][] {
126                        { 2.886778573963039, 1.5257474543463154 },
127                        { 1.5257474543463154, 3.3794567673616918 } });
128
129        // Create components and mixture
130        List<Pair<Double, MultivariateNormalDistribution>> components =
131                new ArrayList<Pair<Double, MultivariateNormalDistribution>>();
132        components.add(new Pair<Double, MultivariateNormalDistribution>(
133                weights[0], mvns[0]));
134        components.add(new Pair<Double, MultivariateNormalDistribution>(
135                weights[1], mvns[1]));
136
137        MixtureMultivariateNormalDistribution badInitialMix
138            = new MixtureMultivariateNormalDistribution(components);
139
140        MultivariateNormalMixtureExpectationMaximization fitter
141            = new MultivariateNormalMixtureExpectationMaximization(data);
142
143        fitter.fit(badInitialMix);
144    }

146    @Test
147    public void testInitialMixture() {
148        // Testing initial mixture estimated from data
149        final double[] correctWeights = new double[] { 0.5, 0.5 };
150
151        final double[][] correctMeans = new double[][] {
152            {-0.0021722935000328823, 3.5432892936887908},
153            {5.090902706507635, 8.68540656355283},
154        };
155
156        final RealMatrix[] correctCovMats = new Array2DRowRealMatrix[2];
157
158        correctCovMats[0] = new Array2DRowRealMatrix(new double[][] {
159                { 4.537422569229048, 3.5266152281729304 },
160                { 3.5266152281729304, 6.175448814169779 } });
161
162        correctCovMats[1] = new Array2DRowRealMatrix( new double[][] {
163                { 2.886778573963039, 1.5257474543463154 },
164                { 1.5257474543463154, 3.3794567673616918 } });
165
166        final MultivariateNormalDistribution[] correctMVNs = new
167                MultivariateNormalDistribution[2];
168
169        correctMVNs[0] = new MultivariateNormalDistribution(correctMeans[0],
170                correctCovMats[0].getData());
171
172        correctMVNs[1] = new MultivariateNormalDistribution(correctMeans[1],
173                correctCovMats[1].getData());
174
175        final MixtureMultivariateNormalDistribution initialMix
176            = MultivariateNormalMixtureExpectationMaximization.estimate(getTestSamples(), 2);
177
178        int i = 0;
179        for (Pair<Double, MultivariateNormalDistribution> component : initialMix
180                .getComponents()) {
181            Assert.assertEquals(correctWeights[i], component.getFirst(),
182                    Math.ulp(1d));
183            
184            final double[] means = component.getValue().getMeans();
185            Assert.assertTrue(Arrays.equals(correctMeans[i], means));
186            
187            final RealMatrix covMat = component.getValue().getCovariances();
188            Assert.assertEquals(correctCovMats[i], covMat);
189            i++;
190        }
191    }

193    @Test
194    public void testFit() {
195        // Test that the loglikelihood, weights, and models are determined and
196        // fitted correctly
197        final double[][] data = getTestSamples();
198        final double correctLogLikelihood = -4.292431006791994;
199        final double[] correctWeights = new double[] { 0.2962324189652912, 0.7037675810347089 };
200        
201        final double[][] correctMeans = new double[][]{
202            {-1.4213112715121132, 1.6924690505757753},
203            {4.213612224374709, 7.975621325853645}
204        };
205        
206        final RealMatrix[] correctCovMats = new Array2DRowRealMatrix[2];
207        correctCovMats[0] = new Array2DRowRealMatrix(new double[][] {
208            { 1.739356907285747, -0.5867644251487614 },
209            { -0.5867644251487614, 1.0232932029324642 } }
210                );
211        correctCovMats[1] = new Array2DRowRealMatrix(new double[][] {
212            { 4.245384898007161, 2.5797798966382155 },
213            { 2.5797798966382155, 3.9200272522448367 } });
214        
215        final MultivariateNormalDistribution[] correctMVNs = new MultivariateNormalDistribution[2];
216        correctMVNs[0] = new MultivariateNormalDistribution(correctMeans[0], correctCovMats[0].getData());
217        correctMVNs[1] = new MultivariateNormalDistribution(correctMeans[1], correctCovMats[1].getData());
218
219        MultivariateNormalMixtureExpectationMaximization fitter
220            = new MultivariateNormalMixtureExpectationMaximization(data);
221
222        MixtureMultivariateNormalDistribution initialMix
223            = MultivariateNormalMixtureExpectationMaximization.estimate(data, 2);
224        fitter.fit(initialMix);
225        MixtureMultivariateNormalDistribution fittedMix = fitter.getFittedModel();
226        List<Pair<Double, MultivariateNormalDistribution>> components = fittedMix.getComponents();
227
228        Assert.assertEquals(correctLogLikelihood,
229                            fitter.getLogLikelihood(),
230                            Math.ulp(1d));
231
232        int i = 0;
233        for (Pair<Double, MultivariateNormalDistribution> component : components) {
234            final double weight = component.getFirst();
235            final MultivariateNormalDistribution mvn = component.getSecond();
236            final double[] mean = mvn.getMeans();
237            final RealMatrix covMat = mvn.getCovariances();
238            Assert.assertEquals(correctWeights[i], weight, Math.ulp(1d));
239            Assert.assertTrue(Arrays.equals(correctMeans[i], mean));
240            Assert.assertEquals(correctCovMats[i], covMat);
241            i++;
242        }
243    }
