083    @Test
084    public void testDfAccessors() {
085        FDistribution dist = (FDistribution) getDistribution();
086        Assert.assertEquals(5d, dist.getNumeratorDegreesOfFreedom(), Double.MIN_VALUE);
087        Assert.assertEquals(6d, dist.getDenominatorDegreesOfFreedom(), Double.MIN_VALUE);
088    }

090    @Test
091    public void testPreconditions() {
092        try {
093            new FDistribution(0, 1);

106    @Test
107    public void testLargeDegreesOfFreedom() {
108        FDistribution fd = new FDistribution(100000, 100000);
109        double p = fd.cumulativeProbability(.999);
110        double x = fd.inverseCumulativeProbability(p);
111        Assert.assertEquals(.999, x, 1.0e-5);
112    }
113
114    @Test
115    public void testSmallDegreesOfFreedom() {
116        FDistribution fd = new FDistribution(1, 1);
117        double p = fd.cumulativeProbability(0.975);
118        double x = fd.inverseCumulativeProbability(p);
119        Assert.assertEquals(0.975, x, 1.0e-5);
120
121        fd = new FDistribution(1, 2);
122        p = fd.cumulativeProbability(0.975);
123        x = fd.inverseCumulativeProbability(p);
124        Assert.assertEquals(0.975, x, 1.0e-5);
125    }
126
127    @Test
128    public void testMoments() {
129        final double tol = 1e-9;
130        FDistribution dist;
131
132        dist = new FDistribution(1, 2);
133        Assert.assertTrue(Double.isNaN(dist.getNumericalMean()));
134        Assert.assertTrue(Double.isNaN(dist.getNumericalVariance()));
135
136        dist = new FDistribution(1, 3);
137        Assert.assertEquals(dist.getNumericalMean(), 3d / (3d - 2d), tol);
138        Assert.assertTrue(Double.isNaN(dist.getNumericalVariance()));
139
140        dist = new FDistribution(1, 5);
141        Assert.assertEquals(dist.getNumericalMean(), 5d / (5d - 2d), tol);
142        Assert.assertEquals(dist.getNumericalVariance(), (2d * 5d * 5d * 4d) / 9d, tol);
143    }
