017package org.apache.commons.math3.distribution;
018
019import org.apache.commons.math3.random.Well19937a;
020import org.apache.commons.math3.util.Precision;
021import org.junit.Assert;
022import org.junit.Test;
023
024public class LevyDistributionTest extends RealDistributionAbstractTest {
025
026    @Test
027    public void testParameters() {
028        LevyDistribution d = makeDistribution();
029        Assert.assertEquals(1.2, d.getLocation(), Precision.EPSILON);
030        Assert.assertEquals(0.4,   d.getScale(),  Precision.EPSILON);
031    }
032
033    @Test
034    public void testSupport() {
035        LevyDistribution d = makeDistribution();
036        Assert.assertEquals(d.getLocation(), d.getSupportLowerBound(), Precision.EPSILON);
037        Assert.assertTrue(Double.isInfinite(d.getSupportUpperBound()));
038        Assert.assertTrue(d.isSupportConnected());
039    }
040
041    @Override
042    public LevyDistribution makeDistribution() {
043        return new LevyDistribution(new Well19937a(0xc5a5506bbb17e57al), 1.2, 0.4);
044    }
045
046    @Override
047    public double[] makeCumulativeTestPoints() {
048        return new double[] {
049            1.2001, 1.21, 1.225, 1.25, 1.3, 1.9, 3.4, 5.6
050        };
051    }
052
053    @Override
054    public double[] makeCumulativeTestValues() {
055        // values computed with R and function plevy from rmutil package

059        };
060    }
