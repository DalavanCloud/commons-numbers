065    @Test
066    public void testInverseCumulativeProbabilitySmallPAccuracy() {
067        WeibullDistribution dist = new WeibullDistribution(2, 3);
068        double t = dist.inverseCumulativeProbability(1e-17);
069        // Analytically, answer is solution to 1e-17 = 1-exp(-(x/3)^2)
070        // x = sqrt(-9*log(1-1e-17))
071        // If we're not careful, answer will be 0. Answer below is computed with care in Octave:
072        Assert.assertEquals(9.48683298050514e-9, t, 1e-17);
073    }

106
107    @Test
108    public void testMoments() {
109        final double tol = 1e-9;
110        WeibullDistribution dist;
111
112        dist = new WeibullDistribution(2.5, 3.5);
113        // In R: 3.5*gamma(1+(1/2.5)) (or emperically: mean(rweibull(10000, 2.5, 3.5)))
114        Assert.assertEquals(dist.getNumericalMean(), 3.5 * FastMath.exp(Gamma.logGamma(1 + (1 / 2.5))), tol);
115        Assert.assertEquals(dist.getNumericalVariance(), (3.5 * 3.5) *
116                FastMath.exp(Gamma.logGamma(1 + (2 / 2.5))) -
117                (dist.getNumericalMean() * dist.getNumericalMean()), tol);
118
119        dist = new WeibullDistribution(10.4, 2.222);
120        Assert.assertEquals(dist.getNumericalMean(), 2.222 * FastMath.exp(Gamma.logGamma(1 + (1 / 10.4))), tol);
121        Assert.assertEquals(dist.getNumericalVariance(), (2.222 * 2.222) *
122                FastMath.exp(Gamma.logGamma(1 + (2 / 10.4))) -
123                (dist.getNumericalMean() * dist.getNumericalMean()), tol);
124    }
