322    @Test
323    public void testAdd() {
324        Fraction a = new Fraction(1, 2);
325        Fraction b = new Fraction(2, 3);
326
327        assertFraction(1, 1, a.add(a));
328        assertFraction(7, 6, a.add(b));
329        assertFraction(7, 6, b.add(a));
330        assertFraction(4, 3, b.add(b));
331
332        Fraction f1 = new Fraction(Integer.MAX_VALUE - 1, 1);
333        Fraction f2 = Fraction.ONE;
334        Fraction f = f1.add(f2);
335        Assert.assertEquals(Integer.MAX_VALUE, f.getNumerator());
336        Assert.assertEquals(1, f.getDenominator());
337        f = f1.add(1);
338        Assert.assertEquals(Integer.MAX_VALUE, f.getNumerator());
339        Assert.assertEquals(1, f.getDenominator());
340
341        f1 = new Fraction(-1, 13*13*2*2);
342        f2 = new Fraction(-2, 13*17*2);
343        f = f1.add(f2);
344        Assert.assertEquals(13*13*17*2*2, f.getDenominator());
345        Assert.assertEquals(-17 - 2*13*2, f.getNumerator());
346
347        try {
348            f.add(null);
349            Assert.fail("expecting MathIllegalArgumentException");
350        } catch (MathIllegalArgumentException ex) {}
351
352        // if this fraction is added naively, it will overflow.
353        // check that it doesn't.
354        f1 = new Fraction(1,32768*3);
355        f2 = new Fraction(1,59049);
356        f = f1.add(f2);
357        Assert.assertEquals(52451, f.getNumerator());
358        Assert.assertEquals(1934917632, f.getDenominator());
359
360        f1 = new Fraction(Integer.MIN_VALUE, 3);
361        f2 = new Fraction(1,3);
362        f = f1.add(f2);
363        Assert.assertEquals(Integer.MIN_VALUE+1, f.getNumerator());
364        Assert.assertEquals(3, f.getDenominator());
365
366        f1 = new Fraction(Integer.MAX_VALUE - 1, 1);
367        f2 = Fraction.ONE;
368        f = f1.add(f2);
369        Assert.assertEquals(Integer.MAX_VALUE, f.getNumerator());
370        Assert.assertEquals(1, f.getDenominator());
371
372        try {
373            f = f.add(Fraction.ONE); // should overflow
374            Assert.fail("expecting MathArithmeticException but got: " + f.toString());
375        } catch (MathArithmeticException ex) {}
376
377        // denominator should not be a multiple of 2 or 3 to trigger overflow
378        f1 = new Fraction(Integer.MIN_VALUE, 5);
379        f2 = new Fraction(-1,5);
380        try {
381            f = f1.add(f2); // should overflow
382            Assert.fail("expecting MathArithmeticException but got: " + f.toString());
383        } catch (MathArithmeticException ex) {}
384
385        try {
386            f= new Fraction(-Integer.MAX_VALUE, 1);
387            f = f.add(f);
