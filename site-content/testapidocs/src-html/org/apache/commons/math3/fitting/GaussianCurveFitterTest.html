378    /**
379     * Adds the specified points to specified <code>GaussianCurveFitter</code>
380     * instance.
381     *
382     * @param points Data points where first dimension is a point index and
383     *        second dimension is an array of length two representing the point
384     *        with the first value corresponding to X and the second value
385     *        corresponding to Y.
386     * @return the collection of observed points.
387     */
388    private static WeightedObservedPoints createDataset(double[][] points) {
389        final WeightedObservedPoints obs = new WeightedObservedPoints();
390        for (int i = 0; i < points.length; i++) {
391            obs.add(points[i][0], points[i][1]);
392        }
393        return obs;
394    }
