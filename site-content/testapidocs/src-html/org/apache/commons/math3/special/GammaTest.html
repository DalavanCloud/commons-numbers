029    private void testRegularizedGamma(double expected, double a, double x) {
030        double actualP = Gamma.regularizedGammaP(a, x);
031        double actualQ = Gamma.regularizedGammaQ(a, x);
032        TestUtils.assertEquals(expected, actualP, 10e-15);
033        TestUtils.assertEquals(actualP, 1.0 - actualQ, 10e-15);
034    }

036    private void testLogGamma(double expected, double x) {
037        double actual = Gamma.logGamma(x);
038        TestUtils.assertEquals(expected, actual, 10e-15);
039    }

932        for (int i = 0; i < GAMMA_REF.length; i++) {
933            final double[] ref = GAMMA_REF[i];
934            final double x = ref[0];
935            final double expected = ref[1];

937            final double absX = FastMath.abs(x);
938            final int ulps;
939            if (absX <= 8.0) {
940                ulps = 3;
941            } else if (absX <= 20.0) {
942                ulps = 5;
943            } else if (absX <= 30.0) {
944                ulps = 50;
945            } else if (absX <= 50.0) {
946                ulps = 180;
947            } else {
948                ulps = 500;
949            }
950            final double tol = ulps * FastMath.ulp(expected);
951            Assert.assertEquals(Double.toString(x), expected, actual, tol);
952        }
