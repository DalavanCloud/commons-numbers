090    @Test
091    public void testRosenbrock() {
092        MultivariateFunction rosenbrock =
093            new MultivariateFunction() {
094                public double value(double[] x) {
095                    ++count;
096                    double a = x[1] - x[0] * x[0];
097                    double b = 1.0 - x[0];
098                    return 100 * a * a + b * b;
099                }
100            };
101
102        count = 0;
103        SimplexOptimizer optimizer = new SimplexOptimizer(-1, 1e-3);
104        optimizer.setSimplex(new MultiDirectionalSimplex(new double[][] {
105                    { -1.2,  1.0 }, { 0.9, 1.2 } , {  3.5, -2.3 }
106                }));
107        PointValuePair optimum =
108            optimizer.optimize(100, rosenbrock, GoalType.MINIMIZE, new double[] { -1.2, 1 });
109
110        Assert.assertEquals(count, optimizer.getEvaluations());
111        Assert.assertTrue(optimizer.getEvaluations() > 50);
112        Assert.assertTrue(optimizer.getEvaluations() < 100);
113        Assert.assertTrue(optimum.getValue() > 1e-2);
114    }

116    @Test
117    public void testPowell() {
118        MultivariateFunction powell =
119            new MultivariateFunction() {
120                public double value(double[] x) {
121                    ++count;
122                    double a = x[0] + 10 * x[1];
123                    double b = x[2] - x[3];
124                    double c = x[1] - 2 * x[2];
125                    double d = x[0] - x[3];
126                    return a * a + 5 * b * b + c * c * c * c + 10 * d * d * d * d;
127                }
128            };
129
130        count = 0;
131        SimplexOptimizer optimizer = new SimplexOptimizer(-1, 1e-3);
132        optimizer.setSimplex(new MultiDirectionalSimplex(4));
133        PointValuePair optimum =
134            optimizer.optimize(1000, powell, GoalType.MINIMIZE, new double[] { 3, -1, 0, 1 });
135        Assert.assertEquals(count, optimizer.getEvaluations());
136        Assert.assertTrue(optimizer.getEvaluations() > 800);
137        Assert.assertTrue(optimizer.getEvaluations() < 900);
138        Assert.assertTrue(optimum.getValue() > 1e-2);
139    }

142    public void testMath283() {
143        // fails because MultiDirectional.iterateSimplex is looping forever
144        // the while(true) should be replaced with a convergence check
145        SimplexOptimizer optimizer = new SimplexOptimizer(1e-14, 1e-14);
146        optimizer.setSimplex(new MultiDirectionalSimplex(2));
147        final Gaussian2D function = new Gaussian2D(0, 0, 1);
148        PointValuePair estimate = optimizer.optimize(1000, function,
149                                                         GoalType.MAXIMIZE, function.getMaximumPosition());
150        final double EPSILON = 1e-5;
151        final double expectedMaximum = function.getMaximum();
152        final double actualMaximum = estimate.getValue();
153        Assert.assertEquals(expectedMaximum, actualMaximum, EPSILON);
154
155        final double[] expectedPosition = function.getMaximumPosition();
156        final double[] actualPosition = estimate.getPoint();
157        Assert.assertEquals(expectedPosition[0], actualPosition[0], EPSILON );
158        Assert.assertEquals(expectedPosition[1], actualPosition[1], EPSILON );
159    }

161    private static class FourExtrema implements MultivariateFunction {
162        // The following function has 4 local extrema.
163        final double xM = -3.841947088256863675365;
164        final double yM = -1.391745200270734924416;
165        final double xP =  0.2286682237349059125691;
166        final double yP = -yM;
167        final double valueXmYm = 0.2373295333134216789769; // Local maximum.
168        final double valueXmYp = -valueXmYm; // Local minimum.
169        final double valueXpYm = -0.7290400707055187115322; // Global minimum.
170        final double valueXpYp = -valueXpYm; // Global maximum.
171
172        public double value(double[] variables) {
173            final double x = variables[0];
174            final double y = variables[1];
175            return (x == 0 || y == 0) ? 0 :
176                FastMath.atan(x) * FastMath.atan(x + 2) * FastMath.atan(y) * FastMath.atan(y) / (x * y);
177        }
178    }
179
180    private static class Gaussian2D implements MultivariateFunction {
181        private final double[] maximumPosition;
182        private final double std;
183
184        public Gaussian2D(double xOpt, double yOpt, double std) {
185            maximumPosition = new double[] { xOpt, yOpt };
186            this.std = std;
187        }
188
189        public double getMaximum() {
190            return value(maximumPosition);
191        }
192
193        public double[] getMaximumPosition() {
194            return maximumPosition.clone();
195        }
196
197        public double value(double[] point) {
198            final double x = point[0], y = point[1];
199            final double twoS2 = 2.0 * std * std;
200            return 1.0 / (twoS2 * FastMath.PI) * FastMath.exp(-(x * x + y * y) / twoS2);
201        }
202    }
