577    /**
578     * Converts a test string to a {@link LinearConstraint}.
579     * Ex: x0 + x1 + x2 + x3 - x12 = 0
580     */
581    private LinearConstraint equationFromString(int numCoefficients, String s) {
582        Relationship relationship;
583        if (s.contains(">=")) {
584            relationship = Relationship.GEQ;
585        } else if (s.contains("<=")) {
586            relationship = Relationship.LEQ;
587        } else if (s.contains("=")) {
588            relationship = Relationship.EQ;
589        } else {
590            throw new IllegalArgumentException();
591        }
592
593        String[] equationParts = s.split("[>|<]?=");
594        double rhs = Double.parseDouble(equationParts[1].trim());
595
596        double[] lhs = new double[numCoefficients];
597        String left = equationParts[0].replaceAll(" ?x", "");
598        String[] coefficients = left.split(" ");
599        for (String coefficient : coefficients) {
600            double value = coefficient.charAt(0) == '-' ? -1 : 1;
601            int index = Integer.parseInt(coefficient.replaceFirst("[+|-]", "").trim());
602            lhs[index] = value;
603        }
604        return new LinearConstraint(lhs, relationship, rhs);
605    }

607    private static boolean validSolution(PointValuePair solution, List<LinearConstraint> constraints, double epsilon) {
608        double[] vals = solution.getPoint();
609        for (LinearConstraint c : constraints) {
610            double[] coeffs = c.getCoefficients().toArray();
611            double result = 0.0d;
612            for (int i = 0; i < vals.length; i++) {
613                result += vals[i] * coeffs[i];
614            }
615            
616            switch (c.getRelationship()) {
617            case EQ:
618                if (!Precision.equals(result, c.getValue(), epsilon)) {
619                    return false;
620                }
621                break;
622                
623            case GEQ:
624                if (Precision.compareTo(result, c.getValue(), epsilon) < 0) {
625                    return false;
626                }
627                break;
628                
629            case LEQ:
630                if (Precision.compareTo(result, c.getValue(), epsilon) > 0) {
631                    return false;
632                }
633                break;
634            }
635        }
636        
637        return true;
638    }
