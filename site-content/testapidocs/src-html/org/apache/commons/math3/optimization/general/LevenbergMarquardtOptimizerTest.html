143    private void checkEstimate(MultivariateDifferentiableVectorFunction problem,
144                               double initialStepBoundFactor, int maxCostEval,
145                               double costRelativeTolerance, double parRelativeTolerance,
146                               double orthoTolerance, boolean shouldFail) {
147        try {
148            LevenbergMarquardtOptimizer optimizer
149                = new LevenbergMarquardtOptimizer(initialStepBoundFactor,
150                                                  costRelativeTolerance,
151                                                  parRelativeTolerance,
152                                                  orthoTolerance,
153                                                  Precision.SAFE_MIN);
154            optimizer.optimize(maxCostEval, problem, new double[] { 0, 0, 0, 0, 0 },
155                               new double[] { 1, 1, 1, 1, 1 },
156                               new double[] { 98.680, 47.345 });
157            Assert.assertTrue(!shouldFail);
158        } catch (DimensionMismatchException ee) {
159            Assert.assertTrue(shouldFail);
160        } catch (TooManyEvaluationsException ee) {
161            Assert.assertTrue(shouldFail);
162        }
163    }

194    @Test
195    public void testBevington() {
196        final double[][] dataPoints = {
197            // column 1 = times
198            { 15, 30, 45, 60, 75, 90, 105, 120, 135, 150,
199              165, 180, 195, 210, 225, 240, 255, 270, 285, 300,
200              315, 330, 345, 360, 375, 390, 405, 420, 435, 450,
201              465, 480, 495, 510, 525, 540, 555, 570, 585, 600,
202              615, 630, 645, 660, 675, 690, 705, 720, 735, 750,
203              765, 780, 795, 810, 825, 840, 855, 870, 885, },
204            // column 2 = measured counts
205            { 775, 479, 380, 302, 185, 157, 137, 119, 110, 89,
206              74, 61, 66, 68, 48, 54, 51, 46, 55, 29,
207              28, 37, 49, 26, 35, 29, 31, 24, 25, 35,
208              24, 30, 26, 28, 21, 18, 20, 27, 17, 17,
209              14, 17, 24, 11, 22, 17, 12, 10, 13, 16,
210              9, 9, 14, 21, 17, 13, 12, 18, 10, },
211        };
212
213        final BevingtonProblem problem = new BevingtonProblem();
214
215        final int len = dataPoints[0].length;
216        final double[] weights = new double[len];
217        for (int i = 0; i < len; i++) {
218            problem.addPoint(dataPoints[0][i],
219                             dataPoints[1][i]);
220
221            weights[i] = 1 / dataPoints[1][i];
222        }
223
224        final LevenbergMarquardtOptimizer optimizer
225            = new LevenbergMarquardtOptimizer();
226
227        final PointVectorValuePair optimum
228            = optimizer.optimize(100, problem, dataPoints[1], weights,
229                               new double[] { 10, 900, 80, 27, 225 });
230
231        final double[] solution = optimum.getPoint();
232        final double[] expectedSolution = { 10.4, 958.3, 131.4, 33.9, 205.0 };
233
234        final double[][] covarMatrix = optimizer.computeCovariances(solution, 1e-14);
235        final double[][] expectedCovarMatrix = {
236            { 3.38, -3.69, 27.98, -2.34, -49.24 },
237            { -3.69, 2492.26, 81.89, -69.21, -8.9 },
238            { 27.98, 81.89, 468.99, -44.22, -615.44 },
239            { -2.34, -69.21, -44.22, 6.39, 53.80 },
240            { -49.24, -8.9, -615.44, 53.8, 929.45 }
241        };
242
243        final int numParams = expectedSolution.length;
244
245        // Check that the computed solution is within the reference error range.
246        for (int i = 0; i < numParams; i++) {
247            final double error = FastMath.sqrt(expectedCovarMatrix[i][i]);
248            Assert.assertEquals("Parameter " + i, expectedSolution[i], solution[i], error);
249        }
250
251        // Check that each entry of the computed covariance matrix is within 10%
252        // of the reference matrix entry.
253        for (int i = 0; i < numParams; i++) {
254            for (int j = 0; j < numParams; j++) {
255                Assert.assertEquals("Covariance matrix [" + i + "][" + j + "]",
256                                    expectedCovarMatrix[i][j],
257                                    covarMatrix[i][j],
258                                    FastMath.abs(0.1 * expectedCovarMatrix[i][j]));
259            }
260        }
261    }

263    @Test
264    public void testCircleFitting2() {
265        final double xCenter = 123.456;
266        final double yCenter = 654.321;
267        final double xSigma = 10;
268        final double ySigma = 15;
269        final double radius = 111.111;
270        // The test is extremely sensitive to the seed.
271        final long seed = 59421061L;
272        final RandomCirclePointGenerator factory
273            = new RandomCirclePointGenerator(xCenter, yCenter, radius,
274                                             xSigma, ySigma,
275                                             seed);
276        final CircleProblem circle = new CircleProblem(xSigma, ySigma);
277
278        final int numPoints = 10;
279        for (Vector2D p : factory.generate(numPoints)) {
280            circle.addPoint(p);
281            // System.out.println(p.x + " " + p.y);
282        }
283
284        // First guess for the center's coordinates and radius.
285        final double[] init = { 90, 659, 115 };
286
287        final LevenbergMarquardtOptimizer optimizer
288            = new LevenbergMarquardtOptimizer();
289        final PointVectorValuePair optimum = optimizer.optimize(100, circle,
290                                                                circle.target(), circle.weight(),
291                                                                init);
292
293        final double[] paramFound = optimum.getPoint();
294
295        // Retrieve errors estimation.
296        final double[][] covMatrix = optimizer.computeCovariances(paramFound, 1e-14);
297        final double[] asymptoticStandardErrorFound = optimizer.guessParametersErrors();
298        final double[] sigmaFound = new double[covMatrix.length];
299        for (int i = 0; i < covMatrix.length; i++) {
300            sigmaFound[i] = FastMath.sqrt(covMatrix[i][i]);
301//             System.out.println("i=" + i + " value=" + paramFound[i]
302//                                + " sigma=" + sigmaFound[i]
303//                                + " ase=" + asymptoticStandardErrorFound[i]);
304        }
305
306        // System.out.println("chi2=" + optimizer.getChiSquare());
307
308        // Check that the parameters are found within the assumed error bars.
309        Assert.assertEquals(xCenter, paramFound[0], asymptoticStandardErrorFound[0]);
310        Assert.assertEquals(yCenter, paramFound[1], asymptoticStandardErrorFound[1]);
311        Assert.assertEquals(radius, paramFound[2], asymptoticStandardErrorFound[2]);
312    }

314    private static class QuadraticProblem implements MultivariateDifferentiableVectorFunction, Serializable {
315
316        private static final long serialVersionUID = 7072187082052755854L;
317        private List<Double> x;
318        private List<Double> y;
319
320        public QuadraticProblem() {
321            x = new ArrayList<Double>();
322            y = new ArrayList<Double>();
323        }
324
325        public void addPoint(double x, double y) {
326            this.x.add(x);
327            this.y.add(y);
328        }
329
330        public double[] value(double[] variables) {
331            double[] values = new double[x.size()];
332            for (int i = 0; i < values.length; ++i) {
333                values[i] = (variables[0] * x.get(i) + variables[1]) * x.get(i) + variables[2];
334            }
335            return values;
336        }
337
338        public DerivativeStructure[] value(DerivativeStructure[] variables) {
339            DerivativeStructure[] values = new DerivativeStructure[x.size()];
340            for (int i = 0; i < values.length; ++i) {
341                values[i] = (variables[0].multiply(x.get(i)).add(variables[1])).multiply(x.get(i)).add(variables[2]);
342            }
343            return values;
344        }
345
346    }

348    private static class BevingtonProblem
349        implements MultivariateDifferentiableVectorFunction {
350        private List<Double> time;
351        private List<Double> count;
352
353        public BevingtonProblem() {
354            time = new ArrayList<Double>();
355            count = new ArrayList<Double>();
356        }
357
358        public void addPoint(double t, double c) {
359            time.add(t);
360            count.add(c);
361        }
362
363        public double[] value(double[] params) {
364            double[] values = new double[time.size()];
365            for (int i = 0; i < values.length; ++i) {
366                final double t = time.get(i);
367                values[i] = params[0]
368                    + params[1] * FastMath.exp(-t / params[3])
369                    + params[2] * FastMath.exp(-t / params[4]);
370            }
371            return values;
372        }
373
374        public DerivativeStructure[] value(DerivativeStructure[] params) {
375            DerivativeStructure[] values = new DerivativeStructure[time.size()];
376            for (int i = 0; i < values.length; ++i) {
377                final double t = time.get(i);
378                values[i] = params[0].add(
379                    params[1].multiply(params[3].reciprocal().multiply(-t).exp())).add(
380                    params[2].multiply(params[4].reciprocal().multiply(-t).exp()));
381            }
382            return values;
383        }
