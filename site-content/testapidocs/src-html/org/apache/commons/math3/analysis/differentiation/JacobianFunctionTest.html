018package org.apache.commons.math3.analysis.differentiation;
019
020import org.junit.Assert;
021
022import org.apache.commons.math3.TestUtils;
023import org.apache.commons.math3.util.FastMath;
024import org.junit.Test;
025
026
027/**
028 * Test for class {@link JacobianFunction}.
029 */
030public class JacobianFunctionTest {
031
032    @Test
033    public void testSphere() {
034        SphereMapping    f = new SphereMapping(10.0);
035        JacobianFunction j = new JacobianFunction(f);
036        for (double latitude = -1.5; latitude < 1.5; latitude += 0.1) {
037            for (double longitude = -3.1; longitude < 3.1; longitude += 0.1) {
038                double[] point = new double[] { latitude, longitude };
039                double[][] referenceJacobian  = f.jacobian(point);
040                double[][] testJacobian       = j.value(point);
041                Assert.assertEquals(referenceJacobian.length, testJacobian.length);
042                for (int i = 0; i < 3; ++i) {
043                    TestUtils.assertEquals(referenceJacobian[i], testJacobian[i], 2.0e-15);
044                }
045            }
046        }
047    }
048
049    /* Maps (latitude, longitude) to (x, y, z) */
050    private static class SphereMapping implements MultivariateDifferentiableVectorFunction {
051
052        private final double radius;
053
054        public SphereMapping(final double radius) {
055            this.radius = radius;
056        }
057        
058        public double[] value(double[] point) {
059            final double cLat = FastMath.cos(point[0]);
060            final double sLat = FastMath.sin(point[0]);
061            final double cLon = FastMath.cos(point[1]);
062            final double sLon = FastMath.sin(point[1]);
063            return new double[] {
064                radius * cLon * cLat,
065                radius * sLon * cLat,
066                radius * sLat
067            };
068        }
069        
070        public DerivativeStructure[] value(DerivativeStructure[] point) {
071            final DerivativeStructure cLat = point[0].cos();
072            final DerivativeStructure sLat = point[0].sin();
073            final DerivativeStructure cLon = point[1].cos();
074            final DerivativeStructure sLon = point[1].sin();
075            return new DerivativeStructure[] {
076                cLon.multiply(cLat).multiply(radius),
077                sLon.multiply(cLat).multiply(radius),
078                sLat.multiply(radius)
079            };
080        }
081
082        public double[][] jacobian(double[] point) {
083            final double cLat = FastMath.cos(point[0]);
084            final double sLat = FastMath.sin(point[0]);
085            final double cLon = FastMath.cos(point[1]);
086            final double sLon = FastMath.sin(point[1]);
087            return new double[][] {
088                { -radius * cLon * sLat, -radius * sLon * cLat },
089                { -radius * sLon * sLat,  radius * cLon * cLat },
090                {  radius * cLat,         0  }
091            };
092        }
093
094    }
095
096}
