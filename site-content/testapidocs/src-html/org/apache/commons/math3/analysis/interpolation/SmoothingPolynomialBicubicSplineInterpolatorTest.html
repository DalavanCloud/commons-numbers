030public final class SmoothingPolynomialBicubicSplineInterpolatorTest {
031    /**
032     * Test preconditions.
033     */
034    @Test
035    public void testPreconditions() {
036        double[] xval = new double[] {3, 4, 5, 6.5, 7.5};
037        double[] yval = new double[] {-4, -3, -1, 2.5, 3};
038        double[][] zval = new double[xval.length][yval.length];
039
040        BivariateGridInterpolator interpolator = new SmoothingPolynomialBicubicSplineInterpolator(0);
041        
042        @SuppressWarnings("unused")
043        BivariateFunction p = interpolator.interpolate(xval, yval, zval);
044        
045        double[] wxval = new double[] {3, 2, 5, 6.5};
046        try {
047            p = interpolator.interpolate(wxval, yval, zval);
048            Assert.fail("an exception should have been thrown");
049        } catch (MathIllegalArgumentException e) {
050            // Expected
051        }
052
053        double[] wyval = new double[] {-4, -3, -1, -1};
054        try {
055            p = interpolator.interpolate(xval, wyval, zval);
056            Assert.fail("an exception should have been thrown");
057        } catch (MathIllegalArgumentException e) {
058            // Expected
059        }
060
061        double[][] wzval = new double[xval.length][yval.length + 1];
062        try {
063            p = interpolator.interpolate(xval, yval, wzval);
064            Assert.fail("an exception should have been thrown");
065        } catch (DimensionMismatchException e) {
066            // Expected
067        }
068        wzval = new double[xval.length - 1][yval.length];
069        try {
070            p = interpolator.interpolate(xval, yval, wzval);
071            Assert.fail("an exception should have been thrown");
072        } catch (DimensionMismatchException e) {
073            // Expected
074        }
075        wzval = new double[xval.length][yval.length - 1];
076        try {
077            p = interpolator.interpolate(xval, yval, wzval);
078            Assert.fail("an exception should have been thrown");
079        } catch (DimensionMismatchException e) {
080            // Expected
081        }
082    }
083
084    /**
085     * Test of interpolator for a plane.
086     * <p>
087     * z = 2 x - 3 y + 5
088     */
089    @Test
090    public void testPlane() {
091        BivariateFunction f = new BivariateFunction() {
092                public double value(double x, double y) {
093                    return 2 * x - 3 * y + 5
094                        + ((int) (FastMath.abs(5 * x + 3 * y)) % 2 == 0 ? 1 : -1);
095                }
096            };
097
098        BivariateGridInterpolator interpolator = new SmoothingPolynomialBicubicSplineInterpolator(1);
099
100        double[] xval = new double[] {3, 4, 5, 6.5, 7.5};
101        double[] yval = new double[] {-4, -3, -1, 2, 2.5, 3.5};
102        double[][] zval = new double[xval.length][yval.length];
103        for (int i = 0; i < xval.length; i++) {
104            for (int j = 0; j < yval.length; j++) {
105                zval[i][j] = f.value(xval[i], yval[j]);
106            }
107        }
108
109        BivariateFunction p = interpolator.interpolate(xval, yval, zval);
110        double x, y;
111        double expected, result;
112        
113        x = 4;
114        y = -3;
115        expected = f.value(x, y);
116        result = p.value(x, y);
117        Assert.assertEquals("On sample point", expected, result, 2);
118
119        x = 4.5;
120        y = -1.5;
121        expected = f.value(x, y);
122        result = p.value(x, y);
123        Assert.assertEquals("half-way between sample points (middle of the patch)", expected, result, 2);
124
125        x = 3.5;
126        y = -3.5;
127        expected = f.value(x, y);
128        result = p.value(x, y);
129        Assert.assertEquals("half-way between sample points (border of the patch)", expected, result, 2);
130    }

132    /**
133     * Test of interpolator for a paraboloid.
134     * <p>
135     * z = 2 x<sup>2</sup> - 3 y<sup>2</sup> + 4 x y - 5
136     */
137    @Test
138    public void testParaboloid() {
139        BivariateFunction f = new BivariateFunction() {
140                public double value(double x, double y) {
141                    return 2 * x * x - 3 * y * y + 4 * x * y - 5
142                        + ((int) (FastMath.abs(5 * x + 3 * y)) % 2 == 0 ? 1 : -1);
143                }
144            };
145
146        BivariateGridInterpolator interpolator = new SmoothingPolynomialBicubicSplineInterpolator(4);
147
148        double[] xval = new double[] {3, 4, 5, 6.5, 7.5, 8};
149        double[] yval = new double[] {-4, -3, -2, -1, 0.5, 2.5};
150        double[][] zval = new double[xval.length][yval.length];
151        for (int i = 0; i < xval.length; i++) {
152            for (int j = 0; j < yval.length; j++) {
153                zval[i][j] = f.value(xval[i], yval[j]);
154            }
155        }
156
157        BivariateFunction p = interpolator.interpolate(xval, yval, zval);
158        double x, y;
159        double expected, result;
160
161        x = 5;
162        y = 0.5;
163        expected = f.value(x, y);
164        result = p.value(x, y);
165        Assert.assertEquals("On sample point", expected, result, 2);
166
167        x = 4.5;
168        y = -1.5;
169        expected = f.value(x, y);
170        result = p.value(x, y);
171        Assert.assertEquals("half-way between sample points (middle of the patch)", expected, result, 2);
172
173        x = 3.5;
174        y = -3.5;
175        expected = f.value(x, y);
176        result = p.value(x, y);
177        Assert.assertEquals("half-way between sample points (border of the patch)", expected, result, 2);
178    }
