018package org.apache.commons.math3.analysis.function;
019
020import org.apache.commons.math3.analysis.UnivariateFunction;
021import org.apache.commons.math3.analysis.differentiation.DerivativeStructure;
022import org.apache.commons.math3.exception.DimensionMismatchException;
023import org.apache.commons.math3.exception.NullArgumentException;
024import org.apache.commons.math3.util.FastMath;
025import org.apache.commons.math3.util.Precision;
026import org.junit.Assert;
027import org.junit.Test;
028
029/**
030 * Test for class {@link HarmonicOscillator}.
031 */
032public class HarmonicOscillatorTest {
033    private final double EPS = Math.ulp(1d);
034
035    @Test
036    public void testSomeValues() {
037        final double a = -1.2;
038        final double w = 0.34;
039        final double p = 5.6;
040        final UnivariateFunction f = new HarmonicOscillator(a, w, p);
041
042        final double d = 0.12345;
043        for (int i = 0; i < 10; i++) {
044            final double v = i * d;
045            Assert.assertEquals(a * FastMath.cos(w * v + p), f.value(v), 0);
046        }
047    }
048
049    @Test
050    public void testDerivative() {
051        final double a = -1.2;
052        final double w = 0.34;
053        final double p = 5.6;
054        final HarmonicOscillator f = new HarmonicOscillator(a, w, p);
055
056        for (int maxOrder = 0; maxOrder < 6; ++maxOrder) {
057            final double d = 0.12345;
058            for (int i = 0; i < 10; i++) {
059                final double v = i * d;
060                final DerivativeStructure h = f.value(new DerivativeStructure(1, maxOrder, 0, v));
061                for (int k = 0; k <= maxOrder; ++k) {
062                    final double trigo;
063                    switch (k % 4) {
064                        case 0:
065                            trigo = +FastMath.cos(w * v + p);
066                            break;
067                        case 1:
068                            trigo = -FastMath.sin(w * v + p);
069                            break;
070                        case 2:
071                            trigo = -FastMath.cos(w * v + p);
072                            break;
073                        default:
074                            trigo = +FastMath.sin(w * v + p);
075                            break;
076                    }
077                    Assert.assertEquals(a * FastMath.pow(w, k) * trigo,
078                                        h.getPartialDerivative(k),
079                                        Precision.EPSILON);
080                }
081            }
082        }
083    }
084
085    @Test(expected=NullArgumentException.class)
086    public void testParametricUsage1() {
087        final HarmonicOscillator.Parametric g = new HarmonicOscillator.Parametric();
088        g.value(0, null);
089    }
090
091    @Test(expected=DimensionMismatchException.class)
092    public void testParametricUsage2() {
093        final HarmonicOscillator.Parametric g = new HarmonicOscillator.Parametric();
094        g.value(0, new double[] {0});
095    }
096
097    @Test(expected=NullArgumentException.class)
098    public void testParametricUsage3() {
099        final HarmonicOscillator.Parametric g = new HarmonicOscillator.Parametric();
100        g.gradient(0, null);
101    }
102
103    @Test(expected=DimensionMismatchException.class)
104    public void testParametricUsage4() {
105        final HarmonicOscillator.Parametric g = new HarmonicOscillator.Parametric();
106        g.gradient(0, new double[] {0});
107    }
108
109    @Test
110    public void testParametricValue() {
111        final double amplitude = 2;
112        final double omega = 3;
113        final double phase = 4;
114        final HarmonicOscillator f = new HarmonicOscillator(amplitude, omega, phase);
115
116        final HarmonicOscillator.Parametric g = new HarmonicOscillator.Parametric();
117        Assert.assertEquals(f.value(-1), g.value(-1, new double[] {amplitude, omega, phase}), 0);
118        Assert.assertEquals(f.value(0), g.value(0, new double[] {amplitude, omega, phase}), 0);
119        Assert.assertEquals(f.value(2), g.value(2, new double[] {amplitude, omega, phase}), 0);
120    }
121
122    @Test
123    public void testParametricGradient() {
124        final double amplitude = 2;
125        final double omega = 3;
126        final double phase = 4;
127        final HarmonicOscillator.Parametric f = new HarmonicOscillator.Parametric();
128
129        final double x = 1;
130        final double[] grad = f.gradient(1, new double[] {amplitude, omega, phase});
131        final double xTimesOmegaPlusPhase = omega * x + phase;
132        final double a = FastMath.cos(xTimesOmegaPlusPhase);
133        Assert.assertEquals(a, grad[0], EPS);
134        final double w = -amplitude * x * FastMath.sin(xTimesOmegaPlusPhase);
135        Assert.assertEquals(w, grad[1], EPS);
136        final double p = -amplitude * FastMath.sin(xTimesOmegaPlusPhase);
137        Assert.assertEquals(p, grad[2], EPS);
138    }
139}
