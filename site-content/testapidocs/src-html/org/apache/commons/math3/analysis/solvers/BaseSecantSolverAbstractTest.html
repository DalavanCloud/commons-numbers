029/**
030 * Base class for root-finding algorithms tests derived from
031 * {@link BaseSecantSolver}.

034public abstract class BaseSecantSolverAbstractTest {
035    /** Returns the solver to use to perform the tests.
036     * @return the solver to use to perform the tests
037     */
038    protected abstract UnivariateSolver getSolver();
039
040    /** Returns the expected number of evaluations for the
041     * {@link #testQuinticZero} unit test. A value of {@code -1} indicates that
042     * the test should be skipped for that solver.
043     * @return the expected number of evaluations for the
044     * {@link #testQuinticZero} unit test
045     */
046    protected abstract int[] getQuinticEvalCounts();

048    @Test
049    public void testSinZero() {
050        // The sinus function is behaved well around the root at pi. The second
051        // order derivative is zero, which means linear approximating methods
052        // still converge quadratically.
053        UnivariateFunction f = new Sin();
054        double result;
055        UnivariateSolver solver = getSolver();
056
057        result = solver.solve(100, f, 3, 4);
058        //System.out.println(
059        //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
060        Assert.assertEquals(result, FastMath.PI, solver.getAbsoluteAccuracy());
061        Assert.assertTrue(solver.getEvaluations() <= 6);
062        result = solver.solve(100, f, 1, 4);
063        //System.out.println(
064        //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
065        Assert.assertEquals(result, FastMath.PI, solver.getAbsoluteAccuracy());
066        Assert.assertTrue(solver.getEvaluations() <= 7);
067    }

069    @Test
070    public void testQuinticZero() {
071        // The quintic function has zeros at 0, +-0.5 and +-1.
072        // Around the root of 0 the function is well behaved, with a second
073        // derivative of zero a 0.
074        // The other roots are less well to find, in particular the root at 1,
075        // because the function grows fast for x>1.
076        // The function has extrema (first derivative is zero) at 0.27195613
077        // and 0.82221643, intervals containing these values are harder for
078        // the solvers.
079        UnivariateFunction f = new QuinticFunction();
080        double result;
081        UnivariateSolver solver = getSolver();
082        double atol = solver.getAbsoluteAccuracy();
083        int[] counts = getQuinticEvalCounts();
084
085        // Tests data: initial bounds, and expected solution, per test case.
086        double[][] testsData = {{-0.2,  0.2,  0.0},
087                                {-0.1,  0.3,  0.0},
088                                {-0.3,  0.45, 0.0},
089                                { 0.3,  0.7,  0.5},
090                                { 0.2,  0.6,  0.5},
091                                { 0.05, 0.95, 0.5},
092                                { 0.85, 1.25, 1.0},
093                                { 0.8,  1.2,  1.0},
094                                { 0.85, 1.75, 1.0},
095                                { 0.55, 1.45, 1.0},
096                                { 0.85, 5.0,  1.0},
097                               };
098        int maxIter = 500;
099
100        for(int i = 0; i < testsData.length; i++) {
101            // Skip test, if needed.
102            if (counts[i] == -1) continue;
103
104            // Compute solution.
105            double[] testData = testsData[i];
106            result = solver.solve(maxIter, f, testData[0], testData[1]);
107            //System.out.println(
108            //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
109
110            // Check solution.
111            Assert.assertEquals(result, testData[2], atol);
112            Assert.assertTrue(solver.getEvaluations() <= counts[i] + 1);
113        }
114    }

116    @Test
117    public void testRootEndpoints() {
118        UnivariateFunction f = new XMinus5Function();
119        UnivariateSolver solver = getSolver();
120
121        // End-point is root. This should be a special case in the solver, and
122        // the initial end-point should be returned exactly.
123        double result = solver.solve(100, f, 5.0, 6.0);
124        Assert.assertEquals(5.0, result, 0.0);
125
126        result = solver.solve(100, f, 4.0, 5.0);
127        Assert.assertEquals(5.0, result, 0.0);
128
129        result = solver.solve(100, f, 5.0, 6.0, 5.5);
130        Assert.assertEquals(5.0, result, 0.0);
131
132        result = solver.solve(100, f, 4.0, 5.0, 4.5);
133        Assert.assertEquals(5.0, result, 0.0);
134    }

138        UnivariateFunction f = new Sin();
139        UnivariateSolver solver = getSolver();
140        try {  // bad interval
141            solver.solve(100, f, 1, -1);
142            Assert.fail("Expecting NumberIsTooLargeException - bad interval");
143        } catch (NumberIsTooLargeException ex) {
144            // expected
145        }
146        try {  // no bracket
147            solver.solve(100, f, 1, 1.5);
148            Assert.fail("Expecting NoBracketingException - non-bracketing");
149        } catch (NoBracketingException ex) {
150            // expected
151        }
152        try {  // no bracket
153            solver.solve(100, f, 1, 1.5, 1.2);
154            Assert.fail("Expecting NoBracketingException - non-bracketing");
155        } catch (NoBracketingException ex) {
156            // expected
157        }
158    }

160    @Test
161    public void testSolutionLeftSide() {
162        UnivariateFunction f = new Sin();
163        UnivariateSolver solver = getSolver();
164        double left = -1.5;
165        double right = 0.05;
166        for(int i = 0; i < 10; i++) {
167            // Test whether the allowed solutions are taken into account.
168            double solution = getSolution(solver, 100, f, left, right, AllowedSolution.LEFT_SIDE);
169            if (!Double.isNaN(solution)) {
170                Assert.assertTrue(solution <= 0.0);
171            }
172
173            // Prepare for next test.
174            left -= 0.1;
175            right += 0.3;
176        }
177    }

235    private double getSolution(UnivariateSolver solver, int maxEval, UnivariateFunction f,
236                               double left, double right, AllowedSolution allowedSolution) {
237        try {
238            @SuppressWarnings("unchecked")
239            BracketedUnivariateSolver<UnivariateFunction> bracketing =
240            (BracketedUnivariateSolver<UnivariateFunction>) solver;
241            return bracketing.solve(100, f, left, right, allowedSolution);
242        } catch (ClassCastException cce) {
243            double baseRoot = solver.solve(maxEval, f, left, right);
244            if ((baseRoot <= left) || (baseRoot >= right)) {
245                // the solution slipped out of interval
246                return Double.NaN;
247            }
248            PegasusSolver bracketing =
249                    new PegasusSolver(solver.getRelativeAccuracy(), solver.getAbsoluteAccuracy(),
250                                      solver.getFunctionValueAccuracy());
251            return UnivariateSolverUtils.forceSide(maxEval - solver.getEvaluations(),
252                                                       f, bracketing, baseRoot, left, right,
253                                                       allowedSolution);
254        }
255    }
