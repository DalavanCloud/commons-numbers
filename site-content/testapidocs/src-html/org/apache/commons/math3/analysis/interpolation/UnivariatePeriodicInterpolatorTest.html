029/**
030 * Test for {@link UnivariatePeriodicInterpolator}.
031 */
032public class UnivariatePeriodicInterpolatorTest {
033    private final Random rng = new Random(1224465L);
034
035    @Test
036    public void testSine() {
037        final int n = 30;
038        final double[] xval = new double[n];
039        final double[] yval = new double[n];
040        final double period = 12.3;
041        final double offset = 45.67;
042
043        double delta = 0;
044        for (int i = 0; i < n; i++) {
045            delta += rng.nextDouble() * period / n;
046            xval[i] = offset + delta;
047            yval[i] = FastMath.sin(xval[i]);
048        }
049
050        final UnivariateInterpolator inter = new LinearInterpolator();
051        final UnivariateFunction f = inter.interpolate(xval, yval);
052
053        final UnivariateInterpolator interP
054            = new UnivariatePeriodicInterpolator(new LinearInterpolator(),
055                                                     period, 1);
056        final UnivariateFunction fP = interP.interpolate(xval, yval);
057
058        // Comparing with original interpolation algorithm.
059        final double xMin = xval[0];
060        final double xMax = xval[n - 1];
061        for (int i = 0; i < n; i++) {
062            final double x = xMin + (xMax - xMin) * rng.nextDouble();
063            final double y = f.value(x);
064            final double yP = fP.value(x);
065
066            Assert.assertEquals("x=" + x, y, yP, Math.ulp(1d));
067        }
068
069        // Test interpolation outside the primary interval.
070        for (int i = 0; i < n; i++) {
071            final double xIn = offset + rng.nextDouble() * period;
072            final double xOut = xIn + rng.nextInt(123456789) * period;
073            final double yIn = fP.value(xIn);
074            final double yOut = fP.value(xOut);
075
076            Assert.assertEquals(yIn, yOut, 1e-7);
077        }
078    }

080    @Test
081    public void testLessThanOnePeriodCoverage() {
082        final int n = 30;
083        final double[] xval = new double[n];
084        final double[] yval = new double[n];
085        final double period = 12.3;
086        final double offset = 45.67;
087
088        double delta = period / 2;
089        for (int i = 0; i < n; i++) {
090            delta += period / (2 * n) * rng.nextDouble();
091            xval[i] = offset + delta;
092            yval[i] = FastMath.sin(xval[i]);
093        }
094
095        final UnivariateInterpolator interP
096            = new UnivariatePeriodicInterpolator(new LinearInterpolator(),
097                                                     period, 1);
098        final UnivariateFunction fP = interP.interpolate(xval, yval);
099
100        // Test interpolation outside the sample data interval.
101        for (int i = 0; i < n; i++) {
102            final double xIn = offset + rng.nextDouble() * period;
103            final double xOut = xIn + rng.nextInt(123456789) * period;
104            final double yIn = fP.value(xIn);
105            final double yOut = fP.value(xOut);
106
107            Assert.assertEquals(yIn, yOut, 1e-7);
108        }
109    }

111    @Test
112    public void testMoreThanOnePeriodCoverage() {
113        final int n = 30;
114        final double[] xval = new double[n];
115        final double[] yval = new double[n];
116        final double period = 12.3;
117        final double offset = 45.67;
118
119        double delta = period / 2;
120        for (int i = 0; i < n; i++) {
121            delta += 10 * period / n * rng.nextDouble();
122            xval[i] = offset + delta;
123            yval[i] = FastMath.sin(xval[i]);
124        }
125
126        final UnivariateInterpolator interP
127            = new UnivariatePeriodicInterpolator(new LinearInterpolator(),
128                                                     period, 1);
129        final UnivariateFunction fP = interP.interpolate(xval, yval);
130
131        // Test interpolation outside the sample data interval.
132        for (int i = 0; i < n; i++) {
133            final double xIn = offset + rng.nextDouble() * period;
134            final double xOut = xIn + rng.nextInt(123456789) * period;
135            final double yIn = fP.value(xIn);
136            final double yOut = fP.value(xOut);
137
138            Assert.assertEquals(yIn, yOut, 1e-6);
139        }
140    }
