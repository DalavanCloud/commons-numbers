314                    }
315                }

320    @Test
321    public void testShift() {
322        // f1(x) = 1 + x + 2 x^2
323        PolynomialFunction f1x = new PolynomialFunction(new double[] { 1, 1, 2 });
324
325        PolynomialFunction f1x1
326            = new PolynomialFunction(PolynomialsUtils.shift(f1x.getCoefficients(), 1));
327        checkPolynomial(f1x1, "4 + 5 x + 2 x^2");
328
329        PolynomialFunction f1xM1
330            = new PolynomialFunction(PolynomialsUtils.shift(f1x.getCoefficients(), -1));
331        checkPolynomial(f1xM1, "2 - 3 x + 2 x^2");
332        
333        PolynomialFunction f1x3
334            = new PolynomialFunction(PolynomialsUtils.shift(f1x.getCoefficients(), 3));
335        checkPolynomial(f1x3, "22 + 13 x + 2 x^2");
336
337        // f2(x) = 2 + 3 x^2 + 8 x^3 + 121 x^5
338        PolynomialFunction f2x = new PolynomialFunction(new double[]{2, 0, 3, 8, 0, 121});
339
340        PolynomialFunction f2x1
341            = new PolynomialFunction(PolynomialsUtils.shift(f2x.getCoefficients(), 1));
342        checkPolynomial(f2x1, "134 + 635 x + 1237 x^2 + 1218 x^3 + 605 x^4 + 121 x^5");
343
344        PolynomialFunction f2x3
345            = new PolynomialFunction(PolynomialsUtils.shift(f2x.getCoefficients(), 3));
346        checkPolynomial(f2x3, "29648 + 49239 x + 32745 x^2 + 10898 x^3 + 1815 x^4 + 121 x^5");
347    }

350    private void checkPolynomial(PolynomialFunction p, long denominator, String reference) {
351        PolynomialFunction q = new PolynomialFunction(new double[] { denominator});
352        Assert.assertEquals(reference, p.multiply(q).toString());
353    }

355    private void checkPolynomial(PolynomialFunction p, String reference) {
356        Assert.assertEquals(reference, p.toString());
357    }

359    private void checkNullPolynomial(PolynomialFunction p) {
360        for (double coefficient : p.getCoefficients()) {
361            Assert.assertEquals(0, coefficient, 1e-13);
362        }
363    }
364
365    private void checkOrthogonality(final PolynomialFunction p1,
366                                    final PolynomialFunction p2,
367                                    final UnivariateFunction weight,
368                                    final double a, final double b,
369                                    final double nonZeroThreshold,
370                                    final double zeroThreshold) {
371        UnivariateFunction f = new UnivariateFunction() {
372            public double value(double x) {
373                return weight.value(x) * p1.value(x) * p2.value(x);
374            }
375        };
376        double dotProduct =
377                new IterativeLegendreGaussIntegrator(5, 1.0e-9, 1.0e-8, 2, 15).integrate(1000000, f, a, b);
378        if (p1.degree() == p2.degree()) {
379            // integral should be non-zero
380            Assert.assertTrue("I(" + p1.degree() + ", " + p2.degree() + ") = "+ dotProduct,
381                              FastMath.abs(dotProduct) > nonZeroThreshold);
382        } else {
383            // integral should be zero
384            Assert.assertEquals("I(" + p1.degree() + ", " + p2.degree() + ") = "+ dotProduct,
385                                0.0, FastMath.abs(dotProduct), zeroThreshold);
386        }
387    }
