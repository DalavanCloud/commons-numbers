228    @Test
229    public void testPowDoubleDS() {
230        for (int maxOrder = 1; maxOrder < 5; ++maxOrder) {
231
232            DerivativeStructure x = new DerivativeStructure(3, maxOrder, 0, 0.1);
233            DerivativeStructure y = new DerivativeStructure(3, maxOrder, 1, 0.2);
234            DerivativeStructure z = new DerivativeStructure(3, maxOrder, 2, 0.3);
235            List<DerivativeStructure> list = Arrays.asList(x, y, z,
236                                                           x.add(y).add(z),
237                                                           x.multiply(y).multiply(z));
238
239            for (DerivativeStructure ds : list) {
240                // the special case a = 0 is included here
241                for (double a : new double[] { 0.0, 0.1, 1.0, 2.0, 5.0 }) {
242                    DerivativeStructure reference = (a == 0) ?
243                                                    x.getField().getZero() :
244                                                    new DerivativeStructure(3, maxOrder, a).pow(ds);
245                    DerivativeStructure result = DerivativeStructure.pow(a, ds);
246                    checkEquals(reference, result, 1.0e-15);
247                }
248
249            }
250
251            // negative base: -1^x can be evaluated for integers only, so value is sometimes OK, derivatives are always NaN
252            DerivativeStructure negEvenInteger = DerivativeStructure.pow(-2.0, new DerivativeStructure(3,  maxOrder, 0, 2.0));
253            Assert.assertEquals(4.0, negEvenInteger.getValue(), 1.0e-15);
254            Assert.assertTrue(Double.isNaN(negEvenInteger.getPartialDerivative(1, 0, 0)));
255            DerivativeStructure negOddInteger = DerivativeStructure.pow(-2.0, new DerivativeStructure(3,  maxOrder, 0, 3.0));
256            Assert.assertEquals(-8.0, negOddInteger.getValue(), 1.0e-15);
257            Assert.assertTrue(Double.isNaN(negOddInteger.getPartialDerivative(1, 0, 0)));
258            DerivativeStructure negNonInteger = DerivativeStructure.pow(-2.0, new DerivativeStructure(3,  maxOrder, 0, 2.001));
259            Assert.assertTrue(Double.isNaN(negNonInteger.getValue()));
260            Assert.assertTrue(Double.isNaN(negNonInteger.getPartialDerivative(1, 0, 0)));
261
262            DerivativeStructure zeroNeg = DerivativeStructure.pow(0.0, new DerivativeStructure(3,  maxOrder, 0, -1.0));
263            Assert.assertTrue(Double.isNaN(zeroNeg.getValue()));
264            Assert.assertTrue(Double.isNaN(zeroNeg.getPartialDerivative(1, 0, 0)));
265            DerivativeStructure posNeg = DerivativeStructure.pow(2.0, new DerivativeStructure(3,  maxOrder, 0, -2.0));
266            Assert.assertEquals(1.0 / 4.0, posNeg.getValue(), 1.0e-15);
267            Assert.assertEquals(FastMath.log(2.0) / 4.0, posNeg.getPartialDerivative(1, 0, 0), 1.0e-15);
268
269            // very special case: a = 0 and power = 0
270            DerivativeStructure zeroZero = DerivativeStructure.pow(0.0, new DerivativeStructure(3,  maxOrder, 0, 0.0));
271
272            // this should be OK for simple first derivative with one variable only ...
273            Assert.assertEquals(1.0, zeroZero.getValue(), 1.0e-15);
274            Assert.assertEquals(Double.NEGATIVE_INFINITY, zeroZero.getPartialDerivative(1, 0, 0), 1.0e-15);
275
276            // the following checks show a LIMITATION of the current implementation
277            // we have no way to tell x is a pure linear variable x = 0
278            // we only say: "x is a structure with value = 0.0,
279            // first derivative with respect to x = 1.0, and all other derivatives
280            // (first order with respect to y and z and higher derivatives) all 0.0.
281            // We have function f(x) = a^x and x = 0 so we compute:
282            // f(0) = 1, f'(0) = ln(a), f''(0) = ln(a)^2. The limit of these values
283            // when a converges to 0 implies all derivatives keep switching between
284            // +infinity and -infinity.
285            //
286            // Function composition rule for first derivatives is:
287            // d[f(g(x,y,z))]/dy = f'(g(x,y,z)) * dg(x,y,z)/dy
288            // so given that in our case x represents g and does not depend
289            // on y or z, we have dg(x,y,z)/dy = 0
290            // applying the composition rules gives:
291            // d[f(g(x,y,z))]/dy = f'(g(x,y,z)) * dg(x,y,z)/dy
292            //                 = -infinity * 0
293            //                 = NaN
294            // if we knew x is really the x variable and not the identity
295            // function applied to x, we would not have computed f'(g(x,y,z)) * dg(x,y,z)/dy
296            // and we would have found that the result was 0 and not NaN
297            Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(0, 1, 0)));
298            Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(0, 0, 1)));
299
300            // Function composition rule for second derivatives is:
301            // d2[f(g(x))]/dx2 = f''(g(x)) * [g'(x)]^2 + f'(g(x)) * g''(x)
302            // when function f is the a^x root and x = 0 we have:
303            // f(0) = 1, f'(0) = ln(a), f''(0) = ln(a)^2 which for a = 0 implies
304            // all derivatives keep switching between +infinity and -infinity
305            // so given that in our case x represents g, we have g(x) = 0,
306            // g'(x) = 1 and g''(x) = 0
307            // applying the composition rules gives:
308            // d2[f(g(x))]/dx2 = f''(g(x)) * [g'(x)]^2 + f'(g(x)) * g''(x)
309            //                 = +infinity * 1^2 + -infinity * 0
310            //                 = +infinity + NaN
311            //                 = NaN
312            // if we knew x is really the x variable and not the identity
313            // function applied to x, we would not have computed f'(g(x)) * g''(x)
314            // and we would have found that the result was +infinity and not NaN
315            if (maxOrder > 1) {
316                Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(2, 0, 0)));
317                Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(0, 2, 0)));
318                Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(0, 0, 2)));
319                Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(1, 1, 0)));
320                Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(0, 1, 1)));
321                Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(1, 1, 0)));
322            }
323
324        }

340                            new DerivativeStructure(1, dsX,

346                            new DerivativeStructure(1, dsX,

412    public void testTrigo() {
413        double epsilon = 2.0e-12;
414        for (int maxOrder = 0; maxOrder < 5; ++maxOrder) {
415            for (double x = 0.1; x < 1.2; x += 0.1) {
416                DerivativeStructure dsX = new DerivativeStructure(3, maxOrder, 0, x);
417                for (double y = 0.1; y < 1.2; y += 0.1) {
418                    DerivativeStructure dsY = new DerivativeStructure(3, maxOrder, 1, y);
419                    for (double z = 0.1; z < 1.2; z += 0.1) {
420                        DerivativeStructure dsZ = new DerivativeStructure(3, maxOrder, 2, z);
421                        DerivativeStructure f = dsX.divide(dsY.cos().add(dsZ.tan())).sin();
422                        double a = FastMath.cos(y) + FastMath.tan(z);
423                        double f0 = FastMath.sin(x / a);
424                        Assert.assertEquals(f0, f.getValue(), FastMath.abs(epsilon * f0));
425                        if (f.getOrder() > 0) {
426                            double dfdx = FastMath.cos(x / a) / a;
427                            Assert.assertEquals(dfdx, f.getPartialDerivative(1, 0, 0), FastMath.abs(epsilon * dfdx));
428                            double dfdy =  x * FastMath.sin(y) * dfdx / a;
429                            Assert.assertEquals(dfdy, f.getPartialDerivative(0, 1, 0), FastMath.abs(epsilon * dfdy));
430                            double cz = FastMath.cos(z);
431                            double cz2 = cz * cz;
432                            double dfdz = -x * dfdx / (a * cz2);
433                            Assert.assertEquals(dfdz, f.getPartialDerivative(0, 0, 1), FastMath.abs(epsilon * dfdz));
434                            if (f.getOrder() > 1) {
435                                double df2dx2 = -(f0 / (a * a));
436                                Assert.assertEquals(df2dx2, f.getPartialDerivative(2, 0, 0), FastMath.abs(epsilon * df2dx2));
437                                double df2dy2 = x * FastMath.cos(y) * dfdx / a -
438                                                x * x * FastMath.sin(y) * FastMath.sin(y) * f0 / (a * a * a * a) +
439                                                2 * FastMath.sin(y) * dfdy / a;
440                                Assert.assertEquals(df2dy2, f.getPartialDerivative(0, 2, 0), FastMath.abs(epsilon * df2dy2));
441                                double c4 = cz2 * cz2;
442                                double df2dz2 = x * (2 * a * (1 - a * cz * FastMath.sin(z)) * dfdx - x * f0 / a ) / (a * a * a * c4);
443                                Assert.assertEquals(df2dz2, f.getPartialDerivative(0, 0, 2), FastMath.abs(epsilon * df2dz2));
444                                double df2dxdy = dfdy / x  - x * FastMath.sin(y) * f0 / (a * a * a);
445                                Assert.assertEquals(df2dxdy, f.getPartialDerivative(1, 1, 0), FastMath.abs(epsilon * df2dxdy));
446                            }
447                        }
448                    }
449                }
450            }

471    public void testRootNSingularity() {
472        for (int n = 2; n < 10; ++n) {
473            for (int maxOrder = 0; maxOrder < 12; ++maxOrder) {
474                DerivativeStructure dsZero = new DerivativeStructure(1, maxOrder, 0, 0.0);
475                DerivativeStructure rootN  = dsZero.rootN(n);
476                Assert.assertEquals(0.0, rootN.getValue(), 1.0e-20);
477                if (maxOrder > 0) {
478                    Assert.assertTrue(Double.isInfinite(rootN.getPartialDerivative(1)));
479                    Assert.assertTrue(rootN.getPartialDerivative(1) > 0);
480                    for (int order = 2; order <= maxOrder; ++order) {
481                        // the following checks shows a LIMITATION of the current implementation
482                        // we have no way to tell dsZero is a pure linear variable x = 0
483                        // we only say: "dsZero is a structure with value = 0.0,
484                        // first derivative = 1.0, second and higher derivatives = 0.0".
485                        // Function composition rule for second derivatives is:
486                        // d2[f(g(x))]/dx2 = f''(g(x)) * [g'(x)]^2 + f'(g(x)) * g''(x)
487                        // when function f is the nth root and x = 0 we have:
488                        // f(0) = 0, f'(0) = +infinity, f''(0) = -infinity (and higher
489                        // derivatives keep switching between +infinity and -infinity)
490                        // so given that in our case dsZero represents g, we have g(x) = 0,
491                        // g'(x) = 1 and g''(x) = 0
492                        // applying the composition rules gives:
493                        // d2[f(g(x))]/dx2 = f''(g(x)) * [g'(x)]^2 + f'(g(x)) * g''(x)
494                        //                 = -infinity * 1^2 + +infinity * 0
495                        //                 = -infinity + NaN
496                        //                 = NaN
497                        // if we knew dsZero is really the x variable and not the identity
498                        // function applied to x, we would not have computed f'(g(x)) * g''(x)
499                        // and we would have found that the result was -infinity and not NaN
500                        Assert.assertTrue(Double.isNaN(rootN.getPartialDerivative(order)));
501                    }
502                }
503
504                // the following shows that the limitation explained above is NOT a bug...
505                // if we set up the higher order derivatives for g appropriately, we do
506                // compute the higher order derivatives of the composition correctly
507                double[] gDerivatives = new double[ 1 + maxOrder];
508                gDerivatives[0] = 0.0;
509                for (int k = 1; k <= maxOrder; ++k) {
510                    gDerivatives[k] = FastMath.pow(-1.0, k + 1);
511                }
512                DerivativeStructure correctRoot = new DerivativeStructure(1, maxOrder, gDerivatives).rootN(n);
513                Assert.assertEquals(0.0, correctRoot.getValue(), 1.0e-20);
514                if (maxOrder > 0) {
515                    Assert.assertTrue(Double.isInfinite(correctRoot.getPartialDerivative(1)));
516                    Assert.assertTrue(correctRoot.getPartialDerivative(1) > 0);
517                    for (int order = 2; order <= maxOrder; ++order) {
518                        Assert.assertTrue(Double.isInfinite(correctRoot.getPartialDerivative(order)));
519                        if ((order % 2) == 0) {
520                            Assert.assertTrue(correctRoot.getPartialDerivative(order) < 0);
521                        } else {
522                            Assert.assertTrue(correctRoot.getPartialDerivative(order) > 0);
523                        }
524                    }
525                }

527            }

1561
1562        // check dimension
1563        Assert.assertEquals(ds1.getFreeParameters(), ds2.getFreeParameters());
1564        Assert.assertEquals(ds1.getOrder(), ds2.getOrder());
1565
1566        int[] derivatives = new int[ds1.getFreeParameters()];
1567        int sum = 0;
1568        while (true) {
1569
1570            if (sum <= ds1.getOrder()) {
1571                Assert.assertEquals(ds1.getPartialDerivative(derivatives),
1572                                    ds2.getPartialDerivative(derivatives),
1573                                    epsilon);
1574            }
1575
1576            boolean increment = true;
1577            sum = 0;
1578            for (int i = derivatives.length - 1; i >= 0; --i) {
1579                if (increment) {
1580                    if (derivatives[i] == ds1.getOrder()) {
1581                        derivatives[i] = 0;
1582                    } else {
1583                        derivatives[i]++;
1584                        increment = false;
1585                    }
1586                }
1587                sum += derivatives[i];
1588            }
1589            if (increment) {
1590                return;
1591            }
1592
1593        }
