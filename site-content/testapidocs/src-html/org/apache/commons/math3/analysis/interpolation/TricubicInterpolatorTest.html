087    public void testIsValid() {
088        double[] xval = new double[] {3, 4, 5, 6.5};
089        double[] yval = new double[] {-4, -3, -1, 2.5};
090        double[] zval = new double[] {-12, -8, -5.5, -3, 0, 2.5};
091        double[][][] fval = new double[xval.length][yval.length][zval.length];
092
093        @SuppressWarnings("unused")
094        TricubicInterpolatingFunction tcf = new TricubicInterpolator().interpolate(xval, yval, zval, fval);
095
096        // Valid.
097        Assert.assertTrue(tcf.isValidPoint(4, -3, -8));
098        Assert.assertTrue(tcf.isValidPoint(5, -3, -8));
099        Assert.assertTrue(tcf.isValidPoint(4, -1, -8));
100        Assert.assertTrue(tcf.isValidPoint(5, -1, -8));
101        Assert.assertTrue(tcf.isValidPoint(4, -3, 0));
102        Assert.assertTrue(tcf.isValidPoint(5, -3, 0));
103        Assert.assertTrue(tcf.isValidPoint(4, -1, 0));
104        Assert.assertTrue(tcf.isValidPoint(5, -1, 0));
105
106        // Invalid.
107        Assert.assertFalse(tcf.isValidPoint(3.5, -3, -8));
108        Assert.assertFalse(tcf.isValidPoint(4.5, -3.1, -8));
109        Assert.assertFalse(tcf.isValidPoint(4.5, -2, 0));
110        Assert.assertFalse(tcf.isValidPoint(4.5, 0, -3.5));
111        Assert.assertFalse(tcf.isValidPoint(-10, 4.1, -1));
112    }

114    /**
115     * Test for a plane.
116     * <p>
117     *  f(x, y, z) = 2 x - 3 y - 4 z + 5
118     * </p>
119     */
120    @Test
121    public void testPlane() {
122        double[] xval = new double[] {3, 4, 5, 6.5};
123        double[] yval = new double[] {-4, -3, -1, 2, 2.5};
124        double[] zval = new double[] {-12, -8, -5.5, -3, 0, 2.5};
125
126        // Function values
127        TrivariateFunction f = new TrivariateFunction() {
128                public double value(double x, double y, double z) {
129                    return 2 * x - 3 * y - 4 * z + 5;
130                }
131            };
132
133        double[][][] fval = new double[xval.length][yval.length][zval.length];
134
135        for (int i = 0; i < xval.length; i++) {
136            for (int j = 0; j < yval.length; j++) {
137                for (int k = 0; k < zval.length; k++) {
138                    fval[i][j][k] = f.value(xval[i], yval[j], zval[k]);
139                }
140            }
141        }
142
143        TrivariateFunction tcf = new TricubicInterpolator().interpolate(xval,
144                                                                        yval,
145                                                                        zval,
146                                                                        fval);
147        double x, y, z;
148        double expected, result;
149
150        x = 4;
151        y = -3;
152        z = 0;
153        expected = f.value(x, y, z);
154        result = tcf.value(x, y, z);
155        Assert.assertEquals("On sample point",
156                            expected, result, 1e-15);
157
158        x = 4.5;
159        y = -1.5;
160        z = -4.25;
161        expected = f.value(x, y, z);
162        result = tcf.value(x, y, z);
163        Assert.assertEquals("Half-way between sample points (middle of the patch)",
164                            expected, result, 1e-14);
165    }

167    /**
168     * Sine wave.
169     * <p>
170     *  f(x, y, z) = a cos [&omega; z - k<sub>y</sub> x - k<sub>y</sub> y]
171     * </p>
172     * with A = 0.2, &omega; = 0.5, k<sub>x</sub> = 2, k<sub>y</sub> = 1.
173     */
174    @Test
175    public void testWave() {
176        double[] xval = new double[] {3, 4, 5, 6.5};
177        double[] yval = new double[] {-4, -3, -1, 2, 2.5};
178        double[] zval = new double[] {-12, -8, -5.5, -3, 0, 4};
179
180        final double a = 0.2;
181        final double omega = 0.5;
182        final double kx = 2;
183        final double ky = 1;
184
185        // Function values
186        TrivariateFunction f = new TrivariateFunction() {
187                public double value(double x, double y, double z) {
188                    return a * FastMath.cos(omega * z - kx * x - ky * y);
189                }
190            };
191
192        double[][][] fval = new double[xval.length][yval.length][zval.length];
193        for (int i = 0; i < xval.length; i++) {
194            for (int j = 0; j < yval.length; j++) {
195                for (int k = 0; k < zval.length; k++) {
196                    fval[i][j][k] = f.value(xval[i], yval[j], zval[k]);
197                }
198            }
199        }
200
201        TrivariateFunction tcf = new TricubicInterpolator().interpolate(xval,
202                                                                        yval,
203                                                                        zval,
204                                                                        fval);
205
206        double x, y, z;
207        double expected, result;
208
209        x = 4;
210        y = -3;
211        z = 0;
212        expected = f.value(x, y, z);
213        result = tcf.value(x, y, z);
214        Assert.assertEquals("On sample point",
215                            expected, result, 1e-14);
216
217        x = 4.5;
218        y = -1.5;
219        z = -4.25;
220        expected = f.value(x, y, z);
221        result = tcf.value(x, y, z);
222        Assert.assertEquals("Half-way between sample points (middle of the patch)",
223                            expected, result, 1e-1); // XXX Too high tolerance!
224    }
