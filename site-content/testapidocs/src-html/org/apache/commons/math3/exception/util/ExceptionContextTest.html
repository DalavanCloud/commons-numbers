017package org.apache.commons.math3.exception.util;
018
019import java.util.Locale;
020import java.util.Arrays;
021import java.io.IOException;
022import java.io.ObjectOutputStream;
023import java.io.ObjectInputStream;
024import java.io.ByteArrayOutputStream;
025import java.io.ByteArrayInputStream;
026
027import org.junit.Assert;
028import org.junit.Test;
029
030/**
031 * Test for {@link ExceptionContext}.
032 * 
033 */
034public class ExceptionContextTest {
035    @Test
036    public void testMessageChain() {
037        final ExceptionContext c = new ExceptionContext(new Exception("oops"));
038        final String sep = " | "; // Non-default separator.
039        final String m1 = "column index (0)";
040        c.addMessage(LocalizedFormats.COLUMN_INDEX, 0);
041        final String m2 = "got 1x2 but expected 3x4";
042        c.addMessage(LocalizedFormats.DIMENSIONS_MISMATCH_2x2, 1, 2, 3, 4);
043        final String m3 = "It didn't work out";
044        c.addMessage(LocalizedFormats.SIMPLE_MESSAGE, m3);
045
046        Assert.assertEquals(c.getMessage(Locale.US, sep),
047                            m1 + sep + m2 + sep + m3);
048    }
049
050    @Test
051    public void testNoArgAddMessage() {
052        final ExceptionContext c = new ExceptionContext(new Exception("hello"));
053        c.addMessage(LocalizedFormats.SIMPLE_MESSAGE);
054        Assert.assertEquals(c.getMessage(), "{0}");
055    }
056
057    @Test
058    public void testContext() {
059        final ExceptionContext c = new ExceptionContext(new Exception("bye"));
060
061        final String[] keys = {"Key 1", "Key 2"};
062        final Object[] values = {"Value 1", Integer.valueOf(2)};
063
064        for (int i = 0; i < keys.length; i++) {
065            c.setValue(keys[i], values[i]);
066        }
067
068        // Check that all keys are present.
069        Assert.assertTrue(c.getKeys().containsAll(Arrays.asList(keys)));
070
071        // Check that all values are correctly stored.
072        for (int i = 0; i < keys.length; i++) {
073            Assert.assertEquals(values[i], c.getValue(keys[i]));
074        }
075
076        // Check behaviour on missing key.
077        Assert.assertNull(c.getValue("xyz"));
078    }
079
080    @Test
081    public void testSerialize()
082        throws IOException,
083               ClassNotFoundException {
084        final ExceptionContext cOut = new ExceptionContext(new Exception("Apache"));
085        cOut.addMessage(LocalizedFormats.COLUMN_INDEX, 0);
086        cOut.setValue("Key 1", Integer.valueOf(0));
087
088        ByteArrayOutputStream bos = new ByteArrayOutputStream();
089        ObjectOutputStream oos = new ObjectOutputStream(bos);
090        oos.writeObject(cOut);
091
092        ByteArrayInputStream bis = new ByteArrayInputStream(bos.toByteArray());
093        ObjectInputStream ois = new ObjectInputStream(bis);
094        ExceptionContext cIn = (ExceptionContext) ois.readObject();
095
096        Assert.assertTrue(cOut.getMessage().equals(cIn.getMessage()));
097        for (String key : cIn.getKeys()) {
098            Assert.assertTrue(cOut.getValue(key).equals(cIn.getValue(key)));
099        }
100    }
101
102    @Test
103    public void testSerializeUnserializable() throws Exception {
104        final ExceptionContext cOut = new ExceptionContext(new Exception("Apache Commons Math"));
105        cOut.addMessage(LocalizedFormats.SIMPLE_MESSAGE, "OK");
106        cOut.addMessage(LocalizedFormats.SIMPLE_MESSAGE, new Unserializable());
107        String key = "Key 1";
108        cOut.setValue(key, new Unserializable());
109
110        {
111            ByteArrayOutputStream bos = new ByteArrayOutputStream();
112            ObjectOutputStream oos = new ObjectOutputStream(bos);
113            oos.writeObject(cOut);
114
115            ByteArrayInputStream bis = new ByteArrayInputStream(bos.toByteArray());
116            ObjectInputStream ois = new ObjectInputStream(bis);
117            ExceptionContext cIn = (ExceptionContext) ois.readObject();
118
119            String nsObjStr = (String) cIn.getValue(key);
120            Assert.assertTrue(nsObjStr.matches(".*could not be serialized.*"));
121        }
122    }
123
124    /**
125     * Class used by {@link #testSerializeUnserializable()}.
126     */
127    private static class Unserializable {
128        Unserializable() {}
129    }
130}
