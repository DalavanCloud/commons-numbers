037    // JIRA: MATH-322
038    @Test
039    public void closeEvents() throws MaxCountExceededException, NoBracketingException {
040
041        final double r1  = 90.0;
042        final double r2  = 135.0;
043        final double gap = r2 - r1;
044        EventHandler closeEventsGenerator = new EventHandler() {
045            public void init(double t0, double[] y0, double t) {
046            }
047            public void resetState(double t, double[] y) {
048            }
049            public double g(double t, double[] y) {
050                return (t - r1) * (r2 - t);
051            }
052            public Action eventOccurred(double t, double[] y, boolean increasing) {
053                return Action.CONTINUE;
054            }
055        };
056
057        final double tolerance = 0.1;
058        EventState es = new EventState(closeEventsGenerator, 1.5 * gap,
059                                       tolerance, 100,
060                                       new BrentSolver(tolerance));
061        es.setExpandable(new ExpandableStatefulODE(new FirstOrderDifferentialEquations() {
062            public int getDimension() {
063                return 0;
064            }
065            public void computeDerivatives(double t, double[] y, double[] yDot) {
066            }
067        }));
068
069        AbstractStepInterpolator interpolator =
070            new DummyStepInterpolator(new double[0], new double[0], true);
071        interpolator.storeTime(r1 - 2.5 * gap);
072        interpolator.shift();
073        interpolator.storeTime(r1 - 1.5 * gap);
074        es.reinitializeBegin(interpolator);
075
076        interpolator.shift();
077        interpolator.storeTime(r1 - 0.5 * gap);
078        Assert.assertFalse(es.evaluateStep(interpolator));
079
080        interpolator.shift();
081        interpolator.storeTime(0.5 * (r1 + r2));
082        Assert.assertTrue(es.evaluateStep(interpolator));
083        Assert.assertEquals(r1, es.getEventTime(), tolerance);
084        es.stepAccepted(es.getEventTime(), new double[0]);
085
086        interpolator.shift();
087        interpolator.storeTime(r2 + 0.4 * gap);
088        Assert.assertTrue(es.evaluateStep(interpolator));
089        Assert.assertEquals(r2, es.getEventTime(), tolerance);
090
091    }

093    // Jira: MATH-695
094    @Test
095    public void testIssue695()
096        throws DimensionMismatchException, NumberIsTooSmallException,
097               MaxCountExceededException, NoBracketingException {
098
099        FirstOrderDifferentialEquations equation = new FirstOrderDifferentialEquations() {
100            
101            public int getDimension() {
102                return 1;
103            }
104            
105            public void computeDerivatives(double t, double[] y, double[] yDot) {
106                yDot[0] = 1.0;
107            }
108        };
109
110        DormandPrince853Integrator integrator = new DormandPrince853Integrator(0.001, 1000, 1.0e-14, 1.0e-14);
111        integrator.addEventHandler(new ResettingEvent(10.99), 0.1, 1.0e-9, 1000);
112        integrator.addEventHandler(new ResettingEvent(11.01), 0.1, 1.0e-9, 1000);
113        integrator.setInitialStepSize(3.0);
114
115        double target = 30.0;
116        double[] y = new double[1];
117        double tEnd = integrator.integrate(equation, 0.0, y, target, y);
118        Assert.assertEquals(target, tEnd, 1.0e-10);
119        Assert.assertEquals(32.0, y[0], 1.0e-10);
120
121    }

123    private static class ResettingEvent implements EventHandler {
124
125        private static double lastTriggerTime = Double.NEGATIVE_INFINITY;
126        private final double tEvent;
127
128        public ResettingEvent(final double tEvent) {
129            this.tEvent = tEvent;
130        }
131
132        public void init(double t0, double[] y0, double t) {
133        }
134
135        public double g(double t, double[] y) {
136            // the bug corresponding to issue 695 causes the g function
137            // to be called at obsolete times t despite an event
138            // occurring later has already been triggered.
139            // When this occurs, the following assertion is violated
140            Assert.assertTrue("going backard in time! (" + t + " < " + lastTriggerTime + ")",
141                              t >= lastTriggerTime);
142            return t - tEvent;
143        }
144
145        public Action eventOccurred(double t, double[] y, boolean increasing) {
146            // remember in a class variable when the event was triggered
147            lastTriggerTime = t;
148            return Action.RESET_STATE;
149        }
150
151        public void resetState(double t, double[] y) {
152            y[0] += 1.0;
153        }
154
155    }

157    // Jira: MATH-965
158    @Test
159    public void testIssue965()
160        throws DimensionMismatchException, NumberIsTooSmallException,
161               MaxCountExceededException, NoBracketingException {
162
163        ExpandableStatefulODE equation =
164                new ExpandableStatefulODE(new FirstOrderDifferentialEquations() {
165            
166            public int getDimension() {
167                return 1;
168            }
169            
170            public void computeDerivatives(double t, double[] y, double[] yDot) {
171                yDot[0] = 2.0;
172            }
173        });
174        equation.setTime(0.0);
175        equation.setPrimaryState(new double[1]);
176        equation.addSecondaryEquations(new SecondaryEquations() {
177            
178            public int getDimension() {
179                return 1;
180            }
181            
182            public void computeDerivatives(double t, double[] primary,
183                                           double[] primaryDot, double[] secondary,
184                                           double[] secondaryDot) {
185                secondaryDot[0] = -3.0;
186            }
187        });
188        int index = equation.getSecondaryMappers()[0].getFirstIndex();
189
190        DormandPrince853Integrator integrator = new DormandPrince853Integrator(0.001, 1000, 1.0e-14, 1.0e-14);
191        integrator.addEventHandler(new SecondaryStateEvent(index, -3.0), 0.1, 1.0e-9, 1000);
192        integrator.setInitialStepSize(3.0);
193
194        integrator.integrate(equation, 30.0);
195        Assert.assertEquals( 1.0, equation.getTime(), 1.0e-10);
196        Assert.assertEquals( 2.0, equation.getPrimaryState()[0], 1.0e-10);
197        Assert.assertEquals(-3.0, equation.getSecondaryState(0)[0], 1.0e-10);
198
199    }

201    private static class SecondaryStateEvent implements EventHandler {
202
203        private int index;
204        private final double target;
205
206        public SecondaryStateEvent(final int index, final double target) {
207            this.index  = index;
208            this.target = target;
209        }
210
211        public void init(double t0, double[] y0, double t) {
212        }
213
214        public double g(double t, double[] y) {
215            return y[index] - target;
216        }
217
218        public Action eventOccurred(double t, double[] y, boolean increasing) {
219            return Action.STOP;
220        }
221
222        public void resetState(double t, double[] y) {
223        }
224
225    }
