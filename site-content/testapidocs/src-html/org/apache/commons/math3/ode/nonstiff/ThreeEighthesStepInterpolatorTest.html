018package org.apache.commons.math3.ode.nonstiff;
019
020
021import java.io.ByteArrayInputStream;
022import java.io.ByteArrayOutputStream;
023import java.io.IOException;
024import java.io.ObjectInputStream;
025import java.io.ObjectOutputStream;
026import java.util.Random;
027
028import org.apache.commons.math3.exception.DimensionMismatchException;
029import org.apache.commons.math3.exception.MaxCountExceededException;
030import org.apache.commons.math3.exception.NoBracketingException;
031import org.apache.commons.math3.exception.NumberIsTooSmallException;
032import org.apache.commons.math3.ode.ContinuousOutputModel;
033import org.apache.commons.math3.ode.TestProblem3;
034import org.apache.commons.math3.ode.sampling.StepHandler;
035import org.apache.commons.math3.ode.sampling.StepInterpolatorTestUtils;
036import org.junit.Assert;
037import org.junit.Test;
038
039public class ThreeEighthesStepInterpolatorTest {
040
041  @Test
042  public void derivativesConsistency()
043      throws DimensionMismatchException, NumberIsTooSmallException,
044             MaxCountExceededException, NoBracketingException {
045    TestProblem3 pb = new TestProblem3();
046    double step = (pb.getFinalTime() - pb.getInitialTime()) * 0.001;
047    ThreeEighthesIntegrator integ = new ThreeEighthesIntegrator(step);
048    StepInterpolatorTestUtils.checkDerivativesConsistency(integ, pb, 1.0e-10);
049  }
050
051  @Test
052  public void serialization()
053    throws IOException, ClassNotFoundException,
054           DimensionMismatchException, NumberIsTooSmallException,
055           MaxCountExceededException, NoBracketingException {
056
057    TestProblem3 pb = new TestProblem3(0.9);
058    double step = (pb.getFinalTime() - pb.getInitialTime()) * 0.0003;
059    ThreeEighthesIntegrator integ = new ThreeEighthesIntegrator(step);
060    integ.addStepHandler(new ContinuousOutputModel());
061    integ.integrate(pb,
062                    pb.getInitialTime(), pb.getInitialState(),
063                    pb.getFinalTime(), new double[pb.getDimension()]);
064
065    ByteArrayOutputStream bos = new ByteArrayOutputStream();
066    ObjectOutputStream    oos = new ObjectOutputStream(bos);
067    for (StepHandler handler : integ.getStepHandlers()) {
068        oos.writeObject(handler);
069    }
070
071    Assert.assertTrue(bos.size () > 880000);
072    Assert.assertTrue(bos.size () < 900000);
073
074    ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
075    ObjectInputStream     ois = new ObjectInputStream(bis);
076    ContinuousOutputModel cm  = (ContinuousOutputModel) ois.readObject();
077
078    Random random = new Random(347588535632l);
079    double maxError = 0.0;
080    for (int i = 0; i < 1000; ++i) {
081      double r = random.nextDouble();
082      double time = r * pb.getInitialTime() + (1.0 - r) * pb.getFinalTime();
083      cm.setInterpolatedTime(time);
084      double[] interpolatedY = cm.getInterpolatedState ();
085      double[] theoreticalY  = pb.computeTheoreticalState(time);
086      double dx = interpolatedY[0] - theoreticalY[0];
087      double dy = interpolatedY[1] - theoreticalY[1];
088      double error = dx * dx + dy * dy;
089      if (error > maxError) {
090        maxError = error;
091      }
092    }
093
094    Assert.assertTrue(maxError > 0.005);
095
096  }
