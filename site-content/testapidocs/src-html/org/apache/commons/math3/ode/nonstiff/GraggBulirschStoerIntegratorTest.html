113  @Test
114  public void testIncreasingTolerance()
115      throws DimensionMismatchException, NumberIsTooSmallException,
116             MaxCountExceededException, NoBracketingException {
117
118    int previousCalls = Integer.MAX_VALUE;
119    for (int i = -12; i < -4; ++i) {
120      TestProblem1 pb     = new TestProblem1();
121      double minStep      = 0;
122      double maxStep      = pb.getFinalTime() - pb.getInitialTime();
123      double absTolerance = FastMath.pow(10.0, i);
124      double relTolerance = absTolerance;
125
126      FirstOrderIntegrator integ =
127        new GraggBulirschStoerIntegrator(minStep, maxStep,
128                                         absTolerance, relTolerance);
129      TestProblemHandler handler = new TestProblemHandler(pb, integ);
130      integ.addStepHandler(handler);
131      integ.integrate(pb,
132                      pb.getInitialTime(), pb.getInitialState(),
133                      pb.getFinalTime(), new double[pb.getDimension()]);
134
135      // the coefficients are only valid for this test
136      // and have been obtained from trial and error
137      // there is no general relation between local and global errors
138      double ratio =  handler.getMaximalValueError() / absTolerance;
139      Assert.assertTrue(ratio < 2.4);
140      Assert.assertTrue(ratio > 0.02);
141      Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-12);
142
143      int calls = pb.getCalls();
144      Assert.assertEquals(integ.getEvaluations(), calls);
145      Assert.assertTrue(calls <= previousCalls);
146      previousCalls = calls;
147
148    }

152  @Test
153  public void testIntegratorControls()
154      throws DimensionMismatchException, NumberIsTooSmallException,
155             MaxCountExceededException, NoBracketingException {
156
157    TestProblem3 pb = new TestProblem3(0.999);
158    GraggBulirschStoerIntegrator integ =
159        new GraggBulirschStoerIntegrator(0, pb.getFinalTime() - pb.getInitialTime(),
160                1.0e-8, 1.0e-10);
161
162    double errorWithDefaultSettings = getMaxError(integ, pb);
163
164    // stability control
165    integ.setStabilityCheck(true, 2, 1, 0.99);
166    Assert.assertTrue(errorWithDefaultSettings < getMaxError(integ, pb));
167    integ.setStabilityCheck(true, -1, -1, -1);
168
169    integ.setControlFactors(0.5, 0.99, 0.1, 2.5);
170    Assert.assertTrue(errorWithDefaultSettings < getMaxError(integ, pb));
171    integ.setControlFactors(-1, -1, -1, -1);
172
173    integ.setOrderControl(10, 0.7, 0.95);
174    Assert.assertTrue(errorWithDefaultSettings < getMaxError(integ, pb));
175    integ.setOrderControl(-1, -1, -1);
176
177    integ.setInterpolationControl(true, 3);
178    Assert.assertTrue(errorWithDefaultSettings < getMaxError(integ, pb));
179    integ.setInterpolationControl(true, -1);
180
181  }

183  private double getMaxError(FirstOrderIntegrator integrator, TestProblemAbstract pb)
184      throws DimensionMismatchException, NumberIsTooSmallException,
185             MaxCountExceededException, NoBracketingException {
186      TestProblemHandler handler = new TestProblemHandler(pb, integrator);

189                           pb.getInitialTime(), pb.getInitialState(),
190                           pb.getFinalTime(), new double[pb.getDimension()]);
191      return handler.getMaximalValueError();
192  }

194  @Test
195  public void testEvents()
196      throws DimensionMismatchException, NumberIsTooSmallException,
197             MaxCountExceededException, NoBracketingException {
198
199    TestProblem4 pb = new TestProblem4();
200    double minStep = 0;
201    double maxStep = pb.getFinalTime() - pb.getInitialTime();
202    double scalAbsoluteTolerance = 1.0e-10;
203    double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
204
205    FirstOrderIntegrator integ = new GraggBulirschStoerIntegrator(minStep, maxStep,
206                                                                  scalAbsoluteTolerance,
207                                                                  scalRelativeTolerance);
208    TestProblemHandler handler = new TestProblemHandler(pb, integ);
209    integ.addStepHandler(handler);
210    EventHandler[] functions = pb.getEventsHandlers();
211    double convergence = 1.0e-8 * maxStep;
212    for (int l = 0; l < functions.length; ++l) {
213      integ.addEventHandler(functions[l], Double.POSITIVE_INFINITY, convergence, 1000);
214    }
215    Assert.assertEquals(functions.length, integ.getEventHandlers().size());
216    integ.integrate(pb,
217                    pb.getInitialTime(), pb.getInitialState(),
218                    pb.getFinalTime(), new double[pb.getDimension()]);
219
220    Assert.assertTrue(handler.getMaximalValueError() < 4.0e-7);
221    Assert.assertEquals(0, handler.getMaximalTimeError(), convergence);
222    Assert.assertEquals(12.0, handler.getLastTime(), convergence);
223    integ.clearEventHandlers();
224    Assert.assertEquals(0, integ.getEventHandlers().size());
225
226  }

349  private static class KeplerStepHandler implements StepHandler {
350    public KeplerStepHandler(TestProblem3 pb) {
351      this.pb = pb;
352    }
353    public void init(double t0, double[] y0, double t) {
354      nbSteps = 0;
355      maxError = 0;
356    }
357    public void handleStep(StepInterpolator interpolator, boolean isLast)
358        throws MaxCountExceededException {
359
360      ++nbSteps;
361      for (int a = 1; a < 100; ++a) {
362
363        double prev   = interpolator.getPreviousTime();
364        double curr   = interpolator.getCurrentTime();
365        double interp = ((100 - a) * prev + a * curr) / 100;
366        interpolator.setInterpolatedTime(interp);
367
368        double[] interpolatedY = interpolator.getInterpolatedState ();
369        double[] theoreticalY  = pb.computeTheoreticalState(interpolator.getInterpolatedTime());
370        double dx = interpolatedY[0] - theoreticalY[0];
371        double dy = interpolatedY[1] - theoreticalY[1];
372        double error = dx * dx + dy * dy;
373        if (error > maxError) {
374          maxError = error;
375        }
376      }
377      if (isLast) {
378        Assert.assertTrue(maxError < 2.7e-6);
379        Assert.assertTrue(nbSteps < 80);
380      }
381    }
382    private int nbSteps;
383    private double maxError;
384    private TestProblem3 pb;
385  }

387  public static class VariableStepHandler implements StepHandler {
388    public VariableStepHandler() {
389        firstTime = true;
390        minStep = 0;
391        maxStep = 0;
392    }
393    public void init(double t0, double[] y0, double t) {
394      firstTime = true;
395      minStep = 0;
396      maxStep = 0;
397    }
398    public void handleStep(StepInterpolator interpolator,
399                           boolean isLast) {
400
401      double step = FastMath.abs(interpolator.getCurrentTime()
402                             - interpolator.getPreviousTime());
403      if (firstTime) {
404        minStep   = FastMath.abs(step);
405        maxStep   = minStep;
406        firstTime = false;
407      } else {
408        if (step < minStep) {
409          minStep = step;
410        }
411        if (step > maxStep) {
412          maxStep = step;
413        }
414      }
415
416      if (isLast) {
417        Assert.assertTrue(minStep < 8.2e-3);
418        Assert.assertTrue(maxStep > 1.5);
419      }
420    }
421    private boolean firstTime;
422    private double  minStep;
423    private double  maxStep;
