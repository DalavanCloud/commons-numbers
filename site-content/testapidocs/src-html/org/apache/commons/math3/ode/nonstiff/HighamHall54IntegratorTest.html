048            public void computeDerivatives(double t, double[] y, double[] dot) {

064        // expected behavior
065      }
066

070      } catch(RuntimeException de) {
071        // expected behavior
072      }

099  @Test
100  public void testIncreasingTolerance()
101      throws DimensionMismatchException, NumberIsTooSmallException,
102             MaxCountExceededException, NoBracketingException {
103
104    int previousCalls = Integer.MAX_VALUE;
105    for (int i = -12; i < -2; ++i) {
106      TestProblem1 pb = new TestProblem1();
107      double minStep = 0;
108      double maxStep = pb.getFinalTime() - pb.getInitialTime();
109      double scalAbsoluteTolerance = FastMath.pow(10.0, i);
110      double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
111
112      FirstOrderIntegrator integ = new HighamHall54Integrator(minStep, maxStep,
113                                                              scalAbsoluteTolerance,
114                                                              scalRelativeTolerance);
115      TestProblemHandler handler = new TestProblemHandler(pb, integ);
116      integ.addStepHandler(handler);
117      integ.integrate(pb,
118                      pb.getInitialTime(), pb.getInitialState(),
119                      pb.getFinalTime(), new double[pb.getDimension()]);
120
121      // the 1.3 factor is only valid for this test
122      // and has been obtained from trial and error
123      // there is no general relation between local and global errors
124      Assert.assertTrue(handler.getMaximalValueError() < (1.3 * scalAbsoluteTolerance));
125      Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-12);
126
127      int calls = pb.getCalls();
128      Assert.assertEquals(integ.getEvaluations(), calls);
129      Assert.assertTrue(calls <= previousCalls);
130      previousCalls = calls;
131
132    }
133
134  }

161  @Test
162  public void testEvents()
163      throws DimensionMismatchException, NumberIsTooSmallException,
164             MaxCountExceededException, NoBracketingException {
165
166    TestProblem4 pb = new TestProblem4();
167    double minStep = 0;
168    double maxStep = pb.getFinalTime() - pb.getInitialTime();
169    double scalAbsoluteTolerance = 1.0e-8;
170    double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
171
172    FirstOrderIntegrator integ = new HighamHall54Integrator(minStep, maxStep,
173                                                            scalAbsoluteTolerance,
174                                                            scalRelativeTolerance);
175    TestProblemHandler handler = new TestProblemHandler(pb, integ);
176    integ.addStepHandler(handler);
177    EventHandler[] functions = pb.getEventsHandlers();
178    double convergence = 1.0e-8 * maxStep;
179    for (int l = 0; l < functions.length; ++l) {
180      integ.addEventHandler(functions[l],
181                                 Double.POSITIVE_INFINITY, convergence, 1000);
182    }
183    Assert.assertEquals(functions.length, integ.getEventHandlers().size());
184    integ.integrate(pb,
185                    pb.getInitialTime(), pb.getInitialState(),
186                    pb.getFinalTime(), new double[pb.getDimension()]);
187
188    Assert.assertTrue(handler.getMaximalValueError() < 1.0e-7);
189    Assert.assertEquals(0, handler.getMaximalTimeError(), convergence);
190    Assert.assertEquals(12.0, handler.getLastTime(), convergence);
191    integ.clearEventHandlers();
192    Assert.assertEquals(0, integ.getEventHandlers().size());
193
194  }

253        new HighamHall54Integrator(minStep, maxStep,

264      public double g(double t, double[] y) {
265        double middle = (pb.getInitialTime() + pb.getFinalTime()) / 2;

269      public void resetState(double t, double[] y) {
270      }
