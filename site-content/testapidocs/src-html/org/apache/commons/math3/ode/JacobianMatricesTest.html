033    @Test
034    public void testLowAccuracyExternalDifferentiation()
035        throws NumberIsTooSmallException, DimensionMismatchException,
036               MaxCountExceededException, NoBracketingException {
037        // this test does not really test JacobianMatrices,
038        // it only shows that WITHOUT this class, attempting to recover
039        // the jacobians from external differentiation on simple integration
040        // results with low accuracy gives very poor results. In fact,
041        // the curves dy/dp = g(b) when b varies from 2.88 to 3.08 are
042        // essentially noise.
043        // This test is taken from Hairer, Norsett and Wanner book
044        // Solving Ordinary Differential Equations I (Nonstiff problems),
045        // the curves dy/dp = g(b) are in figure 6.5
046        FirstOrderIntegrator integ =
047            new DormandPrince54Integrator(1.0e-8, 100.0, new double[] { 1.0e-4, 1.0e-4 }, new double[] { 1.0e-4, 1.0e-4 });
048        double hP = 1.0e-12;
049        SummaryStatistics residualsP0 = new SummaryStatistics();
050        SummaryStatistics residualsP1 = new SummaryStatistics();
051        for (double b = 2.88; b < 3.08; b += 0.001) {
052            Brusselator brusselator = new Brusselator(b);
053            double[] y = { 1.3, b };
054            integ.integrate(brusselator, 0, y, 20.0, y);
055            double[] yP = { 1.3, b + hP };
056            integ.integrate(brusselator, 0, yP, 20.0, yP);
057            residualsP0.addValue((yP[0] - y[0]) / hP - brusselator.dYdP0());
058            residualsP1.addValue((yP[1] - y[1]) / hP - brusselator.dYdP1());
059        }
060        Assert.assertTrue((residualsP0.getMax() - residualsP0.getMin()) > 500);
061        Assert.assertTrue(residualsP0.getStandardDeviation() > 30);
062        Assert.assertTrue((residualsP1.getMax() - residualsP1.getMin()) > 700);
063        Assert.assertTrue(residualsP1.getStandardDeviation() > 40);
064    }

066    @Test
067    public void testHighAccuracyExternalDifferentiation()
068        throws NumberIsTooSmallException, DimensionMismatchException,
069               MaxCountExceededException, NoBracketingException, UnknownParameterException {
070        FirstOrderIntegrator integ =
071            new DormandPrince54Integrator(1.0e-8, 100.0, new double[] { 1.0e-10, 1.0e-10 }, new double[] { 1.0e-10, 1.0e-10 });
072        double hP = 1.0e-12;
073        SummaryStatistics residualsP0 = new SummaryStatistics();
074        SummaryStatistics residualsP1 = new SummaryStatistics();
075        for (double b = 2.88; b < 3.08; b += 0.001) {
076            ParamBrusselator brusselator = new ParamBrusselator(b);
077            double[] y = { 1.3, b };
078            integ.integrate(brusselator, 0, y, 20.0, y);
079            double[] yP = { 1.3, b + hP };
080            brusselator.setParameter("b", b + hP);
081            integ.integrate(brusselator, 0, yP, 20.0, yP);
082            residualsP0.addValue((yP[0] - y[0]) / hP - brusselator.dYdP0());
083            residualsP1.addValue((yP[1] - y[1]) / hP - brusselator.dYdP1());
084        }
085        Assert.assertTrue((residualsP0.getMax() - residualsP0.getMin()) > 0.02);
086        Assert.assertTrue((residualsP0.getMax() - residualsP0.getMin()) < 0.03);
087        Assert.assertTrue(residualsP0.getStandardDeviation() > 0.003);
088        Assert.assertTrue(residualsP0.getStandardDeviation() < 0.004);
089        Assert.assertTrue((residualsP1.getMax() - residualsP1.getMin()) > 0.04);
090        Assert.assertTrue((residualsP1.getMax() - residualsP1.getMin()) < 0.05);
091        Assert.assertTrue(residualsP1.getStandardDeviation() > 0.007);
092        Assert.assertTrue(residualsP1.getStandardDeviation() < 0.008);
093    }

180    @Test
181    public void testFinalResult()
182        throws MaxCountExceededException, DimensionMismatchException,
183               NumberIsTooSmallException, NoBracketingException,
184               UnknownParameterException, MismatchedEquations {
185
186        AbstractIntegrator integ =
187            new DormandPrince54Integrator(1.0e-8, 100.0, new double[] { 1.0e-10, 1.0e-10 }, new double[] { 1.0e-10, 1.0e-10 });
188        double[] y = new double[] { 0.0, 1.0 };
189        Circle circle = new Circle(y, 1.0, 1.0, 0.1);
190
191        JacobianMatrices jacob = new JacobianMatrices(circle, Circle.CX, Circle.CY, Circle.OMEGA);
192        jacob.addParameterJacobianProvider(circle);
193        jacob.setInitialMainStateJacobian(circle.exactDyDy0(0));
194        jacob.setInitialParameterJacobian(Circle.CX, circle.exactDyDcx(0));
195        jacob.setInitialParameterJacobian(Circle.CY, circle.exactDyDcy(0));
196        jacob.setInitialParameterJacobian(Circle.OMEGA, circle.exactDyDom(0));
197
198        ExpandableStatefulODE efode = new ExpandableStatefulODE(circle);
199        efode.setTime(0);
200        efode.setPrimaryState(y);
201        jacob.registerVariationalEquations(efode);
202
203        integ.setMaxEvaluations(5000);
204
205        double t = 18 * FastMath.PI;
206        integ.integrate(efode, t);
207        y = efode.getPrimaryState();
208        for (int i = 0; i < y.length; ++i) {
209            Assert.assertEquals(circle.exactY(t)[i], y[i], 1.0e-9);
210        }
211
212        double[][] dydy0 = new double[2][2];
213        jacob.getCurrentMainSetJacobian(dydy0);
214        for (int i = 0; i < dydy0.length; ++i) {
215            for (int j = 0; j < dydy0[i].length; ++j) {
216                Assert.assertEquals(circle.exactDyDy0(t)[i][j], dydy0[i][j], 1.0e-9);
217            }
218        }
219        double[] dydcx = new double[2];
220        jacob.getCurrentParameterJacobian(Circle.CX, dydcx);
221        for (int i = 0; i < dydcx.length; ++i) {
222            Assert.assertEquals(circle.exactDyDcx(t)[i], dydcx[i], 1.0e-7);
223        }
224        double[] dydcy = new double[2];
225        jacob.getCurrentParameterJacobian(Circle.CY, dydcy);
226        for (int i = 0; i < dydcy.length; ++i) {
227            Assert.assertEquals(circle.exactDyDcy(t)[i], dydcy[i], 1.0e-7);
228        }
229        double[] dydom = new double[2];
230        jacob.getCurrentParameterJacobian(Circle.OMEGA, dydom);
231        for (int i = 0; i < dydom.length; ++i) {
232            Assert.assertEquals(circle.exactDyDom(t)[i], dydom[i], 1.0e-7);
233        }
234    }

297    private static class Brusselator extends AbstractParameterizable
298        implements MainStateJacobianProvider, ParameterJacobianProvider {
299
300        public static final String B = "b";
301
302        private double b;
303
304        public Brusselator(double b) {
305            super(B);
306            this.b = b;
307        }
308
309        public int getDimension() {
310            return 2;
311        }
312
313        public void computeDerivatives(double t, double[] y, double[] yDot) {
314            double prod = y[0] * y[0] * y[1];
315            yDot[0] = 1 + prod - (b + 1) * y[0];
316            yDot[1] = b * y[0] - prod;
317        }
318
319        public void computeMainStateJacobian(double t, double[] y, double[] yDot,
320                                             double[][] dFdY) {
321            double p = 2 * y[0] * y[1];
322            double y02 = y[0] * y[0];
323            dFdY[0][0] = p - (1 + b);
324            dFdY[0][1] = y02;
325            dFdY[1][0] = b - p;
326            dFdY[1][1] = -y02;
327        }
328
329        public void computeParameterJacobian(double t, double[] y, double[] yDot,
330                                             String paramName, double[] dFdP) {
331            if (isSupported(paramName)) {
332                dFdP[0] = -y[0];
333                dFdP[1] = y[0];
334            } else {
335                dFdP[0] = 0;
336                dFdP[1] = 0;
337            }
338        }
339
340        public double dYdP0() {
341            return -1088.232716447743 + (1050.775747149553 + (-339.012934631828 + 36.52917025056327 * b) * b) * b;
342        }
343
344        public double dYdP1() {
345            return 1502.824469929139 + (-1438.6974831849952 + (460.959476642384 - 49.43847385647082 * b) * b) * b;
346        }
347
348    }

396    /** ODE representing a point moving on a circle with provided center and angular rate. */
397    private static class Circle extends AbstractParameterizable
398        implements MainStateJacobianProvider, ParameterJacobianProvider {
399
400        public static final String CX = "cx";
401        public static final String CY = "cy";
402        public static final String OMEGA = "omega";
403
404        private final double[] y0;
405        private double cx;
406        private double cy;
407        private double omega;
408
409        public Circle(double[] y0, double cx, double cy, double omega) {
410            super(CX,CY,OMEGA);
411            this.y0    = y0.clone();
412            this.cx    = cx;
413            this.cy    = cy;
414            this.omega = omega;
415        }
416
417        public int getDimension() {
418            return 2;
419        }
420
421        public void computeDerivatives(double t, double[] y, double[] yDot) {
422            yDot[0] = omega * (cy - y[1]);
423            yDot[1] = omega * (y[0] - cx);
424        }
425
426        public void computeMainStateJacobian(double t, double[] y,
427                                             double[] yDot, double[][] dFdY) {
428            dFdY[0][0] = 0;
429            dFdY[0][1] = -omega;
430            dFdY[1][0] = omega;
431            dFdY[1][1] = 0;
432        }
433
434        public void computeParameterJacobian(double t, double[] y, double[] yDot,
435                                             String paramName, double[] dFdP)
436            throws UnknownParameterException {
437            complainIfNotSupported(paramName);
438            if (paramName.equals(CX)) {
439                dFdP[0] = 0;
440                dFdP[1] = -omega;
441            } else if (paramName.equals(CY)) {
442                dFdP[0] = omega;
443                dFdP[1] = 0;
444            }  else {
445                dFdP[0] = cy - y[1];
446                dFdP[1] = y[0] - cx;
447            }           
448        }
449
450        public double[] exactY(double t) {
451            double cos = FastMath.cos(omega * t);
452            double sin = FastMath.sin(omega * t);
453            double dx0 = y0[0] - cx;
454            double dy0 = y0[1] - cy;
455            return new double[] {
456                cx + cos * dx0 - sin * dy0,
457                cy + sin * dx0 + cos * dy0
458            };
459        }
460
461        public double[][] exactDyDy0(double t) {
462            double cos = FastMath.cos(omega * t);
463            double sin = FastMath.sin(omega * t);
464            return new double[][] {
465                { cos, -sin },
466                { sin,  cos }
467            };
468        }
469
470        public double[] exactDyDcx(double t) {
471            double cos = FastMath.cos(omega * t);
472            double sin = FastMath.sin(omega * t);
473            return new double[] {1 - cos, -sin};
474        }
475
476        public double[] exactDyDcy(double t) {
477            double cos = FastMath.cos(omega * t);
478            double sin = FastMath.sin(omega * t);
479            return new double[] {sin, 1 - cos};
480        }
481
482        public double[] exactDyDom(double t) {
483            double cos = FastMath.cos(omega * t);
484            double sin = FastMath.sin(omega * t);
485            double dx0 = y0[0] - cx;
486            double dy0 = y0[1] - cy;
487            return new double[] { -t * (sin * dx0 + cos * dy0) , t * (cos * dx0 - sin * dy0) };
488        }
489
490    }

492    /** ODE representing a point moving on a circle with provided center and angular rate. */
493    private static class ParameterizedCircle extends AbstractParameterizable
494        implements FirstOrderDifferentialEquations, ParameterizedODE {
495
496        public static final String CX = "cx";
497        public static final String CY = "cy";
498        public static final String OMEGA = "omega";
499
500        private final double[] y0;
501        private double cx;
502        private double cy;
503        private double omega;
504
505        public ParameterizedCircle(double[] y0, double cx, double cy, double omega) {
506            super(CX,CY,OMEGA);
507            this.y0    = y0.clone();
508            this.cx    = cx;
509            this.cy    = cy;
510            this.omega = omega;
511        }
512
513        public int getDimension() {
514            return 2;
515        }
516
517        public void computeDerivatives(double t, double[] y, double[] yDot) {
518            yDot[0] = omega * (cy - y[1]);
519            yDot[1] = omega * (y[0] - cx);
520        }
521
522        public double getParameter(final String name)
523            throws UnknownParameterException {
524            if (name.equals(CX)) {
525                return cx;
526            } else if (name.equals(CY)) {
527                    return cy;
528            } else if (name.equals(OMEGA)) {
529                return omega;
530            } else {
531                throw new UnknownParameterException(name);
532            }
533        }
534
535        public void setParameter(final String name, final double value)
536            throws UnknownParameterException {
537            if (name.equals(CX)) {
538                cx = value;
539            } else if (name.equals(CY)) {
540                cy = value;
541            } else if (name.equals(OMEGA)) {
542                omega = value;
543            } else {
544                throw new UnknownParameterException(name);
545            }
546        }

548        public double[] exactY(double t) {
549            double cos = FastMath.cos(omega * t);
550            double sin = FastMath.sin(omega * t);
551            double dx0 = y0[0] - cx;
552            double dy0 = y0[1] - cy;
553            return new double[] {
554                cx + cos * dx0 - sin * dy0,
555                cy + sin * dx0 + cos * dy0
556            };
557        }

559        public double[][] exactDyDy0(double t) {
560            double cos = FastMath.cos(omega * t);
561            double sin = FastMath.sin(omega * t);
562            return new double[][] {
563                { cos, -sin },
564                { sin,  cos }
565            };
566        }

568        public double[] exactDyDcx(double t) {
569            double cos = FastMath.cos(omega * t);
570            double sin = FastMath.sin(omega * t);
571            return new double[] {1 - cos, -sin};
572        }

574        public double[] exactDyDcy(double t) {
575            double cos = FastMath.cos(omega * t);
576            double sin = FastMath.sin(omega * t);
577            return new double[] {sin, 1 - cos};
578        }

580        public double[] exactDyDom(double t) {
581            double cos = FastMath.cos(omega * t);
582            double sin = FastMath.sin(omega * t);
583            double dx0 = y0[0] - cx;
584            double dy0 = y0[1] - cy;
585            return new double[] { -t * (sin * dx0 + cos * dy0) , t * (cos * dx0 - sin * dy0) };
586        }
