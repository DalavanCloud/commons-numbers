155  @Test
156  public void testIncreasingTolerance()
157      throws DimensionMismatchException, NumberIsTooSmallException,
158             MaxCountExceededException, NoBracketingException {
159
160    int previousCalls = Integer.MAX_VALUE;
161    for (int i = -12; i < -2; ++i) {
162      TestProblem1 pb = new TestProblem1();
163      double minStep = 0;
164      double maxStep = pb.getFinalTime() - pb.getInitialTime();
165      double scalAbsoluteTolerance = FastMath.pow(10.0, i);
166      double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
167
168      EmbeddedRungeKuttaIntegrator integ =
169          new DormandPrince54Integrator(minStep, maxStep,
170                                        scalAbsoluteTolerance, scalRelativeTolerance);
171      TestProblemHandler handler = new TestProblemHandler(pb, integ);
172      integ.setSafety(0.8);
173      integ.setMaxGrowth(5.0);
174      integ.setMinReduction(0.3);
175      integ.addStepHandler(handler);
176      integ.integrate(pb,
177                      pb.getInitialTime(), pb.getInitialState(),
178                      pb.getFinalTime(), new double[pb.getDimension()]);
179      Assert.assertEquals(0.8, integ.getSafety(), 1.0e-12);
180      Assert.assertEquals(5.0, integ.getMaxGrowth(), 1.0e-12);
181      Assert.assertEquals(0.3, integ.getMinReduction(), 1.0e-12);
182
183      // the 0.7 factor is only valid for this test
184      // and has been obtained from trial and error
185      // there is no general relation between local and global errors
186      Assert.assertTrue(handler.getMaximalValueError() < (0.7 * scalAbsoluteTolerance));
187      Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-12);
188
189      int calls = pb.getCalls();
190      Assert.assertEquals(integ.getEvaluations(), calls);
191      Assert.assertTrue(calls <= previousCalls);
192      previousCalls = calls;
193
194    }

198  @Test
199  public void testEvents()
200      throws DimensionMismatchException, NumberIsTooSmallException,
201             MaxCountExceededException, NoBracketingException {
202
203    TestProblem4 pb = new TestProblem4();
204    double minStep = 0;
205    double maxStep = pb.getFinalTime() - pb.getInitialTime();
206    double scalAbsoluteTolerance = 1.0e-8;
207    double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
208
209    FirstOrderIntegrator integ = new DormandPrince54Integrator(minStep, maxStep,
210                                                               scalAbsoluteTolerance,
211                                                               scalRelativeTolerance);
212    TestProblemHandler handler = new TestProblemHandler(pb, integ);
213    integ.addStepHandler(handler);
214    EventHandler[] functions = pb.getEventsHandlers();
215    double convergence = 1.0e-8 * maxStep;
216    for (int l = 0; l < functions.length; ++l) {
217      integ.addEventHandler(functions[l],
218                                 Double.POSITIVE_INFINITY, convergence, 1000);
219    }
220    Assert.assertEquals(functions.length, integ.getEventHandlers().size());
221    integ.integrate(pb,
222                    pb.getInitialTime(), pb.getInitialState(),
223                    pb.getFinalTime(), new double[pb.getDimension()]);
224
225    Assert.assertTrue(handler.getMaximalValueError() < 5.0e-6);
226    Assert.assertEquals(0, handler.getMaximalTimeError(), convergence);
227    Assert.assertEquals(12.0, handler.getLastTime(), convergence);
228    integ.clearEventHandlers();
229    Assert.assertEquals(0, integ.getEventHandlers().size());
230
231  }

277  private static class KeplerHandler implements StepHandler {
278    public KeplerHandler(TestProblem3 pb) {
279      this.pb = pb;
280    }
281    public void init(double t0, double[] y0, double t) {
282      nbSteps = 0;
283      maxError = 0;
284    }
285    public void handleStep(StepInterpolator interpolator, boolean isLast)
286        throws MaxCountExceededException {
287
288      ++nbSteps;
289      for (int a = 1; a < 10; ++a) {
290
291        double prev   = interpolator.getPreviousTime();
292        double curr   = interpolator.getCurrentTime();
293        double interp = ((10 - a) * prev + a * curr) / 10;
294        interpolator.setInterpolatedTime(interp);
295
296        double[] interpolatedY = interpolator.getInterpolatedState ();
297        double[] theoreticalY  = pb.computeTheoreticalState(interpolator.getInterpolatedTime());
298        double dx = interpolatedY[0] - theoreticalY[0];
299        double dy = interpolatedY[1] - theoreticalY[1];
300        double error = dx * dx + dy * dy;
301        if (error > maxError) {
302          maxError = error;
303        }
304      }
305      if (isLast) {
306        Assert.assertTrue(maxError < 7.0e-10);
307        Assert.assertTrue(nbSteps < 400);
308      }
309    }
310    private int nbSteps;
311    private double maxError;
312    private TestProblem3 pb;
313  }

315  private static class VariableHandler implements StepHandler {
316    public VariableHandler() {
317      firstTime = true;
318      minStep = 0;
319      maxStep = 0;
320    }
321    public void init(double t0, double[] y0, double t) {
322      firstTime = true;
323      minStep = 0;
324      maxStep = 0;
325    }
326    public void handleStep(StepInterpolator interpolator,
327                           boolean isLast) {
328
329      double step = FastMath.abs(interpolator.getCurrentTime()
330                             - interpolator.getPreviousTime());
331      if (firstTime) {
332        minStep   = FastMath.abs(step);
333        maxStep   = minStep;
334        firstTime = false;
335      } else {
336        if (step < minStep) {
337          minStep = step;
338        }
339        if (step > maxStep) {
340          maxStep = step;
341        }
342      }
343
344      if (isLast) {
345        Assert.assertTrue(minStep < (1.0 / 450.0));
346        Assert.assertTrue(maxStep > (1.0 / 4.2));
347      }
348    }
349    private boolean firstTime;
350    private double  minStep;
351    private double  maxStep;
352  }
