056  @Test
057  public void testRandomAccess() throws DimensionMismatchException, NumberIsTooSmallException, MaxCountExceededException, NoBracketingException {
058
059    ContinuousOutputModel cm = new ContinuousOutputModel();
060    integ.addStepHandler(cm);
061    integ.integrate(pb,
062                    pb.getInitialTime(), pb.getInitialState(),
063                    pb.getFinalTime(), new double[pb.getDimension()]);
064
065    Random random = new Random(347588535632l);
066    double maxError    = 0.0;
067    double maxErrorDot = 0.0;
068    for (int i = 0; i < 1000; ++i) {
069      double r = random.nextDouble();
070      double time = r * pb.getInitialTime() + (1.0 - r) * pb.getFinalTime();
071      cm.setInterpolatedTime(time);
072      double[] interpolatedY    = cm.getInterpolatedState();
073      double[] interpolatedYDot = cm.getInterpolatedDerivatives();
074      double[] theoreticalY     = pb.computeTheoreticalState(time);
075      double[] theoreticalYDot  = new double[pb.getDimension()];
076      pb.doComputeDerivatives(time, theoreticalY, theoreticalYDot);
077      double dx = interpolatedY[0] - theoreticalY[0];
078      double dy = interpolatedY[1] - theoreticalY[1];
079      double error = dx * dx + dy * dy;
080      maxError = FastMath.max(maxError, error);
081      double dxDot = interpolatedYDot[0] - theoreticalYDot[0];
082      double dyDot = interpolatedYDot[1] - theoreticalYDot[1];
083      double errorDot = dxDot * dxDot + dyDot * dyDot;
084      maxErrorDot = FastMath.max(maxErrorDot, errorDot);
085    }
086
087    Assert.assertEquals(0.0, maxError,    1.0e-9);
088    Assert.assertEquals(0.0, maxErrorDot, 4.0e-7);
089
090  }

092  @Test
093  public void testModelsMerging() throws MaxCountExceededException, MathIllegalArgumentException {
094
095      // theoretical solution: y[0] = cos(t), y[1] = sin(t)
096      FirstOrderDifferentialEquations problem =
097          new FirstOrderDifferentialEquations() {
098              public void computeDerivatives(double t, double[] y, double[] dot) {
099                  dot[0] = -y[1];
100                  dot[1] =  y[0];
101              }
102              public int getDimension() {
103                  return 2;
104              }
105          };
106
107      // integrate backward from &pi; to 0;
108      ContinuousOutputModel cm1 = new ContinuousOutputModel();
109      FirstOrderIntegrator integ1 =
110          new DormandPrince853Integrator(0, 1.0, 1.0e-8, 1.0e-8);
111      integ1.addStepHandler(cm1);
112      integ1.integrate(problem, FastMath.PI, new double[] { -1.0, 0.0 },
113                       0, new double[2]);
114
115      // integrate backward from 2&pi; to &pi;
116      ContinuousOutputModel cm2 = new ContinuousOutputModel();
117      FirstOrderIntegrator integ2 =
118          new DormandPrince853Integrator(0, 0.1, 1.0e-12, 1.0e-12);
119      integ2.addStepHandler(cm2);
120      integ2.integrate(problem, 2.0 * FastMath.PI, new double[] { 1.0, 0.0 },
121                       FastMath.PI, new double[2]);
122
123      // merge the two half circles
124      ContinuousOutputModel cm = new ContinuousOutputModel();
125      cm.append(cm2);
126      cm.append(new ContinuousOutputModel());
127      cm.append(cm1);
128
129      // check circle
130      Assert.assertEquals(2.0 * FastMath.PI, cm.getInitialTime(), 1.0e-12);
131      Assert.assertEquals(0, cm.getFinalTime(), 1.0e-12);
132      Assert.assertEquals(cm.getFinalTime(), cm.getInterpolatedTime(), 1.0e-12);
133      for (double t = 0; t < 2.0 * FastMath.PI; t += 0.1) {
134          cm.setInterpolatedTime(t);
135          double[] y = cm.getInterpolatedState();
136          Assert.assertEquals(FastMath.cos(t), y[0], 1.0e-7);
137          Assert.assertEquals(FastMath.sin(t), y[1], 1.0e-7);
138      }
139
140  }

142  @Test
143  public void testErrorConditions() throws MaxCountExceededException, MathIllegalArgumentException {
144
145      ContinuousOutputModel cm = new ContinuousOutputModel();
146      cm.handleStep(buildInterpolator(0, new double[] { 0.0, 1.0, -2.0 }, 1), true);
147
148      // dimension mismatch
149      Assert.assertTrue(checkAppendError(cm, 1.0, new double[] { 0.0, 1.0 }, 2.0));
150
151      // hole between time ranges
152      Assert.assertTrue(checkAppendError(cm, 10.0, new double[] { 0.0, 1.0, -2.0 }, 20.0));
153
154      // propagation direction mismatch
155      Assert.assertTrue(checkAppendError(cm, 1.0, new double[] { 0.0, 1.0, -2.0 }, 0.0));
156
157      // no errors
158      Assert.assertFalse(checkAppendError(cm, 1.0, new double[] { 0.0, 1.0, -2.0 }, 2.0));
159
160  }
161
162  private boolean checkAppendError(ContinuousOutputModel cm,
163                                   double t0, double[] y0, double t1)
164      throws MaxCountExceededException, MathIllegalArgumentException {
165      try {
166          ContinuousOutputModel otherCm = new ContinuousOutputModel();
167          otherCm.handleStep(buildInterpolator(t0, y0, t1), true);
168          cm.append(otherCm);
169      } catch(IllegalArgumentException iae) {
170          return true; // there was an allowable error
171      }
172      return false; // no allowable error
173  }
174
175  private StepInterpolator buildInterpolator(double t0, double[] y0, double t1) {
176      DummyStepInterpolator interpolator  = new DummyStepInterpolator(y0, new double[y0.length], t1 >= t0);
177      interpolator.storeTime(t0);
178      interpolator.shift();
179      interpolator.storeTime(t1);
180      return interpolator;
181  }

183  public void checkValue(double value, double reference) {
184    Assert.assertTrue(FastMath.abs(value - reference) < 1.0e-10);
