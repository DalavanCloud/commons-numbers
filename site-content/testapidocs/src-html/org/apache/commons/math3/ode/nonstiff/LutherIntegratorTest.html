042    @Test
043    public void testMissedEndEvent()
044            throws DimensionMismatchException, NumberIsTooSmallException,
045            MaxCountExceededException, NoBracketingException {
046        final double   t0     = 1878250320.0000029;
047        final double   tEvent = 1878250379.9999986;
048        final double[] k      = { 1.0e-4, 1.0e-5, 1.0e-6 };
049        FirstOrderDifferentialEquations ode = new FirstOrderDifferentialEquations() {
050
051            public int getDimension() {
052                return k.length;
053            }
054
055            public void computeDerivatives(double t, double[] y, double[] yDot) {
056                for (int i = 0; i < y.length; ++i) {
057                    yDot[i] = k[i] * y[i];
058                }
059            }
060        };
061
062        LutherIntegrator integrator = new LutherIntegrator(60.0);
063
064        double[] y0   = new double[k.length];
065        for (int i = 0; i < y0.length; ++i) {
066            y0[i] = i + 1;
067        }
068        double[] y    = new double[k.length];
069
070        double finalT = integrator.integrate(ode, t0, y0, tEvent, y);
071        Assert.assertEquals(tEvent, finalT, 1.0e-15);
072        for (int i = 0; i < y.length; ++i) {
073            Assert.assertEquals(y0[i] * FastMath.exp(k[i] * (finalT - t0)), y[i], 1.0e-15);
074        }
075
076        integrator.addEventHandler(new EventHandler() {
077
078            public void init(double t0, double[] y0, double t) {
079            }
080
081            public void resetState(double t, double[] y) {
082            }
083
084            public double g(double t, double[] y) {
085                return t - tEvent;
086            }
087
088            public Action eventOccurred(double t, double[] y, boolean increasing) {
089                Assert.assertEquals(tEvent, t, 1.0e-15);
090                return Action.CONTINUE;
091            }
092        }, Double.POSITIVE_INFINITY, 1.0e-20, 100);
093        finalT = integrator.integrate(ode, t0, y0, tEvent + 120, y);
094        Assert.assertEquals(tEvent + 120, finalT, 1.0e-15);
095        for (int i = 0; i < y.length; ++i) {
096            Assert.assertEquals(y0[i] * FastMath.exp(k[i] * (finalT - t0)), y[i], 1.0e-15);
097        }
098
099    }

131    @Test
132    public void testDecreasingSteps()
133            throws DimensionMismatchException, NumberIsTooSmallException,
134            MaxCountExceededException, NoBracketingException {
135
136        TestProblemAbstract[] problems = TestProblemFactory.getProblems();
137        for (int k = 0; k < problems.length; ++k) {
138
139            double previousValueError = Double.NaN;
140            double previousTimeError = Double.NaN;
141            for (int i = 4; i < 10; ++i) {
142
143                TestProblemAbstract pb = problems[k].copy();
144                double step = (pb.getFinalTime() - pb.getInitialTime()) * FastMath.pow(2.0, -i);
145
146                FirstOrderIntegrator integ = new LutherIntegrator(step);
147                TestProblemHandler handler = new TestProblemHandler(pb, integ);
148                integ.addStepHandler(handler);
149                EventHandler[] functions = pb.getEventsHandlers();
150                for (int l = 0; l < functions.length; ++l) {
151                    integ.addEventHandler(functions[l],
152                                          Double.POSITIVE_INFINITY, 1.0e-6 * step, 1000);
153                }
154                Assert.assertEquals(functions.length, integ.getEventHandlers().size());
155                double stopTime = integ.integrate(pb, pb.getInitialTime(), pb.getInitialState(),
156                                                  pb.getFinalTime(), new double[pb.getDimension()]);
157                if (functions.length == 0) {
158                    Assert.assertEquals(pb.getFinalTime(), stopTime, 1.0e-10);
159                }
160
161                double error = handler.getMaximalValueError();
162                if (i > 4) {
163                    Assert.assertTrue(error < 1.01 * FastMath.abs(previousValueError));
164                }
165                previousValueError = error;
166
167                double timeError = handler.getMaximalTimeError();
168                if (i > 4) {
169                    Assert.assertTrue(timeError <= FastMath.abs(previousTimeError));
170                }
171                previousTimeError = timeError;
172
173                integ.clearEventHandlers();
174                Assert.assertEquals(0, integ.getEventHandlers().size());
175            }
176
177        }
178
179    }

183            throws DimensionMismatchException, NumberIsTooSmallException,
184            MaxCountExceededException, NoBracketingException {
185

243            throws DimensionMismatchException, NumberIsTooSmallException,
244            MaxCountExceededException, NoBracketingException {
245
246        final TestProblem3 pb  = new TestProblem3(0.9);
247        double step = (pb.getFinalTime() - pb.getInitialTime()) * 0.0003;
248
249        FirstOrderIntegrator integ = new LutherIntegrator(step);
250        integ.addStepHandler(new KeplerHandler(pb));
251        integ.integrate(pb,
252                        pb.getInitialTime(), pb.getInitialState(),
253                        pb.getFinalTime(), new double[pb.getDimension()]);

255
256    private static class KeplerHandler implements StepHandler {
257        public KeplerHandler(TestProblem3 pb) {
258            this.pb = pb;
259            maxError = 0;
260        }
261        public void init(double t0, double[] y0, double t) {
262            maxError = 0;
263        }
264        public void handleStep(StepInterpolator interpolator, boolean isLast) {
265
266            double[] interpolatedY = interpolator.getInterpolatedState ();
267            double[] theoreticalY  = pb.computeTheoreticalState(interpolator.getCurrentTime());
268            double dx = interpolatedY[0] - theoreticalY[0];
269            double dy = interpolatedY[1] - theoreticalY[1];
270            double error = dx * dx + dy * dy;
271            if (error > maxError) {
272                maxError = error;
273            }
274            if (isLast) {
275                Assert.assertTrue(maxError < 2.2e-7);
276            }
277        }
278        private double maxError = 0;
279        private TestProblem3 pb;
280    }

283    public void testStepSize()
284            throws DimensionMismatchException, NumberIsTooSmallException,
285            MaxCountExceededException, NoBracketingException {
286        final double step = 1.23456;
287        FirstOrderIntegrator integ = new LutherIntegrator(step);
288        integ.addStepHandler(new StepHandler() {
289            public void handleStep(StepInterpolator interpolator, boolean isLast) {
290                if (! isLast) {
291                    Assert.assertEquals(step,
292                                        interpolator.getCurrentTime() - interpolator.getPreviousTime(),
293                                        1.0e-12);
294                }
295            }
296            public void init(double t0, double[] y0, double t) {
297            }
298        });
299        integ.integrate(new FirstOrderDifferentialEquations() {
300            public void computeDerivatives(double t, double[] y, double[] dot) {
301                dot[0] = 1.0;
302            }
303            public int getDimension() {
304                return 1;
305            }
306        }, 0.0, new double[] { 0.0 }, 5.0, new double[1]);
307    }

309    @Test
310    public void testSingleStep() {
311
312        final TestProblem3 pb  = new TestProblem3(0.9);
313        double h = (pb.getFinalTime() - pb.getInitialTime()) * 0.0003;
314
315        RungeKuttaIntegrator integ = new LutherIntegrator(Double.NaN);
316        double   t = pb.getInitialTime();
317        double[] y = pb.getInitialState();
318        for (int i = 0; i < 100; ++i) {
319            y = integ.singleStep(pb, t, y, t + h);
320            t += h;
321        }
322        double[] yth = pb.computeTheoreticalState(t);
323        double dx = y[0] - yth[0];
324        double dy = y[1] - yth[1];
325        double error = dx * dx + dy * dy;
326        Assert.assertEquals(0.0, error, 1.0e-11);
327    }
