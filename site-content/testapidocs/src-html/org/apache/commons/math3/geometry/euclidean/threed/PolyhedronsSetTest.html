129    @Test
130    public void testIsometry() throws MathArithmeticException, MathIllegalArgumentException {
131        Vector3D vertex1 = new Vector3D(1.1, 2.2, 3.3);
132        Vector3D vertex2 = new Vector3D(2.0, 2.4, 4.2);
133        Vector3D vertex3 = new Vector3D(2.8, 3.3, 3.7);
134        Vector3D vertex4 = new Vector3D(1.0, 3.6, 4.5);
135        @SuppressWarnings("unchecked")
136        PolyhedronsSet tree =
137            (PolyhedronsSet) new RegionFactory<Euclidean3D>().buildConvex(
138                new Plane(vertex3, vertex2, vertex1, 1.0e-10),
139                new Plane(vertex2, vertex3, vertex4, 1.0e-10),
140                new Plane(vertex4, vertex3, vertex1, 1.0e-10),
141                new Plane(vertex1, vertex2, vertex4, 1.0e-10));
142        Vector3D barycenter = (Vector3D) tree.getBarycenter();
143        Vector3D s = new Vector3D(10.2, 4.3, -6.7);
144        Vector3D c = new Vector3D(-0.2, 2.1, -3.2);
145        Rotation r = new Rotation(new Vector3D(6.2, -4.4, 2.1), 0.12);
146
147        tree = tree.rotate(c, r).translate(s);
148
149        Vector3D newB =
150            new Vector3D(1.0, s,
151                         1.0, c,
152                         1.0, r.applyTo(barycenter.subtract(c)));
153        Assert.assertEquals(0.0,
154                            newB.subtract((Vector<Euclidean3D>) tree.getBarycenter()).getNorm(),
155                            1.0e-10);
156
157        final Vector3D[] expectedV = new Vector3D[] {
158            new Vector3D(1.0, s,
159                         1.0, c,
160                         1.0, r.applyTo(vertex1.subtract(c))),
161                         new Vector3D(1.0, s,
162                                      1.0, c,
163                                      1.0, r.applyTo(vertex2.subtract(c))),
164                                      new Vector3D(1.0, s,
165                                                   1.0, c,
166                                                   1.0, r.applyTo(vertex3.subtract(c))),
167                                                   new Vector3D(1.0, s,
168                                                                1.0, c,
169                                                                1.0, r.applyTo(vertex4.subtract(c)))
170        };
171        tree.getTree(true).visit(new BSPTreeVisitor<Euclidean3D>() {
172
173            public Order visitOrder(BSPTree<Euclidean3D> node) {
174                return Order.MINUS_SUB_PLUS;
175            }
176
177            public void visitInternalNode(BSPTree<Euclidean3D> node) {
178                @SuppressWarnings("unchecked")
179                BoundaryAttribute<Euclidean3D> attribute =
180                    (BoundaryAttribute<Euclidean3D>) node.getAttribute();
181                if (attribute.getPlusOutside() != null) {
182                    checkFacet((SubPlane) attribute.getPlusOutside());
183                }
184                if (attribute.getPlusInside() != null) {
185                    checkFacet((SubPlane) attribute.getPlusInside());
186                }
187            }
188
189            public void visitLeafNode(BSPTree<Euclidean3D> node) {
190            }
191
192            private void checkFacet(SubPlane facet) {
193                Plane plane = (Plane) facet.getHyperplane();
194                Vector2D[][] vertices =
195                    ((PolygonsSet) facet.getRemainingRegion()).getVertices();
196                Assert.assertEquals(1, vertices.length);
197                for (int i = 0; i < vertices[0].length; ++i) {
198                    Vector3D v = plane.toSpace(vertices[0][i]);
199                    double d = Double.POSITIVE_INFINITY;
200                    for (int k = 0; k < expectedV.length; ++k) {
201                        d = FastMath.min(d, v.subtract(expectedV[k]).getNorm());
202                    }
203                    Assert.assertEquals(0, d, 1.0e-10);
204                }
205            }
206
207        });
208
209    }

249        Assert.assertEquals(8 * w * w * (3 * l - 2 * w), tree.getSize(), 1.0e-10);
250        Assert.assertEquals(24 * w * (2 * l - w), tree.getBoundarySize(), 1.0e-10);

255    public void testIssue780() throws MathArithmeticException {
256        float[] coords = {
257            1.000000f, -1.000000f, -1.000000f, 
258            1.000000f, -1.000000f, 1.000000f, 
259            -1.000000f, -1.000000f, 1.000000f, 
260            -1.000000f, -1.000000f, -1.000000f, 
261            1.000000f, 1.000000f, -1f, 
262            0.999999f, 1.000000f, 1.000000f,   // 1.000000f, 1.000000f, 1.000000f, 
263            -1.000000f, 1.000000f, 1.000000f, 
264            -1.000000f, 1.000000f, -1.000000f};
265        int[] indices = {
266            0, 1, 2, 0, 2, 3, 
267            4, 7, 6, 4, 6, 5, 
268            0, 4, 5, 0, 5, 1, 
269            1, 5, 6, 1, 6, 2, 
270            2, 6, 7, 2, 7, 3, 
271            4, 0, 3, 4, 3, 7};
272        ArrayList<SubHyperplane<Euclidean3D>> subHyperplaneList = new ArrayList<SubHyperplane<Euclidean3D>>();
273        for (int idx = 0; idx < indices.length; idx += 3) {
274            int idxA = indices[idx] * 3;
275            int idxB = indices[idx + 1] * 3;
276            int idxC = indices[idx + 2] * 3;
277            Vector3D v_1 = new Vector3D(coords[idxA], coords[idxA + 1], coords[idxA + 2]);
278            Vector3D v_2 = new Vector3D(coords[idxB], coords[idxB + 1], coords[idxB + 2]);
279            Vector3D v_3 = new Vector3D(coords[idxC], coords[idxC + 1], coords[idxC + 2]);
280            Vector3D[] vertices = {v_1, v_2, v_3};
281            Plane polyPlane = new Plane(v_1, v_2, v_3, 1.0e-10);
282            ArrayList<SubHyperplane<Euclidean2D>> lines = new ArrayList<SubHyperplane<Euclidean2D>>();
283
284            Vector2D[] projPts = new Vector2D[vertices.length];
285            for (int ptIdx = 0; ptIdx < projPts.length; ptIdx++) {
286                projPts[ptIdx] = polyPlane.toSubSpace(vertices[ptIdx]);
287            }
288
289            SubLine lineInPlane = null;
290            for (int ptIdx = 0; ptIdx < projPts.length; ptIdx++) {
291                lineInPlane = new SubLine(projPts[ptIdx], projPts[(ptIdx + 1) % projPts.length], 1.0e-10);
292                lines.add(lineInPlane);
293            }
294            Region<Euclidean2D> polyRegion = new PolygonsSet(lines, 1.0e-10);
295            SubPlane polygon = new SubPlane(polyPlane, polyRegion);
296            subHyperplaneList.add(polygon);
297        }
298        PolyhedronsSet polyhedronsSet = new PolyhedronsSet(subHyperplaneList, 1.0e-10);
299        Assert.assertEquals( 8.0, polyhedronsSet.getSize(), 3.0e-6);
300        Assert.assertEquals(24.0, polyhedronsSet.getBoundarySize(), 5.0e-6);
301    }

432                Assert.assertEquals(expected, list.get(0));
433            } catch (NoSuchFieldException nsfe) {
434                Assert.fail(nsfe.getLocalizedMessage());
435            } catch (IllegalAccessException iae) {
436                Assert.fail(iae.getLocalizedMessage());
437            }
438        }

441    @Test
442    public void testIssue1211() throws IOException, ParseException {
443
444        PolyhedronsSet polyset = RegionParser.parsePolyhedronsSet(loadTestData("issue-1211.bsp"));
445        RandomGenerator random = new Well1024a(0xb97c9d1ade21e40al);
446        int nrays = 1000;
447        for (int i = 0; i < nrays; i++) {
448            Vector3D origin    = Vector3D.ZERO;
449            Vector3D direction = new Vector3D(2 * random.nextDouble() - 1,
450                                              2 * random.nextDouble() - 1,
451                                              2 * random.nextDouble() - 1).normalize();
452            Line line = new Line(origin, origin.add(direction), polyset.getTolerance());
453            SubHyperplane<Euclidean3D> plane = polyset.firstIntersection(origin, line);
454            if (plane != null) {
455                Vector3D intersectionPoint = ((Plane)plane.getHyperplane()).intersection(line);
456                double dotProduct = direction.dotProduct(intersectionPoint.subtract(origin));
457                Assert.assertTrue(dotProduct > 0);
458            }
459        }
460    }
