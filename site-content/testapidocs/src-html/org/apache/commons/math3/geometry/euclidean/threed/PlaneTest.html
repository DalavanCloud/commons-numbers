030    @Test
031    public void testContains() throws MathArithmeticException {
032        Plane p = new Plane(new Vector3D(0, 0, 1), new Vector3D(0, 0, 1), 1.0e-10);
033        Assert.assertTrue(p.contains(new Vector3D(0, 0, 1)));
034        Assert.assertTrue(p.contains(new Vector3D(17, -32, 1)));
035        Assert.assertTrue(! p.contains(new Vector3D(17, -32, 1.001)));
036    }

038    @Test
039    public void testOffset() throws MathArithmeticException {
040        Vector3D p1 = new Vector3D(1, 1, 1);
041        Plane p = new Plane(p1, new Vector3D(0.2, 0, 0), 1.0e-10);
042        Assert.assertEquals(-5.0, p.getOffset(new Vector3D(-4, 0, 0)), 1.0e-10);
043        Assert.assertEquals(+5.0, p.getOffset(new Vector3D(6, 10, -12)), 1.0e-10);
044        Assert.assertEquals(0.3,
045                            p.getOffset(new Vector3D(1.0, p1, 0.3, p.getNormal())),
046                            1.0e-10);
047        Assert.assertEquals(-0.3,
048                            p.getOffset(new Vector3D(1.0, p1, -0.3, p.getNormal())),
049                            1.0e-10);
050    }

069    @Test
070    public void testRotate() throws MathArithmeticException, MathIllegalArgumentException {
071        Vector3D p1 = new Vector3D(1.2, 3.4, -5.8);
072        Vector3D p2 = new Vector3D(3.4, -5.8, 1.2);
073        Vector3D p3 = new Vector3D(-2.0, 4.3, 0.7);
074        Plane    p  = new Plane(p1, p2, p3, 1.0e-10);
075        Vector3D oldNormal = p.getNormal();
076
077        p = p.rotate(p2, new Rotation(p2.subtract(p1), 1.7));
078        Assert.assertTrue(p.contains(p1));
079        Assert.assertTrue(p.contains(p2));
080        Assert.assertTrue(! p.contains(p3));
081
082        p = p.rotate(p2, new Rotation(oldNormal, 0.1));
083        Assert.assertTrue(! p.contains(p1));
084        Assert.assertTrue(p.contains(p2));
085        Assert.assertTrue(! p.contains(p3));
086
087        p = p.rotate(p1, new Rotation(oldNormal, 0.1));
088        Assert.assertTrue(! p.contains(p1));
089        Assert.assertTrue(! p.contains(p2));
090        Assert.assertTrue(! p.contains(p3));
091
092    }

094    @Test
095    public void testTranslate() throws MathArithmeticException {
096        Vector3D p1 = new Vector3D(1.2, 3.4, -5.8);
097        Vector3D p2 = new Vector3D(3.4, -5.8, 1.2);
098        Vector3D p3 = new Vector3D(-2.0, 4.3, 0.7);
099        Plane    p  = new Plane(p1, p2, p3, 1.0e-10);
100
101        p = p.translate(new Vector3D(2.0, p.getU(), -1.5, p.getV()));
102        Assert.assertTrue(p.contains(p1));
103        Assert.assertTrue(p.contains(p2));
104        Assert.assertTrue(p.contains(p3));
105
106        p = p.translate(new Vector3D(-1.2, p.getNormal()));
107        Assert.assertTrue(! p.contains(p1));
108        Assert.assertTrue(! p.contains(p2));
109        Assert.assertTrue(! p.contains(p3));
110
111        p = p.translate(new Vector3D(+1.2, p.getNormal()));
112        Assert.assertTrue(p.contains(p1));
113        Assert.assertTrue(p.contains(p2));
114        Assert.assertTrue(p.contains(p3));
115
116    }

118    @Test
119    public void testIntersection() throws MathArithmeticException, MathIllegalArgumentException {
120        Plane p = new Plane(new Vector3D(1, 2, 3), new Vector3D(-4, 1, -5), 1.0e-10);
121        Line  l = new Line(new Vector3D(0.2, -3.5, 0.7), new Vector3D(1.2, -2.5, -0.3), 1.0e-10);
122        Vector3D point = p.intersection(l);
123        Assert.assertTrue(p.contains(point));
124        Assert.assertTrue(l.contains(point));
125        Assert.assertNull(p.intersection(new Line(new Vector3D(10, 10, 10),
126                                                  new Vector3D(10, 10, 10).add(p.getNormal().orthogonal()),
127                                                  1.0e-10)));
128    }

132        Vector3D p1  = new Vector3D (1.2, 3.4, -5.8);
133        Vector3D p2  = new Vector3D (3.4, -5.8, 1.2);

135        Plane    pB  = new Plane(p1, new Vector3D (11.4, -3.8, 5.1), p2, 1.0e-10);
136        Line     l   = pA.intersection(pB);
137        Assert.assertTrue(l.contains(p1));
138        Assert.assertTrue(l.contains(p2));
139        Assert.assertNull(pA.intersection(pA));

142    @Test
143    public void testIntersection3() throws MathArithmeticException {
144        Vector3D reference = new Vector3D (1.2, 3.4, -5.8);
145        Plane p1 = new Plane(reference, new Vector3D(1, 3, 3), 1.0e-10);
146        Plane p2 = new Plane(reference, new Vector3D(-2, 4, 0), 1.0e-10);
147        Plane p3 = new Plane(reference, new Vector3D(7, 0, -4), 1.0e-10);
148        Vector3D p = Plane.intersection(p1, p2, p3);
149        Assert.assertEquals(reference.getX(), p.getX(), 1.0e-10);
150        Assert.assertEquals(reference.getY(), p.getY(), 1.0e-10);
151        Assert.assertEquals(reference.getZ(), p.getZ(), 1.0e-10);
152    }

154    @Test
155    public void testSimilar() throws MathArithmeticException {
156        Vector3D p1  = new Vector3D (1.2, 3.4, -5.8);
157        Vector3D p2  = new Vector3D (3.4, -5.8, 1.2);
158        Vector3D p3  = new Vector3D (-2.0, 4.3, 0.7);
159        Plane    pA  = new Plane(p1, p2, p3, 1.0e-10);
160        Plane    pB  = new Plane(p1, new Vector3D (11.4, -3.8, 5.1), p2, 1.0e-10);
161        Assert.assertTrue(! pA.isSimilarTo(pB));
162        Assert.assertTrue(pA.isSimilarTo(pA));
163        Assert.assertTrue(pA.isSimilarTo(new Plane(p1, p3, p2, 1.0e-10)));
164        Vector3D shift = new Vector3D(0.3, pA.getNormal());
165        Assert.assertTrue(! pA.isSimilarTo(new Plane(p1.add(shift),
166                                                     p3.add(shift),
167                                                     p2.add(shift),
168                                                     1.0e-10)));
169    }
