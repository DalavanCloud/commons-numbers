112  public void testVectorTwoPairs() throws MathArithmeticException {
113
114    Vector3D u1 = new Vector3D(3, 0, 0);
115    Vector3D u2 = new Vector3D(0, 5, 0);
116    Vector3D v1 = new Vector3D(0, 0, 2);
117    Vector3D v2 = new Vector3D(-2, 0, 2);
118    Rotation r = new Rotation(u1, u2, v1, v2);
119    checkVector(r.applyTo(Vector3D.PLUS_I), Vector3D.PLUS_K);
120    checkVector(r.applyTo(Vector3D.PLUS_J), Vector3D.MINUS_I);
121
122    r = new Rotation(u1, u2, u1.negate(), u2.negate());
123    Vector3D axis = r.getAxis();
124    if (Vector3D.dotProduct(axis, Vector3D.PLUS_K) > 0) {
125      checkVector(axis, Vector3D.PLUS_K);
126    } else {
127      checkVector(axis, Vector3D.MINUS_K);
128    }
129    checkAngle(r.getAngle(), FastMath.PI);
130
131    double sqrt = FastMath.sqrt(2) / 2;
132    r = new Rotation(Vector3D.PLUS_I,  Vector3D.PLUS_J,
133                     new Vector3D(0.5, 0.5,  sqrt),
134                     new Vector3D(0.5, 0.5, -sqrt));
135    checkRotation(r, sqrt, 0.5, 0.5, 0);
136
137    r = new Rotation(u1, u2, u1, Vector3D.crossProduct(u1, u2));
138    checkRotation(r, sqrt, -sqrt, 0, 0);
139
140    checkRotation(new Rotation(u1, u2, u1, u2), 1, 0, 0, 0);
141
142    try {
143        new Rotation(u1, u2, Vector3D.ZERO, v2);
144        Assert.fail("an exception should have been thrown");
145    } catch (MathArithmeticException e) {
146      // expected behavior
147    }
