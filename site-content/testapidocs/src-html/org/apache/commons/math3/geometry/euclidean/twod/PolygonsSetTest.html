150    @Test
151    public void testHole() {
152        Vector2D[][] vertices = new Vector2D[][] {
153            new Vector2D[] {
154                new Vector2D(0.0, 0.0),
155                new Vector2D(3.0, 0.0),
156                new Vector2D(3.0, 3.0),
157                new Vector2D(0.0, 3.0)
158            }, new Vector2D[] {
159                new Vector2D(1.0, 2.0),
160                new Vector2D(2.0, 2.0),
161                new Vector2D(2.0, 1.0),
162                new Vector2D(1.0, 1.0)
163            }
164        };
165        PolygonsSet set = buildSet(vertices);
166        checkPoints(Region.Location.INSIDE, set, new Vector2D[] {
167            new Vector2D(0.5, 0.5),
168            new Vector2D(1.5, 0.5),
169            new Vector2D(2.5, 0.5),
170            new Vector2D(0.5, 1.5),
171            new Vector2D(2.5, 1.5),
172            new Vector2D(0.5, 2.5),
173            new Vector2D(1.5, 2.5),
174            new Vector2D(2.5, 2.5),
175            new Vector2D(0.5, 1.0)
176        });
177        checkPoints(Region.Location.OUTSIDE, set, new Vector2D[] {
178            new Vector2D(1.5, 1.5),
179            new Vector2D(3.5, 1.0),
180            new Vector2D(4.0, 1.5),
181            new Vector2D(6.0, 6.0)
182        });
183        checkPoints(Region.Location.BOUNDARY, set, new Vector2D[] {
184            new Vector2D(1.0, 1.0),
185            new Vector2D(1.5, 0.0),
186            new Vector2D(1.5, 1.0),
187            new Vector2D(1.5, 2.0),
188            new Vector2D(1.5, 3.0),
189            new Vector2D(3.0, 3.0)
190        });
191        checkVertices(set.getVertices(), vertices);
192
193        for (double x = -0.999; x < 3.999; x += 0.11) {
194            Vector2D v = new Vector2D(x, x + 0.5);
195            BoundaryProjection<Euclidean2D> projection = set.projectToBoundary(v);
196            Assert.assertTrue(projection.getOriginal() == v);
197            Vector2D p = (Vector2D) projection.getProjected();
198            if (x < -0.5) {
199                Assert.assertEquals(0.0,      p.getX(), 1.0e-10);
200                Assert.assertEquals(0.0,      p.getY(), 1.0e-10);
201                Assert.assertEquals(+v.distance(Vector2D.ZERO), projection.getOffset(), 1.0e-10);
202            } else if (x < 0.5) {
203                Assert.assertEquals(0.0,      p.getX(), 1.0e-10);
204                Assert.assertEquals(v.getY(), p.getY(), 1.0e-10);
205                Assert.assertEquals(-v.getX(), projection.getOffset(), 1.0e-10);
206            } else if (x < 1.25) {
207                Assert.assertEquals(1.0,      p.getX(), 1.0e-10);
208                Assert.assertEquals(v.getY(), p.getY(), 1.0e-10);
209                Assert.assertEquals(v.getX() - 1.0, projection.getOffset(), 1.0e-10);
210            } else if (x < 2.0) {
211                Assert.assertEquals(v.getX(), p.getX(), 1.0e-10);
212                Assert.assertEquals(2.0,      p.getY(), 1.0e-10);
213                Assert.assertEquals(2.0 - v.getY(), projection.getOffset(), 1.0e-10);
214            } else if (x < 3.0) {
215                Assert.assertEquals(v.getX(), p.getX(), 1.0e-10);
216                Assert.assertEquals(3.0,      p.getY(), 1.0e-10);
217                Assert.assertEquals(v.getY() - 3.0, projection.getOffset(), 1.0e-10);
218            } else {
219                Assert.assertEquals(3.0,      p.getX(), 1.0e-10);
220                Assert.assertEquals(3.0,      p.getY(), 1.0e-10);
221                Assert.assertEquals(+v.distance(new Vector2D(3, 3)), projection.getOffset(), 1.0e-10);
222            }
223            
224        }
225

295    @Test
296    public void testLineIntersection() {
297        Vector2D[][] vertices = new Vector2D[][] {
298            new Vector2D[] {
299                new Vector2D( 0.0,  0.0),
300                new Vector2D( 2.0,  0.0),
301                new Vector2D( 2.0,  1.0),
302                new Vector2D( 3.0,  1.0),
303                new Vector2D( 3.0,  3.0),
304                new Vector2D( 1.0,  3.0),
305                new Vector2D( 1.0,  2.0),
306                new Vector2D( 0.0,  2.0)
307            }
308        };
309        PolygonsSet set = buildSet(vertices);
310
311        Line l1 = new Line(new Vector2D(-1.5, 0.0), FastMath.PI / 4, 1.0e-10);
312        SubLine s1 = (SubLine) set.intersection(l1.wholeHyperplane());
313        List<Interval> i1 = ((IntervalsSet) s1.getRemainingRegion()).asList();
314        Assert.assertEquals(2, i1.size());
315        Interval v10 = i1.get(0);
316        Vector2D p10Lower = l1.toSpace(new Vector1D(v10.getInf()));
317        Assert.assertEquals(0.0, p10Lower.getX(), 1.0e-10);
318        Assert.assertEquals(1.5, p10Lower.getY(), 1.0e-10);
319        Vector2D p10Upper = l1.toSpace(new Vector1D(v10.getSup()));
320        Assert.assertEquals(0.5, p10Upper.getX(), 1.0e-10);
321        Assert.assertEquals(2.0, p10Upper.getY(), 1.0e-10);
322        Interval v11 = i1.get(1);
323        Vector2D p11Lower = l1.toSpace(new Vector1D(v11.getInf()));
324        Assert.assertEquals(1.0, p11Lower.getX(), 1.0e-10);
325        Assert.assertEquals(2.5, p11Lower.getY(), 1.0e-10);
326        Vector2D p11Upper = l1.toSpace(new Vector1D(v11.getSup()));
327        Assert.assertEquals(1.5, p11Upper.getX(), 1.0e-10);
328        Assert.assertEquals(3.0, p11Upper.getY(), 1.0e-10);
329
330        Line l2 = new Line(new Vector2D(-1.0, 2.0), 0, 1.0e-10);
331        SubLine s2 = (SubLine) set.intersection(l2.wholeHyperplane());
332        List<Interval> i2 = ((IntervalsSet) s2.getRemainingRegion()).asList();
333        Assert.assertEquals(1, i2.size());
334        Interval v20 = i2.get(0);
335        Vector2D p20Lower = l2.toSpace(new Vector1D(v20.getInf()));
336        Assert.assertEquals(1.0, p20Lower.getX(), 1.0e-10);
337        Assert.assertEquals(2.0, p20Lower.getY(), 1.0e-10);
338        Vector2D p20Upper = l2.toSpace(new Vector1D(v20.getSup()));
339        Assert.assertEquals(3.0, p20Upper.getX(), 1.0e-10);
340        Assert.assertEquals(2.0, p20Upper.getY(), 1.0e-10);

1267    private boolean checkInSegment(Vector2D p,
1268                                   Vector2D p1, Vector2D p2,
1269                                   double tolerance) {
1270        Line line = new Line(p1, p2, 1.0e-10);
1271        if (line.getOffset(p) < tolerance) {
1272            double x  = (line.toSubSpace(p)).getX();
1273            double x1 = (line.toSubSpace(p1)).getX();
1274            double x2 = (line.toSubSpace(p2)).getX();
1275            return (((x - x1) * (x - x2) <= 0.0)
1276                    || (p1.distance(p) < tolerance)
1277                    || (p2.distance(p) < tolerance));
1278        } else {
1279            return false;
1280        }

1283    private void checkVertices(Vector2D[][] rebuiltVertices,
1284                               Vector2D[][] vertices) {
1285
1286        // each rebuilt vertex should be in a segment joining two original vertices
1287        for (int i = 0; i < rebuiltVertices.length; ++i) {
1288            for (int j = 0; j < rebuiltVertices[i].length; ++j) {
1289                boolean inSegment = false;
1290                Vector2D p = rebuiltVertices[i][j];
1291                for (int k = 0; k < vertices.length; ++k) {
1292                    Vector2D[] loop = vertices[k];
1293                    int length = loop.length;
1294                    for (int l = 0; (! inSegment) && (l < length); ++l) {
1295                        inSegment = checkInSegment(p, loop[l], loop[(l + 1) % length], 1.0e-10);
1296                    }
1297                }
1298                Assert.assertTrue(inSegment);
1299            }
1300        }
1301
1302        // each original vertex should have a corresponding rebuilt vertex
1303        for (int k = 0; k < vertices.length; ++k) {
1304            for (int l = 0; l < vertices[k].length; ++l) {
1305                double min = Double.POSITIVE_INFINITY;

1309                                       min);

1311                }
