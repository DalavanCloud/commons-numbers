044/**
045 * Solves the "Travelling Salesman's Problem" (i.e. trying to find the
046 * sequence of cities that minimizes the travel distance) using a 1D
047 * SOFM.
048 */
049public class TravellingSalesmanSolver {
050    private static final long FIRST_NEURON_ID = 0;
051    /** RNG. */
052    private final RandomGenerator random;
053    /** Set of cities. */
054    private final Set<City> cities = new HashSet<City>();
055    /** SOFM. */
056    private final Network net;
057    /** Distance function. */
058    private final DistanceMeasure distance = new EuclideanDistance();
059    /** Total number of neurons. */
060    private final int numberOfNeurons;
061
062    /**
063     * @param cityList List of cities to visit in a single travel.
064     * @param numNeuronsPerCity Number of neurons per city.
065     */
066    public TravellingSalesmanSolver(City[] cityList,
067                                    double numNeuronsPerCity) {
068        this(cityList, numNeuronsPerCity, new Well44497b().nextLong());
069    }
070
071    /**
072     * @param cityList List of cities to visit in a single travel.
073     * @param numNeuronsPerCity Number of neurons per city.
074     * @param seed Seed for the RNG that is used to present the samples
075     * to the trainer.
076     */
077    public TravellingSalesmanSolver(City[] cityList,
078                                    double numNeuronsPerCity,
079                                    long seed) {
080        random = new Well44497b(seed);
081
082        // Make sure that each city will appear only once in the list.
083        for (City city : cityList) {
084            cities.add(city);
085        }
086
087        // Total number of neurons.
088        numberOfNeurons = (int) numNeuronsPerCity * cities.size();
089
090        // Create a network with circle topology.
091        net = new NeuronString(numberOfNeurons, true, makeInitializers()).getNetwork();
092    }
093
094    /**
095     * Creates training tasks.
096     *
097     * @param numTasks Number of tasks to create.
098     * @param numSamplesPerTask Number of training samples per task.
099     * @return the created tasks.
100     */
101    public Runnable[] createParallelTasks(int numTasks,
102                                          long numSamplesPerTask) {
103        final Runnable[] tasks = new Runnable[numTasks];
104        final LearningFactorFunction learning
105            = LearningFactorFunctionFactory.exponentialDecay(2e-1,
106                                                             5e-2,
107                                                             numSamplesPerTask / 2);
108        final NeighbourhoodSizeFunction neighbourhood
109            = NeighbourhoodSizeFunctionFactory.exponentialDecay(0.5 * numberOfNeurons,
110                                                                0.1 * numberOfNeurons,
111                                                                numSamplesPerTask / 2);
112
113        for (int i = 0; i < numTasks; i++) {
114            final KohonenUpdateAction action = new KohonenUpdateAction(distance,
115                                                                       learning,
116                                                                       neighbourhood);
117            tasks[i] = new KohonenTrainingTask(net,
118                                               createRandomIterator(numSamplesPerTask),
119                                               action);
120        }
121        
122        return tasks;
123    }
124
125    /**
126     * Creates a training task.
127     *
128     * @param numSamples Number of training samples.
129     * @return the created task.
130     */
131    public Runnable createSequentialTask(long numSamples) {
132        return createParallelTasks(1, numSamples)[0];
133    }
134
135    /**
136     * Creates an iterator that will present a series of city's coordinates in
137     * a random order.
138     *
139     * @param numSamples Number of samples.
140     * @return the iterator.
141     */
142    private Iterator<double[]> createRandomIterator(final long numSamples) {
143        final List<City> cityList = new ArrayList<City>();
144        cityList.addAll(cities);
145
146        return new Iterator<double[]>() {
147            /** Number of samples. */
148            private long n = 0;
149            /** {@inheritDoc} */
150            public boolean hasNext() {
151                return n < numSamples;
152            }
153            /** {@inheritDoc} */
154            public double[] next() {
155                ++n;
156                return cityList.get(random.nextInt(cityList.size())).getCoordinates();
157            }
158            /** {@inheritDoc} */
159            public void remove() {
160                throw new MathUnsupportedOperationException();
161            }
162        };
163    }
164
165    /**
166     * @return the list of linked neurons (i.e. the one-dimensional
167     * SOFM).
168     */
169    private List<Neuron> getNeuronList() {
170        // Sequence of coordinates.
171        final List<Neuron> list = new ArrayList<Neuron>();
172
173        // First neuron.
174        Neuron current = net.getNeuron(FIRST_NEURON_ID);
175        while (true) {
176            list.add(current);
177            final Collection<Neuron> neighbours
178                = net.getNeighbours(current, list);
179
180            final Iterator<Neuron> iter = neighbours.iterator();
181            if (!iter.hasNext()) {
182                // All neurons have been visited.
183                break;
184            }
185
186            current = iter.next();
187        }
188
189        return list;
190    }
191
192    /**
193     * @return the list of features (coordinates) of linked neurons.
194     */
195    public List<double[]> getCoordinatesList() {
196        // Sequence of coordinates.
197        final List<double[]> coordinatesList = new ArrayList<double[]>();
198
199        for (Neuron n : getNeuronList()) {
200            coordinatesList.add(n.getFeatures());
201        }
202
203        return coordinatesList;
204    }
205
206    /**
207     * Returns the travel proposed by the solver.
208     * Note: cities can be missing or duplicated.
209     *
210     * @return the list of cities in travel order.
211     */
212    public City[] getCityList() {
213        final List<double[]> coord = getCoordinatesList();
214        final City[] cityList = new City[coord.size()];
215        for (int i = 0; i < cityList.length; i++) {
216            final double[] c = coord.get(i);
217            cityList[i] = getClosestCity(c[0], c[1]);
218        }
219        return cityList;
220    }
221
222    /**
223     * @param x x-coordinate.
224     * @param y y-coordinate.
225     * @return the city whose coordinates are closest to {@code (x, y)}.
226     */
227    public City getClosestCity(double x,
228                               double y) {
229        City closest = null;
230        double min = Double.POSITIVE_INFINITY;
231        for (City c : cities) {
232            final double d = c.distance(x, y);
233            if (d < min) {
234                min = d;
235                closest = c;
236            }
237        }
238        return closest;
239    }
240
241    /**
242     * Computes the barycentre of all city locations.
243     *
244     * @param cities City list.
245     * @return the barycentre.
246     */
247    private static double[] barycentre(Set<City> cities) {
248        double xB = 0;
249        double yB = 0;
250
251        int count = 0;
252        for (City c : cities) {
253            final double[] coord = c.getCoordinates();
254            xB += coord[0];
255            yB += coord[1];
256
257            ++count;
258        }
259
260        return new double[] { xB / count, yB / count };
261    }
262
263    /**
264     * Computes the largest distance between the point at coordinates
265     * {@code (x, y)} and any of the cities.
266     *
267     * @param x x-coodinate.
268     * @param y y-coodinate.
269     * @param cities City list.
270     * @return the largest distance.
271     */
272    private static double largestDistance(double x,
273                                          double y,
274                                          Set<City> cities) {
275        double maxDist = 0;
276        for (City c : cities) {
277            final double dist = c.distance(x, y);
278            if (dist > maxDist) {
279                maxDist = dist;
280            }
281        }
282
283        return maxDist;
284    }
285
286    /**
287     * Creates the features' initializers: an approximate circle around the
288     * barycentre of the cities.
289     *
290     * @return an array containing the two initializers.
291     */
292    private FeatureInitializer[] makeInitializers() {
293        // Barycentre.
294        final double[] centre = barycentre(cities);
295        // Largest distance from centre.
296        final double radius = 0.5 * largestDistance(centre[0], centre[1], cities);
297
298        final double omega = 2 * Math.PI / numberOfNeurons;
299        final UnivariateFunction h1 = new HarmonicOscillator(radius, omega, 0);
300        final UnivariateFunction h2 = new HarmonicOscillator(radius, omega, 0.5 * Math.PI);
301
302        final UnivariateFunction f1 = FunctionUtils.add(h1, new Constant(centre[0]));
303        final UnivariateFunction f2 = FunctionUtils.add(h2, new Constant(centre[1]));
304
305        final RealDistribution u
306            = new UniformRealDistribution(random, -0.05 * radius, 0.05 * radius);
307
308        return new FeatureInitializer[] {
309            FeatureInitializerFactory.randomize(u, FeatureInitializerFactory.function(f1, 0, 1)),
310            FeatureInitializerFactory.randomize(u, FeatureInitializerFactory.function(f2, 0, 1))
311        };
312    }
313}

315/**
316 * A city, represented by a name and two-dimensional coordinates.
317 */
318class City {
319    /** Identifier. */
320    final String name;
321    /** x-coordinate. */
322    final double x;
323    /** y-coordinate. */
324    final double y;
325
326    /**
327     * @param name Name.
328     * @param x Cartesian x-coordinate.
329     * @param y Cartesian y-coordinate.
330     */
331    public City(String name,
332                double x,
333                double y) {
334        this.name = name;
335        this.x = x;
336        this.y = y;
337    }
338
339    /**
340     * @retun the name.
341     */
342    public String getName() {
343        return name;
344    }
345
346    /**
347     * @return the (x, y) coordinates.
348     */
349    public double[] getCoordinates() {
350        return new double[] { x, y };
351    }
352
353    /**
354     * Computes the distance between this city and
355     * the given point.
356     *
357     * @param x x-coodinate.
358     * @param y y-coodinate.
359     * @return the distance between {@code (x, y)} and this
360     * city.
361     */
362    public double distance(double x,
363                           double y) {
364        final double xDiff = this.x - x;
365        final double yDiff = this.y - y;
366
367        return FastMath.sqrt(xDiff * xDiff + yDiff * yDiff);
368    }
369
370    /** {@inheritDoc} */
371    public boolean equals(Object o) {
372        if (o instanceof City) {
373            final City other = (City) o;
374            return x == other.x &&
375                y == other.y;
376        }
377        return false;
378    }
379
380    /** {@inheritDoc} */
381    public int hashCode() {
382        int result = 17;
383
384        final long c1 = Double.doubleToLongBits(x);
385        result = 31 * result + (int) (c1 ^ (c1 >>> 32));
386
387        final long c2 = Double.doubleToLongBits(y);
388        result = 31 * result + (int) (c2 ^ (c2 >>> 32));
389
390        return result;
391    }
