043    @Test
044    public void testTravellerSalesmanSquareTourSequentialSolver() {
045        // Cities (in optimal travel order).
046        final City[] squareOfCities = new City[] {
047            new City("o0", 0, 0),
048            new City("o1", 1, 0),
049            new City("o2", 2, 0),
050            new City("o3", 3, 0),
051            new City("o4", 3, 1),
052            new City("o5", 3, 2),
053            new City("o6", 3, 3),
054            new City("o7", 2, 3),
055            new City("o8", 1, 3),
056            new City("o9", 0, 3),
057            new City("i3", 1, 2),
058            new City("i2", 2, 2),
059            new City("i1", 2, 1),
060            new City("i0", 1, 1),
061        };
062
063        // Seed that allows the unit test to always succeed.
064        final long seed = 1245632379L;
065
066        final TravellingSalesmanSolver solver = new TravellingSalesmanSolver(squareOfCities, 2, seed);
067        // printSummary("before.travel.seq.dat", solver);
068        solver.createSequentialTask(15000).run();
069        // printSummary("after.travel.seq.dat", solver);
070        final City[] result = solver.getCityList();
071        Assert.assertEquals(squareOfCities.length,
072                            uniqueCities(result).size());
073        final double ratio = computeTravelDistance(squareOfCities) / computeTravelDistance(result);
074        Assert.assertEquals(1, ratio, 1e-1); // We do not require the optimal travel.
075    }

078    @Test
079    @Retry
080    public void testTravellerSalesmanSquareTourParallelSolver() throws ExecutionException {
081        // Cities (in optimal travel order).
082        final City[] squareOfCities = new City[] {
083            new City("o0", 0, 0),
084            new City("o1", 1, 0),
085            new City("o2", 2, 0),
086            new City("o3", 3, 0),
087            new City("o4", 3, 1),
088            new City("o5", 3, 2),
089            new City("o6", 3, 3),
090            new City("o7", 2, 3),
091            new City("o8", 1, 3),
092            new City("o9", 0, 3),
093            new City("i3", 1, 2),
094            new City("i2", 2, 2),
095            new City("i1", 2, 1),
096            new City("i0", 1, 1),
097        };
098
099        final TravellingSalesmanSolver solver = new TravellingSalesmanSolver(squareOfCities, 2);
100        // printSummary("before.travel.par.dat", solver);
101
102        // Parallel execution.
103        final ExecutorService service = Executors.newCachedThreadPool();
104        final Runnable[] tasks = solver.createParallelTasks(3, 5000);
105        final List<Future<?>> execOutput = new ArrayList<Future<?>>();
106        // Run tasks.
107        for (Runnable r : tasks) {
108            execOutput.add(service.submit(r));
109        }
110        // Wait for completion (ignoring return value).
111        try {
112            for (Future<?> f : execOutput) {
113                f.get();
114            }
115        } catch (InterruptedException ignored) {}
116        // Terminate all threads.
117        service.shutdown();
118
119        // printSummary("after.travel.par.dat", solver);
120        final City[] result = solver.getCityList();
121        Assert.assertEquals(squareOfCities.length,
122                            uniqueCities(result).size());
123        final double ratio = computeTravelDistance(squareOfCities) / computeTravelDistance(result);
124        Assert.assertEquals(1, ratio, 1e-1); // We do not require the optimal travel.
125    }

127    /**
128     * Creates a map of the travel suggested by the solver.
129     *
130     * @param solver Solver.
131     * @return a 4-columns table: {@code <x (neuron)> <y (neuron)> <x (city)> <y (city)>}.
132     */
133    private String travelCoordinatesTable(TravellingSalesmanSolver solver) {
134        final StringBuilder s = new StringBuilder();
135        for (double[] c : solver.getCoordinatesList()) {
136            s.append(c[0]).append(" ").append(c[1]).append(" ");
137            final City city = solver.getClosestCity(c[0], c[1]);
138            final double[] cityCoord = city.getCoordinates(); 
139            s.append(cityCoord[0]).append(" ").append(cityCoord[1]).append(" ");
140            s.append("   # ").append(city.getName()).append("\n");
141        }
142        return s.toString();
143    }

145    /**
146     * Compute the distance covered by the salesman, including
147     * the trip back (from the last to first city).
148     *
149     * @param cityList List of cities visited during the travel.
150     * @return the total distance.
151     */
152    private Collection<City> uniqueCities(City[] cityList) {
153        final Set<City> unique = new HashSet<City>();
154        for (City c : cityList) {
155            unique.add(c);
156        }
157        return unique;
158    }
159
160    /**
161     * Compute the distance covered by the salesman, including
162     * the trip back (from the last to first city).
163     *
164     * @param cityList List of cities visited during the travel.
165     * @return the total distance.
166     */
167    private double computeTravelDistance(City[] cityList) {
168        double dist = 0;
169        for (int i = 0; i < cityList.length; i++) {
170            final double[] currentCoord = cityList[i].getCoordinates();
171            final double[] nextCoord = cityList[(i + 1) % cityList.length].getCoordinates();
172
173            final double xDiff = currentCoord[0] - nextCoord[0];
174            final double yDiff = currentCoord[1] - nextCoord[1];
175
176            dist += FastMath.sqrt(xDiff * xDiff + yDiff * yDiff);
177        }
178
179        return dist;
180    }

182    /**
183     * Prints a summary of the current state of the solver to the
184     * given filename.
185     *
186     * @param filename File.
187     * @param solver Solver.
188     */
189    @SuppressWarnings("unused")
190    private void printSummary(String filename,
191                              TravellingSalesmanSolver solver) {
192        PrintWriter out = null;
193        try {
194            out = new PrintWriter(filename);
195            out.println(travelCoordinatesTable(solver));
196
197            final City[] result = solver.getCityList();
198            out.println("# Number of unique cities: " + uniqueCities(result).size());
199            out.println("# Travel distance: " + computeTravelDistance(result));
200        } catch (Exception e) {
201            // Do nothing.
202        } finally {
203            if (out != null) {
204                out.close();
205            }
206        }
207    }
