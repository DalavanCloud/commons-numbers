018package org.apache.commons.math3.ml.clustering;
019
020
021import java.util.Arrays;
022import java.util.List;
023
024import org.junit.Assert;
025import org.junit.Test;
026
027public class MultiKMeansPlusPlusClustererTest {
028
029    @Test
030    public void dimension2() {
031        MultiKMeansPlusPlusClusterer<DoublePoint> transformer =
032            new MultiKMeansPlusPlusClusterer<DoublePoint>(
033                    new KMeansPlusPlusClusterer<DoublePoint>(3, 10), 5);
034        
035        DoublePoint[] points = new DoublePoint[] {
036
037                // first expected cluster
038                new DoublePoint(new int[] { -15,  3 }),
039                new DoublePoint(new int[] { -15,  4 }),
040                new DoublePoint(new int[] { -15,  5 }),
041                new DoublePoint(new int[] { -14,  3 }),
042                new DoublePoint(new int[] { -14,  5 }),
043                new DoublePoint(new int[] { -13,  3 }),
044                new DoublePoint(new int[] { -13,  4 }),
045                new DoublePoint(new int[] { -13,  5 }),
046
047                // second expected cluster
048                new DoublePoint(new int[] { -1,  0 }),
049                new DoublePoint(new int[] { -1, -1 }),
050                new DoublePoint(new int[] {  0, -1 }),
051                new DoublePoint(new int[] {  1, -1 }),
052                new DoublePoint(new int[] {  1, -2 }),
053
054                // third expected cluster
055                new DoublePoint(new int[] { 13,  3 }),
056                new DoublePoint(new int[] { 13,  4 }),
057                new DoublePoint(new int[] { 14,  4 }),
058                new DoublePoint(new int[] { 14,  7 }),
059                new DoublePoint(new int[] { 16,  5 }),
060                new DoublePoint(new int[] { 16,  6 }),
061                new DoublePoint(new int[] { 17,  4 }),
062                new DoublePoint(new int[] { 17,  7 })
063
064        };
065        List<CentroidCluster<DoublePoint>> clusters = transformer.cluster(Arrays.asList(points));
066
067        Assert.assertEquals(3, clusters.size());
068        boolean cluster1Found = false;
069        boolean cluster2Found = false;
070        boolean cluster3Found = false;
071        double epsilon = 1e-6;
072        for (CentroidCluster<DoublePoint> cluster : clusters) {
073            Clusterable center = cluster.getCenter();
074            double[] point = center.getPoint();
075            if (point[0] < 0) {
076                cluster1Found = true;
077                Assert.assertEquals(8, cluster.getPoints().size());
078                Assert.assertEquals(-14, point[0], epsilon);
079                Assert.assertEquals( 4, point[1], epsilon);
080            } else if (point[1] < 0) {
081                cluster2Found = true;
082                Assert.assertEquals(5, cluster.getPoints().size());
083                Assert.assertEquals( 0, point[0], epsilon);
084                Assert.assertEquals(-1, point[1], epsilon);
085            } else {
086                cluster3Found = true;
087                Assert.assertEquals(8, cluster.getPoints().size());
088                Assert.assertEquals(15, point[0], epsilon);
089                Assert.assertEquals(5, point[1], epsilon);
090            }
091        }
092        Assert.assertTrue(cluster1Found);
093        Assert.assertTrue(cluster2Found);
094        Assert.assertTrue(cluster3Found);
095
096    }
097
098}
