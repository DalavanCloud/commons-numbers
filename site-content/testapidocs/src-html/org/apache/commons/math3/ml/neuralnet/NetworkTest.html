020import java.io.ByteArrayInputStream;
021import java.io.ByteArrayOutputStream;
022import java.io.IOException;
023import java.io.ObjectInputStream;
024import java.io.ObjectOutputStream;
025import java.util.Collection;

028import org.apache.commons.math3.ml.neuralnet.twod.NeuronSquareMesh2D;
029import org.junit.Assert;
030import org.junit.Test;
031
032/**
033 * Tests for {@link Network}.
034 */
035public class NetworkTest {
036    final FeatureInitializer init = FeatureInitializerFactory.uniform(0, 2);
037
038    @Test
039    public void testGetFeaturesSize() {
040        final FeatureInitializer[] initArray = { init, init, init };
041
042        final Network net = new NeuronSquareMesh2D(2, false,
043                                                   2, false,
044                                                   SquareNeighbourhood.VON_NEUMANN,
045                                                   initArray).getNetwork();
046        Assert.assertEquals(3, net.getFeaturesSize());
047    }
048
049    /*
050     * Test assumes that the network is
051     *
052     *  0-----1
053     *  |     |
054     *  |     |
055     *  2-----3
056     */
057    @Test
058    public void testDeleteLink() {
059        final FeatureInitializer[] initArray = { init };
060        final Network net = new NeuronSquareMesh2D(2, false,
061                                                   2, false,
062                                                   SquareNeighbourhood.VON_NEUMANN,
063                                                   initArray).getNetwork();
064        Collection<Neuron> neighbours;
065
066        // Delete 0 --> 1.
067        net.deleteLink(net.getNeuron(0),
068                       net.getNeuron(1));
069
070        // Link from 0 to 1 was deleted.
071        neighbours = net.getNeighbours(net.getNeuron(0));
072        Assert.assertFalse(neighbours.contains(net.getNeuron(1)));
073        // Link from 1 to 0 still exists.
074        neighbours = net.getNeighbours(net.getNeuron(1));
075        Assert.assertTrue(neighbours.contains(net.getNeuron(0)));
076    }
077
078    /*
079     * Test assumes that the network is
080     *
081     *  0-----1
082     *  |     |
083     *  |     |
084     *  2-----3
085     */
086    @Test
087    public void testDeleteNeuron() {
088        final FeatureInitializer[] initArray = { init };
089        final Network net = new NeuronSquareMesh2D(2, false,
090                                                   2, false,
091                                                   SquareNeighbourhood.VON_NEUMANN,
092                                                   initArray).getNetwork();
093
094        Assert.assertEquals(2, net.getNeighbours(net.getNeuron(0)).size());
095        Assert.assertEquals(2, net.getNeighbours(net.getNeuron(3)).size());
096
097        // Delete neuron 1.
098        net.deleteNeuron(net.getNeuron(1));
099
100        try {
101            net.getNeuron(1);
102        } catch (NoSuchElementException expected) {}
103
104        Assert.assertEquals(1, net.getNeighbours(net.getNeuron(0)).size());
105        Assert.assertEquals(1, net.getNeighbours(net.getNeuron(3)).size());
106    }
107
108    @Test
109    public void testIterationOrder() {
110        final FeatureInitializer[] initArray = { init };
111        final Network net = new NeuronSquareMesh2D(4, false,
112                                                   3, true,
113                                                   SquareNeighbourhood.VON_NEUMANN,
114                                                   initArray).getNetwork();
115
116        // Check that the comparator provides a specific order.
117        boolean isUnspecifiedOrder = false;
118        long previousId = Long.MIN_VALUE;
119        for (Neuron n : net.getNeurons(new Network.NeuronIdentifierComparator())) {
120            final long currentId = n.getIdentifier();
121            if (currentId < previousId) {
122                isUnspecifiedOrder = true;
123                break;
124            }
125            previousId = currentId;
126        }
127        Assert.assertFalse(isUnspecifiedOrder);
128    }
129
130    @Test
131    public void testSerialize()
132        throws IOException,
133               ClassNotFoundException {
134        final FeatureInitializer[] initArray = { init };
135        final Network out = new NeuronSquareMesh2D(4, false,
136                                                   3, true,
137                                                   SquareNeighbourhood.VON_NEUMANN,
138                                                   initArray).getNetwork();
139
140        final ByteArrayOutputStream bos = new ByteArrayOutputStream();
141        final ObjectOutputStream oos = new ObjectOutputStream(bos);
142        oos.writeObject(out);
143
144        final ByteArrayInputStream bis = new ByteArrayInputStream(bos.toByteArray());
145        final ObjectInputStream ois = new ObjectInputStream(bis);
146        final Network in = (Network) ois.readObject();
147
148        for (Neuron nOut : out) {
149            final Neuron nIn = in.getNeuron(nOut.getIdentifier());
150
151            // Same values.
152            final double[] outF = nOut.getFeatures();
153            final double[] inF = nIn.getFeatures();
154            Assert.assertEquals(outF.length, inF.length);
155            for (int i = 0; i < outF.length; i++) {
156                Assert.assertEquals(outF[i], inF[i], 0d);
157            }
158
159            // Same neighbours.
160            final Collection<Neuron> outNeighbours = out.getNeighbours(nOut);
161            final Collection<Neuron> inNeighbours = in.getNeighbours(nIn);
162            Assert.assertEquals(outNeighbours.size(), inNeighbours.size());
163            for (Neuron oN : outNeighbours) {
164                Assert.assertTrue(inNeighbours.contains(in.getNeuron(oN.getIdentifier())));
165            }
166        }
167    }
