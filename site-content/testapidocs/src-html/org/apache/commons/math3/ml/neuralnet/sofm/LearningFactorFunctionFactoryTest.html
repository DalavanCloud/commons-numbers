018package org.apache.commons.math3.ml.neuralnet.sofm;
019
020import org.apache.commons.math3.exception.NotStrictlyPositiveException;
021import org.apache.commons.math3.exception.OutOfRangeException;
022import org.apache.commons.math3.exception.NumberIsTooLargeException;
023import org.junit.Test;
024import org.junit.Assert;
025
026/**
027 * Tests for {@link LearningFactorFunctionFactory} class.
028 */
029public class LearningFactorFunctionFactoryTest {
030    @Test(expected=OutOfRangeException.class)
031    public void testExponentialDecayPrecondition0() {
032        LearningFactorFunctionFactory.exponentialDecay(0d, 0d, 2);
033    }
034    @Test(expected=OutOfRangeException.class)
035    public void testExponentialDecayPrecondition1() {
036        LearningFactorFunctionFactory.exponentialDecay(1 + 1e-10, 0d, 2);
037    }
038    @Test(expected=NotStrictlyPositiveException.class)
039    public void testExponentialDecayPrecondition2() {
040        LearningFactorFunctionFactory.exponentialDecay(1d, 0d, 2);
041    }
042    @Test(expected=NumberIsTooLargeException.class)
043    public void testExponentialDecayPrecondition3() {
044        LearningFactorFunctionFactory.exponentialDecay(1d, 1d, 100);
045    }
046    @Test(expected=NotStrictlyPositiveException.class)
047    public void testExponentialDecayPrecondition4() {
048        LearningFactorFunctionFactory.exponentialDecay(1d, 0.2, 0);
049    }
050
051    @Test
052    public void testExponentialDecayTrivial() {
053        final int n = 65;
054        final double init = 0.5;
055        final double valueAtN = 0.1;
056        final LearningFactorFunction f
057            = LearningFactorFunctionFactory.exponentialDecay(init, valueAtN, n);
058
059        Assert.assertEquals(init, f.value(0), 0d);
060        Assert.assertEquals(valueAtN, f.value(n), 0d);
061        Assert.assertEquals(0, f.value(Long.MAX_VALUE), 0d);
062    }
063
064    @Test(expected=OutOfRangeException.class)
065    public void testQuasiSigmoidDecayPrecondition0() {
066        LearningFactorFunctionFactory.quasiSigmoidDecay(0d, -1d, 2);
067    }
068    @Test(expected=OutOfRangeException.class)
069    public void testQuasiSigmoidDecayPrecondition1() {
070        LearningFactorFunctionFactory.quasiSigmoidDecay(1 + 1e-10, -1d, 2);
071    }
072    @Test(expected=NumberIsTooLargeException.class)
073    public void testQuasiSigmoidDecayPrecondition3() {
074        LearningFactorFunctionFactory.quasiSigmoidDecay(1d, 0d, 100);
075    }
076    @Test(expected=NotStrictlyPositiveException.class)
077    public void testQuasiSigmoidDecayPrecondition4() {
078        LearningFactorFunctionFactory.quasiSigmoidDecay(1d, -1d, 0);
079    }
080
081    @Test
082    public void testQuasiSigmoidDecayTrivial() {
083        final int n = 65;
084        final double init = 0.5;
085        final double slope = -1e-1;
086        final LearningFactorFunction f
087            = LearningFactorFunctionFactory.quasiSigmoidDecay(init, slope, n);
088
089        Assert.assertEquals(init, f.value(0), 0d);
090        // Very approximate derivative.
091        Assert.assertEquals(slope, f.value(n) - f.value(n - 1), 1e-2);
092        Assert.assertEquals(0, f.value(Long.MAX_VALUE), 0d);
093    }
094}
