037    @Test
038    public void testChiSquare() {
039
040        // Target values computed using R version 1.8.1
041        // Some assembly required ;-)
042        //      Use sum((obs - exp)^2/exp) for the chi-square statistic and
043        //      1 - pchisq(sum((obs - exp)^2/exp), length(obs) - 1) for the p-value
044
045        long[] observed = {10, 9, 11};
046        double[] expected = {10, 10, 10};
047        Assert.assertEquals("chi-square statistic", 0.2,  testStatistic.chiSquare(expected, observed), 10E-12);
048        Assert.assertEquals("chi-square p-value", 0.904837418036, testStatistic.chiSquareTest(expected, observed), 1E-10);
049
050        long[] observed1 = { 500, 623, 72, 70, 31 };
051        double[] expected1 = { 485, 541, 82, 61, 37 };
052        Assert.assertEquals( "chi-square test statistic", 9.023307936427388, testStatistic.chiSquare(expected1, observed1), 1E-10);
053        Assert.assertEquals("chi-square p-value", 0.06051952647453607, testStatistic.chiSquareTest(expected1, observed1), 1E-9);
054        Assert.assertTrue("chi-square test reject", testStatistic.chiSquareTest(expected1, observed1, 0.08));
055        Assert.assertTrue("chi-square test accept", !testStatistic.chiSquareTest(expected1, observed1, 0.05));
056
057        try {
058            testStatistic.chiSquareTest(expected1, observed1, 95);
059            Assert.fail("alpha out of range, OutOfRangeException expected");
060        } catch (OutOfRangeException ex) {
061            // expected
062        }
063
064        long[] tooShortObs = { 0 };
065        double[] tooShortEx = { 1 };
066        try {
067            testStatistic.chiSquare(tooShortEx, tooShortObs);
068            Assert.fail("arguments too short, DimensionMismatchException expected");
069        } catch (DimensionMismatchException ex) {
070            // expected
071        }
072
073        // unmatched arrays
074        long[] unMatchedObs = { 0, 1, 2, 3 };
075        double[] unMatchedEx = { 1, 1, 2 };
076        try {
077            testStatistic.chiSquare(unMatchedEx, unMatchedObs);
078            Assert.fail("arrays have different lengths, DimensionMismatchException expected");
079        } catch (DimensionMismatchException ex) {
080            // expected
081        }
082
083        // 0 expected count
084        expected[0] = 0;
085        try {
086            testStatistic.chiSquareTest(expected, observed, .01);
087            Assert.fail("bad expected count, NotStrictlyPositiveException expected");
088        } catch (NotStrictlyPositiveException ex) {
089            // expected
090        }
091
092        // negative observed count
093        expected[0] = 1;
094        observed[0] = -1;
095        try {
096            testStatistic.chiSquareTest(expected, observed, .01);
097            Assert.fail("bad expected count, NotPositiveException expected");
098        } catch (NotPositiveException ex) {
099            // expected
100        }
101
102    }

104    @Test
105    public void testChiSquareIndependence() {
106
107        // Target values computed using R version 1.8.1
108
109        long[][] counts = { {40, 22, 43}, {91, 21, 28}, {60, 10, 22}};
110        Assert.assertEquals( "chi-square test statistic", 22.709027688, testStatistic.chiSquare(counts), 1E-9);
111        Assert.assertEquals("chi-square p-value", 0.000144751460134, testStatistic.chiSquareTest(counts), 1E-9);
112        Assert.assertTrue("chi-square test reject", testStatistic.chiSquareTest(counts, 0.0002));
113        Assert.assertTrue("chi-square test accept", !testStatistic.chiSquareTest(counts, 0.0001));
114
115        long[][] counts2 = {{10, 15}, {30, 40}, {60, 90} };
116        Assert.assertEquals( "chi-square test statistic", 0.168965517241, testStatistic.chiSquare(counts2), 1E-9);
117        Assert.assertEquals("chi-square p-value",0.918987499852, testStatistic.chiSquareTest(counts2), 1E-9);
118        Assert.assertTrue("chi-square test accept", !testStatistic.chiSquareTest(counts2, 0.1));
119
120        // ragged input array
121        long[][] counts3 = { {40, 22, 43}, {91, 21, 28}, {60, 10}};
122        try {
123            testStatistic.chiSquare(counts3);
124            Assert.fail("Expecting DimensionMismatchException");
125        } catch (DimensionMismatchException ex) {
126            // expected
127        }
128
129        // insufficient data
130        long[][] counts4 = {{40, 22, 43}};
131        try {
132            testStatistic.chiSquare(counts4);
133            Assert.fail("Expecting DimensionMismatchException");
134        } catch (DimensionMismatchException ex) {
135            // expected
136        }
137        long[][] counts5 = {{40}, {40}, {30}, {10}};
138        try {
139            testStatistic.chiSquare(counts5);
140            Assert.fail("Expecting DimensionMismatchException");
141        } catch (DimensionMismatchException ex) {
142            // expected
143        }
144
145        // negative counts
146        long[][] counts6 = {{10, -2}, {30, 40}, {60, 90} };
147        try {
148            testStatistic.chiSquare(counts6);
149            Assert.fail("Expecting NotPositiveException");
150        } catch (NotPositiveException ex) {
151            // expected
152        }
153
154        // bad alpha
155        try {
156            testStatistic.chiSquareTest(counts, 0);
157            Assert.fail("Expecting OutOfRangeException");
158        } catch (OutOfRangeException ex) {
159            // expected
160        }
161    }

163    @Test
164    public void testChiSquareLargeTestStatistic() {
165        double[] exp = new double[] {
166            3389119.5, 649136.6, 285745.4, 25357364.76, 11291189.78, 543628.0,
167            232921.0, 437665.75
168        };
169
170        long[] obs = new long[] {
171            2372383, 584222, 257170, 17750155, 7903832, 489265, 209628, 393899
172        };
173        org.apache.commons.math3.stat.inference.ChiSquareTest csti =
174            new org.apache.commons.math3.stat.inference.ChiSquareTest();
175        double cst = csti.chiSquareTest(exp, obs);
176        Assert.assertEquals("chi-square p-value", 0.0, cst, 1E-3);
177        Assert.assertEquals( "chi-square test statistic",
178                114875.90421929007, testStatistic.chiSquare(exp, obs), 1E-9);
179    }

192    /** Target values verified using DATAPLOT version 2006.3 */
193    @Test
194    public void testChiSquareDataSetsComparisonEqualCounts()
195        {
196        long[] observed1 = {10, 12, 12, 10};
197        long[] observed2 = {5, 15, 14, 10};
198        Assert.assertEquals("chi-square p value", 0.541096,
199                testStatistic.chiSquareTestDataSetsComparison(
200                observed1, observed2), 1E-6);
201        Assert.assertEquals("chi-square test statistic", 2.153846,
202                testStatistic.chiSquareDataSetsComparison(
203                observed1, observed2), 1E-6);
204        Assert.assertFalse("chi-square test result",
205                testStatistic.chiSquareTestDataSetsComparison(
206                observed1, observed2, 0.4));
207    }

210    @Test
211    public void testChiSquareDataSetsComparisonUnEqualCounts()
212        {
213        long[] observed1 = {10, 12, 12, 10, 15};
214        long[] observed2 = {15, 10, 10, 15, 5};
215        Assert.assertEquals("chi-square p value", 0.124115,
216                testStatistic.chiSquareTestDataSetsComparison(
217                observed1, observed2), 1E-6);
218        Assert.assertEquals("chi-square test statistic", 7.232189,
219                testStatistic.chiSquareDataSetsComparison(
220                observed1, observed2), 1E-6);
221        Assert.assertTrue("chi-square test result",
222                testStatistic.chiSquareTestDataSetsComparison(
223                observed1, observed2, 0.13));
224        Assert.assertFalse("chi-square test result",
225                testStatistic.chiSquareTestDataSetsComparison(
226                observed1, observed2, 0.12));
