075    @Test
076    public void testPerfectFit() {
077        double[] betaHat = regression.estimateRegressionParameters();
078        TestUtils.assertEquals(betaHat,
079                               new double[]{ 11.0, 1.0 / 2.0, 2.0 / 3.0, 3.0 / 4.0, 4.0 / 5.0, 5.0 / 6.0 },
080                               1e-14);
081        double[] residuals = regression.estimateResiduals();
082        TestUtils.assertEquals(residuals, new double[]{0d,0d,0d,0d,0d,0d},
083                               1e-14);
084        RealMatrix errors =
085            new Array2DRowRealMatrix(regression.estimateRegressionParametersVariance(), false);
086        final double[] s = { 1.0, -1.0 /  2.0, -1.0 /  3.0, -1.0 /  4.0, -1.0 /  5.0, -1.0 /  6.0 };
087        RealMatrix referenceVariance = new Array2DRowRealMatrix(s.length, s.length);
088        referenceVariance.walkInOptimizedOrder(new DefaultRealMatrixChangingVisitor() {
089            @Override
090            public double visit(int row, int column, double value) {
091                if (row == 0) {
092                    return s[column];
093                }
094                double x = s[row] * s[column];
095                return (row == column) ? 2 * x : x;
096            }
097        });
098       Assert.assertEquals(0.0,
099                     errors.subtract(referenceVariance).getNorm(),
100                     5.0e-16 * referenceVariance.getNorm());
101       Assert.assertEquals(1, ((OLSMultipleLinearRegression) regression).calculateRSquared(), 1E-12);
102    }

371    /**
372     * Test hat matrix computation
373     *
374     */
375    @Test
376    public void testHat() {
377
378        /*
379         * This example is from "The Hat Matrix in Regression and ANOVA",
380         * David C. Hoaglin and Roy E. Welsch,
381         * The American Statistician, Vol. 32, No. 1 (Feb., 1978), pp. 17-22.
382         *
383         */
384        double[] design = new double[] {
385                11.14, .499, 11.1,
386                12.74, .558, 8.9,
387                13.13, .604, 8.8,
388                11.51, .441, 8.9,
389                12.38, .550, 8.8,
390                12.60, .528, 9.9,
391                11.13, .418, 10.7,
392                11.7, .480, 10.5,
393                11.02, .406, 10.5,
394                11.41, .467, 10.7
395        };
396
397        int nobs = 10;
398        int nvars = 2;
399
400        // Estimate the model
401        OLSMultipleLinearRegression model = new OLSMultipleLinearRegression();
402        model.newSampleData(design, nobs, nvars);
403
404        RealMatrix hat = model.calculateHat();
405
406        // Reference data is upper half of symmetric hat matrix
407        double[] referenceData = new double[] {
408                .418, -.002,  .079, -.274, -.046,  .181,  .128,  .222,  .050,  .242,
409                       .242,  .292,  .136,  .243,  .128, -.041,  .033, -.035,  .004,
410                              .417, -.019,  .273,  .187, -.126,  .044, -.153,  .004,
411                                     .604,  .197, -.038,  .168, -.022,  .275, -.028,
412                                            .252,  .111, -.030,  .019, -.010, -.010,
413                                                   .148,  .042,  .117,  .012,  .111,
414                                                          .262,  .145,  .277,  .174,
415                                                                 .154,  .120,  .168,
416                                                                        .315,  .148,
417                                                                               .187
418        };
419
420        // Check against reference data and verify symmetry
421        int k = 0;
422        for (int i = 0; i < 10; i++) {
423            for (int j = i; j < 10; j++) {
424                Assert.assertEquals(referenceData[k], hat.getEntry(i, j), 10e-3);
425                Assert.assertEquals(hat.getEntry(i, j), hat.getEntry(j, i), 10e-12);
426                k++;
427            }
428        }
429
430        /*
431         * Verify that residuals computed using the hat matrix are close to
432         * what we get from direct computation, i.e. r = (I - H) y
433         */
434        double[] residuals = model.estimateResiduals();
435        RealMatrix I = MatrixUtils.createRealIdentityMatrix(10);
436        double[] hatResiduals = I.subtract(hat).operate(model.getY()).toArray();
437        TestUtils.assertEquals(residuals, hatResiduals, 10e-12);
438    }

471     * Verifies that setting X and Y separately has the same effect as newSample(X,Y).
472     */
473    @Test
474    public void testNewSample2() {
475        double[] y = new double[] {1, 2, 3, 4}; 
476        double[][] x = new double[][] {
477          {19, 22, 33},
478          {20, 30, 40},
479          {25, 35, 45},
480          {27, 37, 47}   
481        };
482        OLSMultipleLinearRegression regression = new OLSMultipleLinearRegression();
483        regression.newSampleData(y, x);
484        RealMatrix combinedX = regression.getX().copy();
485        RealVector combinedY = regression.getY().copy();
486        regression.newXSampleData(x);
487        regression.newYSampleData(y);
488        Assert.assertEquals(combinedX, regression.getX());
489        Assert.assertEquals(combinedY, regression.getY());
490        
491        // No intercept
492        regression.setNoIntercept(true);
493        regression.newSampleData(y, x);
494        combinedX = regression.getX().copy();
495        combinedY = regression.getY().copy();
496        regression.newXSampleData(x);
