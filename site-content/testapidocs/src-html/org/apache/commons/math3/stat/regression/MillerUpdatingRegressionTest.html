063    @Test
064    public void testAddObsGetNClear() {
065        MillerUpdatingRegression instance = new MillerUpdatingRegression(3, true);
066        double[][] xAll = new double[airdata[0].length][];
067        double[] y = new double[airdata[0].length];
068        for (int i = 0; i < airdata[0].length; i++) {
069            xAll[i] = new double[3];
070            xAll[i][0] = FastMath.log(airdata[3][i]);
071            xAll[i][1] = FastMath.log(airdata[4][i]);
072            xAll[i][2] = airdata[5][i];
073            y[i] = FastMath.log(airdata[2][i]);
074        }
075        instance.addObservations(xAll, y);
076        if (instance.getN() != xAll.length) {
077            Assert.fail("Number of observations not correct in bulk addition");
078        }
079        instance.clear();
080        for (int i = 0; i < xAll.length; i++) {
081            instance.addObservation(xAll[i], y[i]);
082        }
083        if (instance.getN() != xAll.length) {
084            Assert.fail("Number of observations not correct in drip addition");
085        }
086        return;
087    }

757    public void testOneRedundantColumn() {
758        MillerUpdatingRegression instance = new MillerUpdatingRegression(4, false);
759        MillerUpdatingRegression instance2 = new MillerUpdatingRegression(5, false);
760        double[][] x = new double[airdata[0].length][];
761        double[][] x2 = new double[airdata[0].length][];
762        double[] y = new double[airdata[0].length];
763        for (int i = 0; i < airdata[0].length; i++) {
764            x[i] = new double[4];
765            x2[i] = new double[5];
766            x[i][0] = 1.0;
767            x[i][1] = FastMath.log(airdata[3][i]);
768            x[i][2] = FastMath.log(airdata[4][i]);
769            x[i][3] = airdata[5][i];
770
771            x2[i][0] = x[i][0];
772            x2[i][1] = x[i][1];
773            x2[i][2] = x[i][2];
774            x2[i][3] = x[i][3];
775            x2[i][4] = x[i][3];
776
777            y[i] = FastMath.log(airdata[2][i]);
778        }
779
780        instance.addObservations(x, y);
781        RegressionResults result = instance.regress();
782        Assert.assertNotNull("Could not estimate initial regression", result);
783
784        instance2.addObservations(x2, y);
785        RegressionResults resultRedundant = instance2.regress();
786        Assert.assertNotNull("Could not estimate redundant regression", resultRedundant);
787        double[] beta = result.getParameterEstimates();
788        double[] betar = resultRedundant.getParameterEstimates();
789        double[] se = result.getStdErrorOfEstimates();
790        double[] ser = resultRedundant.getStdErrorOfEstimates();
791
792        for (int i = 0; i < beta.length; i++) {
793            if (FastMath.abs(beta[i] - betar[i]) > 1.0e-8) {
794                Assert.fail("Parameters not correctly estimated");
795            }
796            if (FastMath.abs(se[i] - ser[i]) > 1.0e-8) {
797                Assert.fail("Standard errors not correctly estimated");
798            }
799            for (int j = 0; j < i; j++) {
800                if (FastMath.abs(result.getCovarianceOfParameters(i, j)
801                        - resultRedundant.getCovarianceOfParameters(i, j)) > 1.0e-8) {
802                    Assert.fail("Variance Covariance not correct");
803                }
804            }
805        }
806
807
808        TestUtils.assertEquals(result.getAdjustedRSquared(), resultRedundant.getAdjustedRSquared(), 1.0e-8);
809        TestUtils.assertEquals(result.getErrorSumSquares(), resultRedundant.getErrorSumSquares(), 1.0e-8);
810        TestUtils.assertEquals(result.getMeanSquareError(), resultRedundant.getMeanSquareError(), 1.0e-8);
811        TestUtils.assertEquals(result.getRSquared(), resultRedundant.getRSquared(), 1.0e-8);
812        return;
813    }

973                }
974                ++idx;
975                ++off;
976            }
977            ++off;
978            if (FastMath.abs(pc[i+off2] - yxcorr[ i] / (FastMath.sqrt(sumysq) * diag[i])) > 1.0e-8) {
979                Assert.fail("Assert.failed cross product i = " + i + " y");
980            }
981        }
982        double[] pc2 = instance.getPartialCorrelations(1);
983
984        idx = 0;
985
986        for (int i = 1; i < 4; i++) {
987            for (int j = 1; j < i; j++) {

991                ++idx;
992            }
993        }
