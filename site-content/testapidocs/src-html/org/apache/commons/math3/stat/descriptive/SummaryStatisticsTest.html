077        SummaryStatistics u = createSummaryStatistics();
078        Assert.assertTrue("Mean of n = 0 set should be NaN",
079                Double.isNaN( u.getMean() ) );
080        Assert.assertTrue("Standard Deviation of n = 0 set should be NaN",

186    @Test
187    public void testEqualsAndHashCode() {
188        SummaryStatistics u = createSummaryStatistics();
189        SummaryStatistics t = null;
190        int emptyHash = u.hashCode();
191        Assert.assertTrue("reflexive", u.equals(u));
192        Assert.assertFalse("non-null compared to null", u.equals(t));
193        Assert.assertFalse("wrong type", u.equals(Double.valueOf(0)));
194        t = createSummaryStatistics();
195        Assert.assertTrue("empty instances should be equal", t.equals(u));
196        Assert.assertTrue("empty instances should be equal", u.equals(t));
197        Assert.assertEquals("empty hash code", emptyHash, t.hashCode());
198
199        // Add some data to u
200        u.addValue(2d);
201        u.addValue(1d);
202        u.addValue(3d);
203        u.addValue(4d);
204        Assert.assertFalse("different n's should make instances not equal", t.equals(u));
205        Assert.assertFalse("different n's should make instances not equal", u.equals(t));
206        Assert.assertTrue("different n's should make hashcodes different",
207                u.hashCode() != t.hashCode());
208
209        //Add data in same order to t
210        t.addValue(2d);
211        t.addValue(1d);
212        t.addValue(3d);
213        t.addValue(4d);
214        Assert.assertTrue("summaries based on same data should be equal", t.equals(u));
215        Assert.assertTrue("summaries based on same data should be equal", u.equals(t));
216        Assert.assertEquals("summaries based on same data should have same hashcodes",
217                u.hashCode(), t.hashCode());
218
219        // Clear and make sure summaries are indistinguishable from empty summary
220        u.clear();
221        t.clear();
222        Assert.assertTrue("empty instances should be equal", t.equals(u));
223        Assert.assertTrue("empty instances should be equal", u.equals(t));
224        Assert.assertEquals("empty hash code", emptyHash, t.hashCode());
225        Assert.assertEquals("empty hash code", emptyHash, u.hashCode());
226    }

230        SummaryStatistics u = createSummaryStatistics();
231        u.addValue(2d);
232        u.addValue(1d);
233        u.addValue(3d);
234        u.addValue(4d);
235        SummaryStatistics v = new SummaryStatistics(u);
236        Assert.assertEquals(u, v);
237        Assert.assertEquals(v, u);
238
239        // Make sure both behave the same with additional values added
240        u.addValue(7d);
241        u.addValue(9d);
242        u.addValue(11d);
243        u.addValue(23d);
244        v.addValue(7d);
245        v.addValue(9d);
246        v.addValue(11d);
247        v.addValue(23d);
248        Assert.assertEquals(u, v);
249        Assert.assertEquals(v, u);
250
251        // Check implementation pointers are preserved
252        u.clear();
253        u.setSumImpl(new Sum());
254        SummaryStatistics.copy(u,v);
255        Assert.assertEquals(u.getSumImpl(), v.getSumImpl());

300    @Test
301    public void testQuadraticMean() {
302        final double[] values = { 1.2, 3.4, 5.6, 7.89 };
303        final SummaryStatistics stats = createSummaryStatistics();
304
305        final int len = values.length;
306        double expected = 0;
307        for (int i = 0; i < len; i++) {
308            final double v = values[i];
309            expected += v * v / len;
310
311            stats.addValue(v);
312        }
313        expected = Math.sqrt(expected);
314
315        Assert.assertEquals(expected, stats.getQuadraticMean(), Math.ulp(expected));
316    }

354    @Test
355    public void testToString() {
356        SummaryStatistics u = createSummaryStatistics();
357        for (int i = 0; i < 5; i++) {
358            u.addValue(i);
359        }
360        final String[] labels = {"min", "max", "sum", "geometric mean", "variance",
361                "population variance", "second moment", "sum of squares", "standard deviation",
362        "sum of logs"};
363        final double[] values = {u.getMin(), u.getMax(), u.getSum(), u.getGeometricMean(),
364                u.getVariance(), u.getPopulationVariance(), u.getSecondMoment(), u.getSumsq(),
365                u.getStandardDeviation(), u.getSumOfLogs()};
366        final String toString = u.toString();
367        Assert.assertTrue(toString.indexOf("n: " + u.getN()) > 0); // getN() returns a long
368        for (int i = 0; i < values.length; i++) {
369            Assert.assertTrue(toString.indexOf(labels[i] + ": " + String.valueOf(values[i])) > 0);
370        }
