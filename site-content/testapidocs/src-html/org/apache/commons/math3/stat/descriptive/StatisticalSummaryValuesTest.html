017package org.apache.commons.math3.stat.descriptive;
018
019
020import java.util.Locale;
021
022
023import org.apache.commons.math3.TestUtils;
024import org.junit.Assert;
025import org.junit.Test;
026/**
027 * Test cases for the {@link StatisticalSummaryValues} class.
028 *
029 */
030
031public final class StatisticalSummaryValuesTest {
032
033    @Test
034    public void testSerialization() {
035        StatisticalSummaryValues u = new StatisticalSummaryValues(1, 2, 3, 4, 5, 6);
036        TestUtils.checkSerializedEquality(u);
037        StatisticalSummaryValues t = (StatisticalSummaryValues) TestUtils.serializeAndRecover(u);
038        verifyEquality(u, t);
039    }
040
041    @Test
042    public void testEqualsAndHashCode() {
043        StatisticalSummaryValues u  = new StatisticalSummaryValues(1, 2, 3, 4, 5, 6);
044        StatisticalSummaryValues t = null;
045        Assert.assertTrue("reflexive", u.equals(u));
046        Assert.assertFalse("non-null compared to null", u.equals(t));
047        Assert.assertFalse("wrong type", u.equals(Double.valueOf(0)));
048        t = new StatisticalSummaryValues(1, 2, 3, 4, 5, 6);
049        Assert.assertTrue("instances with same data should be equal", t.equals(u));
050        Assert.assertEquals("hash code", u.hashCode(), t.hashCode());
051
052        u = new StatisticalSummaryValues(Double.NaN, 2, 3, 4, 5, 6);
053        t = new StatisticalSummaryValues(1, Double.NaN, 3, 4, 5, 6);
054        Assert.assertFalse("instances based on different data should be different",
055                (u.equals(t) ||t.equals(u)));
056    }
057
058    private void verifyEquality(StatisticalSummaryValues s, StatisticalSummaryValues u) {
059        Assert.assertEquals("N",s.getN(),u.getN());
060        TestUtils.assertEquals("sum",s.getSum(),u.getSum(), 0);
061        TestUtils.assertEquals("var",s.getVariance(),u.getVariance(), 0);
062        TestUtils.assertEquals("std",s.getStandardDeviation(),u.getStandardDeviation(), 0);
063        TestUtils.assertEquals("mean",s.getMean(),u.getMean(), 0);
064        TestUtils.assertEquals("min",s.getMin(),u.getMin(), 0);
065        TestUtils.assertEquals("max",s.getMax(),u.getMax(), 0);
066    }
067    
068    @Test
069    public void testToString() {
070        StatisticalSummaryValues u  = new StatisticalSummaryValues(4.5, 16, 10, 5, 4, 45);
071        Locale d = Locale.getDefault();
072        Locale.setDefault(Locale.US);
073        Assert.assertEquals("StatisticalSummaryValues:\n" +
074                     "n: 10\n" +
075                     "min: 4.0\n" +
076                     "max: 5.0\n" +
077                     "mean: 4.5\n" +
078                     "std dev: 4.0\n" +
079                     "variance: 16.0\n" +
080                     "sum: 45.0\n",  u.toString());
081        Locale.setDefault(d);
082    }
083}
