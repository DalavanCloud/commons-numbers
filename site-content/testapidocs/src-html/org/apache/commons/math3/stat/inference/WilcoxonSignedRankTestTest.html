032public class WilcoxonSignedRankTestTest {

034    protected WilcoxonSignedRankTest testStatistic = new WilcoxonSignedRankTest();
035
036    @Test
037    public void testWilcoxonSignedRankSimple() {
038        /* Target values computed using R version 2.11.1
039         * x <- c(1.83, 0.50, 1.62, 2.48, 1.68, 1.88, 1.55, 3.06, 1.30)
040         * y <- c(0.878, 0.647, 0.598, 2.05, 1.06, 1.29, 1.06, 3.14, 1.29)
041         */
042        final double x[] = {1.83, 0.50, 1.62, 2.48, 1.68, 1.88, 1.55, 3.06, 1.30};
043        final double y[] = {0.878, 0.647, 0.598, 2.05, 1.06, 1.29, 1.06, 3.14, 1.29};
044        
045        /* EXACT:
046         * wilcox.test(x, y, alternative = "two.sided", mu = 0, paired = TRUE, exact = TRUE, correct = FALSE)
047         * V = 40, p-value = 0.03906
048         * 
049         * Corresponds to the value obtained in R.
050         */
051        Assert.assertEquals(40, testStatistic.wilcoxonSignedRank(x, y), 1e-10);
052        Assert.assertEquals(0.03906, testStatistic.wilcoxonSignedRankTest(x, y, true), 1e-5);        
053        
054        /* ASYMPTOTIC:
055         * wilcox.test(x, y, alternative = "two.sided", mu = 0, paired = TRUE, exact = FALSE, correct = FALSE)
056         * V = 40, p-value = 0.03815
057         * 
058         * This is not entirely the same due to different corrects, 
059         * e.g. http://mlsc.lboro.ac.uk/resources/statistics/wsrt.pdf
060         * and src/library/stats/R/wilcox.test.R in the R source
061         */
062        Assert.assertEquals(40, testStatistic.wilcoxonSignedRank(x, y), 1e-10);
063        Assert.assertEquals(0.0329693812, testStatistic.wilcoxonSignedRankTest(x, y, false), 1e-10);
064    }
065    
066    @Test
067    public void testWilcoxonSignedRankInputValidation() {
068        /*
069         * Exact only for sample size <= 30
070         */
071        final double[] x1 = new double[30];
072        final double[] x2 = new double[31];
073        final double[] y1 = new double[30];
074        final double[] y2 = new double[31];
075        for (int i = 0; i < 30; ++i) {
076            x1[i] = x2[i] = y1[i] = y2[i] = i;            
077        }
078        
079        // Exactly 30 is okay
080        //testStatistic.wilcoxonSignedRankTest(x1, y1, true);            
081        
082        try {
083            testStatistic.wilcoxonSignedRankTest(x2, y2, true);
084            Assert.fail("More than 30 samples and exact chosen, NumberIsTooLargeException expected");
085        } catch (NumberIsTooLargeException ex) {
086            // expected
087        }
088        
089        /* Samples must be present, i.e. length > 0
090         */
091        try {
092            testStatistic.wilcoxonSignedRankTest(new double[] { }, new double[] { 1.0 }, true);
093            Assert.fail("x does not contain samples (exact), NoDataException expected");
094        } catch (NoDataException ex) {
095            // expected
096        }
097
098        try {
099            testStatistic.wilcoxonSignedRankTest(new double[] { }, new double[] { 1.0 }, false);
100            Assert.fail("x does not contain samples (asymptotic), NoDataException expected");
101        } catch (NoDataException ex) {
102            // expected
103        }
104
105        try {
106            testStatistic.wilcoxonSignedRankTest(new double[] { 1.0 }, new double[] { }, true);
107            Assert.fail("y does not contain samples (exact), NoDataException expected");
108        } catch (NoDataException ex) {
109            // expected
110        }
111
112        try {
113            testStatistic.wilcoxonSignedRankTest(new double[] { 1.0 }, new double[] { }, false);
114            Assert.fail("y does not contain samples (asymptotic), NoDataException expected");
115        } catch (NoDataException ex) {
116            // expected
117        }
118
119        /* Samples not same size, i.e. cannot be pairred
120         */
121        try {
122            testStatistic.wilcoxonSignedRankTest(new double[] { 1.0, 2.0 }, new double[] { 3.0 }, true);
123            Assert.fail("x and y not same size (exact), DimensionMismatchException expected");
124        } catch (DimensionMismatchException ex) {
125            // expected
126        }
127
128        try {
129            testStatistic.wilcoxonSignedRankTest(new double[] { 1.0, 2.0 }, new double[] { 3.0 }, false);
130            Assert.fail("x and y not same size (asymptotic), DimensionMismatchException expected");
131        } catch (DimensionMismatchException ex) {
132            // expected
133        }
134        
135        /*
136         * x and y is null
137         */
138        try {
139            testStatistic.wilcoxonSignedRankTest(null, null, true);
140            Assert.fail("x and y is null (exact), NullArgumentException expected");
141        } catch (NullArgumentException ex) {
142            // expected
143        }
144        
145        try {
146            testStatistic.wilcoxonSignedRankTest(null, null, false);
147            Assert.fail("x and y is null (asymptotic), NullArgumentException expected");
148        } catch (NullArgumentException ex) {
149            // expected
150        }
151        
152        /*
153         * x or y is null
154         */
155        try {
156            testStatistic.wilcoxonSignedRankTest(null, new double[] { 1.0 }, true);
157            Assert.fail("x is null (exact), NullArgumentException expected");
158        } catch (NullArgumentException ex) {
159            // expected
160        }
161        
162        try {
163            testStatistic.wilcoxonSignedRankTest(null, new double[] { 1.0 }, false);
164            Assert.fail("x is null (asymptotic), NullArgumentException expected");
165        } catch (NullArgumentException ex) {
166            // expected
167        }
168        
169        try {
170            testStatistic.wilcoxonSignedRankTest(new double[] { 1.0 }, null, true);
171            Assert.fail("y is null (exact), NullArgumentException expected");
172        } catch (NullArgumentException ex) {
173            // expected
174        }
175        
176        try {
177            testStatistic.wilcoxonSignedRankTest(new double[] { 1.0 }, null, false);
178            Assert.fail("y is null (asymptotic), NullArgumentException expected");
179        } catch (NullArgumentException ex) {
180            // expected
181        }
182    }
