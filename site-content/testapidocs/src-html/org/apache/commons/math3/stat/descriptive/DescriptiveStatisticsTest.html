065    @Test
066    public void testWindowSize() {
067        DescriptiveStatistics stats = createDescriptiveStatistics();
068        stats.setWindowSize(300);
069        for (int i = 0; i < 100; ++i) {
070            stats.addValue(i + 1);
071        }
072        int refSum = (100 * 101) / 2;
073        Assert.assertEquals(refSum / 100.0, stats.getMean(), 1E-10);
074        Assert.assertEquals(300, stats.getWindowSize());
075        try {
076            stats.setWindowSize(-3);
077            Assert.fail("an exception should have been thrown");
078        } catch (IllegalArgumentException iae) {
079            // expected
080        }
081        Assert.assertEquals(300, stats.getWindowSize());
082        stats.setWindowSize(50);
083        Assert.assertEquals(50, stats.getWindowSize());
084        int refSum2 = refSum - (50 * 51) / 2;
085        Assert.assertEquals(refSum2 / 50.0, stats.getMean(), 1E-10);
086    }

107    @Test
108    public void testQuadraticMean() {
109        final double[] values = { 1.2, 3.4, 5.6, 7.89 };
110        final DescriptiveStatistics stats = new DescriptiveStatistics(values);
111
112        final int len = values.length;
113        double expected = 0;
114        for (int i = 0; i < len; i++) {
115            final double v = values[i];
116            expected += v * v / len;
117        }
118        expected = Math.sqrt(expected);
119
120        Assert.assertEquals(expected, stats.getQuadraticMean(), Math.ulp(expected));
121    }

143    @Test
144    public void testShuffledStatistics() {
145        // the purpose of this test is only to check the get/set methods
146        // we are aware shuffling statistics like this is really not
147        // something sensible to do in production ...
148        DescriptiveStatistics reference = createDescriptiveStatistics();
149        DescriptiveStatistics shuffled  = createDescriptiveStatistics();
150
151        UnivariateStatistic tmp = shuffled.getGeometricMeanImpl();
152        shuffled.setGeometricMeanImpl(shuffled.getMeanImpl());
153        shuffled.setMeanImpl(shuffled.getKurtosisImpl());
154        shuffled.setKurtosisImpl(shuffled.getSkewnessImpl());
155        shuffled.setSkewnessImpl(shuffled.getVarianceImpl());
156        shuffled.setVarianceImpl(shuffled.getMaxImpl());
157        shuffled.setMaxImpl(shuffled.getMinImpl());
158        shuffled.setMinImpl(shuffled.getSumImpl());
159        shuffled.setSumImpl(shuffled.getSumsqImpl());
160        shuffled.setSumsqImpl(tmp);
161
162        for (int i = 100; i > 0; --i) {
163            reference.addValue(i);
164            shuffled.addValue(i);
165        }
166
167        Assert.assertEquals(reference.getMean(),          shuffled.getGeometricMean(), 1.0e-10);
168        Assert.assertEquals(reference.getKurtosis(),      shuffled.getMean(),          1.0e-10);
169        Assert.assertEquals(reference.getSkewness(),      shuffled.getKurtosis(), 1.0e-10);
170        Assert.assertEquals(reference.getVariance(),      shuffled.getSkewness(), 1.0e-10);
171        Assert.assertEquals(reference.getMax(),           shuffled.getVariance(), 1.0e-10);
172        Assert.assertEquals(reference.getMin(),           shuffled.getMax(), 1.0e-10);
173        Assert.assertEquals(reference.getSum(),           shuffled.getMin(), 1.0e-10);
174        Assert.assertEquals(reference.getSumsq(),         shuffled.getSum(), 1.0e-10);
175        Assert.assertEquals(reference.getGeometricMean(), shuffled.getSumsq(), 1.0e-10);
176

228    @Test
229    public void testSummaryConsistency() {
230        final DescriptiveStatistics dstats = new DescriptiveStatistics();
231        final SummaryStatistics sstats = new SummaryStatistics();
232        final int windowSize = 5;
233        dstats.setWindowSize(windowSize);
234        final double tol = 1E-12;
235        for (int i = 0; i < 20; i++) {
236            dstats.addValue(i);
237            sstats.clear();
238            double[] values = dstats.getValues();
239            for (int j = 0; j < values.length; j++) {
240                sstats.addValue(values[j]);
241            }
242            TestUtils.assertEquals(dstats.getMean(), sstats.getMean(), tol);
243            TestUtils.assertEquals(new Mean().evaluate(values), dstats.getMean(), tol);
244            TestUtils.assertEquals(dstats.getMax(), sstats.getMax(), tol);
245            TestUtils.assertEquals(new Max().evaluate(values), dstats.getMax(), tol);
246            TestUtils.assertEquals(dstats.getGeometricMean(), sstats.getGeometricMean(), tol);
247            TestUtils.assertEquals(new GeometricMean().evaluate(values), dstats.getGeometricMean(), tol);
248            TestUtils.assertEquals(dstats.getMin(), sstats.getMin(), tol);
249            TestUtils.assertEquals(new Min().evaluate(values), dstats.getMin(), tol);
250            TestUtils.assertEquals(dstats.getStandardDeviation(), sstats.getStandardDeviation(), tol);
251            TestUtils.assertEquals(dstats.getVariance(), sstats.getVariance(), tol);
252            TestUtils.assertEquals(new Variance().evaluate(values), dstats.getVariance(), tol);
253            TestUtils.assertEquals(dstats.getSum(), sstats.getSum(), tol);
254            TestUtils.assertEquals(new Sum().evaluate(values), dstats.getSum(), tol);
255            TestUtils.assertEquals(dstats.getSumsq(), sstats.getSumsq(), tol);
256            TestUtils.assertEquals(new SumOfSquares().evaluate(values), dstats.getSumsq(), tol);
257            TestUtils.assertEquals(dstats.getPopulationVariance(), sstats.getPopulationVariance(), tol);
258            TestUtils.assertEquals(new Variance(false).evaluate(values), dstats.getPopulationVariance(), tol);
259        }

292    public void checkremoval(DescriptiveStatistics dstat, int wsize,
293                             double mean1, double mean2, double mean3) {
294
295        dstat.setWindowSize(wsize);
296        dstat.clear();
297
298        for (int i = 1 ; i <= 6 ; ++i) {
299            dstat.addValue(i);
300        }
301
302        Assert.assertTrue(Precision.equalsIncludingNaN(mean1, dstat.getMean()));
303        dstat.replaceMostRecentValue(0);
304        Assert.assertTrue(Precision.equalsIncludingNaN(mean2, dstat.getMean()));
305        dstat.removeMostRecentValue();
306        Assert.assertTrue(Precision.equalsIncludingNaN(mean3, dstat.getMean()));
307
308    }
