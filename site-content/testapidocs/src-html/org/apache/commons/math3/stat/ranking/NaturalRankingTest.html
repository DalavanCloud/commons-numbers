035    private final double[] exampleData = { 20, 17, 30, 42.3, 17, 50,
036            Double.NaN, Double.NEGATIVE_INFINITY, 17 };
037    private final double[] tiesFirst = { 0, 0, 2, 1, 4 };
038    private final double[] tiesLast = { 4, 4, 1, 0 };
039    private final double[] multipleNaNs = { 0, 1, Double.NaN, Double.NaN };
040    private final double[] multipleTies = { 3, 2, 5, 5, 6, 6, 1 };
041    private final double[] allSame = { 0, 0, 0, 0 };
042
043    @Test
044    public void testDefault() { // Ties averaged, NaNs failed
045        NaturalRanking ranking = new NaturalRanking();
046        double[] ranks;
047        
048        try {
049            ranks = ranking.rank(exampleData);
050            Assert.fail("expected NotANumberException due to NaNStrategy.FAILED");
051        } catch (NotANumberException e) {
052            // expected
053        }
054        
055        ranks = ranking.rank(tiesFirst);
056        double[] correctRanks = new double[] { 1.5, 1.5, 4, 3, 5 };
057        TestUtils.assertEquals(correctRanks, ranks, 0d);
058        ranks = ranking.rank(tiesLast);
059        correctRanks = new double[] { 3.5, 3.5, 2, 1 };
060        TestUtils.assertEquals(correctRanks, ranks, 0d);
061        
062        try {
063            ranks = ranking.rank(multipleNaNs);
064            Assert.fail("expected NotANumberException due to NaNStrategy.FAILED");
065        } catch (NotANumberException e) {
066            // expected
067        }
068        
069        ranks = ranking.rank(multipleTies);
070        correctRanks = new double[] { 3, 2, 4.5, 4.5, 6.5, 6.5, 1 };
071        TestUtils.assertEquals(correctRanks, ranks, 0d);
072        ranks = ranking.rank(allSame);
073        correctRanks = new double[] { 2.5, 2.5, 2.5, 2.5 };
074        TestUtils.assertEquals(correctRanks, ranks, 0d);
075    }

077    @Test
078    public void testNaNsMaximalTiesMinimum() {
079        NaturalRanking ranking = new NaturalRanking(NaNStrategy.MAXIMAL, TiesStrategy.MINIMUM);
080        double[] ranks = ranking.rank(exampleData);
081        double[] correctRanks = { 5, 2, 6, 7, 2, 8, 9, 1, 2 };
082        TestUtils.assertEquals(correctRanks, ranks, 0d);

085        TestUtils.assertEquals(correctRanks, ranks, 0d);
086        ranks = ranking.rank(tiesLast);
087        correctRanks = new double[] { 3, 3, 2, 1 };
088        TestUtils.assertEquals(correctRanks, ranks, 0d);
089        ranks = ranking.rank(multipleNaNs);
090        correctRanks = new double[] { 1, 2, 3, 3 };
091        TestUtils.assertEquals(correctRanks, ranks, 0d);
092        ranks = ranking.rank(multipleTies);
093        correctRanks = new double[] { 3, 2, 4, 4, 6, 6, 1 };
094        TestUtils.assertEquals(correctRanks, ranks, 0d);

098    }

197    @Test
198    public void testNaNsAndInfs() {
199        double[] data = { 0, Double.POSITIVE_INFINITY, Double.NaN,
200                Double.NEGATIVE_INFINITY };
201        NaturalRanking ranking = new NaturalRanking(NaNStrategy.MAXIMAL);
202        double[] ranks = ranking.rank(data);
203        double[] correctRanks = new double[] { 2, 3.5, 3.5, 1 };
204        TestUtils.assertEquals(correctRanks, ranks, 0d);
205        ranking = new NaturalRanking(NaNStrategy.MINIMAL);
206        ranks = ranking.rank(data);
207        correctRanks = new double[] { 3, 4, 1.5, 1.5 };
208        TestUtils.assertEquals(correctRanks, ranks, 0d);
209    }
