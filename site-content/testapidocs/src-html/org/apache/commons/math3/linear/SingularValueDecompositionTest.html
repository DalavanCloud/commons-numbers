093    @Test
094    public void testHadamard() {
095        RealMatrix matrix = new Array2DRowRealMatrix(new double[][] {
096                {15.0 / 2.0,  5.0 / 2.0,  9.0 / 2.0,  3.0 / 2.0 },
097                { 5.0 / 2.0, 15.0 / 2.0,  3.0 / 2.0,  9.0 / 2.0 },
098                { 9.0 / 2.0,  3.0 / 2.0, 15.0 / 2.0,  5.0 / 2.0 },
099                { 3.0 / 2.0,  9.0 / 2.0,  5.0 / 2.0, 15.0 / 2.0 }
100        }, false);
101        SingularValueDecomposition svd = new SingularValueDecomposition(matrix);
102        Assert.assertEquals(16.0, svd.getSingularValues()[0], 1.0e-14);
103        Assert.assertEquals( 8.0, svd.getSingularValues()[1], 1.0e-14);
104        Assert.assertEquals( 4.0, svd.getSingularValues()[2], 1.0e-14);
105        Assert.assertEquals( 2.0, svd.getSingularValues()[3], 1.0e-14);
106
107        RealMatrix fullCovariance = new Array2DRowRealMatrix(new double[][] {
108                {  85.0 / 1024, -51.0 / 1024, -75.0 / 1024,  45.0 / 1024 },
109                { -51.0 / 1024,  85.0 / 1024,  45.0 / 1024, -75.0 / 1024 },
110                { -75.0 / 1024,  45.0 / 1024,  85.0 / 1024, -51.0 / 1024 },
111                {  45.0 / 1024, -75.0 / 1024, -51.0 / 1024,  85.0 / 1024 }
112        }, false);
113        Assert.assertEquals(0.0,
114                     fullCovariance.subtract(svd.getCovariance(0.0)).getNorm(),
115                     1.0e-14);
116
117        RealMatrix halfCovariance = new Array2DRowRealMatrix(new double[][] {
118                {   5.0 / 1024,  -3.0 / 1024,   5.0 / 1024,  -3.0 / 1024 },
119                {  -3.0 / 1024,   5.0 / 1024,  -3.0 / 1024,   5.0 / 1024 },
120                {   5.0 / 1024,  -3.0 / 1024,   5.0 / 1024,  -3.0 / 1024 },
121                {  -3.0 / 1024,   5.0 / 1024,  -3.0 / 1024,   5.0 / 1024 }
122        }, false);
123        Assert.assertEquals(0.0,
124                     halfCovariance.subtract(svd.getCovariance(6.0)).getNorm(),
125                     1.0e-14);

137    public void checkAEqualUSVt(final RealMatrix matrix) {
138        SingularValueDecomposition svd = new SingularValueDecomposition(matrix);
139        RealMatrix u = svd.getU();
140        RealMatrix s = svd.getS();
141        RealMatrix v = svd.getV();
142        double norm = u.multiply(s).multiply(v.transpose()).subtract(matrix).getNorm();
143        Assert.assertEquals(0, norm, normTolerance);
144
145    }

163    public void checkOrthogonal(final RealMatrix m) {
164        RealMatrix mTm = m.transpose().multiply(m);
165        RealMatrix id  = MatrixUtils.createRealIdentityMatrix(mTm.getRowDimension());
166        Assert.assertEquals(0, mTm.subtract(id).getNorm(), normTolerance);

275    private void loadRealMatrix(RealMatrix m, String resourceName) {
276        try {
277            DataInputStream in = new DataInputStream(getClass().getResourceAsStream(resourceName));
278            BufferedReader br = new BufferedReader(new InputStreamReader(in));
279            String strLine;
280            int row = 0;
281            while ((strLine = br.readLine()) != null) {
282                if (!strLine.startsWith("#")) {
283                    int col = 0;
284                    for (String entry : strLine.split(",")) {
285                        m.setEntry(row, col++, Double.parseDouble(entry));
286                    }
287                    row++;
288                }
289            }
290            in.close();
291        } catch (IOException e) {}
292    }

311                                        final double[] singularValues) {

313        final RealMatrix d = new Array2DRowRealMatrix(rows, columns);
