018package org.apache.commons.math3.linear;
019
020import org.apache.commons.math3.exception.MathIllegalArgumentException;
021
022import org.junit.Assert;
023import org.junit.Test;
024
025public class SingularValueSolverTest {
026
027    private double[][] testSquare = {
028            { 24.0 / 25.0, 43.0 / 25.0 },
029            { 57.0 / 25.0, 24.0 / 25.0 }
030    };
031    private double[][] bigSingular = {
032        { 1.0, 2.0,   3.0,    4.0 },
033        { 2.0, 5.0,   3.0,    4.0 },
034        { 7.0, 3.0, 256.0, 1930.0 },
035        { 3.0, 7.0,   6.0,    8.0 }
036    }; // 4th row = 1st + 2nd
037
038    private static final double normTolerance = 10e-14;
039
040    /** test solve dimension errors */
041    @Test
042    public void testSolveDimensionErrors() {
043        DecompositionSolver solver =
044            new SingularValueDecomposition(MatrixUtils.createRealMatrix(testSquare)).getSolver();
045        RealMatrix b = MatrixUtils.createRealMatrix(new double[3][2]);
046        try {
047            solver.solve(b);
048            Assert.fail("an exception should have been thrown");
049        } catch (MathIllegalArgumentException iae) {
050            // expected behavior
051        }
052        try {
053            solver.solve(b.getColumnVector(0));
054            Assert.fail("an exception should have been thrown");
055        } catch (MathIllegalArgumentException iae) {
056            // expected behavior
057        }
058        try {
059            solver.solve(new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(0)));
060            Assert.fail("an exception should have been thrown");
061        } catch (MathIllegalArgumentException iae) {
062            // expected behavior
063        }
064    }
065
066    /** test least square solve */
067    @Test
068    public void testLeastSquareSolve() {
069        RealMatrix m =
070            MatrixUtils.createRealMatrix(new double[][] {
071                                   { 1.0, 0.0 },
072                                   { 0.0, 0.0 }
073                               });
074        DecompositionSolver solver = new SingularValueDecomposition(m).getSolver();
075        RealMatrix b = MatrixUtils.createRealMatrix(new double[][] {
076            { 11, 12 }, { 21, 22 }
077        });
078        RealMatrix xMatrix = solver.solve(b);
079        Assert.assertEquals(11, xMatrix.getEntry(0, 0), 1.0e-15);
080        Assert.assertEquals(12, xMatrix.getEntry(0, 1), 1.0e-15);
081        Assert.assertEquals(0, xMatrix.getEntry(1, 0), 1.0e-15);
082        Assert.assertEquals(0, xMatrix.getEntry(1, 1), 1.0e-15);
083        RealVector xColVec = solver.solve(b.getColumnVector(0));
084        Assert.assertEquals(11, xColVec.getEntry(0), 1.0e-15);
085        Assert.assertEquals(0, xColVec.getEntry(1), 1.0e-15);
086        RealVector xColOtherVec = solver.solve(new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(0)));
087        Assert.assertEquals(11, xColOtherVec.getEntry(0), 1.0e-15);
088        Assert.assertEquals(0, xColOtherVec.getEntry(1), 1.0e-15);
089    }
090
091    /** test solve */
092    @Test
093    public void testSolve() {
094        DecompositionSolver solver =
095            new SingularValueDecomposition(MatrixUtils.createRealMatrix(testSquare)).getSolver();
096        RealMatrix b = MatrixUtils.createRealMatrix(new double[][] {
097                { 1, 2, 3 }, { 0, -5, 1 }
098        });
099        RealMatrix xRef = MatrixUtils.createRealMatrix(new double[][] {
100                { -8.0 / 25.0, -263.0 / 75.0, -29.0 / 75.0 },
101                { 19.0 / 25.0,   78.0 / 25.0,  49.0 / 25.0 }
102        });
103
104        // using RealMatrix
105        Assert.assertEquals(0, solver.solve(b).subtract(xRef).getNorm(), normTolerance);
106
107        // using ArrayRealVector
108        for (int i = 0; i < b.getColumnDimension(); ++i) {
109            Assert.assertEquals(0,
110                         solver.solve(b.getColumnVector(i)).subtract(xRef.getColumnVector(i)).getNorm(),
111                         1.0e-13);
112        }
113
114        // using RealVector with an alternate implementation
115        for (int i = 0; i < b.getColumnDimension(); ++i) {
116            ArrayRealVectorTest.RealVectorTestImpl v =
117                new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(i));
118            Assert.assertEquals(0,
119                         solver.solve(v).subtract(xRef.getColumnVector(i)).getNorm(),
120                         1.0e-13);
121        }
122
123    }

125    /** test condition number */
126    @Test
127    public void testConditionNumber() {
128        SingularValueDecomposition svd =
129            new SingularValueDecomposition(MatrixUtils.createRealMatrix(testSquare));
130        // replace 1.0e-15 with 1.5e-15
131        Assert.assertEquals(3.0, svd.getConditionNumber(), 1.5e-15);
132    }

134    @Test
135    public void testMath320B() {
136        RealMatrix rm = new Array2DRowRealMatrix(new double[][] {
137            { 1.0, 2.0 }, { 1.0, 2.0 }
138        });
139        SingularValueDecomposition svd =
140            new SingularValueDecomposition(rm);
141        RealMatrix recomposed = svd.getU().multiply(svd.getS()).multiply(svd.getVT());
142        Assert.assertEquals(0.0, recomposed.subtract(rm).getNorm(), 2.0e-15);
143    }
144
145    @Test
146    public void testSingular() {
147      SingularValueDecomposition svd =
148          new SingularValueDecomposition(MatrixUtils.createRealMatrix(bigSingular));
149      RealMatrix pseudoInverse = svd.getSolver().getInverse();
150      RealMatrix expected = new Array2DRowRealMatrix(new double[][] {
151          {-0.0355022687,0.0512742236,-0.0001045523,0.0157719549},
152          {-0.3214992438,0.3162419255,0.0000348508,-0.0052573183},
153          {0.5437098346,-0.4107754586,-0.0008256918,0.132934376},
154          {-0.0714905202,0.053808742,0.0006279816,-0.0176817782}
155      });
156      Assert.assertEquals(0, expected.subtract(pseudoInverse).getNorm(), 1.0e-9);
157    }
