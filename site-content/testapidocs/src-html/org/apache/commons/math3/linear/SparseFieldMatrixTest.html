376    @Test
377    public void testExamples() {
378        // Create a real matrix with two rows and three columns
379        Fraction[][] matrixData = { { new Fraction(1), new Fraction(2), new Fraction(3) }, { new Fraction(2), new Fraction(5), new Fraction(3) } };
380        FieldMatrix<Fraction> m = createSparseMatrix(matrixData);
381        // One more with three rows, two columns
382        Fraction[][] matrixData2 = { { new Fraction(1), new Fraction(2) }, { new Fraction(2), new Fraction(5) }, { new Fraction(1), new Fraction(7) } };
383        FieldMatrix<Fraction> n = createSparseMatrix(matrixData2);
384        // Now multiply m by n
385        FieldMatrix<Fraction> p = m.multiply(n);
386        Assert.assertEquals(2, p.getRowDimension());
387        Assert.assertEquals(2, p.getColumnDimension());
388        // Invert p
389        FieldMatrix<Fraction> pInverse = new FieldLUDecomposition<Fraction>(p).getSolver().getInverse();
390        Assert.assertEquals(2, pInverse.getRowDimension());
391        Assert.assertEquals(2, pInverse.getColumnDimension());
392
393        // Solve example
394        Fraction[][] coefficientsData = { { new Fraction(2), new Fraction(3), new Fraction(-2) }, { new Fraction(-1), new Fraction(7), new Fraction(6) },
395                { new Fraction(4), new Fraction(-3), new Fraction(-5) } };
396        FieldMatrix<Fraction> coefficients = createSparseMatrix(coefficientsData);
397        Fraction[] constants = { new Fraction(1), new Fraction(-2), new Fraction(1) };
398        Fraction[] solution;
399        solution = new FieldLUDecomposition<Fraction>(coefficients)
400            .getSolver()
401            .solve(new ArrayFieldVector<Fraction>(constants, false)).toArray();
402        Assert.assertEquals((new Fraction(2).multiply((solution[0])).add(new Fraction(3).multiply(solution[1])).subtract(new Fraction(2).multiply(solution[2]))).doubleValue(),
403                constants[0].doubleValue(), 1E-12);
404        Assert.assertEquals(((new Fraction(-1).multiply(solution[0])).add(new Fraction(7).multiply(solution[1])).add(new Fraction(6).multiply(solution[2]))).doubleValue(),
405                constants[1].doubleValue(), 1E-12);
406        Assert.assertEquals(((new Fraction(4).multiply(solution[0])).subtract(new Fraction(3).multiply( solution[1])).subtract(new Fraction(5).multiply(solution[2]))).doubleValue(),
407                constants[2].doubleValue(), 1E-12);

675    /** verifies that two matrices are close (1-norm) */
676    protected void assertClose(String msg, FieldMatrix<Fraction> m, FieldMatrix<Fraction> n,
677            double tolerance) {
678        for(int i=0; i < m.getRowDimension(); i++){
679            for(int j=0; j < m.getColumnDimension(); j++){
680                Assert.assertEquals(msg, m.getEntry(i,j).doubleValue(), n.getEntry(i,j).doubleValue(), tolerance);
681            }
682
683        }
684    }
685
686    /** verifies that two vectors are close (sup norm) */
687    protected void assertClose(String msg, Fraction[] m, Fraction[] n,
688            double tolerance) {
689        if (m.length != n.length) {
690            Assert.fail("vectors not same length");
691        }
692        for (int i = 0; i < m.length; i++) {
693            Assert.assertEquals(msg + " " + i + " elements differ", m[i].doubleValue(), n[i].doubleValue(),
694                    tolerance);
695        }
696    }

698    private SparseFieldMatrix<Fraction> createSparseMatrix(Fraction[][] data) {
699        SparseFieldMatrix<Fraction> matrix = new SparseFieldMatrix<Fraction>(field, data.length, data[0].length);
700        for (int row = 0; row < data.length; row++) {
701            for (int col = 0; col < data[row].length; col++) {
702                matrix.setEntry(row, col, data[row][col]);
703            }
704        }
705        return matrix;
706    }
