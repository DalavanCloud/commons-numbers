018package org.apache.commons.math3.linear;

020import org.apache.commons.math3.linear.BiDiagonalTransformer;
021import org.apache.commons.math3.linear.MatrixUtils;
022import org.apache.commons.math3.linear.RealMatrix;
023import org.apache.commons.math3.util.FastMath;
024import org.junit.Assert;
025import org.junit.Test;

027public class BiDiagonalTransformerTest {
028
029    private double[][] testSquare = {
030            { 24.0 / 25.0, 43.0 / 25.0 },
031            { 57.0 / 25.0, 24.0 / 25.0 }
032    };
033
034    private double[][] testNonSquare = {
035        {  -540.0 / 625.0,  963.0 / 625.0, -216.0 / 625.0 },
036        { -1730.0 / 625.0, -744.0 / 625.0, 1008.0 / 625.0 },
037        {  -720.0 / 625.0, 1284.0 / 625.0, -288.0 / 625.0 },
038        {  -360.0 / 625.0,  192.0 / 625.0, 1756.0 / 625.0 },
039    };

041    @Test
042    public void testDimensions() {
043        checkdimensions(MatrixUtils.createRealMatrix(testSquare));
044        checkdimensions(MatrixUtils.createRealMatrix(testNonSquare));
045        checkdimensions(MatrixUtils.createRealMatrix(testNonSquare).transpose());
046    }
047
048    private void checkdimensions(RealMatrix matrix) {
049        final int m = matrix.getRowDimension();
050        final int n = matrix.getColumnDimension();
051        BiDiagonalTransformer transformer = new BiDiagonalTransformer(matrix);
052        Assert.assertEquals(m, transformer.getU().getRowDimension());
053        Assert.assertEquals(m, transformer.getU().getColumnDimension());
054        Assert.assertEquals(m, transformer.getB().getRowDimension());
055        Assert.assertEquals(n, transformer.getB().getColumnDimension());
056        Assert.assertEquals(n, transformer.getV().getRowDimension());
057        Assert.assertEquals(n, transformer.getV().getColumnDimension());
058
059    }
060
061    @Test
062    public void testAEqualUSVt() {
063        checkAEqualUSVt(MatrixUtils.createRealMatrix(testSquare));
064        checkAEqualUSVt(MatrixUtils.createRealMatrix(testNonSquare));
065        checkAEqualUSVt(MatrixUtils.createRealMatrix(testNonSquare).transpose());
066    }
067
068    private void checkAEqualUSVt(RealMatrix matrix) {
069        BiDiagonalTransformer transformer = new BiDiagonalTransformer(matrix);
070        RealMatrix u = transformer.getU();
071        RealMatrix b = transformer.getB();
072        RealMatrix v = transformer.getV();
073        double norm = u.multiply(b).multiply(v.transpose()).subtract(matrix).getNorm();
074        Assert.assertEquals(0, norm, 1.0e-14);
075    }

077    @Test
078    public void testUOrthogonal() {
079        checkOrthogonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare)).getU());
080        checkOrthogonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testNonSquare)).getU());
081        checkOrthogonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testNonSquare).transpose()).getU());
082    }
083
084    @Test
085    public void testVOrthogonal() {
086        checkOrthogonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare)).getV());
087        checkOrthogonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testNonSquare)).getV());
088        checkOrthogonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testNonSquare).transpose()).getV());
089    }
090
091    private void checkOrthogonal(RealMatrix m) {
092        RealMatrix mTm = m.transpose().multiply(m);
093        RealMatrix id  = MatrixUtils.createRealIdentityMatrix(mTm.getRowDimension());
094        Assert.assertEquals(0, mTm.subtract(id).getNorm(), 1.0e-14);
095    }
096
097    @Test
098    public void testBBiDiagonal() {
099        checkBiDiagonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare)).getB());
100        checkBiDiagonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testNonSquare)).getB());
101        checkBiDiagonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testNonSquare).transpose()).getB());
102    }
103
104    private void checkBiDiagonal(RealMatrix m) {
105        final int rows = m.getRowDimension();
106        final int cols = m.getColumnDimension();
107        for (int i = 0; i < rows; ++i) {
108            for (int j = 0; j < cols; ++j) {
109                if (rows < cols) {
110                    if ((i < j) || (i > j + 1)) {
111                        Assert.assertEquals(0, m.getEntry(i, j), 1.0e-16);
112                    }
113                } else {
114                    if ((i < j - 1) || (i > j)) {
115                        Assert.assertEquals(0, m.getEntry(i, j), 1.0e-16);
116                    }
117                }
118            }
119        }
120    }

122    @Test
123    public void testSingularMatrix() {
124       BiDiagonalTransformer transformer =
125            new BiDiagonalTransformer(MatrixUtils.createRealMatrix(new double[][] {
126                { 1.0, 2.0, 3.0 },
127                { 2.0, 3.0, 4.0 },
128                { 3.0, 5.0, 7.0 }
129            }));
130       final double s3  = FastMath.sqrt(3.0);
131       final double s14 = FastMath.sqrt(14.0);
132       final double s1553 = FastMath.sqrt(1553.0);
133       RealMatrix uRef = MatrixUtils.createRealMatrix(new double[][] {
134           {  -1.0 / s14,  5.0 / (s3 * s14),  1.0 / s3 },
135           {  -2.0 / s14, -4.0 / (s3 * s14),  1.0 / s3 },
136           {  -3.0 / s14,  1.0 / (s3 * s14), -1.0 / s3 }
137       });
138       RealMatrix bRef = MatrixUtils.createRealMatrix(new double[][] {
139           { -s14, s1553 / s14,   0.0 },
140           {  0.0, -87 * s3 / (s14 * s1553), -s3 * s14 / s1553 },
141           {  0.0, 0.0, 0.0 }
142       });
143       RealMatrix vRef = MatrixUtils.createRealMatrix(new double[][] {
144           { 1.0,   0.0,         0.0        },
145           { 0.0,  -23 / s1553,  32 / s1553 },
146           { 0.0,  -32 / s1553, -23 / s1553 }
147       });
148
149       // check values against known references
150       RealMatrix u = transformer.getU();
151       Assert.assertEquals(0, u.subtract(uRef).getNorm(), 1.0e-14);
152       RealMatrix b = transformer.getB();
153       Assert.assertEquals(0, b.subtract(bRef).getNorm(), 1.0e-14);
154       RealMatrix v = transformer.getV();
155       Assert.assertEquals(0, v.subtract(vRef).getNorm(), 1.0e-14);
156
157       // check the same cached instance is returned the second time
158       Assert.assertTrue(u == transformer.getU());
159       Assert.assertTrue(b == transformer.getB());
160       Assert.assertTrue(v == transformer.getV());
161
162    }

164    @Test
165    public void testMatricesValues() {
166       BiDiagonalTransformer transformer =
167            new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare));
168       final double s17 = FastMath.sqrt(17.0);
169        RealMatrix uRef = MatrixUtils.createRealMatrix(new double[][] {
170                {  -8 / (5 * s17), 19 / (5 * s17) },
171                { -19 / (5 * s17), -8 / (5 * s17) }
172        });
173        RealMatrix bRef = MatrixUtils.createRealMatrix(new double[][] {
174                { -3 * s17 / 5, 32 * s17 / 85 },
175                {      0.0,     -5 * s17 / 17 }
176        });
177        RealMatrix vRef = MatrixUtils.createRealMatrix(new double[][] {
178                { 1.0,  0.0 },
179                { 0.0, -1.0 }
180        });
181
182        // check values against known references
183        RealMatrix u = transformer.getU();
184        Assert.assertEquals(0, u.subtract(uRef).getNorm(), 1.0e-14);
185        RealMatrix b = transformer.getB();
186        Assert.assertEquals(0, b.subtract(bRef).getNorm(), 1.0e-14);
187        RealMatrix v = transformer.getV();
188        Assert.assertEquals(0, v.subtract(vRef).getNorm(), 1.0e-14);
189
190        // check the same cached instance is returned the second time
191        Assert.assertTrue(u == transformer.getU());
192        Assert.assertTrue(b == transformer.getB());
193        Assert.assertTrue(v == transformer.getV());
194
195    }

197    @Test
198    public void testUpperOrLower() {
199        Assert.assertTrue(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare)).isUpperBiDiagonal());
200        Assert.assertTrue(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testNonSquare)).isUpperBiDiagonal());
201        Assert.assertFalse(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testNonSquare).transpose()).isUpperBiDiagonal());
202    }
