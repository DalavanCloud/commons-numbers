040    @Test
041    public void testConstructor2() {
042        final double[] d = { -1.2, 3.4, 5 };
043        final DiagonalMatrix m = new DiagonalMatrix(d);
044        for (int i = 0; i < m.getRowDimension(); i++) {
045            for (int j = 0; j < m.getRowDimension(); j++) {
046                if (i == j) {
047                    Assert.assertEquals(d[i], m.getEntry(i, j), 0d);
048                } else {
049                    Assert.assertEquals(0d, m.getEntry(i, j), 0d);
050                }
051            }
052        }
053
054        // Check that the underlying was copied.
055        d[0] = 0;
056        Assert.assertFalse(d[0] == m.getEntry(0, 0));
057    }

118            Assert.assertEquals(dim, out[i].length);
119            for (int j = 0; j < m.getRowDimension(); j++) {
120                if (i == j) {

125            }
126        }

214    public void testMultiply2() {
215        final double[] data1 = { -1.2, 3.4, 5 };
216        final DiagonalMatrix diag1 = new DiagonalMatrix(data1);
217
218        final double[][] data2 = { { -1.2, 3.4 },
219                                   { -5.6, 7.8 },
220                                   {  9.1, 2.3 } };
221        final RealMatrix dense2 = new Array2DRowRealMatrix(data2);
222        final RealMatrix dense1 = new Array2DRowRealMatrix(diag1.getData());
223
224        final RealMatrix diagResult = diag1.multiply(dense2);
225        final RealMatrix denseResult = dense1.multiply(dense2);
226
227        for (int i = 0; i < dense1.getRowDimension(); i++) {
228            for (int j = 0; j < dense2.getColumnDimension(); j++) {
229                Assert.assertEquals(denseResult.getEntry(i, j),
230                                    diagResult.getEntry(i, j), 0d);
231            }
232        }
