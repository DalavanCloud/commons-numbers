167    private RealMatrix checkAEqualPHPt(RealMatrix matrix) {
168        HessenbergTransformer transformer = new HessenbergTransformer(matrix);
169        RealMatrix p  = transformer.getP();
170        RealMatrix pT = transformer.getPT();
171        RealMatrix h  = transformer.getH();
172        
173        RealMatrix result = p.multiply(h).multiply(pT);
174        double norm = result.subtract(matrix).getNorm();
175        Assert.assertEquals(0, norm, 1.0e-10);
176        
177        for (int i = 0; i < matrix.getRowDimension(); ++i) {
178            for (int j = 0; j < matrix.getColumnDimension(); ++j) {
179                if (i > j + 1) {
180                    Assert.assertEquals(matrix.getEntry(i, j), result.getEntry(i, j), 1.0e-12);
181                }
182            }
183        }
184        
185        return transformer.getH();
186    }
187
188    private void checkOrthogonal(RealMatrix m) {
189        RealMatrix mTm = m.transpose().multiply(m);
190        RealMatrix id  = MatrixUtils.createRealIdentityMatrix(mTm.getRowDimension());
191        Assert.assertEquals(0, mTm.subtract(id).getNorm(), 1.0e-14);
192    }
193
194    private void checkHessenbergForm(RealMatrix m) {
195        final int rows = m.getRowDimension();
196        final int cols = m.getColumnDimension();
197        for (int i = 0; i < rows; ++i) {
198            for (int j = 0; j < cols; ++j) {
199                if (i > j + 1) {
200                    Assert.assertEquals(0, m.getEntry(i, j), 1.0e-16);
201                }
202            }
203        }
204    }
205    
206    private void checkMatricesValues(double[][] matrix, double[][] pRef, double[][] hRef) {
207        HessenbergTransformer transformer =
208            new HessenbergTransformer(MatrixUtils.createRealMatrix(matrix));
209
210        // check values against known references
211        RealMatrix p = transformer.getP();
212        Assert.assertEquals(0, p.subtract(MatrixUtils.createRealMatrix(pRef)).getNorm(), 1.0e-14);
213
214        RealMatrix h = transformer.getH();
215        Assert.assertEquals(0, h.subtract(MatrixUtils.createRealMatrix(hRef)).getNorm(), 1.0e-14);
216
217        // check the same cached instance is returned the second time
218        Assert.assertTrue(p == transformer.getP());
219        Assert.assertTrue(h == transformer.getH());
220    }
