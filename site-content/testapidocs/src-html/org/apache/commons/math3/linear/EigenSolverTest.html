037    /** test non invertible matrix */
038    @Test
039    public void testNonInvertible() {
040        Random r = new Random(9994100315209l);
041        RealMatrix m =
042            EigenDecompositionTest.createTestMatrix(r, new double[] { 1.0, 0.0, -1.0, -2.0, -3.0 });
043        DecompositionSolver es = new EigenDecomposition(m).getSolver();
044        Assert.assertFalse(es.isNonSingular());
045        try {
046            es.getInverse();
047            Assert.fail("an exception should have been thrown");
048        } catch (SingularMatrixException ime) {
049            // expected behavior
050        }
051    }

053    /** test invertible matrix */
054    @Test
055    public void testInvertible() {
056        Random r = new Random(9994100315209l);
057        RealMatrix m =
058            EigenDecompositionTest.createTestMatrix(r, new double[] { 1.0, 0.5, -1.0, -2.0, -3.0 });
059        DecompositionSolver es = new EigenDecomposition(m).getSolver();
060        Assert.assertTrue(es.isNonSingular());
061        RealMatrix inverse = es.getInverse();
062        RealMatrix error =
063            m.multiply(inverse).subtract(MatrixUtils.createRealIdentityMatrix(m.getRowDimension()));
064        Assert.assertEquals(0, error.getNorm(), 4.0e-15);
065    }

067    /**
068     * Verifies operation on very small values.
069     * Matrix with eigenvalues {8e-100, -1e-100, -1e-100}
070     */
071    @Test
072    public void testInvertibleTinyValues() {
073        final double tiny = 1e-100;
074        RealMatrix m = MatrixUtils.createRealMatrix(new double[][] {
075                {3,  2,  4},
076                {2,  0,  2},
077                {4,  2,  3}
078        });
079        m = m.scalarMultiply(tiny);
080
081        final EigenDecomposition ed = new EigenDecomposition(m);
082        RealMatrix inv = ed.getSolver().getInverse();
083
084        final RealMatrix id = m.multiply(inv);
085        for (int i = 0; i < m.getRowDimension(); i++) {
086            for (int j = 0; j < m.getColumnDimension(); j++) {
087                if (i == j) {
088                    Assert.assertTrue(Precision.equals(1, id.getEntry(i, j), 1e-15));
089                } else {
090                    Assert.assertTrue(Precision.equals(0, id.getEntry(i, j), 1e-15));
091                }
092            }
093        }
094    }

121    /** test solve dimension errors */
122    @Test
123    public void testSolveDimensionErrors() {
124        final double[] refValues = new double[] {
125            2.003, 2.002, 2.001, 1.001, 1.000, 0.001
126        };
127        final RealMatrix matrix = EigenDecompositionTest.createTestMatrix(new Random(35992629946426l), refValues);
128
129        DecompositionSolver es = new EigenDecomposition(matrix).getSolver();
130        RealMatrix b = MatrixUtils.createRealMatrix(new double[2][2]);
131        try {
132            es.solve(b);
133            Assert.fail("an exception should have been thrown");
134        } catch (MathIllegalArgumentException iae) {
135            // expected behavior
136        }
137        try {
138            es.solve(b.getColumnVector(0));
139            Assert.fail("an exception should have been thrown");
140        } catch (MathIllegalArgumentException iae) {
141            // expected behavior
142        }
143        try {
144            es.solve(new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(0)));
145            Assert.fail("an exception should have been thrown");
146        } catch (MathIllegalArgumentException iae) {
147            // expected behavior
148        }
149    }

151    /** test solve */
152    @Test
153    public void testSolve() {
154        RealMatrix m = MatrixUtils.createRealMatrix(new double[][] {
155                { 91,  5, 29, 32, 40, 14 },
156                {  5, 34, -1,  0,  2, -1 },
157                { 29, -1, 12,  9, 21,  8 },
158                { 32,  0,  9, 14,  9,  0 },
159                { 40,  2, 21,  9, 51, 19 },
160                { 14, -1,  8,  0, 19, 14 }
161        });
162        DecompositionSolver es = new EigenDecomposition(m).getSolver();
163        RealMatrix b = MatrixUtils.createRealMatrix(new double[][] {
164                { 1561, 269, 188 },
165                {   69, -21,  70 },
166                {  739, 108,  63 },
167                {  324,  86,  59 },
168                { 1624, 194, 107 },
169                {  796,  69,  36 }
170        });
171        RealMatrix xRef = MatrixUtils.createRealMatrix(new double[][] {
172                { 1,   2, 1 },
173                { 2,  -1, 2 },
174                { 4,   2, 3 },
175                { 8,  -1, 0 },
176                { 16,  2, 0 },
177                { 32, -1, 0 }
178        });
179
180        // using RealMatrix
181        RealMatrix solution=es.solve(b);
182        Assert.assertEquals(0, solution.subtract(xRef).getNorm(), 2.5e-12);
183
184        // using RealVector
185        for (int i = 0; i < b.getColumnDimension(); ++i) {
186            Assert.assertEquals(0,
187                         es.solve(b.getColumnVector(i)).subtract(xRef.getColumnVector(i)).getNorm(),
188                         2.0e-11);
189        }
190
191        // using RealVector with an alternate implementation
192        for (int i = 0; i < b.getColumnDimension(); ++i) {
193            ArrayRealVectorTest.RealVectorTestImpl v =
194                new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(i));
195            Assert.assertEquals(0,
196                         es.solve(v).subtract(xRef.getColumnVector(i)).getNorm(),
197                         2.0e-11);
198        }
199    }
