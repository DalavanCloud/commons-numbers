111    /** test solve */
112    @Test
113    public void testSolve() {
114        QRDecomposition decomposition =
115            new QRDecomposition(MatrixUtils.createRealMatrix(testData3x3NonSingular));
116        DecompositionSolver solver = decomposition.getSolver();
117        RealMatrix b = MatrixUtils.createRealMatrix(new double[][] {
118                { -102, 12250 }, { 544, 24500 }, { 167, -36750 }
119        });
120        RealMatrix xRef = MatrixUtils.createRealMatrix(new double[][] {
121                { 1, 2515 }, { 2, 422 }, { -3, 898 }
122        });
123
124        // using RealMatrix
125        Assert.assertEquals(0, solver.solve(b).subtract(xRef).getNorm(), 2.0e-16 * xRef.getNorm());
126
127        // using ArrayRealVector
128        for (int i = 0; i < b.getColumnDimension(); ++i) {
129            final RealVector x = solver.solve(b.getColumnVector(i));
130            final double error = x.subtract(xRef.getColumnVector(i)).getNorm();
131            Assert.assertEquals(0, error, 3.0e-16 * xRef.getColumnVector(i).getNorm());
132        }
133
134        // using RealVector with an alternate implementation
135        for (int i = 0; i < b.getColumnDimension(); ++i) {
136            ArrayRealVectorTest.RealVectorTestImpl v =
137                new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(i));
138            final RealVector x = solver.solve(v);
139            final double error = x.subtract(xRef.getColumnVector(i)).getNorm();
140            Assert.assertEquals(0, error, 3.0e-16 * xRef.getColumnVector(i).getNorm());
141        }
142
143    }

145    @Test
146    public void testOverdetermined() {
147        final Random r    = new Random(5559252868205245l);
148        int          p    = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
149        int          q    = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
150        RealMatrix   a    = createTestMatrix(r, p, q);
151        RealMatrix   xRef = createTestMatrix(r, q, BlockRealMatrix.BLOCK_SIZE + 3);
152
153        // build a perturbed system: A.X + noise = B
154        RealMatrix b = a.multiply(xRef);
155        final double noise = 0.001;
156        b.walkInOptimizedOrder(new DefaultRealMatrixChangingVisitor() {
157            @Override
158            public double visit(int row, int column, double value) {
159                return value * (1.0 + noise * (2 * r.nextDouble() - 1));
160            }
161        });
162
163        // despite perturbation, the least square solution should be pretty good
164        RealMatrix x = new QRDecomposition(a).getSolver().solve(b);
165        Assert.assertEquals(0, x.subtract(xRef).getNorm(), 0.01 * noise * p * q);
166
167    }
168
169    @Test
170    public void testUnderdetermined() {
171        final Random r    = new Random(42185006424567123l);
172        int          p    = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
173        int          q    = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
174        RealMatrix   a    = createTestMatrix(r, p, q);
175        RealMatrix   xRef = createTestMatrix(r, q, BlockRealMatrix.BLOCK_SIZE + 3);
176        RealMatrix   b    = a.multiply(xRef);
177        RealMatrix   x = new QRDecomposition(a).getSolver().solve(b);
178
179        // too many equations, the system cannot be solved at all
180        Assert.assertTrue(x.subtract(xRef).getNorm() / (p * q) > 0.01);
181
182        // the last unknown should have been set to 0
183        Assert.assertEquals(0.0, x.getSubMatrix(p, q - 1, 0, x.getColumnDimension() - 1).getNorm(), 0);
184    }
185
186    private RealMatrix createTestMatrix(final Random r, final int rows, final int columns) {
187        RealMatrix m = MatrixUtils.createRealMatrix(rows, columns);
188        m.walkInOptimizedOrder(new DefaultRealMatrixChangingVisitor() {
189                @Override
190                    public double visit(int row, int column, double value) {
191                    return 2.0 * r.nextDouble() - 1.0;
192                }
193            });
194        return m;
195    }
