226    public void testPower() {
227        Array2DRowRealMatrix m = new Array2DRowRealMatrix(testData);
228        Array2DRowRealMatrix mInv = new Array2DRowRealMatrix(testDataInv);
229        Array2DRowRealMatrix mPlusInv = new Array2DRowRealMatrix(testDataPlusInv);
230        Array2DRowRealMatrix identity = new Array2DRowRealMatrix(id);
231
232        TestUtils.assertEquals("m^0", m.power(0),
233            identity, entryTolerance);
234        TestUtils.assertEquals("mInv^0", mInv.power(0),
235                identity, entryTolerance);
236        TestUtils.assertEquals("mPlusInv^0", mPlusInv.power(0),
237                identity, entryTolerance);
238
239        TestUtils.assertEquals("m^1", m.power(1),
240                m, entryTolerance);
241        TestUtils.assertEquals("mInv^1", mInv.power(1),
242                mInv, entryTolerance);
243        TestUtils.assertEquals("mPlusInv^1", mPlusInv.power(1),
244                mPlusInv, entryTolerance);
245
246        RealMatrix C1 = m.copy();
247        RealMatrix C2 = mInv.copy();
248        RealMatrix C3 = mPlusInv.copy();
249
250        for (int i = 2; i <= 10; ++i) {
251            C1 = C1.multiply(m);
252            C2 = C2.multiply(mInv);
253            C3 = C3.multiply(mPlusInv);
254
255            TestUtils.assertEquals("m^" + i, m.power(i),
256                    C1, entryTolerance);
257            TestUtils.assertEquals("mInv^" + i, mInv.power(i),
258                    C2, entryTolerance);
259            TestUtils.assertEquals("mPlusInv^" + i, mPlusInv.power(i),
260                    C3, entryTolerance);
261        }
262

264            Array2DRowRealMatrix mNotSquare = new Array2DRowRealMatrix(testData2T);
265            mNotSquare.power(2);
266            Assert.fail("Expecting NonSquareMatrixException");
267        } catch (NonSquareMatrixException ex) {
268            // ignored
269        }

417    /** test examples in user guide */
418    @Test
419    public void testExamples() {
420        // Create a real matrix with two rows and three columns
421        double[][] matrixData = { {1d,2d,3d}, {2d,5d,3d}};
422        RealMatrix m = new Array2DRowRealMatrix(matrixData);
423        // One more with three rows, two columns
424        double[][] matrixData2 = { {1d,2d}, {2d,5d}, {1d, 7d}};
425        RealMatrix n = new Array2DRowRealMatrix(matrixData2);
426        // Now multiply m by n
427        RealMatrix p = m.multiply(n);
428        Assert.assertEquals(2, p.getRowDimension());
429        Assert.assertEquals(2, p.getColumnDimension());
430        // Invert p
431        RealMatrix pInverse = new LUDecomposition(p).getSolver().getInverse();
432        Assert.assertEquals(2, pInverse.getRowDimension());
433        Assert.assertEquals(2, pInverse.getColumnDimension());
434
435        // Solve example
436        double[][] coefficientsData = {{2, 3, -2}, {-1, 7, 6}, {4, -3, -5}};
437        RealMatrix coefficients = new Array2DRowRealMatrix(coefficientsData);
438        RealVector constants = new ArrayRealVector(new double[]{1, -2, 1}, false);
439        RealVector solution = new LUDecomposition(coefficients).getSolver().solve(constants);
440        final double cst0 = constants.getEntry(0);
441        final double cst1 = constants.getEntry(1);
442        final double cst2 = constants.getEntry(2);
443        final double sol0 = solution.getEntry(0);
444        final double sol1 = solution.getEntry(1);
445        final double sol2 = solution.getEntry(2);
446        Assert.assertEquals(2 * sol0 + 3 * sol1 -2 * sol2, cst0, 1E-12);
447        Assert.assertEquals(-1 * sol0 + 7 * sol1 + 6 * sol2, cst1, 1E-12);
448        Assert.assertEquals(4 * sol0 - 3 * sol1 -5 * sol2, cst2, 1E-12);
449    }

1105    /** extracts the l  and u matrices from compact lu representation */
1106    protected void splitLU(RealMatrix lu, double[][] lowerData, double[][] upperData) {
1107        if (!lu.isSquare()) {
1108            throw new NonSquareMatrixException(lu.getRowDimension(), lu.getColumnDimension());
1109        }
1110        if (lowerData.length != lowerData[0].length) {
1111            throw new DimensionMismatchException(lowerData.length, lowerData[0].length);
1112        }
1113        if (upperData.length != upperData[0].length) {
1114            throw new DimensionMismatchException(upperData.length, upperData[0].length);
1115        }
1116        if (lowerData.length != upperData.length) {
1117            throw new DimensionMismatchException(lowerData.length, upperData.length);
1118        }
1119        if (lowerData.length != lu.getRowDimension()) {
1120            throw new DimensionMismatchException(lowerData.length, lu.getRowDimension());
1121        }
1122
1123        int n = lu.getRowDimension();
1124        for (int i = 0; i < n; i++) {
1125            for (int j = 0; j < n; j++) {
1126                if (j < i) {
1127                    lowerData[i][j] = lu.getEntry(i, j);
1128                    upperData[i][j] = 0d;
1129                } else if (i == j) {
1130                    lowerData[i][j] = 1d;
1131                    upperData[i][j] = lu.getEntry(i, j);
1132                } else {
1133                    lowerData[i][j] = 0d;
1134                    upperData[i][j] = lu.getEntry(i, j);
1135                }
1136            }
1137        }
1138    }

1140    /** Returns the result of applying the given row permutation to the matrix */
1141    protected RealMatrix permuteRows(RealMatrix matrix, int[] permutation) {
1142        if (!matrix.isSquare()) {
1143            throw new NonSquareMatrixException(matrix.getRowDimension(),
1144                                               matrix.getColumnDimension());
1145        }
1146        if (matrix.getRowDimension() != permutation.length) {
1147            throw new DimensionMismatchException(matrix.getRowDimension(), permutation.length);
1148        }
1149
1150        int n = matrix.getRowDimension();
1151        int m = matrix.getColumnDimension();
1152        double out[][] = new double[m][n];
1153        for (int i = 0; i < n; i++) {
1154            for (int j = 0; j < m; j++) {
1155                out[i][j] = matrix.getEntry(permutation[i], j);
1156            }
1157        }
1158        return new Array2DRowRealMatrix(out);
1159    }
