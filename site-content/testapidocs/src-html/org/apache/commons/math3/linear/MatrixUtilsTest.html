197     * Verifies that the matrix is an identity matrix
198     */
199    protected void checkIdentityMatrix(RealMatrix m) {
200        for (int i = 0; i < m.getRowDimension(); i++) {
201            for (int j =0; j < m.getColumnDimension(); j++) {
202                if (i == j) {
203                    Assert.assertEquals(m.getEntry(i, j), 1d, 0);
204                } else {
205                    Assert.assertEquals(m.getEntry(i, j), 0d, 0);
206                }
207            }
208        }
209    }

278    public static final Fraction[][] asFraction(double[][] data) {
279        Fraction d[][] = new Fraction[data.length][];
280        try {
281            for (int i = 0; i < data.length; ++i) {
282                double[] dataI = data[i];
283                Fraction[] dI  = new Fraction[dataI.length];
284                for (int j = 0; j < dataI.length; ++j) {
285                    dI[j] = new Fraction(dataI[j]);
286                }
287                d[i] = dI;
288            }
289        } catch (FractionConversionException fce) {
290            Assert.fail(fce.getMessage());
291        }
292        return d;
293    }
294
295    public static final Fraction[] asFraction(double[] data) {
296        Fraction d[] = new Fraction[data.length];
297        try {
298            for (int i = 0; i < data.length; ++i) {
299                d[i] = new Fraction(data[i]);
300            }
301        } catch (FractionConversionException fce) {
302            Assert.fail(fce.getMessage());
303        }
304        return d;
305    }

336    @Test
337    public void testBlockInverse() {
338        final double[][] data = {
339            { -1, 0, 123, 4 },
340            { -56, 78.9, -0.1, -23.4 },
341            { 5.67, 8, -9, 1011 },
342            { 12, 345, -67.8, 9 },
343        };
344
345        final RealMatrix m = new Array2DRowRealMatrix(data);
346        final int len = data.length;
347        final double tol = 1e-14;
348
349        for (int splitIndex = 0; splitIndex < 3; splitIndex++) {
350            final RealMatrix mInv = MatrixUtils.blockInverse(m, splitIndex);
351            final RealMatrix id = m.multiply(mInv);
352
353            // Check that we recovered the identity matrix.
354            for (int i = 0; i < len; i++) {
355                for (int j = 0; j < len; j++) {
356                    final double entry = id.getEntry(i, j);
357                    if (i == j) {
358                        Assert.assertEquals("[" + i + "][" + j + "]",
359                                            1, entry, tol);
360                    } else {
361                        Assert.assertEquals("[" + i + "][" + j + "]",
362                                            0, entry, tol);
363                    }
364                }
365            }
366        }
367    }

400    @Test
401    public void testIsSymmetricTolerance() {
402        final double eps = 1e-4;
403
404        final double[][] dataSym1 = {
405            { 1,   1, 1.00009 },
406            { 1,   1, 1       },
407            { 1.0, 1, 1       },
408        };
409        Assert.assertTrue(MatrixUtils.isSymmetric(MatrixUtils.createRealMatrix(dataSym1), eps));
410        final double[][] dataSym2 = {
411            { 1,   1, 0.99990 },
412            { 1,   1, 1       },
413            { 1.0, 1, 1       },
414        };
415        Assert.assertTrue(MatrixUtils.isSymmetric(MatrixUtils.createRealMatrix(dataSym2), eps));
416
417        final double[][] dataNonSym1 = {
418            { 1,   1, 1.00011 },
419            { 1,   1, 1       },
420            { 1.0, 1, 1       },
421        };
422        Assert.assertFalse(MatrixUtils.isSymmetric(MatrixUtils.createRealMatrix(dataNonSym1), eps));
423        final double[][] dataNonSym2 = {
424            { 1,   1, 0.99989 },
425            { 1,   1, 1       },
426            { 1.0, 1, 1       },
427        };
428        Assert.assertFalse(MatrixUtils.isSymmetric(MatrixUtils.createRealMatrix(dataNonSym2), eps));
429    }
