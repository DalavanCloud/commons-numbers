367    @Test
368    public void testExamples() {
369        // Create a real matrix with two rows and three columns
370        double[][] matrixData = { { 1d, 2d, 3d }, { 2d, 5d, 3d } };
371        RealMatrix m = createSparseMatrix(matrixData);
372        // One more with three rows, two columns
373        double[][] matrixData2 = { { 1d, 2d }, { 2d, 5d }, { 1d, 7d } };
374        RealMatrix n = createSparseMatrix(matrixData2);
375        // Now multiply m by n
376        RealMatrix p = m.multiply(n);
377        Assert.assertEquals(2, p.getRowDimension());
378        Assert.assertEquals(2, p.getColumnDimension());
379        // Invert p
380        RealMatrix pInverse = new LUDecomposition(p).getSolver().getInverse();
381        Assert.assertEquals(2, pInverse.getRowDimension());
382        Assert.assertEquals(2, pInverse.getColumnDimension());
383
384        // Solve example
385        double[][] coefficientsData = { { 2, 3, -2 }, { -1, 7, 6 },
386                { 4, -3, -5 } };
387        RealMatrix coefficients = createSparseMatrix(coefficientsData);
388        RealVector constants = new ArrayRealVector(new double[]{ 1, -2, 1 }, false);
389        RealVector solution = new LUDecomposition(coefficients).getSolver().solve(constants);
390        final double cst0 = constants.getEntry(0);
391        final double cst1 = constants.getEntry(1);
392        final double cst2 = constants.getEntry(2);
393        final double sol0 = solution.getEntry(0);
394        final double sol1 = solution.getEntry(1);
395        final double sol2 = solution.getEntry(2);
396        Assert.assertEquals(2 * sol0 + 3 * sol1 - 2 * sol2, cst0, 1E-12);
397        Assert.assertEquals(-1 * sol0 + 7 * sol1 + 6 * sol2, cst1, 1E-12);
398        Assert.assertEquals(4 * sol0 - 3 * sol1 - 5 * sol2, cst2, 1E-12);
399
400    }

671    /** verifies that two matrices are close (1-norm) */
672    protected void assertClose(String msg, RealMatrix m, RealMatrix n,
673            double tolerance) {
674        Assert.assertTrue(msg, m.subtract(n).getNorm() < tolerance);
675    }
676
677    /** verifies that two vectors are close (sup norm) */
678    protected void assertClose(String msg, double[] m, double[] n,
679            double tolerance) {
680        if (m.length != n.length) {
681            Assert.fail("vectors not same length");
682        }
683        for (int i = 0; i < m.length; i++) {
684            Assert.assertEquals(msg + " " + i + " elements differ", m[i], n[i],
685                    tolerance);
686        }
687    }
688
689    private OpenMapRealMatrix createSparseMatrix(double[][] data) {
690        OpenMapRealMatrix matrix = new OpenMapRealMatrix(data.length, data[0].length);
691        for (int row = 0; row < data.length; row++) {
692            for (int col = 0; col < data[row].length; col++) {
693                matrix.setEntry(row, col, data[row][col]);
694            }
695        }
696        return matrix;
697    }
