053    /** test dimensions */
054    @Test
055    public void testDimensions() {
056        checkDimension(MatrixUtils.createRealMatrix(testData3x3NonSingular));
057
058        checkDimension(MatrixUtils.createRealMatrix(testData4x3));
059
060        checkDimension(MatrixUtils.createRealMatrix(testData3x4));
061
062        Random r = new Random(643895747384642l);
063        int    p = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
064        int    q = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
065        checkDimension(createTestMatrix(r, p, q));
066        checkDimension(createTestMatrix(r, q, p));
067
068    }
069
070    private void checkDimension(RealMatrix m) {
071        int rows = m.getRowDimension();
072        int columns = m.getColumnDimension();
073        QRDecomposition qr = new QRDecomposition(m);
074        Assert.assertEquals(rows,    qr.getQ().getRowDimension());
075        Assert.assertEquals(rows,    qr.getQ().getColumnDimension());
076        Assert.assertEquals(rows,    qr.getR().getRowDimension());
077        Assert.assertEquals(columns, qr.getR().getColumnDimension());
078    }

081    @Test
082    public void testAEqualQR() {
083        checkAEqualQR(MatrixUtils.createRealMatrix(testData3x3NonSingular));
084
085        checkAEqualQR(MatrixUtils.createRealMatrix(testData3x3Singular));
086
087        checkAEqualQR(MatrixUtils.createRealMatrix(testData3x4));
088
089        checkAEqualQR(MatrixUtils.createRealMatrix(testData4x3));

091        Random r = new Random(643895747384642l);
092        int    p = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
093        int    q = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
094        checkAEqualQR(createTestMatrix(r, p, q));
095
096        checkAEqualQR(createTestMatrix(r, q, p));
097
098    }

100    private void checkAEqualQR(RealMatrix m) {
101        QRDecomposition qr = new QRDecomposition(m);
102        double norm = qr.getQ().multiply(qr.getR()).subtract(m).getNorm();
103        Assert.assertEquals(0, norm, normTolerance);
104    }

126    private void checkQOrthogonal(RealMatrix m) {
127        QRDecomposition qr = new QRDecomposition(m);
128        RealMatrix eye = MatrixUtils.createRealIdentityMatrix(m.getRowDimension());
129        double norm = qr.getQT().multiply(qr.getQ()).subtract(eye).getNorm();
130        Assert.assertEquals(0, norm, normTolerance);
131    }

159    private void checkUpperTriangular(RealMatrix m) {
160        m.walkInOptimizedOrder(new DefaultRealMatrixPreservingVisitor() {
161            @Override
162            public void visit(int row, int column, double value) {
163                if (column < row) {
164                    Assert.assertEquals(0.0, value, entryTolerance);
165                }
166            }
167        });
168    }

206    /** test matrices values */
207    @Test
208    public void testMatricesValues() {
209        QRDecomposition qr =
210            new QRDecomposition(MatrixUtils.createRealMatrix(testData3x3NonSingular));
211        RealMatrix qRef = MatrixUtils.createRealMatrix(new double[][] {
212                { -12.0 / 14.0,   69.0 / 175.0,  -58.0 / 175.0 },
213                {  -6.0 / 14.0, -158.0 / 175.0,    6.0 / 175.0 },
214                {   4.0 / 14.0,  -30.0 / 175.0, -165.0 / 175.0 }
215        });
216        RealMatrix rRef = MatrixUtils.createRealMatrix(new double[][] {
217                { -14.0,  -21.0, 14.0 },
218                {   0.0, -175.0, 70.0 },
219                {   0.0,    0.0, 35.0 }
220        });
221        RealMatrix hRef = MatrixUtils.createRealMatrix(new double[][] {
222                { 26.0 / 14.0, 0.0, 0.0 },
223                {  6.0 / 14.0, 648.0 / 325.0, 0.0 },
224                { -4.0 / 14.0,  36.0 / 325.0, 2.0 }
225        });
226
227        // check values against known references
228        RealMatrix q = qr.getQ();
229        Assert.assertEquals(0, q.subtract(qRef).getNorm(), 1.0e-13);
230        RealMatrix qT = qr.getQT();
231        Assert.assertEquals(0, qT.subtract(qRef.transpose()).getNorm(), 1.0e-13);
232        RealMatrix r = qr.getR();
233        Assert.assertEquals(0, r.subtract(rRef).getNorm(), 1.0e-13);
234        RealMatrix h = qr.getH();
235        Assert.assertEquals(0, h.subtract(hRef).getNorm(), 1.0e-13);
236
237        // check the same cached instance is returned the second time
238        Assert.assertTrue(q == qr.getQ());
239        Assert.assertTrue(r == qr.getR());
240        Assert.assertTrue(h == qr.getH());
241
242    }

251    @Test
252    public void testInvertTallSkinny() {
253        RealMatrix a     = MatrixUtils.createRealMatrix(testData4x3);
254        RealMatrix pinv  = new QRDecomposition(a).getSolver().getInverse();
255        Assert.assertEquals(0, pinv.multiply(a).subtract(MatrixUtils.createRealIdentityMatrix(3)).getNorm(), 1.0e-6);
256    }

258    @Test
259    public void testInvertShortWide() {
260        RealMatrix a = MatrixUtils.createRealMatrix(testData3x4);
261        RealMatrix pinv  = new QRDecomposition(a).getSolver().getInverse();
262        Assert.assertEquals(0, a.multiply(pinv).subtract(MatrixUtils.createRealIdentityMatrix(3)).getNorm(), 1.0e-6);
263        Assert.assertEquals(0, pinv.multiply(a).getSubMatrix(0, 2, 0, 2).subtract(MatrixUtils.createRealIdentityMatrix(3)).getNorm(), 1.0e-6);
264    }

266    private RealMatrix createTestMatrix(final Random r, final int rows, final int columns) {
267        RealMatrix m = MatrixUtils.createRealMatrix(rows, columns);
268        m.walkInOptimizedOrder(new DefaultRealMatrixChangingVisitor(){
269            @Override
270            public double visit(int row, int column, double value) {
271                return 2.0 * r.nextDouble() - 1.0;
272            }
273        });
274        return m;
275    }
