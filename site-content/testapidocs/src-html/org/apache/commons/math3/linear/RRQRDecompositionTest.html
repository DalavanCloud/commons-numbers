018package org.apache.commons.math3.linear;

020import java.util.Random;
021
022import org.junit.Assert;
023import org.junit.Test;
024
025
026public class RRQRDecompositionTest {
027    private double[][] testData3x3NonSingular = {
028            { 12, -51, 4 },
029            { 6, 167, -68 },
030            { -4, 24, -41 }, };
031
032    private double[][] testData3x3Singular = {
033            { 1, 4, 7, },
034            { 2, 5, 8, },
035            { 3, 6, 9, }, };
036
037    private double[][] testData3x4 = {
038            { 12, -51, 4, 1 },
039            { 6, 167, -68, 2 },
040            { -4, 24, -41, 3 }, };
041
042    private double[][] testData4x3 = {
043            { 12, -51, 4, },
044            { 6, 167, -68, },
045            { -4, 24, -41, },
046            { -5, 34, 7, }, };
047
048    private static final double entryTolerance = 10e-16;
049
050    private static final double normTolerance = 10e-14;
051
052    /** test dimensions */
053    @Test
054    public void testDimensions() {
055        checkDimension(MatrixUtils.createRealMatrix(testData3x3NonSingular));
056
057        checkDimension(MatrixUtils.createRealMatrix(testData4x3));
058
059        checkDimension(MatrixUtils.createRealMatrix(testData3x4));
060
061        Random r = new Random(643895747384642l);
062        int    p = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
063        int    q = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
064        checkDimension(createTestMatrix(r, p, q));
065        checkDimension(createTestMatrix(r, q, p));
066
067    }
068
069    private void checkDimension(RealMatrix m) {
070        int rows = m.getRowDimension();
071        int columns = m.getColumnDimension();
072        RRQRDecomposition qr = new RRQRDecomposition(m);
073        Assert.assertEquals(rows,    qr.getQ().getRowDimension());
074        Assert.assertEquals(rows,    qr.getQ().getColumnDimension());
075        Assert.assertEquals(rows,    qr.getR().getRowDimension());
076        Assert.assertEquals(columns, qr.getR().getColumnDimension());
077    }
078
079    /** test AP = QR */
080    @Test
081    public void testAPEqualQR() {
082        checkAPEqualQR(MatrixUtils.createRealMatrix(testData3x3NonSingular));
083
084        checkAPEqualQR(MatrixUtils.createRealMatrix(testData3x3Singular));
085
086        checkAPEqualQR(MatrixUtils.createRealMatrix(testData3x4));
087
088        checkAPEqualQR(MatrixUtils.createRealMatrix(testData4x3));
089
090        Random r = new Random(643895747384642l);
091        int    p = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
092        int    q = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
093        checkAPEqualQR(createTestMatrix(r, p, q));
094
095        checkAPEqualQR(createTestMatrix(r, q, p));
096
097    }
098
099    private void checkAPEqualQR(RealMatrix m) {
100        RRQRDecomposition rrqr = new RRQRDecomposition(m);
101        double norm = rrqr.getQ().multiply(rrqr.getR()).subtract(m.multiply(rrqr.getP())).getNorm();
102        Assert.assertEquals(0, norm, normTolerance);
103    }
104
105    /** test the orthogonality of Q */
106    @Test
107    public void testQOrthogonal() {
108        checkQOrthogonal(MatrixUtils.createRealMatrix(testData3x3NonSingular));
109
110        checkQOrthogonal(MatrixUtils.createRealMatrix(testData3x3Singular));
111
112        checkQOrthogonal(MatrixUtils.createRealMatrix(testData3x4));
113
114        checkQOrthogonal(MatrixUtils.createRealMatrix(testData4x3));
115
116        Random r = new Random(643895747384642l);
117        int    p = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
118        int    q = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
119        checkQOrthogonal(createTestMatrix(r, p, q));
120
121        checkQOrthogonal(createTestMatrix(r, q, p));
122
123    }
124
125    private void checkQOrthogonal(RealMatrix m) {
126        RRQRDecomposition qr = new RRQRDecomposition(m);
127        RealMatrix eye = MatrixUtils.createRealIdentityMatrix(m.getRowDimension());
128        double norm = qr.getQT().multiply(qr.getQ()).subtract(eye).getNorm();
129        Assert.assertEquals(0, norm, normTolerance);
130    }

132    /** test that R is upper triangular */
133    @Test
134    public void testRUpperTriangular() {
135        RealMatrix matrix = MatrixUtils.createRealMatrix(testData3x3NonSingular);
136        checkUpperTriangular(new RRQRDecomposition(matrix).getR());
137
138        matrix = MatrixUtils.createRealMatrix(testData3x3Singular);
139        checkUpperTriangular(new RRQRDecomposition(matrix).getR());
140
141        matrix = MatrixUtils.createRealMatrix(testData3x4);
142        checkUpperTriangular(new RRQRDecomposition(matrix).getR());
143
144        matrix = MatrixUtils.createRealMatrix(testData4x3);
145        checkUpperTriangular(new RRQRDecomposition(matrix).getR());
146
147        Random r = new Random(643895747384642l);
148        int    p = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
149        int    q = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
150        matrix = createTestMatrix(r, p, q);
151        checkUpperTriangular(new RRQRDecomposition(matrix).getR());
152
153        matrix = createTestMatrix(r, p, q);
154        checkUpperTriangular(new RRQRDecomposition(matrix).getR());
155
156    }
157
158    private void checkUpperTriangular(RealMatrix m) {
159        m.walkInOptimizedOrder(new DefaultRealMatrixPreservingVisitor() {
160            @Override
161            public void visit(int row, int column, double value) {
162                if (column < row) {
163                    Assert.assertEquals(0.0, value, entryTolerance);
164                }
165            }
166        });
167    }

169    /** test that H is trapezoidal */
170    @Test
171    public void testHTrapezoidal() {
172        RealMatrix matrix = MatrixUtils.createRealMatrix(testData3x3NonSingular);
173        checkTrapezoidal(new RRQRDecomposition(matrix).getH());
174
175        matrix = MatrixUtils.createRealMatrix(testData3x3Singular);
176        checkTrapezoidal(new RRQRDecomposition(matrix).getH());
177
178        matrix = MatrixUtils.createRealMatrix(testData3x4);
179        checkTrapezoidal(new RRQRDecomposition(matrix).getH());
180
181        matrix = MatrixUtils.createRealMatrix(testData4x3);
182        checkTrapezoidal(new RRQRDecomposition(matrix).getH());
183
184        Random r = new Random(643895747384642l);
185        int    p = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
186        int    q = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
187        matrix = createTestMatrix(r, p, q);
188        checkTrapezoidal(new RRQRDecomposition(matrix).getH());
189
190        matrix = createTestMatrix(r, p, q);
191        checkTrapezoidal(new RRQRDecomposition(matrix).getH());
192
193    }
194
195    private void checkTrapezoidal(RealMatrix m) {
196        m.walkInOptimizedOrder(new DefaultRealMatrixPreservingVisitor() {
197            @Override
198            public void visit(int row, int column, double value) {
199                if (column > row) {
200                    Assert.assertEquals(0.0, value, entryTolerance);
201                }
202            }
203        });
204    }
205
206    @Test(expected=SingularMatrixException.class)
207    public void testNonInvertible() {
208        RRQRDecomposition qr =
209            new RRQRDecomposition(MatrixUtils.createRealMatrix(testData3x3Singular), 3.0e-16);
210        qr.getSolver().getInverse();
211    }
212
213    private RealMatrix createTestMatrix(final Random r, final int rows, final int columns) {
214        RealMatrix m = MatrixUtils.createRealMatrix(rows, columns);
215        m.walkInOptimizedOrder(new DefaultRealMatrixChangingVisitor(){
216            @Override
217            public double visit(int row, int column, double value) {
218                return 2.0 * r.nextDouble() - 1.0;
219            }
220        });
221        return m;
222    }

225    @Test
226    public void testRank() {
227        double[][] d = { { 1, 1, 1 }, { 0, 0, 0 }, { 1, 2, 3 } };
228        RealMatrix m = new Array2DRowRealMatrix(d);
229        RRQRDecomposition qr = new RRQRDecomposition(m);
