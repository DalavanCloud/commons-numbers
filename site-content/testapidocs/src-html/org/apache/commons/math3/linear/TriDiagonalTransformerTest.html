043    @Test
044    public void testNonSquare() {
045        try {
046            new TriDiagonalTransformer(MatrixUtils.createRealMatrix(new double[3][2]));
047            Assert.fail("an exception should have been thrown");
048        } catch (NonSquareMatrixException ime) {
049            // expected behavior
050        }

053    @Test
054    public void testAEqualQTQt() {
055        checkAEqualQTQt(MatrixUtils.createRealMatrix(testSquare5));
056        checkAEqualQTQt(MatrixUtils.createRealMatrix(testSquare3));
057    }
058
059    private void checkAEqualQTQt(RealMatrix matrix) {
060        TriDiagonalTransformer transformer = new TriDiagonalTransformer(matrix);
061        RealMatrix q  = transformer.getQ();
062        RealMatrix qT = transformer.getQT();
063        RealMatrix t  = transformer.getT();
064        double norm = q.multiply(t).multiply(qT).subtract(matrix).getNorm();
065        Assert.assertEquals(0, norm, 4.0e-15);
066    }
067
068    @Test
069    public void testNoAccessBelowDiagonal() {
070        checkNoAccessBelowDiagonal(testSquare5);
071        checkNoAccessBelowDiagonal(testSquare3);
072    }
073
074    private void checkNoAccessBelowDiagonal(double[][] data) {
075        double[][] modifiedData = new double[data.length][];
076        for (int i = 0; i < data.length; ++i) {
077            modifiedData[i] = data[i].clone();
078            Arrays.fill(modifiedData[i], 0, i, Double.NaN);
079        }
080        RealMatrix matrix = MatrixUtils.createRealMatrix(modifiedData);
081        TriDiagonalTransformer transformer = new TriDiagonalTransformer(matrix);
082        RealMatrix q  = transformer.getQ();
083        RealMatrix qT = transformer.getQT();
084        RealMatrix t  = transformer.getT();
085        double norm = q.multiply(t).multiply(qT).subtract(MatrixUtils.createRealMatrix(data)).getNorm();
086        Assert.assertEquals(0, norm, 4.0e-15);
087    }

089    @Test
090    public void testQOrthogonal() {
091        checkOrthogonal(new TriDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare5)).getQ());
092        checkOrthogonal(new TriDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare3)).getQ());
093    }
094
095    @Test
096    public void testQTOrthogonal() {
097        checkOrthogonal(new TriDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare5)).getQT());
098        checkOrthogonal(new TriDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare3)).getQT());
099    }
100
101    private void checkOrthogonal(RealMatrix m) {
102        RealMatrix mTm = m.transpose().multiply(m);
103        RealMatrix id  = MatrixUtils.createRealIdentityMatrix(mTm.getRowDimension());
104        Assert.assertEquals(0, mTm.subtract(id).getNorm(), 1.0e-15);
105    }
106
107    @Test
108    public void testTTriDiagonal() {
109        checkTriDiagonal(new TriDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare5)).getT());
110        checkTriDiagonal(new TriDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare3)).getT());
111    }
112
113    private void checkTriDiagonal(RealMatrix m) {
114        final int rows = m.getRowDimension();
115        final int cols = m.getColumnDimension();
116        for (int i = 0; i < rows; ++i) {
117            for (int j = 0; j < cols; ++j) {
118                if ((i < j - 1) || (i > j + 1)) {
119                    Assert.assertEquals(0, m.getEntry(i, j), 1.0e-16);
120                }
121            }
122        }
123    }

139    @Test
140    public void testMatricesValues3() {
141        checkMatricesValues(testSquare3,
142                            new double[][] {
143                                {  1.0,  0.0,  0.0 },
144                                {  0.0, -0.6,  0.8 },
145                                {  0.0, -0.8, -0.6 },
146                            },
147                            new double[] { 1, 2.64, -0.64 },
148                            new double[] { -5, -1.52 });
149    }
150
151    private void checkMatricesValues(double[][] matrix, double[][] qRef,
152                                     double[] mainDiagnonal,
153                                     double[] secondaryDiagonal) {
154        TriDiagonalTransformer transformer =
155            new TriDiagonalTransformer(MatrixUtils.createRealMatrix(matrix));
156
157        // check values against known references
158        RealMatrix q = transformer.getQ();
159        Assert.assertEquals(0, q.subtract(MatrixUtils.createRealMatrix(qRef)).getNorm(), 1.0e-14);
160
161        RealMatrix t = transformer.getT();
162        double[][] tData = new double[mainDiagnonal.length][mainDiagnonal.length];
163        for (int i = 0; i < mainDiagnonal.length; ++i) {
164            tData[i][i] = mainDiagnonal[i];
165            if (i > 0) {
166                tData[i][i - 1] = secondaryDiagonal[i - 1];
167            }
168            if (i < secondaryDiagonal.length) {
169                tData[i][i + 1] = secondaryDiagonal[i];
170            }
171        }
172        Assert.assertEquals(0, t.subtract(MatrixUtils.createRealMatrix(tData)).getNorm(), 1.0e-14);
173
174        // check the same cached instance is returned the second time
175        Assert.assertTrue(q == transformer.getQ());
176        Assert.assertTrue(t == transformer.getT());
177    }
