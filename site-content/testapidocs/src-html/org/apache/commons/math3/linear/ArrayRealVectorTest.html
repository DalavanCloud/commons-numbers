017package org.apache.commons.math3.linear;
018
019import org.apache.commons.math3.exception.MathIllegalArgumentException;
020import org.junit.Assert;
021import org.junit.Test;
022
023/**
024 * Test cases for the {@link ArrayRealVector} class.
025 *
026 */
027public class ArrayRealVectorTest extends RealVectorAbstractTest {
028
029
030    @Override
031    public RealVector create(final double[] data) {
032        return new ArrayRealVector(data, true);
033    }
034
035    @Test
036    public void testConstructors() {
037        final double[] vec1 = {1d, 2d, 3d};
038        final double[] vec3 = {7d, 8d, 9d};
039        final double[] vec4 = {1d, 2d, 3d, 4d, 5d, 6d, 7d, 8d, 9d};
040        final Double[] dvec1 = {1d, 2d, 3d, 4d, 5d, 6d, 7d, 8d, 9d};
041
042        ArrayRealVector v0 = new ArrayRealVector();
043        Assert.assertEquals("testData len", 0, v0.getDimension());
044
045        ArrayRealVector v1 = new ArrayRealVector(7);
046        Assert.assertEquals("testData len", 7, v1.getDimension());
047        Assert.assertEquals("testData is 0.0 ", 0.0, v1.getEntry(6), 0);
048
049        ArrayRealVector v2 = new ArrayRealVector(5, 1.23);
050        Assert.assertEquals("testData len", 5, v2.getDimension());
051        Assert.assertEquals("testData is 1.23 ", 1.23, v2.getEntry(4), 0);
052
053        ArrayRealVector v3 = new ArrayRealVector(vec1);
054        Assert.assertEquals("testData len", 3, v3.getDimension());
055        Assert.assertEquals("testData is 2.0 ", 2.0, v3.getEntry(1), 0);
056
057        ArrayRealVector v3_bis = new ArrayRealVector(vec1, true);
058        Assert.assertEquals("testData len", 3, v3_bis.getDimension());
059        Assert.assertEquals("testData is 2.0 ", 2.0, v3_bis.getEntry(1), 0);
060        Assert.assertNotSame(v3_bis.getDataRef(), vec1);
061        Assert.assertNotSame(v3_bis.toArray(), vec1);
062
063        ArrayRealVector v3_ter = new ArrayRealVector(vec1, false);
064        Assert.assertEquals("testData len", 3, v3_ter.getDimension());
065        Assert.assertEquals("testData is 2.0 ", 2.0, v3_ter.getEntry(1), 0);
066        Assert.assertSame(v3_ter.getDataRef(), vec1);
067        Assert.assertNotSame(v3_ter.toArray(), vec1);
068
069        ArrayRealVector v4 = new ArrayRealVector(vec4, 3, 2);
070        Assert.assertEquals("testData len", 2, v4.getDimension());
071        Assert.assertEquals("testData is 4.0 ", 4.0, v4.getEntry(0), 0);
072        try {
073            new ArrayRealVector(vec4, 8, 3);
074            Assert.fail("MathIllegalArgumentException expected");
075        } catch (MathIllegalArgumentException ex) {
076            // expected behavior
077        }
078
079        RealVector v5_i = new ArrayRealVector(dvec1);
080        Assert.assertEquals("testData len", 9, v5_i.getDimension());
081        Assert.assertEquals("testData is 9.0 ", 9.0, v5_i.getEntry(8), 0);
082
083        ArrayRealVector v5 = new ArrayRealVector(dvec1);
084        Assert.assertEquals("testData len", 9, v5.getDimension());
085        Assert.assertEquals("testData is 9.0 ", 9.0, v5.getEntry(8), 0);
086
087        ArrayRealVector v6 = new ArrayRealVector(dvec1, 3, 2);
088        Assert.assertEquals("testData len", 2, v6.getDimension());
089        Assert.assertEquals("testData is 4.0 ", 4.0, v6.getEntry(0), 0);
090        try {
091            new ArrayRealVector(dvec1, 8, 3);
092            Assert.fail("MathIllegalArgumentException expected");
093        } catch (MathIllegalArgumentException ex) {
094            // expected behavior
095        }
096
097        ArrayRealVector v7 = new ArrayRealVector(v1);
098        Assert.assertEquals("testData len", 7, v7.getDimension());
099        Assert.assertEquals("testData is 0.0 ", 0.0, v7.getEntry(6), 0);
100
101        RealVectorTestImpl v7_i = new RealVectorTestImpl(vec1);
102
103        ArrayRealVector v7_2 = new ArrayRealVector(v7_i);
104        Assert.assertEquals("testData len", 3, v7_2.getDimension());
105        Assert.assertEquals("testData is 0.0 ", 2.0d, v7_2.getEntry(1), 0);
106
107        ArrayRealVector v8 = new ArrayRealVector(v1, true);
108        Assert.assertEquals("testData len", 7, v8.getDimension());
109        Assert.assertEquals("testData is 0.0 ", 0.0, v8.getEntry(6), 0);
110        Assert.assertNotSame("testData not same object ", v1.getDataRef(), v8.getDataRef());
111
112        ArrayRealVector v8_2 = new ArrayRealVector(v1, false);
113        Assert.assertEquals("testData len", 7, v8_2.getDimension());
114        Assert.assertEquals("testData is 0.0 ", 0.0, v8_2.getEntry(6), 0);
115        Assert.assertEquals("testData same object ", v1.getDataRef(), v8_2.getDataRef());
116
117        ArrayRealVector v9 = new ArrayRealVector(v1, v3);
118        Assert.assertEquals("testData len", 10, v9.getDimension());
119        Assert.assertEquals("testData is 1.0 ", 1.0, v9.getEntry(7), 0);
120
121        ArrayRealVector v10 = new ArrayRealVector(v2, new RealVectorTestImpl(vec3));
122        Assert.assertEquals("testData len", 8, v10.getDimension());
123        Assert.assertEquals("testData is 1.23 ", 1.23, v10.getEntry(4), 0);
124        Assert.assertEquals("testData is 7.0 ", 7.0, v10.getEntry(5), 0);
125
126        ArrayRealVector v11 = new ArrayRealVector(new RealVectorTestImpl(vec3), v2);
127        Assert.assertEquals("testData len", 8, v11.getDimension());
128        Assert.assertEquals("testData is 9.0 ", 9.0, v11.getEntry(2), 0);
129        Assert.assertEquals("testData is 1.23 ", 1.23, v11.getEntry(3), 0);
130
131        ArrayRealVector v12 = new ArrayRealVector(v2, vec3);
132        Assert.assertEquals("testData len", 8, v12.getDimension());
133        Assert.assertEquals("testData is 1.23 ", 1.23, v12.getEntry(4), 0);
134        Assert.assertEquals("testData is 7.0 ", 7.0, v12.getEntry(5), 0);
135
136        ArrayRealVector v13 = new ArrayRealVector(vec3, v2);
137        Assert.assertEquals("testData len", 8, v13.getDimension());
138        Assert.assertEquals("testData is 9.0 ", 9.0, v13.getEntry(2), 0);
139        Assert.assertEquals("testData is 1.23 ", 1.23, v13.getEntry(3), 0);
140
141        ArrayRealVector v14 = new ArrayRealVector(vec3, vec4);
142        Assert.assertEquals("testData len", 12, v14.getDimension());
143        Assert.assertEquals("testData is 9.0 ", 9.0, v14.getEntry(2), 0);
144        Assert.assertEquals("testData is 1.0 ", 1.0, v14.getEntry(3), 0);
145
146    }
147
148    @Test
149    public void testGetDataRef() {
150        final double[] data = {1d, 2d, 3d, 4d};
151        final ArrayRealVector v = new ArrayRealVector(data);
152        v.getDataRef()[0] = 0d;
153        Assert.assertEquals("", 0d, v.getEntry(0), 0);
154    }
155
156    @Test
157    public void testPredicates() {
158
159        Assert.assertEquals(create(new double[] { Double.NaN, 1, 2 }).hashCode(),
160                     create(new double[] { 0, Double.NaN, 2 }).hashCode());
161
162        Assert.assertTrue(create(new double[] { Double.NaN, 1, 2 }).hashCode() !=
163                   create(new double[] { 0, 1, 2 }).hashCode());
164    }
165
166    @Test
167    public void testZeroVectors() {
168        Assert.assertEquals(0, new ArrayRealVector(new double[0]).getDimension());
169        Assert.assertEquals(0, new ArrayRealVector(new double[0], true).getDimension());
170        Assert.assertEquals(0, new ArrayRealVector(new double[0], false).getDimension());
171    }
