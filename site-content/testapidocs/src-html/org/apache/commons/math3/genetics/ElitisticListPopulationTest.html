017package org.apache.commons.math3.genetics;
018
019
020import java.util.Collections;
021import java.util.List;
022
023import org.apache.commons.math3.exception.OutOfRangeException;
024import org.junit.Assert;
025import org.junit.Test;
026
027public class ElitisticListPopulationTest {
028
029    private static int counter = 0;
030
031    @Test
032    public void testNextGeneration() {
033        ElitisticListPopulation pop = new ElitisticListPopulation(100, 0.203);
034
035        for (int i=0; i<pop.getPopulationLimit(); i++) {
036            pop.addChromosome(new DummyChromosome());
037        }
038
039        Population nextGeneration = pop.nextGeneration();
040
041        Assert.assertEquals(20, nextGeneration.getPopulationSize());
042    }
043    
044    @Test
045    public void testSetElitismRate() {
046        final double rate = 0.25;
047        final ElitisticListPopulation pop = new ElitisticListPopulation(100, 0.203);
048        pop.setElitismRate(rate);
049        Assert.assertEquals(rate, pop.getElitismRate(), 1e-6);
050    }
051    
052    @Test(expected = OutOfRangeException.class)
053    public void testSetElitismRateTooLow() {
054        final double rate = -0.25;
055        final ElitisticListPopulation pop = new ElitisticListPopulation(100, 0.203);
056        pop.setElitismRate(rate);
057    }
058    
059    @Test(expected = OutOfRangeException.class)
060    public void testSetElitismRateTooHigh() {
061        final double rate = 1.25;
062        final ElitisticListPopulation pop = new ElitisticListPopulation(100, 0.203);
063        pop.setElitismRate(rate);
064    }
065    
066    @Test(expected = OutOfRangeException.class)
067    public void testConstructorTooLow() {
068        final double rate = -0.25;
069        new ElitisticListPopulation(100, rate);
070    }
071    
072    @Test(expected = OutOfRangeException.class)
073    public void testConstructorTooHigh() {
074        final double rate = 1.25;
075        new ElitisticListPopulation(100, rate);
076    }
077
078    @Test(expected = OutOfRangeException.class)
079    public void testChromosomeListConstructorTooLow() {
080        final List<Chromosome> chromosomes = Collections.emptyList();
081        final double rate = -0.25;
082        new ElitisticListPopulation(chromosomes, 100, rate);
083    }
084
085    @Test(expected = OutOfRangeException.class)
086    public void testChromosomeListConstructorTooHigh() {
087        final List<Chromosome> chromosomes = Collections.emptyList();
088        final double rate = 1.25;
089        new ElitisticListPopulation(chromosomes, 100, rate);
090    }
091
092    private static class DummyChromosome extends Chromosome {
093        private final int fitness;
094
095        public DummyChromosome() {
096            this.fitness = counter;
097            counter++;
098        }
099
100        public double fitness() {
101            return this.fitness;
102        }
103    }
104
105}
