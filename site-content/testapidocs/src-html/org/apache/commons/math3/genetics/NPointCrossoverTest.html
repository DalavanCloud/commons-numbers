083    @Test
084    public void testCrossover() {
085        Integer[] p1 = new Integer[] {1,0,1,0,1,0,1,0,1};
086        Integer[] p2 = new Integer[] {0,1,0,1,0,1,0,1,0};
087
088        BinaryChromosome p1c = new DummyBinaryChromosome(p1);
089        BinaryChromosome p2c = new DummyBinaryChromosome(p2);
090
091        final int order = 3;
092        NPointCrossover<Integer> npc = new NPointCrossover<Integer>(order);
093
094        // the two parent chromosomes are different at each position, so it is easy to detect
095        // the number of crossovers that happened for each child
096        for (int i=0; i<20; i++) {
097            ChromosomePair pair = npc.crossover(p1c,p2c);
098
099            Integer[] c1 = new Integer[p1.length];
100            Integer[] c2 = new Integer[p2.length];
101
102            c1 = ((BinaryChromosome) pair.getFirst()).getRepresentation().toArray(c1);
103            c2 = ((BinaryChromosome) pair.getSecond()).getRepresentation().toArray(c2);
104
105            Assert.assertEquals(order, detectCrossoverPoints(p1c, p2c, (BinaryChromosome) pair.getFirst()));
106            Assert.assertEquals(order, detectCrossoverPoints(p2c, p1c, (BinaryChromosome) pair.getSecond()));            
107        }
108    }
109    
110    private int detectCrossoverPoints(BinaryChromosome p1, BinaryChromosome p2, BinaryChromosome c) {
111        int crossovers = 0;
112        final int length = p1.getLength();
113
114        final List<Integer> p1Rep = p1.getRepresentation();
115        final List<Integer> p2Rep = p2.getRepresentation();
116        final List<Integer> cRep = c.getRepresentation();
117        
118        List<Integer> rep = p1Rep;
119        for (int i = 0; i < length; i++) {
120            if (rep.get(i) != cRep.get(i)) {
121                crossovers++;
122                rep = rep == p1Rep ? p2Rep : p1Rep;
123            }
124        }
125        
126        return crossovers;
127    }
