27  /**
28   * This class represents an interpolator over the last step during an
29   * ODE integration for the 8(5,3) Dormand-Prince integrator.

31   * @see DormandPrince853FieldIntegrator

190     /** {@inheritDoc} */
191     @Override
192     protected DormandPrince853FieldStepInterpolator<T> create(final Field<T> newField, final boolean newForward, final T[][] newYDotK,
193                                                                final FieldODEStateAndDerivative<T> newGlobalPreviousState,
194                                                                final FieldODEStateAndDerivative<T> newGlobalCurrentState,
195                                                                final FieldODEStateAndDerivative<T> newSoftPreviousState,
196                                                                final FieldODEStateAndDerivative<T> newSoftCurrentState,
197                                                                final FieldEquationsMapper<T> newMapper) {
198         return new DormandPrince853FieldStepInterpolator<T>(newField, newForward, newYDotK,
199                                                             newGlobalPreviousState, newGlobalCurrentState,
200                                                             newSoftPreviousState, newSoftCurrentState,
201                                                             newMapper);
202     }

204     /** Create a fraction.
205      * @param field field to which the elements belong
206      * @param p numerator
207      * @param q denominator
208      * @return p/q computed in the instance field
209      */
210     private T fraction(final Field<T> field, final double p, final double q) {
211         return field.getZero().add(p).divide(q);
212     }

214     /** {@inheritDoc} */
215     @SuppressWarnings("unchecked")
216     @Override
217     protected FieldODEStateAndDerivative<T> computeInterpolatedStateAndDerivatives(final FieldEquationsMapper<T> mapper,
218                                                                                    final T time, final T theta,
219                                                                                    final T thetaH, final T oneMinusThetaH)
220         throws MaxCountExceededException {
221 
222         final T one      = time.getField().getOne();
223         final T eta      = one.subtract(theta);
224         final T twoTheta = theta.multiply(2);
225         final T theta2   = theta.multiply(theta);
226         final T dot1     = one.subtract(twoTheta);
227         final T dot2     = theta.multiply(theta.multiply(-3).add(2));
228         final T dot3     = twoTheta.multiply(theta.multiply(twoTheta.subtract(3)).add(1));
229         final T dot4     = theta2.multiply(theta.multiply(theta.multiply(5).subtract(8)).add(3));
230         final T dot5     = theta2.multiply(theta.multiply(theta.multiply(theta.multiply(-6).add(15)).subtract(12)).add(3));
231         final T dot6     = theta2.multiply(theta.multiply(theta.multiply(theta.multiply(theta.multiply(-7).add(18)).subtract(15)).add(4)));
232         final T[] interpolatedState;
233         final T[] interpolatedDerivatives;
234 
235 
236         if (getGlobalPreviousState() != null && theta.getReal() <= 0.5) {
237             final T f0 = thetaH;
238             final T f1 = f0.multiply(eta);
239             final T f2 = f1.multiply(theta);
240             final T f3 = f2.multiply(eta);
241             final T f4 = f3.multiply(theta);
242             final T f5 = f4.multiply(eta);
243             final T f6 = f5.multiply(theta);
244             final T[] p = MathArrays.buildArray(time.getField(), 16);
245             final T[] q = MathArrays.buildArray(time.getField(), 16);
246             for (int i = 0; i < p.length; ++i) {
247                 p[i] =     f0.multiply(d[0][i]).
248                        add(f1.multiply(d[1][i])).
249                        add(f2.multiply(d[2][i])).
250                        add(f3.multiply(d[3][i])).
251                        add(f4.multiply(d[4][i])).
252                        add(f5.multiply(d[5][i])).
253                        add(f6.multiply(d[6][i]));
254                 q[i] =                    d[0][i].
255                         add(dot1.multiply(d[1][i])).
256                         add(dot2.multiply(d[2][i])).
257                         add(dot3.multiply(d[3][i])).
258                         add(dot4.multiply(d[4][i])).
259                         add(dot5.multiply(d[5][i])).
260                         add(dot6.multiply(d[6][i]));
261             }
262             interpolatedState       = previousStateLinearCombination(p[0], p[1], p[ 2], p[ 3], p[ 4], p[ 5], p[ 6], p[ 7],
263                                                                      p[8], p[9], p[10], p[11], p[12], p[13], p[14], p[15]);
264             interpolatedDerivatives = derivativeLinearCombination(q[0], q[1], q[ 2], q[ 3], q[ 4], q[ 5], q[ 6], q[ 7],
265                                                                   q[8], q[9], q[10], q[11], q[12], q[13], q[14], q[15]);
266         } else {
267             final T f0 = oneMinusThetaH.negate();
268             final T f1 = f0.multiply(theta).negate();
269             final T f2 = f1.multiply(theta);
270             final T f3 = f2.multiply(eta);
271             final T f4 = f3.multiply(theta);
272             final T f5 = f4.multiply(eta);
273             final T f6 = f5.multiply(theta);
274             final T[] p = MathArrays.buildArray(time.getField(), 16);
275             final T[] q = MathArrays.buildArray(time.getField(), 16);
276             for (int i = 0; i < p.length; ++i) {
277                 p[i] =     f0.multiply(d[0][i]).
278                        add(f1.multiply(d[1][i])).
279                        add(f2.multiply(d[2][i])).
280                        add(f3.multiply(d[3][i])).
281                        add(f4.multiply(d[4][i])).
282                        add(f5.multiply(d[5][i])).
283                        add(f6.multiply(d[6][i]));
284                 q[i] =                    d[0][i].
285                         add(dot1.multiply(d[1][i])).
286                         add(dot2.multiply(d[2][i])).
287                         add(dot3.multiply(d[3][i])).
288                         add(dot4.multiply(d[4][i])).
289                         add(dot5.multiply(d[5][i])).
290                         add(dot6.multiply(d[6][i]));
291             }
292             interpolatedState       = currentStateLinearCombination(p[0], p[1], p[ 2], p[ 3], p[ 4], p[ 5], p[ 6], p[ 7],
293                                                                     p[8], p[9], p[10], p[11], p[12], p[13], p[14], p[15]);
294             interpolatedDerivatives = derivativeLinearCombination(q[0], q[1], q[ 2], q[ 3], q[ 4], q[ 5], q[ 6], q[ 7],
295                                                                   q[8], q[9], q[10], q[11], q[12], q[13], q[14], q[15]);
296         }
297 
298         return new FieldODEStateAndDerivative<T>(time, interpolatedState, interpolatedDerivatives);
299
