24   * Enumeration for localized messages formats used in exceptions messages.

380     /** Simple constructor.
381      * @param sourceFormat source English format to use when no

383      */
384     LocalizedFormats(final String sourceFormat) {
385         this.sourceFormat = sourceFormat;
386     }
387 
388     /** {@inheritDoc} */
389     public String getSourceString() {
390         return sourceFormat;
391     }
392 
393     /** {@inheritDoc} */
394     public String getLocalizedString(final Locale locale) {
395         try {
396             final String path = LocalizedFormats.class.getName().replaceAll("\\.", "/");
397             ResourceBundle bundle =
398                     ResourceBundle.getBundle("assets/" + path, locale);
399             if (bundle.getLocale().getLanguage().equals(locale.getLanguage())) {
400                 // the value of the resource is the translated format
401                 return bundle.getString(toString());
402             }
403 
404         } catch (MissingResourceException mre) { // NOPMD
405             // do nothing here
406         }
407 
408         // either the locale is not supported or the resource is unknown
409         // don't translate and fall back to using the source format
410         return sourceFormat;
411 
412     }
