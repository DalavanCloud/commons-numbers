17  package org.apache.commons.math3.analysis.interpolation;
18  
19  import org.apache.commons.math3.analysis.polynomials.PolynomialFunction;
20  import org.apache.commons.math3.analysis.polynomials.PolynomialSplineFunction;
21  import org.apache.commons.math3.exception.DimensionMismatchException;
22  import org.apache.commons.math3.exception.NonMonotonicSequenceException;
23  import org.apache.commons.math3.exception.NumberIsTooSmallException;
24  import org.apache.commons.math3.exception.util.LocalizedFormats;
25  import org.apache.commons.math3.util.MathArrays;
26  
27  /**
28   * Computes a natural (also known as "free", "unclamped") cubic spline interpolation for the data set.
29   * <p>
30   * The {@link #interpolate(double[], double[])} method returns a {@link PolynomialSplineFunction}
31   * consisting of n cubic polynomials, defined over the subintervals determined by the x values,
32   * {@code x[0] < x[i] ... < x[n].}  The x values are referred to as "knot points."
33   * <p>
34   * The value of the PolynomialSplineFunction at a point x that is greater than or equal to the smallest
35   * knot point and strictly less than the largest knot point is computed by finding the subinterval to which
36   * x belongs and computing the value of the corresponding polynomial at <code>x - x[i] </code> where
37   * <code>i</code> is the index of the subinterval.  See {@link PolynomialSplineFunction} for more details.
38   * </p>
39   * <p>
40   * The interpolating polynomials satisfy: <ol>
41   * <li>The value of the PolynomialSplineFunction at each of the input x values equals the
42   *  corresponding y value.</li>
43   * <li>Adjacent polynomials are equal through two derivatives at the knot points (i.e., adjacent polynomials
44   *  "match up" at the knot points, as do their first and second derivatives).</li>
45   * </ol>
46   * <p>
47   * The cubic spline interpolation algorithm implemented is as described in R.L. Burden, J.D. Faires,
48   * <u>Numerical Analysis</u>, 4th Ed., 1989, PWS-Kent, ISBN 0-53491-585-X, pp 126-131.
49   * </p>
50   *
51   */
52  public class SplineInterpolator implements UnivariateInterpolator {
53      /**
54       * Computes an interpolating function for the data set.
55       * @param x the arguments for the interpolation points
56       * @param y the values for the interpolation points
57       * @return a function which interpolates the data set
58       * @throws DimensionMismatchException if {@code x} and {@code y}
59       * have different sizes.
60       * @throws NonMonotonicSequenceException if {@code x} is not sorted in
61       * strict increasing order.
62       * @throws NumberIsTooSmallException if the size of {@code x} is smaller
63       * than 3.
64       */
65      public PolynomialSplineFunction interpolate(double x[], double y[])
66          throws DimensionMismatchException,
67                 NumberIsTooSmallException,
68                 NonMonotonicSequenceException {
69          if (x.length != y.length) {
70              throw new DimensionMismatchException(x.length, y.length);
71          }
72  
73          if (x.length < 3) {
74              throw new NumberIsTooSmallException(LocalizedFormats.NUMBER_OF_POINTS,
75                                                  x.length, 3, true);
76          }
77  
78          // Number of intervals.  The number of data points is n + 1.
79          final int n = x.length - 1;
80  
81          MathArrays.checkOrder(x);
82  
83          // Differences between knot points
84          final double h[] = new double[n];
85          for (int i = 0; i < n; i++) {
86              h[i] = x[i + 1] - x[i];
87          }
88  
89          final double mu[] = new double[n];
90          final double z[] = new double[n + 1];
91          mu[0] = 0d;
92          z[0] = 0d;
93          double g = 0;
94          for (int i = 1; i < n; i++) {
95              g = 2d * (x[i+1]  - x[i - 1]) - h[i - 1] * mu[i -1];
96              mu[i] = h[i] / g;
97              z[i] = (3d * (y[i + 1] * h[i - 1] - y[i] * (x[i + 1] - x[i - 1])+ y[i - 1] * h[i]) /
98                      (h[i - 1] * h[i]) - h[i - 1] * z[i - 1]) / g;
99          }
100 
101         // cubic spline coefficients --  b is linear, c quadratic, d is cubic (original y's are constants)
102         final double b[] = new double[n];
103         final double c[] = new double[n + 1];
104         final double d[] = new double[n];
105 
106         z[n] = 0d;
107         c[n] = 0d;
108 
109         for (int j = n -1; j >=0; j--) {
110             c[j] = z[j] - mu[j] * c[j + 1];
111             b[j] = (y[j + 1] - y[j]) / h[j] - h[j] * (c[j + 1] + 2d * c[j]) / 3d;
112             d[j] = (c[j + 1] - c[j]) / (3d * h[j]);
113         }
114 
115         final PolynomialFunction polynomials[] = new PolynomialFunction[n];
116         final double coefficients[] = new double[4];
117         for (int i = 0; i < n; i++) {
118             coefficients[0] = y[i];
119             coefficients[1] = b[i];
120             coefficients[2] = c[i];
121             coefficients[3] = d[i];
122             polynomials[i] = new PolynomialFunction(coefficients);
123         }
124 
125         return new PolynomialSplineFunction(x, polynomials);
126     }
127 }
