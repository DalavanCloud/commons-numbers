137     @Test
138     public void testSingleLink() {
139         final EuclideanIntegerPoint[] points = {
140                 new EuclideanIntegerPoint(new int[] {10, 10}), // A
141                 new EuclideanIntegerPoint(new int[] {12, 9}),
142                 new EuclideanIntegerPoint(new int[] {10, 8}),
143                 new EuclideanIntegerPoint(new int[] {8, 8}),
144                 new EuclideanIntegerPoint(new int[] {8, 6}),
145                 new EuclideanIntegerPoint(new int[] {7, 7}),
146                 new EuclideanIntegerPoint(new int[] {5, 6}),  // B
147                 new EuclideanIntegerPoint(new int[] {14, 8}), // C
148                 new EuclideanIntegerPoint(new int[] {7, 15}), // N - Noise, should not be present
149                 new EuclideanIntegerPoint(new int[] {17, 8}), // D - single-link connected to C should not be present
150 
151         };
152 
153         final DBSCANClusterer<EuclideanIntegerPoint> clusterer = new DBSCANClusterer<EuclideanIntegerPoint>(3, 3);
154         List<Cluster<EuclideanIntegerPoint>> clusters = clusterer.cluster(Arrays.asList(points));
155 
156         Assert.assertEquals(1, clusters.size());
157 
158         final List<EuclideanIntegerPoint> clusterOne =
159                 Arrays.asList(points[0], points[1], points[2], points[3], points[4], points[5], points[6], points[7]);
160         Assert.assertTrue(clusters.get(0).getPoints().containsAll(clusterOne));
161     }
