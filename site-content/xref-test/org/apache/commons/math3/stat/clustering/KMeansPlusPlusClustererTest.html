34      @Test
35      public void dimension2() {
36          KMeansPlusPlusClusterer<EuclideanIntegerPoint> transformer =
37              new KMeansPlusPlusClusterer<EuclideanIntegerPoint>(new Random(1746432956321l));
38          EuclideanIntegerPoint[] points = new EuclideanIntegerPoint[] {
39  
40                  // first expected cluster
41                  new EuclideanIntegerPoint(new int[] { -15,  3 }),
42                  new EuclideanIntegerPoint(new int[] { -15,  4 }),
43                  new EuclideanIntegerPoint(new int[] { -15,  5 }),
44                  new EuclideanIntegerPoint(new int[] { -14,  3 }),
45                  new EuclideanIntegerPoint(new int[] { -14,  5 }),
46                  new EuclideanIntegerPoint(new int[] { -13,  3 }),
47                  new EuclideanIntegerPoint(new int[] { -13,  4 }),
48                  new EuclideanIntegerPoint(new int[] { -13,  5 }),
49  
50                  // second expected cluster
51                  new EuclideanIntegerPoint(new int[] { -1,  0 }),
52                  new EuclideanIntegerPoint(new int[] { -1, -1 }),
53                  new EuclideanIntegerPoint(new int[] {  0, -1 }),
54                  new EuclideanIntegerPoint(new int[] {  1, -1 }),
55                  new EuclideanIntegerPoint(new int[] {  1, -2 }),
56  
57                  // third expected cluster
58                  new EuclideanIntegerPoint(new int[] { 13,  3 }),
59                  new EuclideanIntegerPoint(new int[] { 13,  4 }),
60                  new EuclideanIntegerPoint(new int[] { 14,  4 }),
61                  new EuclideanIntegerPoint(new int[] { 14,  7 }),
62                  new EuclideanIntegerPoint(new int[] { 16,  5 }),
63                  new EuclideanIntegerPoint(new int[] { 16,  6 }),
64                  new EuclideanIntegerPoint(new int[] { 17,  4 }),
65                  new EuclideanIntegerPoint(new int[] { 17,  7 })
66  
67          };
68          List<Cluster<EuclideanIntegerPoint>> clusters =
69              transformer.cluster(Arrays.asList(points), 3, 5, 10);
70  
71          Assert.assertEquals(3, clusters.size());
72          boolean cluster1Found = false;
73          boolean cluster2Found = false;
74          boolean cluster3Found = false;
75          for (Cluster<EuclideanIntegerPoint> cluster : clusters) {
76              int[] center = cluster.getCenter().getPoint();
77              if (center[0] < 0) {
78                  cluster1Found = true;
79                  Assert.assertEquals(8, cluster.getPoints().size());
80                  Assert.assertEquals(-14, center[0]);
81                  Assert.assertEquals( 4, center[1]);
82              } else if (center[1] < 0) {
83                  cluster2Found = true;
84                  Assert.assertEquals(5, cluster.getPoints().size());
85                  Assert.assertEquals( 0, center[0]);
86                  Assert.assertEquals(-1, center[1]);
87              } else {
88                  cluster3Found = true;
89                  Assert.assertEquals(8, cluster.getPoints().size());
90                  Assert.assertEquals(15, center[0]);
91                  Assert.assertEquals(5, center[1]);
92              }
93          }
94          Assert.assertTrue(cluster1Found);
95          Assert.assertTrue(cluster2Found);
96          Assert.assertTrue(cluster3Found);
97  
98      }

122     @Test
123     public void testCertainSpace() {
124         KMeansPlusPlusClusterer.EmptyClusterStrategy[] strategies = {
125             KMeansPlusPlusClusterer.EmptyClusterStrategy.LARGEST_VARIANCE,
126             KMeansPlusPlusClusterer.EmptyClusterStrategy.LARGEST_POINTS_NUMBER,
127             KMeansPlusPlusClusterer.EmptyClusterStrategy.FARTHEST_POINT
128         };
129         for (KMeansPlusPlusClusterer.EmptyClusterStrategy strategy : strategies) {
130             KMeansPlusPlusClusterer<EuclideanIntegerPoint> transformer =
131                 new KMeansPlusPlusClusterer<EuclideanIntegerPoint>(new Random(1746432956321l), strategy);
132             int numberOfVariables = 27;
133             // initialise testvalues
134             int position1 = 1;
135             int position2 = position1 + numberOfVariables;
136             int position3 = position2 + numberOfVariables;
137             int position4 = position3 + numberOfVariables;
138             // testvalues will be multiplied
139             int multiplier = 1000000;
140 
141             EuclideanIntegerPoint[] breakingPoints = new EuclideanIntegerPoint[numberOfVariables];
142             // define the space which will break the cluster algorithm
143             for (int i = 0; i < numberOfVariables; i++) {
144                 int points[] = { position1, position2, position3, position4 };
145                 // multiply the values
146                 for (int j = 0; j < points.length; j++) {
147                     points[j] *= multiplier;
148                 }
149                 EuclideanIntegerPoint euclideanIntegerPoint = new EuclideanIntegerPoint(points);
150                 breakingPoints[i] = euclideanIntegerPoint;
151                 position1 += numberOfVariables;
152                 position2 += numberOfVariables;
153                 position3 += numberOfVariables;
154                 position4 += numberOfVariables;
155             }
156 
157             for (int n = 2; n < 27; ++n) {
158                 List<Cluster<EuclideanIntegerPoint>> clusters =
159                     transformer.cluster(Arrays.asList(breakingPoints), n, 100);
160                 Assert.assertEquals(n, clusters.size());
161                 int sum = 0;
162                 for (Cluster<EuclideanIntegerPoint> cluster : clusters) {
163                     sum += cluster.getPoints().size();
164                 }
165                 Assert.assertEquals(numberOfVariables, sum);
166             }
167         }

171     /**
172      * A helper class for testSmallDistances(). This class is similar to EuclideanIntegerPoint, but
173      * it defines a different distanceFrom() method that tends to return distances less than 1.
174      */
175     private class CloseIntegerPoint implements Clusterable<CloseIntegerPoint> {
176         public CloseIntegerPoint(EuclideanIntegerPoint point) {
177             euclideanPoint = point;
178         }
179 
180         public double distanceFrom(CloseIntegerPoint p) {
181             return euclideanPoint.distanceFrom(p.euclideanPoint) * 0.001;
182         }
183 
184         public CloseIntegerPoint centroidOf(Collection<CloseIntegerPoint> p) {
185             Collection<EuclideanIntegerPoint> euclideanPoints =
186                 new ArrayList<EuclideanIntegerPoint>();
187             for (CloseIntegerPoint point : p) {
188                 euclideanPoints.add(point.euclideanPoint);
189             }
190             return new CloseIntegerPoint(euclideanPoint.centroidOf(euclideanPoints));
191         }
192 
193         @Override
194         public boolean equals(Object o) {
195             if (!(o instanceof CloseIntegerPoint)) {
196                 return false;
197             }
198             CloseIntegerPoint p = (CloseIntegerPoint) o;
199 
200             return euclideanPoint.equals(p.euclideanPoint);
201         }
202 
203         @Override
204         public int hashCode() {
205             return euclideanPoint.hashCode();
206         }
207 
208         private EuclideanIntegerPoint euclideanPoint;
209     }

211     /**
212      * Test points that are very close together. See issue MATH-546.
213      */
214     @Test
215     public void testSmallDistances() {
216         // Create a bunch of CloseIntegerPoints. Most are identical, but one is different by a
217         // small distance.
218         int[] repeatedArray = { 0 };
219         int[] uniqueArray = { 1 };
220         CloseIntegerPoint repeatedPoint =
221             new CloseIntegerPoint(new EuclideanIntegerPoint(repeatedArray));
222         CloseIntegerPoint uniquePoint =
223             new CloseIntegerPoint(new EuclideanIntegerPoint(uniqueArray));
224 
225         Collection<CloseIntegerPoint> points = new ArrayList<CloseIntegerPoint>();
226         final int NUM_REPEATED_POINTS = 10 * 1000;
227         for (int i = 0; i < NUM_REPEATED_POINTS; ++i) {
228             points.add(repeatedPoint);
229         }
230         points.add(uniquePoint);
231 
232         // Ask a KMeansPlusPlusClusterer to run zero iterations (i.e., to simply choose initial
233         // cluster centers).
234         final long RANDOM_SEED = 0;
235         final int NUM_CLUSTERS = 2;
236         final int NUM_ITERATIONS = 0;
237         KMeansPlusPlusClusterer<CloseIntegerPoint> clusterer =
238             new KMeansPlusPlusClusterer<CloseIntegerPoint>(new Random(RANDOM_SEED));
239         List<Cluster<CloseIntegerPoint>> clusters =
240             clusterer.cluster(points, NUM_CLUSTERS, NUM_ITERATIONS);
241 
242         // Check that one of the chosen centers is the unique point.
243         boolean uniquePointIsCenter = false;
244         for (Cluster<CloseIntegerPoint> cluster : clusters) {
245             if (cluster.getCenter().equals(uniquePoint)) {
246                 uniquePointIsCenter = true;
247             }
248         }
249         Assert.assertTrue(uniquePointIsCenter);
250     }
