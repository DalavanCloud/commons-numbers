55          Assert.assertEquals("min", MIN, StatUtils.min(values), TOLERANCE);
56          Assert.assertEquals("max", MAX, StatUtils.max(values), TOLERANCE);

59      @Test
60      public void testN0andN1Conditions() {
61          double[] values = new double[0];
62  
63          Assert.assertTrue(
64              "Mean of n = 0 set should be NaN",
65              Double.isNaN(StatUtils.mean(values)));
66          Assert.assertTrue(
67              "Variance of n = 0 set should be NaN",
68              Double.isNaN(StatUtils.variance(values)));
69  
70          values = new double[] { ONE };
71  
72          Assert.assertTrue(
73              "Mean of n = 1 set should be value of single item n1",
74              StatUtils.mean(values) == ONE);
75          Assert.assertTrue(
76              "Variance of n = 1 set should be zero",
77              StatUtils.variance(values) == 0);

418     public void testDifferenceStats() {
419         double sample1[] = {1d, 2d, 3d, 4d};
420         double sample2[] = {1d, 3d, 4d, 2d};
421         double diff[] = {0d, -1d, -1d, 2d};
422         double small[] = {1d, 4d};
423         double meanDifference = StatUtils.meanDifference(sample1, sample2);
424         Assert.assertEquals(StatUtils.sumDifference(sample1, sample2), StatUtils.sum(diff), TOLERANCE);
425         Assert.assertEquals(meanDifference, StatUtils.mean(diff), TOLERANCE);
426         Assert.assertEquals(StatUtils.varianceDifference(sample1, sample2, meanDifference),
427                 StatUtils.variance(diff), TOLERANCE);
428         try {
429             StatUtils.meanDifference(sample1, small);
430             Assert.fail("Expecting MathIllegalArgumentException");
431         } catch (MathIllegalArgumentException ex) {
432             // expected
433         }
434         try {
435             StatUtils.varianceDifference(sample1, small, meanDifference);
436             Assert.fail("Expecting MathIllegalArgumentException");
437         } catch (MathIllegalArgumentException ex) {
438             // expected
439         }
440         try {
441             double[] single = {1.0};
442             StatUtils.varianceDifference(single, single, meanDifference);
443             Assert.fail("Expecting MathIllegalArgumentException");
444         } catch (MathIllegalArgumentException ex) {
445             // expected
446         }
447     }

481     /**
482      * Run with 77 random values, assuming that the outcome has a mean of 0 and a standard deviation of 1 with a
483      * precision of 1E-10.
484      */
485 
486     @Test
487     public void testNormalize2() {
488         // create an sample with 77 values
489         int length = 77;
490         double sample[] = new double[length];
491         for (int i = 0; i < length; i++) {
492             sample[i] = FastMath.random();
493         }
494         // normalize this sample
495         double standardizedSample[] = StatUtils.normalize(sample);
496 
497         DescriptiveStatistics stats = new DescriptiveStatistics();
498         // Add the data from the array
499         for (int i = 0; i < length; i++) {
500             stats.addValue(standardizedSample[i]);
501         }
502         // the calculations do have a limited precision
503         double distance = 1E-10;
504         // check the mean an standard deviation
505         Assert.assertEquals(0.0, stats.getMean(), distance);
506         Assert.assertEquals(1.0, stats.getStandardDeviation(), distance);
507 
508     }

511     public void testMode() {
512         final double[] singleMode = {0, 1, 0, 2, 7, 11, 12};
513         final double[] modeSingle = StatUtils.mode(singleMode);
514         Assert.assertEquals(0, modeSingle[0], Double.MIN_VALUE);
515         Assert.assertEquals(1, modeSingle.length);
516 
517         final double[] twoMode = {0, 1, 2, 0, 2, 3, 7, 11};
518         final double[] modeDouble = StatUtils.mode(twoMode);
519         Assert.assertEquals(0, modeDouble[0], Double.MIN_VALUE);
520         Assert.assertEquals(2, modeDouble[1], Double.MIN_VALUE);
521         Assert.assertEquals(2, modeDouble.length);
522 
523         final double[] nanInfested = {0, 0, 0, Double.NaN, Double.NaN, Double.NaN, Double.NaN, 2, 2, 2, 3, 5};
524         final double[] modeNan = StatUtils.mode(nanInfested);
525         Assert.assertEquals(0, modeNan[0], Double.MIN_VALUE);
526         Assert.assertEquals(2, modeNan[1], Double.MIN_VALUE);
527         Assert.assertEquals(2, modeNan.length);
528 
529         final double[] infInfested = {0, 0, Double.POSITIVE_INFINITY, Double.POSITIVE_INFINITY,
530             Double.NEGATIVE_INFINITY, Double.NEGATIVE_INFINITY, 2, 2, 3, 5};
531         final double[] modeInf = StatUtils.mode(infInfested);
532         Assert.assertEquals(Double.NEGATIVE_INFINITY, modeInf[0], Double.MIN_VALUE);
533         Assert.assertEquals(0, modeInf[1], Double.MIN_VALUE);
534         Assert.assertEquals(2, modeInf[2], Double.MIN_VALUE);
535         Assert.assertEquals(Double.POSITIVE_INFINITY, modeInf[3], Double.MIN_VALUE);
536         Assert.assertEquals(4, modeInf.length);
537 
538         final double[] noData = {};
539         final double[] modeNodata = StatUtils.mode(noData);
540         Assert.assertEquals(0, modeNodata.length);
541 
542         final double[] nansOnly = {Double.NaN, Double.NaN};
543         final double[] modeNansOnly = StatUtils.mode(nansOnly);
544         Assert.assertEquals(0, modeNansOnly.length);
