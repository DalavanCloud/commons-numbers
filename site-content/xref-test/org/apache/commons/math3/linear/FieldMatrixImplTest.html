448     private void checkGetSubMatrix(FieldMatrix<Fraction> m, Fraction[][] reference,
449                                    int startRow, int endRow, int startColumn, int endColumn) {
450         try {
451             FieldMatrix<Fraction> sub = m.getSubMatrix(startRow, endRow, startColumn, endColumn);
452             if (reference != null) {
453                 Assert.assertEquals(new Array2DRowFieldMatrix<Fraction>(reference), sub);
454             } else {
455                 Assert.fail("Expecting OutOfRangeException or NotStrictlyPositiveException"
456                      + " or NumberIsTooSmallException or NoDataException");
457             }
458         } catch (OutOfRangeException e) {
459             if (reference != null) {
460                 throw e;
461             }
462         } catch (NotStrictlyPositiveException e) {
463             if (reference != null) {
464                 throw e;
465             }
466         } catch (NumberIsTooSmallException e) {
467             if (reference != null) {

1090     /** extracts the l  and u matrices from compact lu representation */
1091     protected void splitLU(FieldMatrix<Fraction> lu,
1092                            Fraction[][] lowerData,
1093                            Fraction[][] upperData) {
1094         if (!lu.isSquare()) {
1095             throw new NonSquareMatrixException(lu.getRowDimension(), lu.getColumnDimension());
1096         }
1097         if (lowerData.length != lowerData[0].length) {
1098             throw new DimensionMismatchException(lowerData.length, lowerData[0].length);
1099         }
1100         if (upperData.length != upperData[0].length) {
1101             throw new DimensionMismatchException(upperData.length, upperData[0].length);
1102         }
1103         if (lowerData.length != upperData.length) {
1104             throw new DimensionMismatchException(lowerData.length, upperData.length);
1105         }
1106         if (lowerData.length != lu.getRowDimension()) {
1107             throw new DimensionMismatchException(lowerData.length, lu.getRowDimension());
1108         }
1109         int n = lu.getRowDimension();
1110         for (int i = 0; i < n; i++) {
1111             for (int j = 0; j < n; j++) {
1112                 if (j < i) {
1113                     lowerData[i][j] = lu.getEntry(i, j);
1114                     upperData[i][j] = Fraction.ZERO;
1115                 } else if (i == j) {
1116                     lowerData[i][j] = Fraction.ONE;
1117                     upperData[i][j] = lu.getEntry(i, j);
1118                 } else {
1119                     lowerData[i][j] = Fraction.ZERO;
1120                     upperData[i][j] = lu.getEntry(i, j);
1121                 }
1122             }
1123         }
1124     }

1126     /** Returns the result of applying the given row permutation to the matrix */
1127     protected FieldMatrix<Fraction> permuteRows(FieldMatrix<Fraction> matrix, int[] permutation) {
1128         if (!matrix.isSquare()) {
1129             throw new NonSquareMatrixException(matrix.getRowDimension(),
1130                                                matrix.getColumnDimension());
1131         }
1132         if (matrix.getRowDimension() != permutation.length) {
1133             throw new DimensionMismatchException(matrix.getRowDimension(), permutation.length);
1134         }
1135         int n = matrix.getRowDimension();
1136         int m = matrix.getColumnDimension();
1137         Fraction out[][] = new Fraction[m][n];
1138         for (int i = 0; i < n; i++) {
1139             for (int j = 0; j < m; j++) {
1140                 out[i][j] = matrix.getEntry(permutation[i], j);
1141             }
1142         }
1143         return new Array2DRowFieldMatrix<Fraction>(out);
1144     }
