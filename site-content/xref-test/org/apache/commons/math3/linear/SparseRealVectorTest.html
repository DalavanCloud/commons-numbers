104     /* Check that the operations do not throw an exception (cf. MATH-645). */
105     @Test
106     public void testConcurrentModification() {
107         final RealVector u = new OpenMapRealVector(3, 1e-6);
108         u.setEntry(0, 1);
109         u.setEntry(1, 0);
110         u.setEntry(2, 2);
111 
112         final RealVector v1 = new OpenMapRealVector(3, 1e-6);
113         v1.setEntry(0, 0);
114         v1.setEntry(1, 3);
115         v1.setEntry(2, 0);
116 
117         u.ebeMultiply(v1);
118         u.ebeDivide(v1);
119     }
