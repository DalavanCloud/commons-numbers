60      private Map<Integer, Double> generate() {
61          Map<Integer, Double> map = new HashMap<Integer, Double>();
62          Random r = new Random();
63          for (int i = 0; i < 2000; ++i)
64              map.put(r.nextInt(), r.nextDouble());
65          return map;
66      }

68      private OpenIntToDoubleHashMap createFromJavaMap() {
69          OpenIntToDoubleHashMap map = new OpenIntToDoubleHashMap();
70          for (Map.Entry<Integer, Double> mapEntry : javaMap.entrySet()) {
71              map.put(mapEntry.getKey(), mapEntry.getValue());
72          }
73          return map;
74      }

98      private void assertPutAndGet(OpenIntToDoubleHashMap map, int mapSize,
99              Set<Integer> keysInMap) {
100         Assert.assertEquals(mapSize, map.size());
101         for (Map.Entry<Integer, Double> mapEntry : javaMap.entrySet()) {
102             map.put(mapEntry.getKey(), mapEntry.getValue());
103             if (!keysInMap.contains(mapEntry.getKey()))
104                 ++mapSize;
105             Assert.assertEquals(mapSize, map.size());
106             Assert.assertTrue(Precision.equals(mapEntry.getValue(), map.get(mapEntry.getKey()), 1));
107         }
108     }

203     /**
204      * Returns a map with at least 100 elements where each element is absent from javaMap.
205      */
206     private Map<Integer, Double> generateAbsent() {
207         Map<Integer, Double> generated = new HashMap<Integer, Double>();
208         do {
209             generated.putAll(generate());
210             for (Integer key : javaMap.keySet())
211                 generated.remove(key);
212         } while (generated.size() < 100);
213         return generated;
214     }

226     @Test
227     public void testContainsKey() {
228         OpenIntToDoubleHashMap map = createFromJavaMap();
229         for (Map.Entry<Integer, Double> mapEntry : javaMap.entrySet()) {
230             Assert.assertTrue(map.containsKey(mapEntry.getKey()));
231         }
232         for (Map.Entry<Integer, Double> mapEntry : generateAbsent().entrySet()) {
233             Assert.assertFalse(map.containsKey(mapEntry.getKey()));
234         }
235         for (Map.Entry<Integer, Double> mapEntry : javaMap.entrySet()) {
236             int key = mapEntry.getKey();
237             Assert.assertTrue(map.containsKey(key));
238             map.remove(key);
239             Assert.assertFalse(map.containsKey(key));
240         }
241     }

243     @Test
244     public void testIterator() {
245         OpenIntToDoubleHashMap map = createFromJavaMap();
246         OpenIntToDoubleHashMap.Iterator iterator = map.iterator();
247         for (int i = 0; i < map.size(); ++i) {
248             Assert.assertTrue(iterator.hasNext());
249             iterator.advance();
250             int key = iterator.key();
251             Assert.assertTrue(map.containsKey(key));
252             Assert.assertEquals(javaMap.get(key), map.get(key), 0);
253             Assert.assertEquals(javaMap.get(key), iterator.value(), 0);
254             Assert.assertTrue(javaMap.containsKey(key));
255         }
256         Assert.assertFalse(iterator.hasNext());
257         try {
258             iterator.advance();
259             Assert.fail("an exception should have been thrown");
260         } catch (NoSuchElementException nsee) {
261             // expected
262         }
263     }

278     /**
279      * Regression test for a bug in findInsertionIndex where the hashing in the second probing
280      * loop was inconsistent with the first causing duplicate keys after the right sequence
281      * of puts and removes.
282      */
283     @Test
284     public void testPutKeysWithCollisions() {
285         OpenIntToDoubleHashMap map = new OpenIntToDoubleHashMap();
286         int key1 = -1996012590;
287         double value1 = 1.0;
288         map.put(key1, value1);
289         int key2 = 835099822;
290         map.put(key2, value1);
291         int key3 = 1008859686;
292         map.put(key3, value1);
293         Assert.assertTrue(Precision.equals(value1, map.get(key3), 1));
294         Assert.assertEquals(3, map.size());
295 
296         map.remove(key2);
297         double value2 = 2.0;
298         map.put(key3, value2);
299         Assert.assertTrue(Precision.equals(value2, map.get(key3), 1));
300         Assert.assertEquals(2, map.size());
301     }
