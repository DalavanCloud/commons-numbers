18  package org.apache.commons.math3.distribution;
19  
20  import org.apache.commons.math3.exception.NumberIsTooLargeException;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  /**
25   * Test cases for UniformRealDistribution. See class javadoc for
26   * {@link RealDistributionAbstractTest} for further details.
27   */
28  public class UniformRealDistributionTest extends RealDistributionAbstractTest {
29  
30      // --- Override tolerance -------------------------------------------------
31  
32      @Override
33      public void setUp() {
34          super.setUp();
35          setTolerance(1e-4);
36      }
37  
38      //--- Implementations for abstract methods --------------------------------
39  
40      /** Creates the default uniform real distribution instance to use in tests. */
41      @Override
42      public UniformRealDistribution makeDistribution() {
43          return new UniformRealDistribution(-0.5, 1.25);
44      }
45  
46      /** Creates the default cumulative probability distribution test input values */
47      @Override
48      public double[] makeCumulativeTestPoints() {
49          return new double[] {-0.5001, -0.5, -0.4999, -0.25, -0.0001, 0.0,
50                               0.0001, 0.25, 1.0, 1.2499, 1.25, 1.2501};
51      }
52  
53      /** Creates the default cumulative probability density test expected values */
54      @Override
55      public double[] makeCumulativeTestValues() {
56          return new double[] {0.0, 0.0, 0.0001, 0.25/1.75, 0.4999/1.75,
57                               0.5/1.75, 0.5001/1.75, 0.75/1.75, 1.5/1.75,
58                               1.7499/1.75, 1.0, 1.0};
59      }
60  
61      /** Creates the default probability density test expected values */
62      @Override
63      public double[] makeDensityTestValues() {
64          double d = 1 / 1.75;
65          return new double[] {0, d, d, d, d, d, d, d, d, d, d, 0};
66      }
67  
68      //--- Additional test cases -----------------------------------------------
69  
70      /** Test lower bound getter. */
71      @Test
72      public void testGetLowerBound() {
73          UniformRealDistribution distribution = makeDistribution();
74          Assert.assertEquals(-0.5, distribution.getSupportLowerBound(), 0);
75      }
76  
77      /** Test upper bound getter. */
78      @Test
79      public void testGetUpperBound() {
80          UniformRealDistribution distribution = makeDistribution();
81          Assert.assertEquals(1.25, distribution.getSupportUpperBound(), 0);
82      }
83  
84      /** Test pre-condition for equal lower/upper bound. */
85      @Test(expected=NumberIsTooLargeException.class)
86      public void testPreconditions1() {
87          new UniformRealDistribution(0, 0);
88      }
89  
90      /** Test pre-condition for lower bound larger than upper bound. */
91      @Test(expected=NumberIsTooLargeException.class)
92      public void testPreconditions2() {
93          new UniformRealDistribution(1, 0);
94      }
95  
96      /** Test mean/variance. */
97      @Test
98      public void testMeanVariance() {
99          UniformRealDistribution dist;
100 
101         dist = new UniformRealDistribution(0, 1);
102         Assert.assertEquals(dist.getNumericalMean(), 0.5, 0);
103         Assert.assertEquals(dist.getNumericalVariance(), 1/12.0, 0);
104 
105         dist = new UniformRealDistribution(-1.5, 0.6);
106         Assert.assertEquals(dist.getNumericalMean(), -0.45, 0);
107         Assert.assertEquals(dist.getNumericalVariance(), 0.3675, 0);
108 
109         dist = new UniformRealDistribution(-0.5, 1.25);
110         Assert.assertEquals(dist.getNumericalMean(), 0.375, 0);
111         Assert.assertEquals(dist.getNumericalVariance(), 0.2552083333333333, 0);
112     }
113 
114     /**
115      * Check accuracy of analytical inverse CDF. Fails if a solver is used
116      * with the default accuracy.
117      */
118     @Test
119     public void testInverseCumulativeDistribution() {
120         UniformRealDistribution dist = new UniformRealDistribution(0, 1e-9);
121 
122         Assert.assertEquals(2.5e-10, dist.inverseCumulativeProbability(0.25), 0);
123     }
124 }
