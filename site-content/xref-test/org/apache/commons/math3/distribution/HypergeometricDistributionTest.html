121         verifyCumulativeProbabilities();

163     @Test
164     public void testPreconditions() {
165         try {
166             new HypergeometricDistribution(0, 3, 5);
167             Assert.fail("negative population size. NotStrictlyPositiveException expected");
168         } catch(NotStrictlyPositiveException ex) {
169             // Expected.
170         }
171         try {
172             new HypergeometricDistribution(5, -1, 5);
173             Assert.fail("negative number of successes. NotPositiveException expected");
174         } catch(NotPositiveException ex) {
175             // Expected.
176         }
177         try {
178             new HypergeometricDistribution(5, 3, -1);
179             Assert.fail("negative sample size. NotPositiveException expected");
180         } catch(NotPositiveException ex) {
181             // Expected.
182         }
183         try {
184             new HypergeometricDistribution(5, 6, 5);
185             Assert.fail("numberOfSuccesses > populationSize. NumberIsTooLargeException expected");
186         } catch(NumberIsTooLargeException ex) {
187             // Expected.
188         }
189         try {
190             new HypergeometricDistribution(5, 3, 6);
191             Assert.fail("sampleSize > populationSize. NumberIsTooLargeException expected");
192         } catch(NumberIsTooLargeException ex) {
193             // Expected.
194         }

234     private void testHypergeometricDistributionProbabilities(int populationSize, int sampleSize, int numberOfSucceses, double[][] data) {
235         HypergeometricDistribution dist = new HypergeometricDistribution(populationSize, numberOfSucceses, sampleSize);
236         for (int i = 0; i < data.length; ++i) {
237             int x = (int)data[i][0];
238             double pmf = data[i][1];
239             double actualPmf = dist.probability(x);
240             TestUtils.assertRelativelyEquals("Expected equals for <"+x+"> pmf",pmf, actualPmf, 1.0e-9);
241 
242             double cdf = data[i][2];
243             double actualCdf = dist.cumulativeProbability(x);
244             TestUtils.assertRelativelyEquals("Expected equals for <"+x+"> cdf",cdf, actualCdf, 1.0e-9);
245 
246             double cdf1 = data[i][3];
247             double actualCdf1 = dist.upperCumulativeProbability(x);
248             TestUtils.assertRelativelyEquals("Expected equals for <"+x+"> cdf1",cdf1, actualCdf1, 1.0e-9);
249         }
250     }

280     @Test
281     public void testMoments() {
282         final double tol = 1e-9;
283         HypergeometricDistribution dist;
284 
285         dist = new HypergeometricDistribution(1500, 40, 100);
286         Assert.assertEquals(dist.getNumericalMean(), 40d * 100d / 1500d, tol);
287         Assert.assertEquals(dist.getNumericalVariance(), ( 100d * 40d * (1500d - 100d) * (1500d - 40d) ) / ( (1500d * 1500d * 1499d) ), tol);
288 
289         dist = new HypergeometricDistribution(3000, 55, 200);
290         Assert.assertEquals(dist.getNumericalMean(), 55d * 200d / 3000d, tol);
291         Assert.assertEquals(dist.getNumericalVariance(), ( 200d * 55d * (3000d - 200d) * (3000d - 55d) ) / ( (3000d * 3000d * 2999d) ), tol);
292     }

294     @Test
295     public void testMath644() {
296         int N = 14761461;  // population
297         int m = 1035;      // successes in population
298         int n = 1841;      // number of trials
299 
300         int k = 0;
301         final HypergeometricDistribution dist = new HypergeometricDistribution(N, m, n);
302 
303         Assert.assertTrue(Precision.compareTo(1.0, dist.upperCumulativeProbability(k), 1) == 0);
304         Assert.assertTrue(Precision.compareTo(dist.cumulativeProbability(k), 0.0, 1) > 0);
305 
306         // another way to calculate the upper cumulative probability
307         double upper = 1.0 - dist.cumulativeProbability(k) + dist.probability(k);
308         Assert.assertTrue(Precision.compareTo(1.0, upper, 1) == 0);
309     }
