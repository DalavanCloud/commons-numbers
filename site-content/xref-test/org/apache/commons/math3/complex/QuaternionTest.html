105     @Test
106     public final void testProductQuaternionQuaternion() {
107 
108         // Case : analytic test case
109 
110         final Quaternion qA = new Quaternion(1, 0.5, -3, 4);
111         final Quaternion qB = new Quaternion(6, 2, 1, -9);
112         final Quaternion qResult = Quaternion.multiply(qA, qB);
113 
114         Assert.assertEquals(44, qResult.getQ0(), EPS);
115         Assert.assertEquals(28, qResult.getQ1(), EPS);
116         Assert.assertEquals(-4.5, qResult.getQ2(), EPS);
117         Assert.assertEquals(21.5, qResult.getQ3(), EPS);
118 
119         // comparison with the result given by the formula :
120         // qResult = (scalarA * scalarB - vectorA . vectorB) + (scalarA * vectorB + scalarB * vectorA + vectorA ^
121         // vectorB)
122 
123         final Vector3D vectorA = new Vector3D(qA.getVectorPart());
124         final Vector3D vectorB = new Vector3D(qB.getVectorPart());
125         final Vector3D vectorResult = new Vector3D(qResult.getVectorPart());
126 
127         final double scalarPartRef = qA.getScalarPart() * qB.getScalarPart() - Vector3D.dotProduct(vectorA, vectorB);
128 
129         Assert.assertEquals(scalarPartRef, qResult.getScalarPart(), EPS);
130 
131         final Vector3D vectorPartRef = ((vectorA.scalarMultiply(qB.getScalarPart())).add(vectorB.scalarMultiply(qA
132                 .getScalarPart()))).add(Vector3D.crossProduct(vectorA, vectorB));
133         final double norm = (vectorResult.subtract(vectorPartRef)).getNorm();
134 
135         Assert.assertEquals(0, norm, EPS);
136 
137         // Conjugate of the product of two quaternions and product of their conjugates :
138         // Conj(qA * qB) = Conj(qB) * Conj(qA)
139 
140         final Quaternion conjugateOfProduct = qB.getConjugate().multiply(qA.getConjugate());
141         final Quaternion productOfConjugate = (qA.multiply(qB)).getConjugate();
142 
143         Assert.assertEquals(conjugateOfProduct.getQ0(), productOfConjugate.getQ0(), EPS);
144         Assert.assertEquals(conjugateOfProduct.getQ1(), productOfConjugate.getQ1(), EPS);
145         Assert.assertEquals(conjugateOfProduct.getQ2(), productOfConjugate.getQ2(), EPS);
146         Assert.assertEquals(conjugateOfProduct.getQ3(), productOfConjugate.getQ3(), EPS);
147     }

149     @Test
150     public final void testProductQuaternionVector() {
151 
152         // Case : Product between a vector and a quaternion : QxV
153 
154         final Quaternion quaternion = new Quaternion(4, 7, -1, 2);
155         final double[] vector = {2.0, 1.0, 3.0};
156         final Quaternion qResultQxV = Quaternion.multiply(quaternion, new Quaternion(vector));
157 
158         Assert.assertEquals(-19, qResultQxV.getQ0(), EPS);
159         Assert.assertEquals(3, qResultQxV.getQ1(), EPS);
160         Assert.assertEquals(-13, qResultQxV.getQ2(), EPS);
161         Assert.assertEquals(21, qResultQxV.getQ3(), EPS);
162 
163         // comparison with the result given by the formula :
164         // qResult = (- vectorQ . vector) + (scalarQ * vector + vectorQ ^ vector)
165 
166         final double[] vectorQ = quaternion.getVectorPart();
167         final double[] vectorResultQxV = qResultQxV.getVectorPart();
168 
169         final double scalarPartRefQxV = -Vector3D.dotProduct(new Vector3D(vectorQ), new Vector3D(vector));
170         Assert.assertEquals(scalarPartRefQxV, qResultQxV.getScalarPart(), EPS);
171 
172         final Vector3D vectorPartRefQxV = (new Vector3D(vector).scalarMultiply(quaternion.getScalarPart())).add(Vector3D
173                 .crossProduct(new Vector3D(vectorQ), new Vector3D(vector)));
174         final double normQxV = (new Vector3D(vectorResultQxV).subtract(vectorPartRefQxV)).getNorm();
175         Assert.assertEquals(0, normQxV, EPS);
176 
177         // Case : Product between a vector and a quaternion : VxQ
178 
179         final Quaternion qResultVxQ = Quaternion.multiply(new Quaternion(vector), quaternion);
180 
181         Assert.assertEquals(-19, qResultVxQ.getQ0(), EPS);
182         Assert.assertEquals(13, qResultVxQ.getQ1(), EPS);
183         Assert.assertEquals(21, qResultVxQ.getQ2(), EPS);
184         Assert.assertEquals(3, qResultVxQ.getQ3(), EPS);
185 
186         final double[] vectorResultVxQ = qResultVxQ.getVectorPart();
187 
188         // comparison with the result given by the formula :
189         // qResult = (- vector . vectorQ) + (scalarQ * vector + vector ^ vectorQ)
190 
191         final double scalarPartRefVxQ = -Vector3D.dotProduct(new Vector3D(vectorQ), new Vector3D(vector));
192         Assert.assertEquals(scalarPartRefVxQ, qResultVxQ.getScalarPart(), EPS);
193 
194         final Vector3D vectorPartRefVxQ = (new Vector3D(vector).scalarMultiply(quaternion.getScalarPart())).add(Vector3D
195                 .crossProduct(new Vector3D(vector), new Vector3D(vectorQ)));
196         final double normVxQ = (new Vector3D(vectorResultVxQ).subtract(vectorPartRefVxQ)).getNorm();
197         Assert.assertEquals(0, normVxQ, EPS);
198     }

400     @Test
401     public final void testPolarForm() {
402         final Random r = new Random(48);
403         final int numberOfTrials = 1000;
404         for (int i = 0; i < numberOfTrials; i++) {
405             final Quaternion q = new Quaternion(2 * (r.nextDouble() - 0.5), 2 * (r.nextDouble() - 0.5),
406                                                 2 * (r.nextDouble() - 0.5), 2 * (r.nextDouble() - 0.5));
407             final Quaternion qP = q.getPositivePolarForm();
408 
409             Assert.assertTrue(qP.isUnitQuaternion(COMPARISON_EPS));
410             Assert.assertTrue(qP.getQ0() >= 0);
411 
412             final Rotation rot = new Rotation(q.getQ0(), q.getQ1(), q.getQ2(), q.getQ3(), true);
413             final Rotation rotP = new Rotation(qP.getQ0(), qP.getQ1(), qP.getQ2(), qP.getQ3(), true);
414 
415             Assert.assertEquals(rot.getAngle(), rotP.getAngle(), COMPARISON_EPS);
416             Assert.assertEquals(rot.getAxis(RotationConvention.VECTOR_OPERATOR).getX(),
417                                 rot.getAxis(RotationConvention.VECTOR_OPERATOR).getX(),
418                                 COMPARISON_EPS);
419             Assert.assertEquals(rot.getAxis(RotationConvention.VECTOR_OPERATOR).getY(),
420                                 rot.getAxis(RotationConvention.VECTOR_OPERATOR).getY(),
421                                 COMPARISON_EPS);
422             Assert.assertEquals(rot.getAxis(RotationConvention.VECTOR_OPERATOR).getZ(),
423                                 rot.getAxis(RotationConvention.VECTOR_OPERATOR).getZ(),
424                                 COMPARISON_EPS);
425         }

428     @Test
429     public final void testGetInverse() {
430         final Quaternion q = new Quaternion(1.5, 4, 2, -2.5);
431 
432         final Quaternion inverseQ = q.getInverse();
433         Assert.assertEquals(1.5 / 28.5, inverseQ.getQ0(), 0);
434         Assert.assertEquals(-4.0 / 28.5, inverseQ.getQ1(), 0);
435         Assert.assertEquals(-2.0 / 28.5, inverseQ.getQ2(), 0);
436         Assert.assertEquals(2.5 / 28.5, inverseQ.getQ3(), 0);
437 
438         final Quaternion product = Quaternion.multiply(inverseQ, q);
439         Assert.assertEquals(1, product.getQ0(), EPS);
440         Assert.assertEquals(0, product.getQ1(), EPS);
441         Assert.assertEquals(0, product.getQ2(), EPS);
442         Assert.assertEquals(0, product.getQ3(), EPS);
443 
444         final Quaternion qNul = new Quaternion(0, 0, 0, 0);
445         try {
446             final Quaternion inverseQNul = qNul.getInverse();
447             Assert.fail("expecting ZeroException but got : " + inverseQNul);
448         } catch (ZeroException ex) {
449             // expected
450         }
