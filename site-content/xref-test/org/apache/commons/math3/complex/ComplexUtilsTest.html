18  package org.apache.commons.math3.complex;
19  
20  import org.apache.commons.math3.TestUtils;
21  import org.apache.commons.math3.util.FastMath;
22  import org.junit.Test;
23  import org.junit.Assert;
24  
25  /**
26   */
27  public class ComplexUtilsTest {
28  
29      private double inf = Double.POSITIVE_INFINITY;
30      private double negInf = Double.NEGATIVE_INFINITY;
31      private double nan = Double.NaN;
32      private double pi = FastMath.PI;
33  
34      private Complex negInfInf = new Complex(negInf, inf);
35      private Complex infNegInf = new Complex(inf, negInf);
36      private Complex infInf = new Complex(inf, inf);
37      private Complex negInfNegInf = new Complex(negInf, negInf);
38      private Complex infNaN = new Complex(inf, nan);
39  
40      @Test
41      public void testPolar2Complex() {
42          TestUtils.assertEquals(Complex.ONE,
43                  ComplexUtils.polar2Complex(1, 0), 10e-12);
44          TestUtils.assertEquals(Complex.ZERO,
45                  ComplexUtils.polar2Complex(0, 1), 10e-12);
46          TestUtils.assertEquals(Complex.ZERO,
47                  ComplexUtils.polar2Complex(0, -1), 10e-12);
48          TestUtils.assertEquals(Complex.I,
49                  ComplexUtils.polar2Complex(1, pi/2), 10e-12);
50          TestUtils.assertEquals(Complex.I.negate(),
51                  ComplexUtils.polar2Complex(1, -pi/2), 10e-12);
52          double r = 0;
53          for (int i = 0; i < 5; i++) {
54            r += i;
55            double theta = 0;
56            for (int j =0; j < 20; j++) {
57                theta += pi / 6;
58                TestUtils.assertEquals(altPolar(r, theta),
59                        ComplexUtils.polar2Complex(r, theta), 10e-12);
60            }
61            theta = -2 * pi;
62            for (int j =0; j < 20; j++) {
63                theta -= pi / 6;
64                TestUtils.assertEquals(altPolar(r, theta),
65                        ComplexUtils.polar2Complex(r, theta), 10e-12);
66            }
67          }
68      }
69  
70      protected Complex altPolar(double r, double theta) {
71          return Complex.I.multiply(new Complex(theta, 0)).exp().multiply(new Complex(r, 0));
72      }
73  
74      @Test(expected=IllegalArgumentException.class)
75      public void testPolar2ComplexIllegalModulus() {
76          ComplexUtils.polar2Complex(-1, 0);
77      }
78  
79      @Test
80      public void testPolar2ComplexNaN() {
81          TestUtils.assertSame(Complex.NaN, ComplexUtils.polar2Complex(nan, 1));
82          TestUtils.assertSame(Complex.NaN, ComplexUtils.polar2Complex(1, nan));
83          TestUtils.assertSame(Complex.NaN,
84                  ComplexUtils.polar2Complex(nan, nan));
85      }
86  
87      @Test
88      public void testPolar2ComplexInf() {
89          TestUtils.assertSame(Complex.NaN, ComplexUtils.polar2Complex(1, inf));
90          TestUtils.assertSame(Complex.NaN,
91                  ComplexUtils.polar2Complex(1, negInf));
92          TestUtils.assertSame(Complex.NaN, ComplexUtils.polar2Complex(inf, inf));
93          TestUtils.assertSame(Complex.NaN,
94                  ComplexUtils.polar2Complex(inf, negInf));
95          TestUtils.assertSame(infInf, ComplexUtils.polar2Complex(inf, pi/4));
96          TestUtils.assertSame(infNaN, ComplexUtils.polar2Complex(inf, 0));
97          TestUtils.assertSame(infNegInf, ComplexUtils.polar2Complex(inf, -pi/4));
98          TestUtils.assertSame(negInfInf, ComplexUtils.polar2Complex(inf, 3*pi/4));
99          TestUtils.assertSame(negInfNegInf, ComplexUtils.polar2Complex(inf, 5*pi/4));
100     }
101 
102     @Test
103     public void testConvertToComplex() {
104         final double[] real = new double[] { negInf, -123.45, 0, 1, 234.56, pi, inf };
105         final Complex[] complex = ComplexUtils.convertToComplex(real);
106 
107         for (int i = 0; i < real.length; i++) {
108             Assert.assertEquals(real[i], complex[i].getReal(), 0d);
109         }
110     }
111 }
