38      @Test
39      public void testEquator() {
40          Circle circle = new Circle(new Vector3D(0, 0, 1000), 1.0e-10).copySelf();
41          Assert.assertEquals(Vector3D.PLUS_K, circle.getPole());
42          Assert.assertEquals(1.0e-10, circle.getTolerance(), 1.0e-20);
43          circle.revertSelf();
44          Assert.assertEquals(Vector3D.MINUS_K, circle.getPole());
45          Assert.assertEquals(Vector3D.PLUS_K, circle.getReverse().getPole());
46          Assert.assertEquals(Vector3D.MINUS_K, circle.getPole());
47      }

49      @Test
50      public void testXY() {
51          Circle circle = new Circle(new S2Point(1.2, 2.5), new S2Point(-4.3, 0), 1.0e-10);
52          Assert.assertEquals(0.0, circle.getPointAt(0).distance(circle.getXAxis()), 1.0e-10);
53          Assert.assertEquals(0.0, circle.getPointAt(0.5 * FastMath.PI).distance(circle.getYAxis()), 1.0e-10);
54          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(circle.getXAxis(), circle.getYAxis()), 1.0e-10);
55          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(circle.getXAxis(), circle.getPole()), 1.0e-10);
56          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(circle.getPole(), circle.getYAxis()), 1.0e-10);
57          Assert.assertEquals(0.0,
58                              circle.getPole().distance(Vector3D.crossProduct(circle.getXAxis(), circle.getYAxis())),
59                              1.0e-10);
60      }

62      @Test
63      public void testReverse() {
64          Circle circle = new Circle(new S2Point(1.2, 2.5), new S2Point(-4.3, 0), 1.0e-10);
65          Circle reversed = circle.getReverse();
66          Assert.assertEquals(0.0, reversed.getPointAt(0).distance(reversed.getXAxis()), 1.0e-10);
67          Assert.assertEquals(0.0, reversed.getPointAt(0.5 * FastMath.PI).distance(reversed.getYAxis()), 1.0e-10);
68          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(reversed.getXAxis(), reversed.getYAxis()), 1.0e-10);
69          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(reversed.getXAxis(), reversed.getPole()), 1.0e-10);
70          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(reversed.getPole(), reversed.getYAxis()), 1.0e-10);
71          Assert.assertEquals(0.0,
72                              reversed.getPole().distance(Vector3D.crossProduct(reversed.getXAxis(), reversed.getYAxis())),
73                              1.0e-10);
74  
75          Assert.assertEquals(0, Vector3D.angle(circle.getXAxis(), reversed.getXAxis()), 1.0e-10);
76          Assert.assertEquals(FastMath.PI, Vector3D.angle(circle.getYAxis(), reversed.getYAxis()), 1.0e-10);
77          Assert.assertEquals(FastMath.PI, Vector3D.angle(circle.getPole(), reversed.getPole()), 1.0e-10);
78  
79          Assert.assertTrue(circle.sameOrientationAs(circle));
80          Assert.assertFalse(circle.sameOrientationAs(reversed));
81  
82      }

84      @Test
85      public void testPhase() {
86          Circle circle = new Circle(new S2Point(1.2, 2.5), new S2Point(-4.3, 0), 1.0e-10);
87          Vector3D p = new Vector3D(1, 2, -4);
88          Vector3D samePhase = circle.getPointAt(circle.getPhase(p));
89          Assert.assertEquals(0.0,
90                              Vector3D.angle(Vector3D.crossProduct(circle.getPole(), p),
91                                             Vector3D.crossProduct(circle.getPole(), samePhase)),
92                              1.0e-10);
93          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(circle.getPole(), samePhase), 1.0e-10);
94          Assert.assertEquals(circle.getPhase(p), circle.getPhase(samePhase), 1.0e-10);
95          Assert.assertEquals(0.0, circle.getPhase(circle.getXAxis()), 1.0e-10);
96          Assert.assertEquals(0.5 * FastMath.PI, circle.getPhase(circle.getYAxis()), 1.0e-10);
97  
98      }

109     @Test
110     public void testSpace() {
111         Circle circle = new Circle(new S2Point(1.2, 2.5), new S2Point(-4.3, 0), 1.0e-10);
112         for (double alpha = 0; alpha < MathUtils.TWO_PI; alpha += 0.1) {
113             Vector3D p = new Vector3D(FastMath.cos(alpha), circle.getXAxis(),
114                                       FastMath.sin(alpha), circle.getYAxis());
115             Vector3D q = circle.toSpace(new S1Point(alpha)).getVector();
116             Assert.assertEquals(0.0, p.distance(q), 1.0e-10);
117             Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(circle.getPole(), q), 1.0e-10);
118         }

133     @Test
134     public void testInsideArc() {
135         RandomGenerator random = new Well1024a(0xbfd34e92231bbcfel);
136         UnitSphereRandomVectorGenerator sphRandom = new UnitSphereRandomVectorGenerator(3, random);
137         for (int i = 0; i < 100; ++i) {
138             Circle c1 = new Circle(new Vector3D(sphRandom.nextVector()), 1.0e-10);
139             Circle c2 = new Circle(new Vector3D(sphRandom.nextVector()), 1.0e-10);
140             checkArcIsInside(c1, c2);
141             checkArcIsInside(c2, c1);
142         }
143     }
144 
145     private void checkArcIsInside(final Circle arcCircle, final Circle otherCircle) {
146         Arc arc = arcCircle.getInsideArc(otherCircle);
147         Assert.assertEquals(FastMath.PI, arc.getSize(), 1.0e-10);
148         for (double alpha = arc.getInf(); alpha < arc.getSup(); alpha += 0.1) {
149             Assert.assertTrue(otherCircle.getOffset(arcCircle.getPointAt(alpha)) <= 2.0e-15);
150         }
151         for (double alpha = arc.getSup(); alpha < arc.getInf() + MathUtils.TWO_PI; alpha += 0.1) {
152             Assert.assertTrue(otherCircle.getOffset(arcCircle.getPointAt(alpha)) >= -2.0e-15);
153         }
154     }

156     @Test
157     public void testTransform() {
158         RandomGenerator random = new Well1024a(0x16992fc4294bf2f1l);
159         UnitSphereRandomVectorGenerator sphRandom = new UnitSphereRandomVectorGenerator(3, random);
160         for (int i = 0; i < 100; ++i) {
161 
162             Rotation r = new Rotation(new Vector3D(sphRandom.nextVector()),
163                                       FastMath.PI * random.nextDouble(),
164                                       RotationConvention.VECTOR_OPERATOR);
165             Transform<Sphere2D, Sphere1D> t = Circle.getTransform(r);
166 
167             S2Point  p = new S2Point(new Vector3D(sphRandom.nextVector()));
168             S2Point tp = (S2Point) t.apply(p);
169             Assert.assertEquals(0.0, r.applyTo(p.getVector()).distance(tp.getVector()), 1.0e-10);
170 
171             Circle  c = new Circle(new Vector3D(sphRandom.nextVector()), 1.0e-10);
172             Circle tc = (Circle) t.apply(c);
173             Assert.assertEquals(0.0, r.applyTo(c.getPole()).distance(tc.getPole()),   1.0e-10);
174             Assert.assertEquals(0.0, r.applyTo(c.getXAxis()).distance(tc.getXAxis()), 1.0e-10);
175             Assert.assertEquals(0.0, r.applyTo(c.getYAxis()).distance(tc.getYAxis()), 1.0e-10);
176             Assert.assertEquals(c.getTolerance(), ((Circle) t.apply(c)).getTolerance(), 1.0e-10);
177 
178             SubLimitAngle  sub = new LimitAngle(new S1Point(MathUtils.TWO_PI * random.nextDouble()),
179                                                 random.nextBoolean(), 1.0e-10).wholeHyperplane();
180             Vector3D psub = c.getPointAt(((LimitAngle) sub.getHyperplane()).getLocation().getAlpha());
181             SubLimitAngle tsub = (SubLimitAngle) t.apply(sub, c, tc);
182             Vector3D ptsub = tc.getPointAt(((LimitAngle) tsub.getHyperplane()).getLocation().getAlpha());
183             Assert.assertEquals(0.0, r.applyTo(psub).distance(ptsub), 1.0e-10);
184 
185         }
186     }
