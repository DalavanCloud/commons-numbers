30      @Test
31      public void testEndPoints() throws MathIllegalArgumentException {
32          Vector3D p1 = new Vector3D(-1, -7, 2);
33          Vector3D p2 = new Vector3D(7, -1, 0);
34          Segment segment = new Segment(p1, p2, new Line(p1, p2, 1.0e-10));
35          SubLine sub = new SubLine(segment);
36          List<Segment> segments = sub.getSegments();
37          Assert.assertEquals(1, segments.size());
38          Assert.assertEquals(0.0, new Vector3D(-1, -7, 2).distance(segments.get(0).getStart()), 1.0e-10);
39          Assert.assertEquals(0.0, new Vector3D( 7, -1, 0).distance(segments.get(0).getEnd()), 1.0e-10);
40      }

42      @Test
43      public void testNoEndPoints() throws MathIllegalArgumentException {
44          SubLine wholeLine = new Line(new Vector3D(-1, 7, 2), new Vector3D(7, 1, 0), 1.0e-10).wholeLine();
45          List<Segment> segments = wholeLine.getSegments();
46          Assert.assertEquals(1, segments.size());
47          Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getX()) &&
48                            segments.get(0).getStart().getX() < 0);
49          Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getY()) &&
50                            segments.get(0).getStart().getY() > 0);
51          Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getZ()) &&
52                            segments.get(0).getStart().getZ() > 0);
53          Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getX()) &&
54                            segments.get(0).getEnd().getX() > 0);
55          Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getY()) &&
56                            segments.get(0).getEnd().getY() < 0);
57          Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getZ()) &&
58                            segments.get(0).getEnd().getZ() < 0);
59      }

78      @Test
79      public void testHalfInfiniteNeg() throws MathIllegalArgumentException {
80          SubLine empty = new SubLine(new Line(new Vector3D(-1, -7, 2), new Vector3D(7, -1, -2), 1.0e-10),
81                                      new IntervalsSet(Double.NEGATIVE_INFINITY, 0.0, 1.0e-10));
82          List<Segment> segments = empty.getSegments();
83          Assert.assertEquals(1, segments.size());
84          Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getX()) &&
85                            segments.get(0).getStart().getX() < 0);
86          Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getY()) &&
87                            segments.get(0).getStart().getY() < 0);
88          Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getZ()) &&
89                            segments.get(0).getStart().getZ() > 0);
90          Assert.assertEquals(0.0, new Vector3D(3, -4, 0).distance(segments.get(0).getEnd()), 1.0e-10);
91      }

93      @Test
94      public void testHalfInfinitePos() throws MathIllegalArgumentException {
95          SubLine empty = new SubLine(new Line(new Vector3D(-1, -7, 2), new Vector3D(7, -1, -2), 1.0e-10),
96                                      new IntervalsSet(0.0, Double.POSITIVE_INFINITY, 1.0e-10));
97          List<Segment> segments = empty.getSegments();
98          Assert.assertEquals(1, segments.size());
99          Assert.assertEquals(0.0, new Vector3D(3, -4, 0).distance(segments.get(0).getStart()), 1.0e-10);
100         Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getX()) &&
101                           segments.get(0).getEnd().getX() > 0);
102         Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getY()) &&
103                           segments.get(0).getEnd().getY() > 0);
104         Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getZ()) &&
105                           segments.get(0).getEnd().getZ() < 0);

108     @Test
109     public void testIntersectionInsideInside() throws MathIllegalArgumentException {
110         SubLine sub1 = new SubLine(new Vector3D(1, 1, 1), new Vector3D(3, 1, 1), 1.0e-10);
111         SubLine sub2 = new SubLine(new Vector3D(2, 0, 0), new Vector3D(2, 2, 2), 1.0e-10);
112         Assert.assertEquals(0.0, new Vector3D(2, 1, 1).distance(sub1.intersection(sub2, true)),  1.0e-12);
113         Assert.assertEquals(0.0, new Vector3D(2, 1, 1).distance(sub1.intersection(sub2, false)), 1.0e-12);
114     }
