41    @Test
42    public void testMissedEndEvent()
43        throws DimensionMismatchException, NumberIsTooSmallException,
44               MaxCountExceededException, NoBracketingException {
45        final double   t0     = 1878250320.0000029;
46        final double   tEvent = 1878250379.9999986;
47        final double[] k  = { 1.0e-4, 1.0e-5, 1.0e-6 };
48        FirstOrderDifferentialEquations ode = new FirstOrderDifferentialEquations() {
49  
50            public int getDimension() {
51                return k.length;
52            }
53  
54            public void computeDerivatives(double t, double[] y, double[] yDot) {
55                for (int i = 0; i < y.length; ++i) {
56                    yDot[i] = k[i] * y[i];
57                }
58            }
59        };
60  
61        DormandPrince853Integrator integrator = new DormandPrince853Integrator(0.0, 100.0,
62                                                                               1.0e-10, 1.0e-10);
63  
64        double[] y0   = new double[k.length];
65        for (int i = 0; i < y0.length; ++i) {
66            y0[i] = i + 1;
67        }
68        double[] y    = new double[k.length];
69  
70        integrator.setInitialStepSize(60.0);
71        double finalT = integrator.integrate(ode, t0, y0, tEvent, y);
72        Assert.assertEquals(tEvent, finalT, 5.0e-6);
73        for (int i = 0; i < y.length; ++i) {
74            Assert.assertEquals(y0[i] * FastMath.exp(k[i] * (finalT - t0)), y[i], 1.0e-9);
75        }
76  
77        integrator.setInitialStepSize(60.0);
78        integrator.addEventHandler(new EventHandler() {
79  
80            public void init(double t0, double[] y0, double t) {
81            }
82  
83            public void resetState(double t, double[] y) {
84            }
85  
86            public double g(double t, double[] y) {
87                return t - tEvent;
88            }
89  
90            public Action eventOccurred(double t, double[] y, boolean increasing) {
91                Assert.assertEquals(tEvent, t, 5.0e-6);
92                return Action.CONTINUE;
93            }
94        }, Double.POSITIVE_INFINITY, 1.0e-20, 100);
95        finalT = integrator.integrate(ode, t0, y0, tEvent + 120, y);
96        Assert.assertEquals(tEvent + 120, finalT, 5.0e-6);
97        for (int i = 0; i < y.length; ++i) {
98            Assert.assertEquals(y0[i] * FastMath.exp(k[i] * (finalT - t0)), y[i], 1.0e-9);
99        }
100 
101   }

152   @Test
153   public void testIncreasingTolerance()
154       throws DimensionMismatchException, NumberIsTooSmallException,
155              MaxCountExceededException, NoBracketingException {
156 
157     int previousCalls = Integer.MAX_VALUE;
158     AdaptiveStepsizeIntegrator integ =
159         new DormandPrince853Integrator(0, Double.POSITIVE_INFINITY,
160                                        Double.NaN, Double.NaN);
161     for (int i = -12; i < -2; ++i) {
162       TestProblem1 pb = new TestProblem1();
163       double minStep = 0;
164       double maxStep = pb.getFinalTime() - pb.getInitialTime();
165       double scalAbsoluteTolerance = FastMath.pow(10.0, i);
166       double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
167       integ.setStepSizeControl(minStep, maxStep, scalAbsoluteTolerance, scalRelativeTolerance);
168 
169       TestProblemHandler handler = new TestProblemHandler(pb, integ);
170       integ.addStepHandler(handler);
171       integ.integrate(pb,
172                       pb.getInitialTime(), pb.getInitialState(),
173                       pb.getFinalTime(), new double[pb.getDimension()]);
174 
175       // the 1.3 factor is only valid for this test
176       // and has been obtained from trial and error
177       // there is no general relation between local and global errors
178       Assert.assertTrue(handler.getMaximalValueError() < (1.3 * scalAbsoluteTolerance));
179       Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-12);
180 
181       int calls = pb.getCalls();
182       Assert.assertEquals(integ.getEvaluations(), calls);
183       Assert.assertTrue(calls <= previousCalls);
184       previousCalls = calls;
185 
186     }

243   @Test
244   public void testEvents()
245       throws DimensionMismatchException, NumberIsTooSmallException,
246              MaxCountExceededException, NoBracketingException {
247 
248     TestProblem4 pb = new TestProblem4();
249     double minStep = 0;
250     double maxStep = pb.getFinalTime() - pb.getInitialTime();
251     double scalAbsoluteTolerance = 1.0e-9;
252     double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
253 
254     FirstOrderIntegrator integ = new DormandPrince853Integrator(minStep, maxStep,
255                                                                 scalAbsoluteTolerance,
256                                                                 scalRelativeTolerance);
257     TestProblemHandler handler = new TestProblemHandler(pb, integ);
258     integ.addStepHandler(handler);
259     EventHandler[] functions = pb.getEventsHandlers();
260     double convergence = 1.0e-8 * maxStep;
261     for (int l = 0; l < functions.length; ++l) {
262       integ.addEventHandler(functions[l], Double.POSITIVE_INFINITY, convergence, 1000);
263     }
264     Assert.assertEquals(functions.length, integ.getEventHandlers().size());
265     integ.integrate(pb,
266                     pb.getInitialTime(), pb.getInitialState(),
267                     pb.getFinalTime(), new double[pb.getDimension()]);
268 
269     Assert.assertEquals(0, handler.getMaximalValueError(), 2.1e-7);
270     Assert.assertEquals(0, handler.getMaximalTimeError(), convergence);
271     Assert.assertEquals(12.0, handler.getLastTime(), convergence);
272     integ.clearEventHandlers();
273     Assert.assertEquals(0, integ.getEventHandlers().size());
274 
275   }

336   @Test
337   public void testEventsScheduling() {
338 
339       FirstOrderDifferentialEquations sincos = new FirstOrderDifferentialEquations() {
340 
341           public int getDimension() {
342               return 2;
343           }
344 
345           public void computeDerivatives(double t, double[] y, double[] yDot) {
346               yDot[0] =  y[1];
347               yDot[1] = -y[0];
348           }
349 
350       };
351 
352       SchedulingChecker sinChecker = new SchedulingChecker(0); // events at 0, PI, 2PI ...
353       SchedulingChecker cosChecker = new SchedulingChecker(1); // events at PI/2, 3PI/2, 5PI/2 ...
354 
355       FirstOrderIntegrator integ =
356               new DormandPrince853Integrator(0.001, 1.0, 1.0e-12, 0.0);
357       integ.addEventHandler(sinChecker, 0.01, 1.0e-7, 100);
358       integ.addStepHandler(sinChecker);
359       integ.addEventHandler(cosChecker, 0.01, 1.0e-7, 100);
360       integ.addStepHandler(cosChecker);
361       double   t0 = 0.5;
362       double[] y0 = new double[] { FastMath.sin(t0), FastMath.cos(t0) };
363       double   t  = 10.0;
364       double[] y  = new double[2];
365       integ.integrate(sincos, t0, y0, t, y);
366 
367   }

369   private static class SchedulingChecker implements StepHandler, EventHandler {
370 
371       int index;
372       double tMin;
373 
374       public SchedulingChecker(int index) {
375           this.index = index;
376       }
377 
378       public void init(double t0, double[] y0, double t) {
379           tMin = t0;
380       }
381 
382       public void handleStep(StepInterpolator interpolator, boolean isLast) {
383           tMin = interpolator.getCurrentTime();
384       }
385 
386       public double g(double t, double[]  y) {
387           // once a step has been handled by handleStep,
388           // events checking should only refer to dates after the step
389           Assert.assertTrue(t >= tMin);
390           return y[index];
391       }
392 
393       public Action eventOccurred(double t, double[] y, boolean increasing) {
394           return Action.RESET_STATE;
395       }
396 
397       public void resetState(double t, double[] y) {
398           // in fact, we don't need to reset anything for the test
399       }

403   private static class KeplerHandler implements StepHandler {
404     public KeplerHandler(TestProblem3 pb) {
405       this.pb = pb;
406     }
407     public void init(double t0, double[] y0, double t) {
408       nbSteps = 0;
409       maxError = 0;
410     }
411     public void handleStep(StepInterpolator interpolator, boolean isLast)
412         throws MaxCountExceededException {
413 
414       ++nbSteps;
415       for (int a = 1; a < 10; ++a) {
416 
417         double prev   = interpolator.getPreviousTime();
418         double curr   = interpolator.getCurrentTime();
419         double interp = ((10 - a) * prev + a * curr) / 10;
420         interpolator.setInterpolatedTime(interp);
421 
422         double[] interpolatedY = interpolator.getInterpolatedState ();
423         double[] theoreticalY  = pb.computeTheoreticalState(interpolator.getInterpolatedTime());
424         double dx = interpolatedY[0] - theoreticalY[0];
425         double dy = interpolatedY[1] - theoreticalY[1];
426         double error = dx * dx + dy * dy;
427         if (error > maxError) {
428           maxError = error;
429         }
430       }
431       if (isLast) {
432         Assert.assertTrue(maxError < 2.4e-10);
433         Assert.assertTrue(nbSteps < 150);
434       }
435     }
436     private int nbSteps;
437     private double maxError;
438     private TestProblem3 pb;
439   }

441   private static class VariableHandler implements StepHandler {
442     public VariableHandler() {
443         firstTime = true;
444         minStep = 0;
445         maxStep = 0;
446     }
447     public void init(double t0, double[] y0, double t) {
448       firstTime = true;
449       minStep = 0;
450       maxStep = 0;
451     }
452     public void handleStep(StepInterpolator interpolator,
453                            boolean isLast) {
454 
455       double step = FastMath.abs(interpolator.getCurrentTime()
456                              - interpolator.getPreviousTime());
457       if (firstTime) {
458         minStep   = FastMath.abs(step);
459         maxStep   = minStep;
460         firstTime = false;
461       } else {
462         if (step < minStep) {
463           minStep = step;
464         }
465         if (step > maxStep) {
466           maxStep = step;
467         }
468       }
469 
470       if (isLast) {
471         Assert.assertTrue(minStep < (1.0 / 100.0));
472         Assert.assertTrue(maxStep > (1.0 / 2.0));
473       }
474     }
475     private boolean firstTime = true;
476     private double  minStep = 0;
477     private double  maxStep = 0;
478   }
