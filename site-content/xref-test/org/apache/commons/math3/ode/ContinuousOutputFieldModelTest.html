42      private <T extends RealFieldElement<T>> void doTestBoundaries(final Field<T> field) {
43          TestFieldProblem3<T> pb = new TestFieldProblem3<T>(field, field.getZero().add(0.9));
44          double minStep = 0;
45          double maxStep = pb.getFinalTime().subtract(pb.getInitialState().getTime()).getReal();
46          FirstOrderFieldIntegrator<T> integ = new DormandPrince54FieldIntegrator<T>(field, minStep, maxStep, 1.0e-8, 1.0e-8);
47          integ.addStepHandler(new ContinuousOutputFieldModel<T>());
48          integ.integrate(new FieldExpandableODE<T>(pb), pb.getInitialState(), pb.getFinalTime());
49          ContinuousOutputFieldModel<T> cm = (ContinuousOutputFieldModel<T>) integ.getStepHandlers().iterator().next();
50          cm.getInterpolatedState(pb.getInitialState().getTime().multiply(2).subtract(pb.getFinalTime()));
51          cm.getInterpolatedState(pb.getFinalTime().multiply(2).subtract(pb.getInitialState().getTime()));
52          cm.getInterpolatedState(pb.getInitialState().getTime().add(pb.getFinalTime()).multiply(0.5));
53      }

55      @Test
56      public void testRandomAccess() {
57          doTestRandomAccess(Decimal64Field.getInstance());
58      }
59  
60      private <T extends RealFieldElement<T>> void doTestRandomAccess(final Field<T> field)  {
61  
62          TestFieldProblem3<T> pb = new TestFieldProblem3<T>(field, field.getZero().add(0.9));
63          double minStep = 0;
64          double maxStep = pb.getFinalTime().subtract(pb.getInitialState().getTime()).getReal();
65          FirstOrderFieldIntegrator<T> integ = new DormandPrince54FieldIntegrator<T>(field, minStep, maxStep, 1.0e-8, 1.0e-8);
66          ContinuousOutputFieldModel<T> cm = new ContinuousOutputFieldModel<T>();
67          integ.addStepHandler(cm);
68          integ.integrate(new FieldExpandableODE<T>(pb), pb.getInitialState(), pb.getFinalTime());
69  
70          Random random = new Random(347588535632l);
71          T maxError    = field.getZero();
72          T maxErrorDot = field.getZero();
73          for (int i = 0; i < 1000; ++i) {
74              double r = random.nextDouble();
75              T time = pb.getInitialState().getTime().multiply(r).add(pb.getFinalTime().multiply(1.0 - r));
76              FieldODEStateAndDerivative<T> interpolated = cm.getInterpolatedState(time);
77              T[] theoreticalY = pb.computeTheoreticalState(time);
78              T[] theoreticalYDot  = pb.doComputeDerivatives(time, theoreticalY);
79              T dx = interpolated.getState()[0].subtract(theoreticalY[0]);
80              T dy = interpolated.getState()[1].subtract(theoreticalY[1]);
81              T error = dx.multiply(dx).add(dy.multiply(dy));
82              maxError = MathUtils.max(maxError, error);
83              T dxDot = interpolated.getDerivative()[0].subtract(theoreticalYDot[0]);
84              T dyDot = interpolated.getDerivative()[1].subtract(theoreticalYDot[1]);
85              T errorDot = dxDot.multiply(dxDot).add(dyDot.multiply(dyDot));
86              maxErrorDot = MathUtils.max(maxErrorDot, errorDot);
87          }
88  
89          Assert.assertEquals(0.0, maxError.getReal(),    1.0e-9);
90          Assert.assertEquals(0.0, maxErrorDot.getReal(), 4.0e-7);
91  
92      }

94      @Test
95      public void testModelsMerging() {
96          doTestModelsMerging(Decimal64Field.getInstance());
97      }
98  
99      private <T extends RealFieldElement<T>> void doTestModelsMerging(final Field<T> field) {
100 
101         // theoretical solution: y[0] = cos(t), y[1] = sin(t)
102         FirstOrderFieldDifferentialEquations<T> problem =
103                         new FirstOrderFieldDifferentialEquations<T>() {
104             public T[] computeDerivatives(T t, T[] y) {
105                 T[] yDot = MathArrays.buildArray(field, 2);
106                 yDot[0] = y[1].negate();
107                 yDot[1] = y[0];
108                 return yDot;
109             }
110             public int getDimension() {
111                 return 2;
112             }
113             public void init(T t0, T[] y0, T finalTime) {
114             }
115         };
116 
117         // integrate backward from &pi; to 0;
118         ContinuousOutputFieldModel<T> cm1 = new ContinuousOutputFieldModel<T>();
119         FirstOrderFieldIntegrator<T> integ1 =
120                         new DormandPrince853FieldIntegrator<T>(field, 0, 1.0, 1.0e-8, 1.0e-8);
121         integ1.addStepHandler(cm1);
122         T t0 = field.getZero().add(FastMath.PI);
123         T[] y0 = MathArrays.buildArray(field, 2);
124         y0[0] = field.getOne().negate();
125         y0[1] = field.getZero();
126         integ1.integrate(new FieldExpandableODE<T>(problem),
127                          new FieldODEState<T>(t0, y0),
128                          field.getZero());
129 
130         // integrate backward from 2&pi; to &pi;
131         ContinuousOutputFieldModel<T> cm2 = new ContinuousOutputFieldModel<T>();
132         FirstOrderFieldIntegrator<T> integ2 =
133                         new DormandPrince853FieldIntegrator<T>(field, 0, 0.1, 1.0e-12, 1.0e-12);
134         integ2.addStepHandler(cm2);
135         t0 = field.getZero().add(2.0 * FastMath.PI);
136         y0[0] = field.getOne();
137         y0[1] = field.getZero();
138         integ2.integrate(new FieldExpandableODE<T>(problem),
139                          new FieldODEState<T>(t0, y0),
140                          field.getZero().add(FastMath.PI));
141 
142         // merge the two half circles
143         ContinuousOutputFieldModel<T> cm = new ContinuousOutputFieldModel<T>();
144         cm.append(cm2);
145         cm.append(new ContinuousOutputFieldModel<T>());
146         cm.append(cm1);
147 
148         // check circle
149         Assert.assertEquals(2.0 * FastMath.PI, cm.getInitialTime().getReal(), 1.0e-12);
150         Assert.assertEquals(0, cm.getFinalTime().getReal(), 1.0e-12);
151         for (double t = 0; t < 2.0 * FastMath.PI; t += 0.1) {
152             FieldODEStateAndDerivative<T> interpolated = cm.getInterpolatedState(field.getZero().add(t));
153             Assert.assertEquals(FastMath.cos(t), interpolated.getState()[0].getReal(), 1.0e-7);
154             Assert.assertEquals(FastMath.sin(t), interpolated.getState()[1].getReal(), 1.0e-7);
155         }
156 
157     }

159     @Test
160     public void testErrorConditions() {
161         doTestErrorConditions(Decimal64Field.getInstance());
162     }
163 
164     private <T extends RealFieldElement<T>> void doTestErrorConditions(final Field<T> field) {
165         ContinuousOutputFieldModel<T> cm = new ContinuousOutputFieldModel<T>();
166         cm.handleStep(buildInterpolator(field, 0, 1, new double[] { 0.0, 1.0, -2.0 }), true);
167 
168         // dimension mismatch
169         Assert.assertTrue(checkAppendError(field, cm, 1.0, 2.0, new double[] { 0.0, 1.0 }));
170 
171         // hole between time ranges
172         Assert.assertTrue(checkAppendError(field, cm, 10.0, 20.0, new double[] { 0.0, 1.0, -2.0 }));
173 
174         // propagation direction mismatch
175         Assert.assertTrue(checkAppendError(field, cm, 1.0, 0.0, new double[] { 0.0, 1.0, -2.0 }));
176 
177         // no errors
178         Assert.assertFalse(checkAppendError(field, cm, 1.0, 2.0, new double[] { 0.0, 1.0, -2.0 }));
179 
180     }

182     private <T extends RealFieldElement<T>> boolean checkAppendError(Field<T> field, ContinuousOutputFieldModel<T> cm,
183                                                                      double t0, double t1, double[] y) {
184         try {
185             ContinuousOutputFieldModel<T> otherCm = new ContinuousOutputFieldModel<T>();
186             otherCm.handleStep(buildInterpolator(field, t0, t1, y), true);
187             cm.append(otherCm);
188         } catch(IllegalArgumentException iae) {
189             return true; // there was an allowable error
190         }
191         return false; // no allowable error
192     }

194     private <T extends RealFieldElement<T>> FieldStepInterpolator<T> buildInterpolator(Field<T> field,
195                                                                                        double t0, double t1, double[] y) {
196         T[] fieldY = MathArrays.buildArray(field, y.length);
197         for (int i = 0; i < y.length; ++i) {
198             fieldY[i] = field.getZero().add(y[i]);
199         }
200         final FieldODEStateAndDerivative<T> s0 = new FieldODEStateAndDerivative<T>(field.getZero().add(t0), fieldY, fieldY);
201         final FieldODEStateAndDerivative<T> s1 = new FieldODEStateAndDerivative<T>(field.getZero().add(t1), fieldY, fieldY);
202         final FieldEquationsMapper<T> mapper   = new FieldExpandableODE<T>(new FirstOrderFieldDifferentialEquations<T>() {
203             public int getDimension() {
204                 return s0.getStateDimension();
205             }
206             public void init(T t0, T[] y0, T finalTime) {
207             }
208             public T[] computeDerivatives(T t, T[] y) {
209                 return y;
210             }
211         }).getMapper();
212         return new DummyFieldStepInterpolator<T>(t1 >= t0, s0, s1, s0, s1, mapper);
213     }

215     public void checkValue(double value, double reference) {
216         Assert.assertTrue(FastMath.abs(value - reference) < 1.0e-10);
217     }
