117     @Test
118     public void testTrivial() {
119         LinearProblem problem
120             = new LinearProblem(new double[][] { { 2 } }, new double[] { 3 });
121         JacobianMultivariateVectorOptimizer underlyingOptimizer
122             = new GaussNewtonOptimizer(true, new SimpleVectorValueChecker(1e-6, 1e-6));
123         JDKRandomGenerator g = new JDKRandomGenerator();
124         g.setSeed(16069223052l);
125         RandomVectorGenerator generator
126             = new UncorrelatedRandomVectorGenerator(1, new GaussianRandomGenerator(g));
127         MultiStartMultivariateVectorOptimizer optimizer
128             = new MultiStartMultivariateVectorOptimizer(underlyingOptimizer, 10, generator);
129 
130         PointVectorValuePair optimum
131             = optimizer.optimize(new MaxEval(100),
132                                  problem.getModelFunction(),
133                                  problem.getModelFunctionJacobian(),
134                                  problem.getTarget(),
135                                  new Weight(new double[] { 1 }),
136                                  new InitialGuess(new double[] { 0 }));
137         Assert.assertEquals(1.5, optimum.getPoint()[0], 1e-10);
138         Assert.assertEquals(3.0, optimum.getValue()[0], 1e-10);
139         PointVectorValuePair[] optima = optimizer.getOptima();
140         Assert.assertEquals(10, optima.length);
141         for (int i = 0; i < optima.length; i++) {
142             Assert.assertEquals(1.5, optima[i].getPoint()[0], 1e-10);
143             Assert.assertEquals(3.0, optima[i].getValue()[0], 1e-10);
144         }
145         Assert.assertTrue(optimizer.getEvaluations() > 20);
146         Assert.assertTrue(optimizer.getEvaluations() < 50);
147         Assert.assertEquals(100, optimizer.getMaxEvaluations());
148     }

150     @Test
151     public void testIssue914() {
152         LinearProblem problem = new LinearProblem(new double[][] { { 2 } }, new double[] { 3 });
153         JacobianMultivariateVectorOptimizer underlyingOptimizer =
154                 new GaussNewtonOptimizer(true, new SimpleVectorValueChecker(1e-6, 1e-6)) {
155             @Override
156             public PointVectorValuePair optimize(OptimizationData... optData) {
157                 // filter out simple bounds, as they are not supported
158                 // by the underlying optimizer, and we don't really care for this test
159                 OptimizationData[] filtered = optData.clone();
160                 for (int i = 0; i < filtered.length; ++i) {
161                     if (filtered[i] instanceof SimpleBounds) {
162                         filtered[i] = null;
163                     }
164                 }
165                 return super.optimize(filtered);
166             }
167         };
168         JDKRandomGenerator g = new JDKRandomGenerator();
169         g.setSeed(16069223052l);
170         RandomVectorGenerator generator =
171                 new UncorrelatedRandomVectorGenerator(1, new GaussianRandomGenerator(g));
172         MultiStartMultivariateVectorOptimizer optimizer =
173                 new MultiStartMultivariateVectorOptimizer(underlyingOptimizer, 10, generator);
174 
175         optimizer.optimize(new MaxEval(100),
176                            problem.getModelFunction(),
177                            problem.getModelFunctionJacobian(),
178                            problem.getTarget(),
179                            new Weight(new double[] { 1 }),
180                            new InitialGuess(new double[] { 0 }),
181                            new SimpleBounds(new double[] { -1.0e-10 }, new double[] {  1.0e-10 }));
182         PointVectorValuePair[] optima = optimizer.getOptima();
183         // only the first start should have succeeded
184         Assert.assertEquals(1, optima.length);
185 
186     }

217     private static class LinearProblem {
218         private final RealMatrix factors;
219         private final double[] target;
220 
221         public LinearProblem(double[][] factors,
222                              double[] target) {
223             this.factors = new BlockRealMatrix(factors);
224             this.target  = target;
225         }
226 
227         public Target getTarget() {
228             return new Target(target);
229         }
230 
231         public ModelFunction getModelFunction() {
232             return new ModelFunction(new MultivariateVectorFunction() {
233                     public double[] value(double[] variables) {
234                         return factors.operate(variables);
235                     }
236                 });
237         }
238 
239         public ModelFunctionJacobian getModelFunctionJacobian() {
240             return new ModelFunctionJacobian(new MultivariateMatrixFunction() {
241                     public double[][] value(double[] point) {
242                         return factors.getData();
243                     }
244                 });
245         }
