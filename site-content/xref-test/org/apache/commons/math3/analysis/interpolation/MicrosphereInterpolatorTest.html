17  package org.apache.commons.math3.analysis.interpolation;
18  
19  import org.apache.commons.math3.analysis.MultivariateFunction;
20  import org.apache.commons.math3.util.FastMath;
21  import org.junit.Assert;
22  import org.junit.Test;

24  /**
25   * Test case for the "microsphere projection" interpolator.
26   *
27   * @deprecated Code will be removed in 4.0.  Use {@link InterpolatingMicrosphere}
28   * and {@link MicrosphereProjectionInterpolator} instead.
29   */
30  @Deprecated
31  public final class MicrosphereInterpolatorTest {
32      /**
33       * Test of interpolator for a plane.
34       * <p>
35       * y = 2 x<sub>1</sub> - 3 x<sub>2</sub> + 5
36       */
37      @Test
38      public void testLinearFunction2D() {
39          MultivariateFunction f = new MultivariateFunction() {
40                  public double value(double[] x) {
41                      if (x.length != 2) {
42                          throw new IllegalArgumentException();
43                      }
44                      return 2 * x[0] - 3 * x[1] + 5;
45                  }
46              };
47  
48          MultivariateInterpolator interpolator = new MicrosphereInterpolator();
49  
50          // Interpolating points in [-1, 1][-1, 1] by steps of 1.
51          final int n = 9;
52          final int dim = 2;
53          double[][] x = new double[n][dim];
54          double[] y = new double[n];
55          int index = 0;
56          for (int i = -1; i <= 1; i++) {
57              for (int j = -1; j <= 1; j++) {
58                  x[index][0] = i;
59                  x[index][1] = j;
60                  y[index] = f.value(x[index]);
61                  ++index;
62              }
63          }
64  
65          MultivariateFunction p = interpolator.interpolate(x, y);
66  
67          double[] c = new double[dim];
68          double expected, result;
69  
70          c[0] = 0;
71          c[1] = 0;
72          expected = f.value(c);
73          result = p.value(c);
74          Assert.assertEquals("On sample point", expected, result, FastMath.ulp(1d));
75  
76          c[0] = 0 + 1e-5;
77          c[1] = 1 - 1e-5;
78          expected = f.value(c);
79          result = p.value(c);
80          Assert.assertEquals("1e-5 away from sample point", expected, result, 1e-4);
81      }
82  
83      /**
84       * Test of interpolator for a quadratic function.
85       * <p>
86       * y = 2 x<sub>1</sub><sup>2</sup> - 3 x<sub>2</sub><sup>2</sup>
87       *     + 4 x<sub>1</sub> x<sub>2</sub> - 5
88       */
89      @Test
90      public void testParaboloid2D() {
91          MultivariateFunction f = new MultivariateFunction() {
92                  public double value(double[] x) {
93                      if (x.length != 2) {
94                          throw new IllegalArgumentException();
95                      }
96                      return 2 * x[0] * x[0] - 3 * x[1] * x[1] + 4 * x[0] * x[1] - 5;
97                  }
98              };
99  
100         MultivariateInterpolator interpolator = new MicrosphereInterpolator();
101 
102         // Interpolating points in [-10, 10][-10, 10] by steps of 2.
103         final int n = 121;
104         final int dim = 2;
105         double[][] x = new double[n][dim];
106         double[] y = new double[n];
107         int index = 0;
108         for (int i = -10; i <= 10; i += 2) {
109             for (int j = -10; j <= 10; j += 2) {
110                 x[index][0] = i;
111                 x[index][1] = j;
112                 y[index] = f.value(x[index]);
113                 ++index;
114             }
115         }
116 
117         MultivariateFunction p = interpolator.interpolate(x, y);
118 
119         double[] c = new double[dim];
120         double expected, result;
121 
122         c[0] = 0;
123         c[1] = 0;
124         expected = f.value(c);
125         result = p.value(c);
126         Assert.assertEquals("On sample point", expected, result, FastMath.ulp(1d));
127 
128         c[0] = 2 + 1e-5;
129         c[1] = 2 - 1e-5;
130         expected = f.value(c);
131         result = p.value(c);
132         Assert.assertEquals("1e-5 away from sample point", expected, result, 1e-3);
133     }
